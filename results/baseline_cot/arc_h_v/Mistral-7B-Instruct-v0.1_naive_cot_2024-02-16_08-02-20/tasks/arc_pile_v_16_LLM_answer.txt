def transform(input_grid, output_grid):
    # Perform the transformation on each element of the input grid
    for i in range(len(input_grid)):
        for j in range(len(input_grid[i])):
            if input_grid[i][j] == 0:
                output_grid[i][j] = 0
            else:
                # Check if the current pixel has any touching objects
                touching_objects = []
                for k in range(-1, 2):
                    for l in range(-1, 2):
                        if k!= 0 or l!= 0:
                            x = i + k
                            y = j + l
                            if x >= 0 and x < len(input_grid) and y >= 0 and y < len(input_grid[x]) and input_grid[x][y]!= 0:
                                touching_objects.append((x, y))
                # If there are no touching objects, set the output pixel to the input pixel
                if len(touching_objects) == 0:
                    output_grid[i][j] = input_grid[i][j]
                # Otherwise, set the output pixel to the most common color among the touching objects
                else:
                    colors = {}
                    for obj in touching_objects:
                        colors[input_grid[obj[0]][obj[1]]] = (colors.get(input_grid[obj[0]][obj[1]], 0) + 1) // len(touching_objects)
                    output_grid[i][j] = max(colors, key=colors.get)
