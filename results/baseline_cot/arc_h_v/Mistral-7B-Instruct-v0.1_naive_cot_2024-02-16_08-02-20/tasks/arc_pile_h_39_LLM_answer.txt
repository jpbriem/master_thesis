    # Iterate over each row of the input grid
    for i in range(input_dim):
        # Initialize variables for counting and sorting
        count = 0
        sorted_values = []

            # Check if the current pixel has a value greater than 0 (i.e., it's not black)

                # Increment the count variable
                count += 1
                
                # Append the current pixel value to the sorted_values list
                sorted_values.append(input_grid[i][j])
                
        # If there are no non-black pixels in this row, set the output value to 0
        if count == 0:
            output_grid[i][j] = 0
