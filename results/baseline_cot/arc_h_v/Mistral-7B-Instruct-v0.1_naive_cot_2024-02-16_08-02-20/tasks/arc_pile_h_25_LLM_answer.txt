import numpy as np

def transform(input_grid, output_grid):
    # Get the dimensions of the input and output grids
    input_dim = len(input_grid)
    output_dim = len(output_grid)
    
    # Create an empty output grid with zeros
    output_array = np.zeros((output_dim, output_dim))
    
    # Iterate over each row of the input grid
    for i in range(input_dim):
        # Get the current row of the input grid
        row = input_grid[i]
        
        # Initialize variables for counting and sorting
        count = 0
        sorted_row = []
        
        # Iterate over each element of the row
        for j in range(len(row)):
            # If the element is not zero (background), increment the count and add it to the sorted row
            if row[j]!= 0:
                count += 1
                sorted_row.append(row[j])
                
        # If there are no non-zero elements in the row, set the output value to zero
        if count == 0:
            output_array[i][j] = 0
        else:
            # Otherwise, calculate the average of the non-zero elements and round it to the nearest integer
            avg = int(np.mean(sorted_row))
            
            # Set the output value to the average rounded to the nearest integer
            output_array[i][j] = avg
    
    return output_array
