import numpy as np

def transform(input_grid, output_grid):
    # Perform the transformation on each element of the input grid
    for i in range(len(input_grid)):
        for j in range(len(input_grid[i])):
            if input_grid[i][j] == 0:
                output_grid[i][j] = 0
            else:
                # Check if the current pixel has any neighboring pixels that are also non-zero
                neighbors = []
                if i > 0 and input_grid[i-1][j]!= 0:
                    neighbors.append((i-1, j))
                if j > 0 and input_grid[i][j-1]!= 0:
                    neighbors.append((i, j-1))
                if i < len(input_grid)-1 and input_grid[i+1][j]!= 0:
                    neighbors.append((i+1, j))
                if j < len(input_grid[i])-1 and input_grid[i][j+1]!= 0:
                    neighbors.append((i, j+1))
                # If there are no neighbors, set the output pixel to the current pixel value
                if len(neighbors) == 0:
                    output_grid[i][j] = input_grid[i][j]
                # Otherwise, calculate the average of the neighboring pixels and round it to the nearest integer
                else:
                    avg = sum([input_grid[n[0]][n[1]] for n in neighbors]) / len(neighbors)
                    output_grid[i][j] = int(round(avg))
