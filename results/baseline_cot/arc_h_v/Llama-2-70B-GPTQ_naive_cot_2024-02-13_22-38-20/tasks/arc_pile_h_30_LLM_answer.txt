    neighbors = []

            if ((x+dx >= 0) & (y+dy >= 0) & (x+dx < len(arr)) & (y+dy < len(arr[0]) ) & (arr[x+dx][y+dy]!= 0)):
                neighbors.append((x+dx, y+dy))
                
    return neighbors

def update_objects(objects, arr):
