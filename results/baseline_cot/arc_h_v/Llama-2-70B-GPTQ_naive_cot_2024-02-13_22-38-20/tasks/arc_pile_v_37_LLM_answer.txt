def get_objects(img):
    # Get all nonzero elements
    nonzeros = np.nonzero(img)
    # Find unique colors
    colors = set([i for i in range(len(np.unique(img))) if i!= 0])
    # Initialize empty lists
    objects = []
    coords = []
    sizes = []
    # Loop through each color
    for color in colors:
        # Get indices of current color
        inds = np.where(img == color)[0]
        # Check if there's more than one element
        if len(inds) > 1:
            # Add new object
            objects.append({"color": color})
