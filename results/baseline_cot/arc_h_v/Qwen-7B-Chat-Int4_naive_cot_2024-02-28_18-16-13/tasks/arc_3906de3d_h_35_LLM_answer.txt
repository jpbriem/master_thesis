                    if (dx != 0 and dy != 0) or (dx == 0 and dy == 0):

                        y = col + dy
                        if 0 <= x < rows and 0 <= y < cols:
                            if input[x][y] == input[row][col]:

            # If the current pixel matches the desired value, set it to 1 in the result grid
