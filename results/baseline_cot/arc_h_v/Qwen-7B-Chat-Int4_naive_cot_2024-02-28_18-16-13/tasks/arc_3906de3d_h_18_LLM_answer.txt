def transform(input):
    # Step 1: Sort objects by their size
    sorted_objects = sorted(input, key=lambda x: x[6])

    # Step 2: Identify connected objects and sort them together
    groups = []
    current_group = []
    for obj in sorted_objects:
        if len(current_group) > 0 and (obj[0] == current_group[-1][0] or obj[1] == current_group[-1][1]):
            current_group.append(obj)
        else:
            groups.append(current_group)
            current_group = [obj]

    # Step 3: Remove any isolated objects
    for group in groups:
        if len(group) == 1:
            input.remove(group[0])

    # Step 4: Create new objects from merged groups
    for i in range(len(groups)-1):
        for j in range(i+1, len(groups)):
            merged_obj = [(min(groups[i][0], groups[j][0]), max(groups[i][1], groups[j][1]))]
            merged_obj.extend(groups[i])
            merged_obj.extend(groups[j])
            input.append(merged_obj)

    return input
