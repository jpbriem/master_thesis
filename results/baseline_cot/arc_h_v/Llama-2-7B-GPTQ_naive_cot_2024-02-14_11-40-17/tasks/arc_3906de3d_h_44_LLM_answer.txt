def example_1(input):
    # Partially inspired by https://github.com/matthewrockliff/pygame-examples/blob/master/src/examples/06_transformations.py
    
    # Input
    width = len(input)
    height = len(input[0])
    pixel_count = width * height
    pixel_colors = []
    for i in range(width):
        pixel_colors += input[i][:]
        
    # Output
    output = np.zeros((height, width), dtype=np.uint8)
    for y in range(height):
        for x in range(width):
            if (x + y) % 2 == 0:
                continue
            
            # Get the current pixel's color
            pixel_color = pixel_colors[(y - 1) * width + x]
                
            # Determine the new color
            if pixel_color < 4:
                output[y, x] = 0
            elif pixel_color > 7:
                output[y, x] = 255
            else:
                # If the pixel has an odd value, it will be flipped horizontally
                if pixel_color & 1:
                    output[y, x] = 255
                else:
                    output[y, x] = 0
                
    return output
