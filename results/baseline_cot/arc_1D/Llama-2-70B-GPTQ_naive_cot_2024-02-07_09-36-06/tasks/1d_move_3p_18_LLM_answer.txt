from PIL import Image
import matplotlib.pyplot as plt

def get_image(img):

def get_array(img):

def get_pixels(arr):
    pix = arr[::-1].flatten()
    return pix

def get_noise(pix):
    noise = []
    for i in range(len(pix)):
        if (i+1) % 8 == 0:
            noise.append([])
        noise[-1].append(pix[i])
    return noise

def get_objects(noise):
    obj = {}
    for i in range(len(noise)-1):
        if len(noise[i]) > 1:
            obj['Object_{}'.format(i)] = {
               'start_index' : i*8 + 1,
                'end_index'   : i*8 + len(noise[i]),
               'size'        : len(noise[i]),
                'color'       : noise[i][0]}
    return obj

def get_relationships(obj):
    rel = {}
    for k in sorted(list(obj.keys()))[:-1]:
        v = obj[k]['color']
        for l in sorted(list(obj.keys()))[1:]:
            w = obj[l]['color']
            if v!= w:
                rel[k+'-'+l] = {
                    'distance'     : abs(int(k.split('_')[1]) - int(l.split('_')[1])) * 8,
                    'alignment'    : 'horizontal'}
    return rel

def get_conditional(rel):

    for k in sorted(list(rel.keys()))[:-1]:
        v = rel[k]['distance']
        for l in sorted(list(rel.keys()))[1:]:
            w = rel[l]['distance']

                cond[k+'-'+l] = {

                    'comparison'     : v,
                   'reference'      : w}
    return cond

def get_actions(cond):
    act = {}
    for k in sorted(list(cond.keys()))[:-1]:
        v = cond[k]['condition']
        for l in sorted(list(cond.keys()))[1:]:
            w = cond[l]['condition']

                    'action'         :'move',

                    'target'         : cond[l]['reference']}
    return act

    res = {}
    for k in sorted(list(act.keys()))[:-1]:
        v = act[k]['action']
        for l in sorted(list(act.keys()))[1:]:
            w = act[l]['action']

def get_final(res):
    fin = {}
    for k in sorted(list(res.keys()))[:-1]:
        v = res[k]['result']

        v = fin[k]['final']

                seq += [fin[k]['target']]
    return seq

        if (i+1) % 8 == 0:
            out.append([])
        out[-1].append(seq[i])
    return out

    img = get_image("./images/task_1.png")
