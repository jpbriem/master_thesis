def get_pixel(img):

def get_size(img):

def get_neighbours(img, x, y):
    neighbors = []
    if (x > 0):
        neighbors.append((x - 1, y))
    if (y > 0):
        neighbors.append((x, y - 1))
    if ((x < len(img)-1)):
        neighbors.append((x + 1, y))
    if ((y < len(img[0])-1)):
        neighbors.append((x, y + 1))
    return neighbors

def get_adjacent_pixels(img, x, y):
    adjacentPixels = []
    for neighbor in get_neighbours(img, x, y):
        if (get_pixel(img) == get_pixel(img, neighbor[0], neighbor[1])):
            adjacentPixels.append(neighbor)
    return adjacentPixels

def get_connected_components(img, x, y):
    connectedComponents = []
    currentComponent = [(x, y)]
    while True:
        nextPixel = None
        for neighbor in get_neighbours(img, currentComponent[-1][0], currentComponent[-1][1]):
            if (get_pixel(img, neighbor[0], neighbor[1]) == get_pixel(img, currentComponent[-1][0], currentComponent[-1][1])):
                nextPixel = neighbor
                break
        if (nextPixel!= None):
            currentComponent.append(nextPixel)
        else:
            connectedComponents.append(currentComponent)
            currentComponent = []
    return connectedComponents

def get_largest_component(img, components):
    largestComponent = None
    maxSize = 0
    for component in components:
        if (len(component) > maxSize):
            largestComponent = component
            maxSize = len(component)
    return largestComponent

def get_smaller_components(img, components):

    for component in components:

def get_noisy_pixels(img, components):
    noisyPixels = []
