def create_examples(train, target):
    '''
    Creates examples from the given training data.
    
    Parameters:
        - train (list): list of training inputs.
        - target (list): list of target outputs.
        
    Returns:
        - X (numpy.ndarray): matrix of shape (n_samples, n_features).
        - y (numpy.ndarray): vector of shape (n_samples, ).
    '''
    X = []
