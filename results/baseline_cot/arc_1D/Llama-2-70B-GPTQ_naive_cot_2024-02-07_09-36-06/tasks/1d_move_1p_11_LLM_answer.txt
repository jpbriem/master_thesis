def get_connected_components(img, x, y, radius=1):
    connected_component = [[x, y]]
    visited = set()
    queue = [(x, y)]
    while len(queue)!= 0:
        current_node = queue.pop(0)
        if (current_node not in visited):
            visited.add(current_node)
