def create_mask(img, mask_size=None):
    '''
    Creates a binary mask out of an image.
    
    Parameters:
        img (numpy.ndarray): Image.
        mask_size (int, optional): Size of the mask. Defaults to None.
    
    Returns:
        - mask (numpy.ndarray): Binary mask.
    '''
    mask = np.zeros((img.shape[0], img.shape[1]), dtype='uint8')
    if mask_size!= None:
        mask[:mask_size, :mask_size] = 1
    else:
        mask[:,:] = 1
    return mask

    '''
    Calculates center point of an object.
    
    Parameters:
        bp (list): Boundary points.
    
    Returns:
        - cp (tuple): Center point.
    '''
    cx = sum([x for (x,y) in bp]) / len(bp)
    cy = sum([y for (x,y) in bp]) / len(bp)
