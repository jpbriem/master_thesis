def transform_grid(input_grid, output_grid):

    input_dim = len(input_grid)

        if np.any(input_grid[i]!= 0):

            start_idx = np.argmax(np.nonzero(input_grid[i]))

            # Repeat the non-zero elements until they fill the entire row
            while len(non_zero_elements) < output_dim:
