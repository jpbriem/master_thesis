def get_neighbors(matrix):
    neighbors = []
    for row in range(len(matrix)):
        for col in range(len(matrix[row])):
            # Get the current value at that position
            val = matrix[row][col]
            # Check all the neighboring positions
            for r in [-1, 0, 1]:
                for c in [-1, 0, 1]:
                    # Skip the current position
                    if r == 0 and c == 0:
                        continue
                    # Make sure we don't go out of bounds
                    if not (0 <= row + r < len(matrix)) or \
                       not (0 <= col + c < len(matrix[row])):
                        continue
                    # Add the neighboring values to our list
                    neighbors.append((val, matrix[row+r][col+c]))
    return neighbors
