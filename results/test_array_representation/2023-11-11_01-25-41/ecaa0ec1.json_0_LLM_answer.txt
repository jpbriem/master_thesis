```
import numpy as np

# Create a random array with some zeros and ones
arr = np.random.randint(0, 2, size=(5, 5))

non_zeros = np.where(arr!= 0)

# Use those indices to index into the original array
new_array = arr[non_zeros]
print(new_array)
