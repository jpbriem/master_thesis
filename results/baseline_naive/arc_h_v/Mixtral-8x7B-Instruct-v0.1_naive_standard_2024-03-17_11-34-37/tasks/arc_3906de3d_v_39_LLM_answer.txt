You can use a recursive function to find all paths from top-left corner to bottom right. If you reach an obstacle or go out of bounds then return False. Otherwise if you reached the end point then return True and also keep track of maximum depth so far. In the main part of the code just check whether there exists any path that reaches the end point with max depth greater than current depth. Here's how it looks like in Python:

```
def dfs(grid, i, j, n, m, curr_depth):
    if i == n - 1 and j == m - 1:
        global max_depth
        max_depth = max(max_depth, curr_depth + grid[i][j])
        return True
    elif i >= n or j >= m or grid[i][j] == 0:
        return False
    else:
        up = dfs(grid, i + 1, j, n, m, curr_depth + grid[i][j])
        down = dfs(grid, i, j + 1, n, m, curr_depth + grid[i][j])
        left = dfs(grid, i + 1, j, n, m, curr_depth + grid[i][j])
        right = dfs(grid, i, j + 1, n, m, curr_depth + grid[i][j])
        return up or down or left or right

def solve(grid):
    n, m = len(grid), len(grid[0])
    max_depth = 0
    for i in range(n):
        for j in range(m):
            if grid[i][j]!= 0 and dfs(grid, i, j, n, m, 0):
                grid[i][j] += max_depth
                break
    return grid
```
