void printBoard(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		cout << "[";
		for (int j = 0; j < n; ++j) {

		}

}

bool isValid(int** board, int x, int y, int num, int n) {
	// check row and column
	for (int i = 0; i < n; ++i) {
		if (board[x][i] == num || board[i][y] == num) return false;
	}

	// check box
	int startX = x / 3 * 3;
	int startY = y / 3 * 3;
	for (int i = startX; i < startX + 3; ++i) {
		for (int j = startY; j < startY + 3; ++j) {
			if (board[i][j] == num) return false;
		}
	}

	return true;
}

bool solveSudoku(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			if (board[i][j]!= 0 && board[i][j]!= -1) continue; // already filled or empty cell

			for (int k = 1; k <= n; ++k) {
				if (!isValid(board, i, j, k, n)) continue;

				board[i][j] = k;
				if (solveSudoku(board, n)) return true;
				board[i][j] = 0;
			}

			return false;
		}
	}

	return true;
}

void fillEmptyCell(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			if (board[i][j] == 0) board[i][j] = -1;
		}
	}
}

void convertToOriginal(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			if (board[i][j] > 9) board[i][j] -= 9;
		}
	}
}
