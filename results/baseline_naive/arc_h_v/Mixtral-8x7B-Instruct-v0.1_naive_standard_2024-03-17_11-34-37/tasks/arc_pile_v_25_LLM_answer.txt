public class MoveZerosToEnd {
    public static void main(String[] args) {
        int[][] arr = {{5, 7, 0, 0}, {0, 0, 0, 3}, {0, 7, 2, 3}, {5, 7, 0, 3}};
        moveZeroesToEnd(arr);
        printArray(arr);
    }

    private static void moveZeroesToEnd(int[][] arr) {
        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[i].length - 1; j++) {
                if (arr[i][j] == 0 && arr[i][j + 1]!= 0) {
                    swap(arr, j, j + 1);
                }
            }
        }
    }

    private static void swap(int[][] arr, int firstIndex, int secondIndex) {
        int temp = arr[firstIndex][secondIndex];
        arr[firstIndex][secondIndex] = arr[firstIndex][firstIndex];
        arr[firstIndex][firstIndex] = temp;
    }

    private static void printArray(int[][] arr) {
        System.out.println("--------------------");
        for (int[] a : arr) {
            System.out.print("[ ");
            for (int b : a) {
                System.out.print(b + " ");
            }
            System.out.println("]");
        }
        System.out.println("--------------------");
    }
}
