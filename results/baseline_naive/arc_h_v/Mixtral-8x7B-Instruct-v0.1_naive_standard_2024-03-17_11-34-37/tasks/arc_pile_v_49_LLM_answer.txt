public class Solution {
    public int[][] Fill(int[][] matrix) {
        if (matrix == null || matrix.Length == 0 || matrix[0].Length == 0) return new int[0][];

        for (int i = 0; i < matrix.Length; ++i) {
            bool hasZeroInRow = false;
            for (int j = 0; j < matrix[0].Length; ++j) {
                if (matrix[i][j] == 0) {
                    hasZeroInRow = true;
                    continue;
                }
                if (hasZeroInRow) {
                    matrix[i][j] = 0;
                }
            }
        }

        for (int j = 0; j < matrix[0].Length; ++j) {
            bool hasZeroInCol = false;
            for (int i = 0; i < matrix.Length; ++i) {
                if (matrix[i][j] == 0) {
                    hasZeroInCol = true;
                    continue;
                }
                if (hasZeroInCol) {
                    matrix[i][j] = 0;
                }
            }
        }

        return matrix;
    }
}
