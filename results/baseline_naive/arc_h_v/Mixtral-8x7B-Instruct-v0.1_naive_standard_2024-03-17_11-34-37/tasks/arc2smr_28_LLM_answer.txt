bool isValid(int x, int y, int n) {
	if (x >= 0 && x < n && y >= 0 && y < n) return true;
	return false;
}

void solve(int** board, int n, int m, int x, int y) {
	// base condition
	if (m == 0 ||!isValid(x, y, n)) return;

	// if current cell has mine then mark it and return
	if (board[x][y] == -1) {
		board[x][y] = 9;
		return;
	}

	// calculate number of mines in adjacent cells
	int count = 0;
	for (int dx = -1; dx <= 1; ++dx) {
		for (int dy = -1; dy <= 1; ++dy) {
			int adjX = x + dx;
			int adjY = y + dy;
			if (!isValid(adjX, adjY, n) || board[adjX][adjY]!= -1) continue;
			++count;
		}
	}

	// if no mine found then just return
	if (count == 0) {
		board[x][y] = 0;
		for (int dx = -1; dx <= 1; ++dx) {
			for (int dy = -1; dy <= 1; ++dy) {
				int adjX = x + dx;
				int adjY = y + dy;
				if (!isValid(adjX, adjY, n) || board[adjX][adjY]!= -1) continue;
				solve(board, n, m-1, adjX, adjY);
			}
		}
	} else {
		board[x][y] = count;
	}
}

void clearMines(int** board, int n, int m) {

		for (int j = 0; j < n; ++j) {

			}
		}
	}
}

void placeMines(int** board, int n, int m) {

}
