// Time complexity : O(m*n), Space complexity : O(m*n)
void floodFill(int **arr, int m, int n, int x, int y, int prevColor, int newColor) {
    if (x<0 || x>=m || y<0 || y>=n || arr[x][y]!= prevColor) return;

    arr[x][y] = newColor;
    floodFill(arr, m, n, x+1, y, prevColor, newColor);
    floodFill(arr, m, n, x-1, y, prevColor, newColor);
    floodFill(arr, m, n, x, y+1, prevColor, newColor);
    floodFill(arr, m, n, x, y-1, prevColor, newColor);
}
