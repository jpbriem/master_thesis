def find(grid, x, y, target):
    if not grid[x][y]: return False
    if grid[x][y] == target: return True
    temp = grid[x][y]
    grid[x][y] = 0
    res = False
    if x > 0: res |= find(grid, x-1, y, target - temp)
    if x < len(grid)-1: res |= find(grid, x+1, y, target - temp)
    if y > 0: res |= find(grid, x, y-1, target - temp)
    if y < len(grid[0])-1: res |= find(grid, x, y+1, target - temp)
    grid[x][y] = temp
    return res

def solve(grid):
