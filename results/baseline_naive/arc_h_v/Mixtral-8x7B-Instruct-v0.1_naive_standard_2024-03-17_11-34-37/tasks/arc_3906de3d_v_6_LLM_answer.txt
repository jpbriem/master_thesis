import numpy as np
def func(m):
    m = np.array(m) # convert list to array
    changed = True
    while changed:
        changed = False
        for i in range(len(m)):
            for j in range(len(m[i])):
                if m[i][j] == 0:
                    m[i, :] = 0
                    m[:, j] = 0
                    changed = True
                    break
            if changed:
                break
    return m.tolist()
