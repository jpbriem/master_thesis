You keep track of how many points you have visited during your movement and if it's an odd number, you add one to the current position before moving on to the next direction. This way you will always end up adding two to the original point when you return from the recursion. Here is some code that does what I described above:

def flood(grid):
    def _flood(x, y, dx, dy):
        count = 0
        while True:
            x += dx
            y += dy
            if not (0 <= x < len(grid) and 0 <= y < len(grid[0])):
                break
            if grid[x][y] == 2:
                grid[x][y] += 2 * ((count + 1) % 2)
                count += 1
            elif grid[x][y]!= 0:
                break
        return count

    for i, row in enumerate(grid):
        for j, val in enumerate(row):
            if val == 2:
                grid[i][j] += _flood(i, j,  1, 0) \
                             + _flood(i, j, -1, 0) \
                             + _flood(i, j,  0,  1) \
                             + _flood(i, j,  0, -1)
