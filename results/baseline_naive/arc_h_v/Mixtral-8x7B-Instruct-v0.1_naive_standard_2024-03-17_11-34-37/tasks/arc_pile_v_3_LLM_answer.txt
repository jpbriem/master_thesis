#include <iostream>
using namespace std;

void moveZeroesToEnd(int arr[][5], int n) {
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n - 1; ++j) {
            if (arr[i][j] == 0 && arr[i][j + 1]!= 0) {
                swap(arr[i][j], arr[i][j + 1]);
            }
        }
    }
}
