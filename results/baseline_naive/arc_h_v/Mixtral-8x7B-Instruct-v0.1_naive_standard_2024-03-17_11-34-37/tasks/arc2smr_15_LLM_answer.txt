```
def fill(grid):
    def dfs(r, c):
        if r == len(grid) or c == len(grid[0]) or grid[r][c] == -1:
            return
        if grid[r][c] == 4:
            grid[r][c] = 1
            dfs(r + 1, c)
            dfs(r, c + 1)
            return
        grid[r][c] = -1
        dfs(r + 1, c)
        dfs(r, c + 1)
        grid[r][c] = 0

    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if not i and not j:
                dfs(i, j)

    return grid
