def update(grid):
    changed = False
    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j]!= 0:
                total = grid[i][j]
                k = i + 1
                while k < len(grid) and grid[k][j] == 0:
                    grid[k][j] = total
                    total += grid[k][j]
                    k += 1
                l = j + 1
                while l < len(grid[i]) and grid[i][l] == 0:
                    grid[i][l] = total
                    total += grid[i][l]
                    l += 1
                m = i - 1
                while m >= 0 and grid[m][j] == 0:
                    grid[m][j] = total
                    total += grid[m][j]
                    m -= 1
                n = j - 1
                while n >= 0 and grid[i][n] == 0:
                    grid[i][n] = total
                    total += grid[i][n]
                    n -= 1
                if grid[i][j]!= total:
                    changed = True
                    grid[i][j] = total
    return changed
