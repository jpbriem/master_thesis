def fill(grid):
    def _fill(x, y):
        if x == len(grid) or y == len(grid[0]) or grid[x][y]!= 0:
            return
        grid[x][y] = min(grid[x][y], 2 - sum(grid[i][y] for i in range(x)))
        _fill(x + 1, y)
        _fill(x, y + 1)

    for y in range(len(grid[0])):
        if grid[0][y]:
            break
        else:
            grid[0][y] = 2
    _fill(1, 0)
