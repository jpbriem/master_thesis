#include <iostream>
using namespace std;

void moveZeroes(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    for (int i = 0; i < m; ++i) {
        bool hasNonZero = false;
        for (int j = 0; j < n; ++j) {
            if (grid[i][j]!= 0) {
                swap(grid[i][j], grid[i][hasNonZero]);
                hasNonZero++;
            }
        }
    }
}
