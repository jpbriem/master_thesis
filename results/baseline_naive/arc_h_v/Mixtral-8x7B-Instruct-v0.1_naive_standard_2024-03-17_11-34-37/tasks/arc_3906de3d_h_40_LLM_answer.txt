You can use a recursive function to find all paths from top-left corner to bottom-right. For each path you will check if it's valid or not and update your answer accordingly. Here is an example of how this could be done in Python:

```
def solve(grid):
    def dfs(x, y):
        if x == n - 1 and y == m - 1: # base case
            return grid[x][y]!= 0

        if x < 0 or x >= n or y < 0 or y >= m:
            return False

        if grid[x][y] == 0:
            return False

        temp = grid[x][y]
        grid[x][y] = 0

        res = dfs(x + 1, y) or dfs(x, y + 1)

        grid[x][y] = temp

        return res

    n, m = len(grid), len(grid[0])
    ans = []
    for i in range(n):
        for j in range(m):
            if grid[i][j]!= 0:
                grid[i][j] = 0
                if dfs(i, j):
                    ans.append([list(row) for row in grid])
                grid[i][j] = temp

    return ans
```
