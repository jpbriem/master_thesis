```
def max_sum(grid):
    def dfs(x, y, total=0):
        if not (0 <= x < len(grid) and 0 <= y < len(grid[0])):
            return total
        elif grid[x][y] == 2:
            return max(total + abs(x - i) * 2 + abs(y - j) * 2 for i in range(len(grid)) for j in range(len(grid[0])) if grid[i][j] == 2)
