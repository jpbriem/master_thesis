def solve(matrix):
    def dfs(x, y):
        if x == n - 1 and y == m - 1:
            return True
        elif x >= n or y >= m:
            return False
        else:
            if matrix[x][y]!= 0:
                matrix[x][y] = '.'
                res = dfs(x + 1, y) or dfs(x, y + 1)
                if res:
                    matrix[x][y] = 'X'
                else:
                    matrix[x][y] = 0
                return res
            else:
                return False

    n, m = len(matrix), len(matrix[0])
    for i in range(n):
        for j in range(m):
            if matrix[i][j]!= 0:
                dfs(i, j)
