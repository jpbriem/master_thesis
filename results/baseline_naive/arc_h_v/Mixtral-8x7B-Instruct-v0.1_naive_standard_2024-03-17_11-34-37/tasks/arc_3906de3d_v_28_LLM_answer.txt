def solve(grid):
    def dfs(i, j):
        if i == n or j == m:
            return grid[n-1][m-1]==4
        if grid[i][j]==9:
            return False
        temp = grid[i][j]
        grid[i][j]=0
        res = dfs(i+1, j) or dfs(i, j+1)
        grid[i][j]=temp
        return res

    n, m = len(grid), len(grid[0])
    for i in range(n):
        for j in range(m):
            if grid[i][j]==4 and dfs(i, j):
                return 'YES'
    return 'NO'
