We also need to keep track of visited cells to avoid visiting them again. Here's how I would implement it using Python:

def dfs(grid, i, j):
    if not grid or i < 0 or j < 0 or i >= len(grid) or j >= len(grid[i]) or grid[i][j]!= 5:
        return
    grid[i][j] = 6 # Mark as visited
    dfs(grid, i + 1, j)
    dfs(grid, i - 1, j)
    dfs(grid, i, j + 1)
    dfs(grid, i, j - 1)

def fillIslands(grid):
