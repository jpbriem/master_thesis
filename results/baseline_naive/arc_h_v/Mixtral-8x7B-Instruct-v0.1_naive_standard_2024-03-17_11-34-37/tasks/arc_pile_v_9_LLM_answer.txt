#include <iostream>
using namespace std;

void moveZeroesToEnd(int arr[][5]) {
    int row = 5, col = 5;
    for (int i = 0; i < row; ++i) {
        if (arr[i][col - 1] == 0) {
            bool flag = false;
            for (int j = 0; j < col; ++j) {
                if (arr[row - 1][j]!= 0 &&!flag) {
                    swap(arr[i][j], arr[row - 1][j]);
                    flag = true;
                }
            }
        }
    }
}

    int inputArr[][5] = {{7, 0, 3, 0}, {7, 2, 0, 8}, {0, 0, 0, 0}, {0, 0, 0, 0}};

    printArray(inputArr);
