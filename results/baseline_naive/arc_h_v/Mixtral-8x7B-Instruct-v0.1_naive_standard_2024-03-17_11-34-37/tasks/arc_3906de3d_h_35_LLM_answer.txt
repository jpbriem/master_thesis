If you reach at any point where value of matrix is not zero then check if there exists another path which does not pass through this cell or not. You can do it by checking that whether there exist some other path in submatrix formed after removing current row and column or not.

If no such path exists then mark this cell as visited else leave it unvisited.

Here's an implementation using DFS approach -

```
def dfs(i, j, m, n):
    # base cases
    if i == m or j == n:
        return False
    if m-i <= 1 and n-j <= 1:
        return True

    # marking current cell as visited
    m[i][j] = 'x'

    # trying all four directions
    res = False
    for di, dj in ((1, 0), (-1, 0), (0, 1), (0, -1)):
        ni, nj = i + di, j + dj
        if 0 <= ni < m and 0 <= nj < n and m[ni][nj]!= 'x':
            temp = dfs(ni, nj, m, n)
            if temp:
                res |= temp

    # backtracking
    if not res:
        m[i][j] = 0

    return res

def solve(m):
    r, c = len(m), len(m[0])
    for i in range(r):
        for j in range(c):
            if m[i][j]!= 0 and dfs(i, j, m, c):
                break
        else:
            continue
        break
