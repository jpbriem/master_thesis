You can use a recursive function to find all paths from top-left corner to bottom-right. For each path you will check if it's valid or not and update your answer accordingly.

Here is an implementation in Python:

```
def solve(grid):
    def dfs(x, y):
        if x == n - 1 and y == m - 1: # base case
            return grid[x][y]!= 0

        ans = False
        if x + 1 < n and grid[x+1][y]!= 0:
            grid[x][y] = 0
            ans |= dfs(x+1, y)
            grid[x][y] = 7

        if y + 1 < m and grid[x][y+1]!= 0:
            grid[x][y] = 0
            ans |= dfs(x, y+1)
            grid[x][y] = 7

        return ans

    n, m = len(grid), len(grid[0])
    res = []
    for i in range(n):
        for j in range(m):
            if grid[i][j]!= 0:
                grid[i][j] = 0
                if dfs(i, j):
                    res.append([list(row) for row in grid])
                grid[i][j] = 7

    return res
```
