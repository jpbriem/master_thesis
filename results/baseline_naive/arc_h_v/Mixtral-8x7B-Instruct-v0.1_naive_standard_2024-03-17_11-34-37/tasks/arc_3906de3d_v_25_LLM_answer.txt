def remove(grid, r, c):
    # Base case: return early if either the row or column contains only zeros
    if not any(x[c] for x in grid[:r]) and not any(grid[i][c] for i in range(r)):
        return

    # Set all non-zero elements in this row and column to zero
    for i in range(len(grid)):
        if grid[i][c]:
            grid[i][c] = 0
        if grid[r][i]:
            grid[r][i] = 0

    # Recursively call the function on the next row/column
    remove(grid, r + 1, c)
    remove(grid, r, c + 1)
