The idea is that you start at one island and then look for another island by checking all directions from your current position until you find it or reach the end of the grid. Once you have found an island, you calculate how far away it is from the first island using BFS and then replace all cells along this path with the calculated value. def bfs(grid, i, j):
    queue = deque([(i, j)])
    dist = 0
    while queue:
        size = len(queue)
        dist += 1
        for _ in range(size):
            x, y = queue.popleft()
            if not (0 <= x < n and 0 <= y < m) or grid[x][y]!= 0:
                continue
            grid[x][y] = dist
            queue.append((x + 1, y))
            queue.append((x - 1, y))
            queue.append((x, y + 1))
            queue.append((x, y - 1))

def solve(grid):

    n, m = len(grid), len(grid[0])
    for i in range(n):
