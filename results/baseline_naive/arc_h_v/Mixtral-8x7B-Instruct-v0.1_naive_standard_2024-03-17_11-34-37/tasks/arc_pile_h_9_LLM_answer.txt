#include <iostream>
using namespace std;

void moveZeroesToEnd(int arr[][5]) {
    int row = 5, col = 5;
    for (int i = 0; i < row; ++i) {
        bool flag = false;
        for (int j = 0; j < col; ++j) {
            if (arr[i][j]!= 0) {
                swap(arr[i][j], arr[i][col - 1]);
                --col;
                flag = true;
                break;
            }
        }
        if (!flag)
            break;
    }
}
