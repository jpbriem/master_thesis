void printBoard(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		cout << "[";
		for (int j = 0; j < n; ++j) {
			if (board[i][j] == -1) cout << ",";
			else cout << board[i][j] << ",";
		}
		cout << " ]" << endl;
	}
}

void fillBoard(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			board[i][j] = 0;
		}
	}
}

bool isValid(int** board, int x, int y, int num, int n) {
	// check row and column
	for (int i = 0; i < n; ++i) {
		if (board[x][i] == num || board[i][y] == num) return false;
	}

	// check box
	int startX = x / 3 * 3;
	int startY = y / 3 * 3;
	for (int i = startX; i < startX + 3; ++i) {
		for (int j = startY; j < startY + 3; ++j) {
			if (board[i][j] == num) return false;
		}
	}

	return true;
}

bool solveSudoku(int** board, int n, int idx) {
	if (idx >= n*n) return true;

	int x = idx / n;
	int y = idx % n;

	if (board[x][y]!= 0 && board[x][y]!= -1) return solveSudoku(board, n, idx+1);

	for (int num = 1; num <= n; ++num) {
		if (!isValid(board, x, y, num, n)) continue;

		board[x][y] = num;
		if (solveSudoku(board, n, idx+1)) return true;
		board[x][y] = 0; // backtracking
	}

	return false;
}

void sudokuFloodFill(int** board, int n) {

}
