bool isValid(int** board, int x, int y, int val, int n) {
	// check row and column
	for (int i = 0; i < n; ++i) {
		if ((board[x][i]!= 0 && board[x][i] == val) || (board[i][y]!= 0 && board[i][y] == val)) return false;
	}

		for (int j = startY; j < startY + 3; ++j) {

		}
	}

	return true;
}

bool solveSudoku(int** board, int n) {
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < n; ++j) {
			if (board[i][j] == 0) {
				for (int k = 1; k <= n; ++k) {
					if (isValid(board, i, j, k, n)) {
						board[i][j] = k;
						if (solveSudoku(board, n)) return true;
						board[i][j] = 0; // backtracking
					}
				}
				return false;
			}
		}
	}
	return true;
}
