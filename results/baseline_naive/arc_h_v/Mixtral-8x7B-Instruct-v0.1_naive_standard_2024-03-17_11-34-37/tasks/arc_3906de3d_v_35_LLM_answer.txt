You can use a recursive function to find all possible paths from top-left corner to bottom right. If you reach at any point where value of cell is not equal to current path's number then return false and if we have reached at last cell with correct number in our path then update answer grid accordingly.

Here is C++ code :

```
#include<bits/stdc++.h>
using namespace std;

int n = 5, m = 6 ; // size of input matrix
vector<vector<bool>> vis(n+1, vector<bool>(m+1,false));
vector<vector<int>> mat = {
    {0, 0, 0, 4, 4, 4},
    {0, 0, 0, 4, 4, 4},
    {0, 0, 0, 4, 4, 0},
    {0, 0, 0, 4, 4, 0},
    {0, 0, 0, 4, 0, 0}
};

void solve(int i, int j, int num){
    if(i == n || j == m || i < 0 || j < 0 || mat[i][j]!= num || vis[i][j])
        return ;
    vis[i][j] = true ;
    if(i==n-1 && j==m-1)
        mat[i][j] = num ;
    else{
        solve(i+1,j,num);
        solve(i,j+1,num);
    }
    vis[i][j] = false ;
}

void printPath(vector<vector<int>> &mat){
    for(auto it : mat){
        for(auto x : it)
            cout << x << " ";
        cout << "\n";
    }
}

int main(){
    for(int i=1;i<=9;i++){
        memset(vis,false,sizeof(vis)) ;
        solve(0,0,i);
        printPath(mat);
        cout << "\n\n" ;
    }
}
