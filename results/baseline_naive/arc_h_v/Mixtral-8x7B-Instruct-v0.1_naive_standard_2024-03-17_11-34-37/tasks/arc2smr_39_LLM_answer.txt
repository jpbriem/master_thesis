public class Solution {
    public int[][] UpdateMatrix(int[][] mat) {
        if (mat == null || mat.Length == 0) return new int[0][];
        var m = mat.Length;
        var n = mat[0].Length;
        var queue = new Queue<Tuple<int, int>>();
        for (var i = 0; i < m; i++) {
            for (var j = 0; j < n; j++) {
                if (mat[i][j] == 0) {
                    queue.Enqueue(new Tuple<int, int>(i, j));
                } else {
                    mat[i][j] = -1;
                }
            }
        }

        while (queue.Count > 0) {
            var tpl = queue.Dequeue();
            var x = tpl.Item1;
            var y = tpl.Item2;
            if (x + 1 < m && mat[x+1][y] == -1) {
                mat[x+1][y] = mat[x][y] + 1;
                queue.Enqueue(new Tuple<int, int>(x+1, y));
            }
            if (x - 1 >= 0 && mat[x-1][y] == -1) {
                mat[x-1][y] = mat[x][y] + 1;
                queue.Enqueue(new Tuple<int, int>(x-1, y));
            }
            if (y + 1 < n && mat[x][y+1] == -1) {
                mat[x][y+1] = mat[x][y] + 1;
                queue.Enqueue(new Tuple<int, int>(x, y+1));
            }
            if (y - 1 >= 0 && mat[x][y-1] == -1) {
                mat[x][y-1] = mat[x][y] + 1;
                queue.Enqueue(new Tuple<int, int>(x, y-1));
            }
        }

        return mat;
    }
}
