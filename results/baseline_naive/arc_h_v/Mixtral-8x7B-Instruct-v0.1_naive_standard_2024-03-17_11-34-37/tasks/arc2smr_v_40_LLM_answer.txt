Then you move right one column and repeat this process. If there are no more diamonds left on the current row, you go back to the previous row and continue from where you left off. ```
def flood(grid):
    def _flood(x, y):
        if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[0]):
            return
        if grid[x][y] == '.':
            return
        grid[x][y] = '.'
        if grid[x - 1][y] == 'D':
            _flood(x - 1, y)
        else:
            _flood(x - 1, y + 1)

    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j] == 'D':
                _flood(i, j)

def solve(grid):
    flood(grid)
    for i in range(len(grid)):
        for j in range(len(grid[i])):
