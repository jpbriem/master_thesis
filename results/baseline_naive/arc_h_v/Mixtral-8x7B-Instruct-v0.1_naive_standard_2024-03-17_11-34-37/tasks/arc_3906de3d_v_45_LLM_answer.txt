def dfs(grid, i, j):
    if i < 0 or j < 0 or i == len(grid) or j == len(grid[0]) or grid[i][j] == 0:
        return False
    if grid[i][j] == 3:
        return True
    temp = grid[i][j]
    grid[i][j] = 0
    res = dfs(grid, i + 1, j) or dfs(grid, i - 1, j) or dfs(grid, i, j + 1) or dfs(grid, i, j - 1)
    grid[i][j] = temp
    return res

def solve(grid):
    nrows = len(grid)
    ncols = len(grid[0])
    for r in range(nrows):
        for c in range(ncols):
            if dfs(grid, r, c):
                grid[r][c] = 0
    return grid
