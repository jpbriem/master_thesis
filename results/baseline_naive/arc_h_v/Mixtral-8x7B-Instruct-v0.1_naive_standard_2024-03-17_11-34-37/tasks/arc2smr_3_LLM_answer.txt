You can use a recursive function to find all paths from top-left corner to bottom right. If you reach an obstacle or go out of bounds return False. Otherwise if you reached the end return True and fill in the path with 1s. Then at each step check if there are any other valid paths that haven't been filled yet by checking if the current cell has not been visited before. Here is some code that does this:

```
def dfs(grid, i, j):
    # Base cases
    if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[i]) or grid[i][j] == -1:
        return False
    if i == len(grid)-1 and j == len(grid[i])-1:
        grid[i][j] = 1
        return True

    # Mark as visited
    grid[i][j] = -1

    # Check if we have found another valid path
    res = dfs(grid, i+1, j) or dfs(grid, i-1, j) or dfs(grid, i, j+1) or dfs(grid, i, j-1)

    # Backtracking
    if not res:
        grid[i][j] = 0

    return res

def solve(grid):
    nRows = len(grid)
    nCols = len(grid[0])

    # Fill in first row and column
    for col in range(nCols):
        if grid[0][col]!= 0:
            break
        grid[0][col] = 1

    for row in range(nRows):
        if grid[row][0]!= 0:
            break
        grid[row][0] = 1

    # Find all paths using DFS
    for row in range(1, nRows):
        for col in range(1, nCols):
            if grid[row][col]!= 0:
                continue
            dfs(grid, row, col)

    return grid
```
