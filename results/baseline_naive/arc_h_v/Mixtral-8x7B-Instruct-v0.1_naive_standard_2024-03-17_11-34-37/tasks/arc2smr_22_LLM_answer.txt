```
def flood(grid, i, j):
    if i < 0 or i >= len(grid) or j < 0 or j >= len(grid[i]) or grid[i][j]!= 6 or grid[i][j] == 9:
        return
    grid[i][j] = 9
    flood(grid, i - 1, j)
    flood(grid, i + 1, j)
    flood(grid, i, j - 1)
    flood(grid, i, j + 1)

for i in range(len(grid)):
    for j in range(len(grid[i])):
        if grid[i][j] == 6:
            flood(grid, i, j)
