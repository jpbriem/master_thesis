def getPaths(grid):
    def dfs(x, y, grid, currPath=[]):
        if x == len(grid) or y == len(grid[0]): # out of bounds
            return []
        elif grid[x][y] == 0: # cannot move here
            return []
        else:
            currPath += [grid[x][y]]
            if x + 1 == len(grid) and y + 1 == len(grid[0]): # end point reached
                return [currPath]
            elif x + 1 < len(grid): # moving down
                res = dfs(x+1, y, grid, currPath[:])
            elif y + 1 < len(grid[0]): # moving right
                res = dfs(x, y+1, grid, currPath[:])
            else:
                assert False
            return [path + p for path in res for p in currPath[-1:]]

    ans = []
    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j]!= 0:
                ans += dfs(i, j, grid)
    return ans
