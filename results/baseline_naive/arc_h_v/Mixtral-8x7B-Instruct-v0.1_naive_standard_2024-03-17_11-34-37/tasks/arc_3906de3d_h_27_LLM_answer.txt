## Answer (4)

You can use a recursive function to find all paths from top-left corner to bottom-right. If you reach a cell with value 0 or 9 then return False because there's no path in that direction. When you reach a cell with value 5 and it has not been visited before add it to the current path and continue searching in its neighbours. If you visit a cell twice during search then remove it from the current path and backtrack to previous cells. Finally if we have reached the last row and column of matrix without returning false then append the final path to result list. Here is the code:

```
def solve(i, j, grid, curr_path):
    if i == len(grid)-1 and j == len(grid[0])-1: # base condition
        res.append(curr_path[:])
        return True

    if i >= len(grid) or j >= len(grid[0]): # out of bounds
        return False

    if grid[i][j] == 0 or grid[i][j] == 9: # dead end
        return False

    temp = grid[i][j]
    grid[i][j] = 0
    curr_path.append((i, j))

    up = solve(i - 1, j, grid, curr_path)
    down = solve(i + 1, j, grid, curr_path)
    left = solve(i, j - 1, grid, curr_path)
    right = solve(i, j + 1, grid, curr_path)

    if any([up, down, left, right]):
        res.append(curr_path[:])
    else:
        grid[i][j] = temp
        curr_path.pop()

    return False

res = []
for i in range(len(grid)):
    for j in range(len(grid[0])):
        if grid[i][j]!= 0 and grid[i][j]!= 9:
            solve(i, j, grid, [])
print(res)

from collections import deque

def solve(grid):
    q = deque()
    q.append((0, 0, [], 0))
    while q:
        i, j, curr_path, count = q.popleft()
        if i == len(grid)-1 and j == len(grid[0])-1:
            if count % 2 == 0:
                res.append(curr_path[:])
            continue

        if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[0]):
            continue

        if grid[i][j] == 0 or grid[i][j] == 9:
            continue

        temp = grid[i][j]
        grid[i][j] = 0
        curr_path.append((i, j))

        q.append((i - 1, j, curr_path[:], count+1))
        q.append((i + 1, j, curr_path[:], count+1))
        q.append((i, j - 1, curr_path[:], count+1))
        q.append((i, j + 1, curr_path[:], count+1))

        grid[i][j] = temp
        curr_path.pop()

res = []
solve(grid)
print(res)
