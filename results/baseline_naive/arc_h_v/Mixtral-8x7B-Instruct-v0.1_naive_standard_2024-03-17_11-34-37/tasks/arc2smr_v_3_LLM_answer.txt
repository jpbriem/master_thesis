def flood(grid):
    def _flood(x, y):
        if x < 0 or len(grid) <= x or y < 0 or len(grid[0]) <= y:
            # Out of bounds
            return
        elif grid[x][y] == 0:
            # Empty space
            return
        elif grid[x][y] == 3:
            # Start filling from here
            grid[x][y] = 1
            _flood(x + 1, y)
            _flood(x - 1, y)
            _flood(x, y + 1)
            _flood(x, y - 1)
        else:
            # Already filled
            pass

    # Find starting point
    for i, row in enumerate(grid):
        try:
            j = row.index(3)
            break
        except ValueError:
            continue
    else:
        # No more cells with value of 3
        return grid

    # Fill in grid
    _flood(i, j)

    # Change all remaining 3s to 0s
    for row in grid:

    return grid
