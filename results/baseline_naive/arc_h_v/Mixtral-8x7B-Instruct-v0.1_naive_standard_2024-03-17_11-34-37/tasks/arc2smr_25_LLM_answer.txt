def solve(board):
    def dfs(i, j, board, path):
        if i == len(board)-1 and j == len(board[0])-1:
            return True
        if i >= len(board) or j >= len(board[0]):
            return False
        if board[i][j]!= 6:
            return False
        temp = board[i][j]
        board[i][j] = 0
        path.append((i, j))
        res = dfs(i+1, j, board, path) or dfs(i, j+1, board, path)
        if res:
            return True
        del path[-1]
        board[i][j] = temp
        return False

    n = len(board)
    m = len(board[0])
    for i in range(n):
        for j in range(m):
            if board[i][j] == 6:
                path = []
                if dfs(i, j, board, path):
                    for k in path:
                        x, y = k
                        board[x][y] = 7
    return board
