def solve(grid, i, j, curr_path):
    n = len(grid)
    m = len(grid[0])
    # base condition when we are out of bounds or on boundary cells
    if i < 0 or j < 0 or i == n or j == m:
        return False
    # if we have reached at last cell with correct path then update answer grid accordingly
    elif i == n - 1 and j == m - 1 and grid[i][j] == curr_path[0]:
        grid[i][j] = 'P'
        return True
    # if value of cell does not match with first element of current path then return false
    elif grid[i][j]!= curr_path[0]:
        return False
    else:
        temp = grid[i][j]
        grid[i][j] = 'P'
        res = False
        # check for four directions
        res |= solve(grid, i + 1, j, [temp, temp])
        res |= solve(grid, i - 1, j, [temp, temp])
        res |= solve(grid, i, j + 1, [temp, temp])
        res |= solve(grid, i, j - 1, [temp, temp])
        grid[i][j] = temp
        return res

def fillPaths(grid):
    n = len(grid)
    m = len(grid[0])
    # create an empty list which will contain all possible paths
    paths = []
    for i in range(n):
        for j in range(m):
            if grid[i][j] > 0:
                paths.append([grid[i][j]])
    # iterate over each path and call our helper function
    for path in paths:
        solve(grid, 0, 0, path)
