You can use a recursive function to find all possible paths from top-left corner to bottom right. For each path you will have two possibilities - either go down or go right. If current cell contains number then add it to sum and check if new sum equals given target value. If yes then update result with this path. After that continue searching in both directions. Here is an example implementation of such algorithm:

```
def search(grid, x, y, cur_sum, target, path):
    if not grid[x][y]: # empty cell
        return []

    cur_sum += grid[x][y]
    path = path[:-1] + str(grid[x][y]) + path[-1]

    if cur_sum == target:
        return [path]

    res = []
    if x < len(grid)-1:
        res += search(grid, x+1, y, cur_sum, target, path+"D")
    if y < len(grid[0])-1:
        res += search(grid, x, y+1, cur_sum, target, path+"R")

    return res

def getPaths(grid, target):
    res = []
    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j]:
                res += search(grid, i, j, 0, target, "L"+str(grid[i][j])+"U")

    return res
