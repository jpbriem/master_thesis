def countPoints(grid):
    def dfs(x, y):
        if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[0]) or grid[x][y]!= 2:
            return 0
        grid[x][y] = -1
        total = 1 + dfs(x+1, y) + dfs(x-1, y) + dfs(x, y+1) + dfs(x, y-1)
        return total

    ans = []
    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j] == 2:
                ans.append(dfs(i, j))
    return ans
