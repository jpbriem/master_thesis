import re

def dfs(x, y):
    if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[0]):
        return False
    if grid[x][y] == 0:
        return True
    temp = grid[x][y]
    grid[x][y] = 0
    up = dfs(x - 1, y)
    down = dfs(x + 1, y)
    left = dfs(x, y - 1)
    right = dfs(x, y + 1)
    if not up:
        grid[x - 1][y] = temp
    if not down:
        grid[x + 1][y] = temp
    if not left:
        grid[x][y - 1] = temp
    if not right:
        grid[x][y + 1] = temp
    return temp!= 0

def replaceIslandsWithEight(grid):
