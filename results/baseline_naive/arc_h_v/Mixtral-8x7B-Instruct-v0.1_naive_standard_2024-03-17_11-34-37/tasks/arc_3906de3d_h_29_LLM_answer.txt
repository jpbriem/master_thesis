void fillWater(int arr[][10], int n, int m) {
    // find the first row with water and its column index
    bool foundFirstRowWithWater = false;
    int colIndexOfFirstRowWithWater = -1;
    for (int j = 0; j < m &&!foundFirstRowWithWater; ++j) {
        if (arr[0][j] == 0) {
            foundFirstRowWithWater = true;
            colIndexOfFirstRowWithWater = j;
        }
    }

    for (int i = 1; i < n; ++i) {
        // if there is no water in this row then continue to next row
        if (!foundFirstRowWithWater) {
            bool hasWaterInThisRow = false;
            for (int j = 0; j < m &&!hasWaterInThisRow; ++j) {
                if (arr[i][j] == 0) {
                    hasWaterInThisRow = true;
                }
            }
            if (!hasWaterInThisRow)
                continue;

        // so let's start filling from leftmost cell of current row

            // then just copy value from first row with water

            }
        }
    }
}
