void moveZeroes(int arr[][6]) {
    int row = 6, col = 6;
    for (int i = 0; i < row && i < col; ++i) {
        if (arr[i][i] == 0) {
            bool flag = false;
            for (int j = i + 1; j < row &&!flag; ++j) {
                if (arr[j][i]!= 0) {
                    swap(arr[i], arr[j]);
                    flag = true;
                }
            }
        }
    }
}
