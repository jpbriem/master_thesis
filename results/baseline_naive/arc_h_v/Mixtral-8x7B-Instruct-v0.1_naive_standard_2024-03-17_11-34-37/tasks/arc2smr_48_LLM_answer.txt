def flood(grid, x, y):
    if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[x]): # out of bounds
        return
    if grid[x][y]!= 5: # already filled
        return
    grid[x][y] = 7
    flood(grid, x + 1, y)
    flood(grid, x - 1, y)
    flood(grid, x, y + 1)
    flood(grid, x, y - 1)
