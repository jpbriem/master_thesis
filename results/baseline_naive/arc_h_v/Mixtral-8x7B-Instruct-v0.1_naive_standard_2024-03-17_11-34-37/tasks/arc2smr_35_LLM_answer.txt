def fill(grid):
    def recurse(x, y):
        if not grid[y][x]: return
        count = sum(not grid[n][m] for n in range(y-1, y+2) for m in range(x-1, x+2)) - 1
        if count <= 0: return
        grid[y][x] = count
        for dy, dx in ((-1, 0), (1, 0), (0, -1), (0, 1)):
            recurse(x + dx, y + dy)

    for i, row in enumerate(grid):
        if any(row):
            recurse(*next((j,i) for j, e in enumerate(row) if e))
