You can use a recursive function to find all possible paths from top-left corner to bottom right. For each path you will have two options - either move down or move diagonally up and then right. If at any point of time you encounter `0`, return False as there are no valid paths after that cell. When you reach the last row/column, check if current value in matrix is equal to target value. If yes, increment count by one else decrement target value by one. Finally, return True when you reach end of matrix.

def solve(matrix, i, j, n, m, target):
    if not target:
        return 1
    elif i >= n or j >= m or matrix[i][j] == 0:
        return 0
    temp = matrix[i][j]
    matrix[i][j] = 0
    ans = solve(matrix, i + 1, j, n, m, target - temp) + solve(matrix, i, j + 1, n, m, target)
    matrix[i][j] = temp
    return ans

def numOfPaths(matrix):
    n, m = len(matrix), len(matrix[0])
    total = sum([sum(x) for x in matrix])
    return solve(matrix, 0, 0, n, m, total // 6)
