void moveZeroes(int arr[][6]) {
    int row = 6, col = 6;
    for (int i = 0; i < row; ++i) {
        if (arr[i][col - 1] == 0) {
            bool flag = false;
            for (int j = 0; j < col; ++j) {
                if (arr[row - 1][j]!= 0 &&!flag) {
                    swap(arr[i][j], arr[row - 1][j]);
                    flag = true;
                }
            }
            --row;
        }
    }
}
