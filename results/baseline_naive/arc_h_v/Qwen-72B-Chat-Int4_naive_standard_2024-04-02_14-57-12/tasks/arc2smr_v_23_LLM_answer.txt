def update_grid(grid):
    n = len(grid)
    m = len(grid[0])
    
    # Iterate over each row in the grid
    for i in range(n):
        # Check if there is a '.' cell that has two adjacent cells with value 'd'
        for j in range(m-1):
            if grid[i][j] == '.' and (grid[i][j+1] == 'd' and grid[i][j+2] == 'd'):
                # Replace the '.' cell with an 'e'
                grid[i][j] = 'e'
                
        # Check if there is a '.' cell at the rightmost edge that has a left adjacent cell with value 'd'
