def paint_grid(grid):
    rows = len(grid)
    cols = len(grid[0])
    
    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == 'b':
                # Check left and right neighbors
                if j > 0 and grid[i][j-1] == 'b':
                    grid[i][j-1] = 'g'
                if j < cols - 1 and grid[i][j+1] == 'b':
                    grid[i][j+1] = 'g'

                # Check top and bottom neighbors
                if i > 0 and grid[i-1][j] == 'b':
                    grid[i-1][j] = 'g'
                if i < rows - 1 and grid[i+1][j] == 'b':
                    grid[i+1][j] = 'g'
                    
    return grid
