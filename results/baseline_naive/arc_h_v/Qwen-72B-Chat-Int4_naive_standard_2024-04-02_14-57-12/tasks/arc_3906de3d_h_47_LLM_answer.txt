def merge_bullets(grid):

    cols = len(grid[0])
    
    for i in range(rows):

                for dx in [-1, 1]:
                    nx = j + dx
                    if 0 <= nx < cols and grid[i][nx] == 'd':
                        grid[i][nx] = 'f'
                        break
                
                # Check up and down neighbors
                for dy in [-1, 1]:
                    ny = i + dy
                    if 0 <= ny < rows and grid[ny][j] == 'd':
                        grid[ny][j] = 'f'
                        break
    
    return grid
