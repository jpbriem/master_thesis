def flip_b_to_f(grid):
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == 'b':
                num_neighbors = 0
                
                # check left neighbor
                if j > 0 and grid[i][j-1] == 'b':
                    num_neighbors += 1
                    
                # check right neighbor
                if j < len(grid[0])-1 and grid[i][j+1] == 'b':
                    num_neighbors += 1
                    
                # check top neighbor
                if i > 0 and grid[i-1][j] == 'b':
                    num_neighbors += 1
                    
                # check bottom neighbor
                if i < len(grid)-1 and grid[i+1][j] == 'b':
                    num_neighbors += 1
                    
                # if exactly two neighbors are black, change it to a flower
                if num_neighbors == 2:
                    grid[i][j] = 'f'
                    
    return grid
