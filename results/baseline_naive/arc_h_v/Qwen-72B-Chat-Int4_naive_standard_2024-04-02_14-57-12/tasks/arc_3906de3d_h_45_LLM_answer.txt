def merge_holes(arr):
    # Create a dictionary to store the indices of each character in the array
    char_indices = {}
    for i, row in enumerate(arr):
        for j, c in enumerate(row):
            if c != '.':
                if c not in char_indices:
                    char_indices[c] = []
                char_indices[c].append((i, j))

    # Iterate through the characters in alphabetical order and merge their holes
    merged_arr = arr.copy()
    sorted_chars = sorted(char_indices.keys())
    for c in sorted_chars:
        indices = char_indices[c]
        prev_i, prev_j = None, None
        for i, j in indices:
            if prev_i is not None and abs(i - prev_i) > 1 or abs(j - prev_j) > 1:
                # If this character is too far away from the previous one,
                # create a new hole between them
                merged_arr[prev_i][prev_j+1:j] = ['.' for _ in range(prev_j+1, j)]
            prev_i, prev_j = i, j

    return merged_arr
