def connect_bees(grid):
    n = len(grid)
    m = len(grid[0])
    for i in range(n):
        for j in range(m):
            if grid[i][j] == 'b':
                count = 1
                dx, dy = [0, 0], [-1, 1]
                while True:
                    x, y = i + dx[count % 4], j + dy[count % 4]
                    if not (0 <= x < n and 0 <= y < m) or grid[x][y] != '.':
                        break
                    count += 1
                if count >= 5:
                    for k in range(count // 2 - 1, count):
                        x, y = i + dx[k % 4], j + dy[k % 4]
                        grid[x][y] = 'd'
    return grid
