def remove_duplicates(matrix):
    for i in range(len(matrix)):
        row = list(matrix[i])
        duplicates = set([char for char in row if row.count(char) > 1])
        for dup_char in duplicates:
            row = [char if char != dup_char else '' for char in row]
        matrix[i] = ''.join(row)
    return matrix
