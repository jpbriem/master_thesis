```python
def move_islands(grid):
    m, n = len(grid), len(grid[0])
    directions = [(0, 1), (1, 0)]
    
    for i in range(m):
        for j in range(n):
            if grid[i][j] == 'i':
                island_count = 0
                queue = [(i, j)]
                
                while queue:
                    x, y = queue.pop(0)
                    
                    if grid[x][y] != 'i':
                        continue
                    
                    grid[x][y] = 'd'
                    island_count += 1
                    
                    for dx, dy in directions:
                        nx, ny = x + dx, y + dy
                        
                        if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] == 'i':
                            queue.append((nx, ny))
                            
                if island_count > 1:
                    return False
    
    return grid
```
