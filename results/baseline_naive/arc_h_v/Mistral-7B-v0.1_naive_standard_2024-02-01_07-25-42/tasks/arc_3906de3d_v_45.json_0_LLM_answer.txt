class Solution {
public:
    vector<vector<char>> updateBoard(vector<vector<char>>& board, vector<int>& click) {
        int m = board.size();
        if (m == 0) return {};
        int n = board[0].size();
        if (n == 0) return {};
        // 将二维数组转换为一维数组
        vector<char> vec;
        for (auto &row : board) {
            for (auto c : row) {
                vec.push_back(c);
            }
        }
        // 遍历一维数组，如果当前元素是'h'，则将其替换成'c'；否则不做任何操作。
        for (int i = 0; i < vec.size(); ++i) {
            if (vec[i] == 'M') continue;
            if (vec[i] == 'E') {
                vec[i] = 'B';
                int x = i / n;
                int y = i % n;
                int count = getCount(board, x, y);
                if (count > 0) {
                    vec[i] = char('0' + count);
                }
            } else if (vec[i] == 'B') {
                vec[i] = 'X';
            }
        }
        // 将一维数组转换回二维数组并返回。
        vector<vector<char>> res;
        for (int i = 0; i < m; ++i) {
            vector<char> tmp;
            for (int j = 0; j < n; ++j) {
                tmp.push_back(vec[(i * n) + j]);
            }
            res.push_back(tmp);
        }
        return res;
    }
private:
    int getCount(const vector<vector<char>> &board, const int x, const int y) {
        int m = board.size();
        if (m == 0) return 0;
        int n = board[0].size();
        if (n == 0) return 0;
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if ((x + i >= 0 && x + i < m) && (y + j >= 0 && y + j < n)) {
                    if (board[x + i][y + j] == 'M') {
                        ++count;
                    }
                }
            }
        }
        return count;
    }
};
