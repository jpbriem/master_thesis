    vector<vector<char>> updateMatrix(vector<vector<char>>& mat) {
        int m = mat.size();

        int n = mat[0].size();
        queue<pair<int, int>> q;
        vector<vector<bool>> visited(m, vector<bool>(n));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (mat[i][j] == '0') {
                    q.push({i, j});
                    visited[i][j] = true;
                }
            }
        }
        while (!q.empty()) {
            auto cur = q.front();

            for (auto dir : {{0, -1}, {0, 1}, {-1, 0}, {1, 0}}) {
                int newX = x + dir[0];

        }
