    def islands(self, matrix: List[List[str]]) -> int:
        def dfs(x, y):
            if x < 0 or y < 0 or x >= len(matrix) or y >= len(matrix[0]) or matrix[x][y] != '.':

                return True

            self.visited[x][y] = True
            for dx, dy in [(0, -1), (0, 1), (-1, 0), (1, 0)]:
                dfs(x + dx, y + dy)

            return True

        visited = [[False] * len(matrix[0]) for _ in range(len(matrix))]

            for j in range(len(matrix[i])):

                    res += dfs(i, j)
                    
        return res

In the `dfs` function, we first check if the current cell has already been visited. If it has, we simply return True since we have already processed this cell before. Otherwise, we mark the current cell as visited and recursively process its neighbors.
