class Solution:
    def removeInvalidParentheses(self, s: str) -> str:
        stack = []
        for i in range(len(s)):
            if s[i] == '(':
                stack.append(i)
            elif s[i] == ')':
                index = stack.pop()
                while index >= 0 and s[index] != '(':
                    index -= 1
                if index < 0:
                    return "Invalid Parentheses"
        return ''.join(s)
