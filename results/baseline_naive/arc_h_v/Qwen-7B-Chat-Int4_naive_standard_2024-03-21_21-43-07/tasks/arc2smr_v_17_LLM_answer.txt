class Solution:
    def convert(self, matrix):
        """
        :type matrix: List[List[str]]
        :rtype: List[List[str]]
        """
        if not matrix or len(matrix) == 0:
            return []
        
        # check if the input is a valid matrix
        for row in matrix:
            if len(row) != len(matrix[0]):
                return []
        
        # create an empty list to store the converted matrix
        converted_matrix = [[None] * len(matrix[0]) for _ in range(len(matrix))]
        
        # iterate through each element of the original matrix
        for i in range(len(matrix)):
            for j in range(len(matrix[0])):
                # handle special cases where there are only '.'s or f's in the current cell
                if matrix[i][j] == '.':
                    converted_matrix[i][j] = ''
                elif matrix[i][j] == 'f':
                    converted_matrix[i][j] = 'f'
                else:
                    # otherwise, find the index of the first occurrence of the character in the current row and column
                    count = 0
                    for k in range(i, -1, -1):
                        if matrix[k][j] == matrix[i][j]:
                            count += 1
                    for l in range(j, -1, -1):
                        if matrix[i][l] == matrix[i][j]:
                            count += 1
                    
                    # if the character has been found in both directions, it means that it occurs twice in the same position
                    if count > 1:
                        converted_matrix[i][j] = matrix[i][j]
                    
                    # if the character has been found only in one direction, it means that it occurs once in the current position
                    else:
                        converted_matrix[i][j] = matrix[i][j].upper()
            
        return converted_matrix
