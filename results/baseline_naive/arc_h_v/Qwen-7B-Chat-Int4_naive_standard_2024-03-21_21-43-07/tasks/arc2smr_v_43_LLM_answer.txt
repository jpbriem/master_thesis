    def expand(left, right):
        if left >= right:
            return s[left:right+1]
        else:
            mid = (left + right) // 2
            lps = expand(left, mid)
            rps = expand(mid+1, right)
            return max(lps, rps)

    s = list(s)
    n = len(s)
    if n < 2:
        return s

    start = end = -1
    for i in range(n):
        if s[i] == s[0]:
            start = i
        elif start != -1 and s[i] == s[start+1]:
            start += 1

    if start == -1:
