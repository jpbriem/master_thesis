class Solution:
    def convert(self, s: str) -> str:
        res = []
        for i in range(len(s)):
            if s[i] == '.':
                res.append('.')
            else:
                temp = ''
                while temp != '' and i < len(s)-1 and s[i+1] != '.':
                    temp += s[i]
                    i+=1
                res.append(temp)
        
        return ''.join(res)
