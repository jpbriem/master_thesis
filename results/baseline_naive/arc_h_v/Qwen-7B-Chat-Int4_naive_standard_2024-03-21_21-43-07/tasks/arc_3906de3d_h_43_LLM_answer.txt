def longest_palindrome(input):
    def helper(s, l, r):
        if l > r:
            return None

        mid = (l + r) // 2
        curr = s[mid]
        left = mid - 1
        right = mid + 1

        while left >= 0 and right < len(s) and s[left] == s[right]:
            curr = s[left]
            left -= 1
            right += 1

        if curr != '.':
            return max(max(helper(s, l, mid), helper(s, mid+1, r)), curr)

        return None

    return max(helper(input[0], 0, len(input[0])-1), helper(input[1], 0, len(input[1])-1))
