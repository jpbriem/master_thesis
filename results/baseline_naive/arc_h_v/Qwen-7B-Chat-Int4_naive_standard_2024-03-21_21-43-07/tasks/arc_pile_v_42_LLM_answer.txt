class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs or len(strs) == 0:
            return []
        
        ans = [[] for _ in range(len(set(strs)))]
        
        for s in strs:
            key = ''.join(sorted(s))
            ans[key].append(s)
            
        return ans
