def longest_common_subsequence(str1, str2):
    m = len(str1)
    n = len(str2)

    # Initialize a 2D array to store the lengths of LCS for each pair of characters
    dp = [[0] * (n+1) for _ in range(m+1)]

    # Fill in the dp table from left to right and top to bottom
    for i in range(1, m+1):
        for j in range(1, n+1):
            if str1[i-1] == str2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])

    # Build the LCS by backtracking through the dp table
    lcs = ''
    i = m
    j = n
    while i > 0 and j > 0:
        if str1[i-1] == str2[j-1]:
            lcs += str1[i-1]
            i -= 1
            j -= 1
        elif dp[i-1][j] >= dp[i][j-1]:
            i -= 1
        else:
            j -= 1

    return lcs[::-1]
