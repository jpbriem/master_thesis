def longest_palindrome(input):
    n = len(input)
    dp = [[0] * (n + 1) for _ in range(n + 1)]
    ans = []
    for i in range(1, n + 1):
        dp[i][i] = True

    # check if substring of length 2 is palindrome or not
    for i in range(3, n + 1):
        j = i - 2
        while j >= 0 and input[j] == input[i]:
            dp[i][j] = True
            j -= 1

    # check if substring of length 3 is palindrome or not
    for k in range(4, n + 1):
        l = k - 3
        m = k - 2
        while l >= 0 and m >= 0 and input[l] == input[m] == input[k]:
            dp[k][l] = dp[k][m]
            dp[k][m] = dp[k][k]
            l -= 1
            m -= 1

    # check if substring of length 4 or more is palindrome or not
    for p in range(5, n + 1):
        q = p - 4
        r = p - 3
        s = p - 2
        t = p - 1
        while q >= 0 and r >= 0 and s >= 0 and t >= 0 and input[q] == input[r] == input[s] == input[t] == input[p]:
            dp[p][q] = dp[p][r] = dp[p][s] = dp[p][t] = dp[p][p]
            q -= 1
            r -= 1
            s -= 1
            t -= 1

    for k in range(n + 1):
        for i in range(k, n + 1):
            if dp[i][k]:
                ans.append(i)

    return ans
