def find_path(board):
    # Base case: all cells are filled or there are no more empty cells
    if not any(c != '.' for c in board) or len({c for c in board if c == '.'}) == 0:
        return []

    result = []
    marked = set()

    def dfs(row, col, steps=()):
        nonlocal result

        # Check if the cell is already visited
        if row < 0 or row >= len(board) or col < 0 or col >= len(board[0]) or board[row][col] != '.' or board[row][col] == 'h':
            return

        # Mark the cell as visited
        marked.add((row, col))

        # Add the current step to the result
        result.append(steps)

        # Recursively search for paths in all four directions
        dfs(row-1, col)
        dfs(row+1, col)
        dfs(row, col-1)

    dfs(0, 0)

    return result
