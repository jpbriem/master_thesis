class Solution:
    def numIsland(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: int
        """
        if not grid or len(grid) == 0 or len(grid[0]) == 0:
            return 0
        
        directions = [[-1, 0], [1, 0], [0, -1], [0, 1]]
        
        m = len(grid)
        n = len(grid[0])
        
        seen = set([(m*n)])
        visited = {i: False for i in range(m*n)}
        
        count = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '.' and (i,j) not in seen:
                    count += self.dfs(i, j, grid, seen, visited)
                    
        return count
    
    def dfs(self, x, y, grid, seen, visited):
        if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[0]):
            return 0
            
        if grid[x][y] != '.' or (x,y) in visited:
            return 0
            
        seen.add((x*y))
        visited[x*y] = True
        
        count = 1 + self.dfs(x+directions[0][0], y+directions[0][1], grid, seen, visited)
        count += self.dfs(x-directions[1][0], y-directions[1][1], grid, seen, visited)
        count += self.dfs(x-directions[2][0], y-directions[2][1], grid, seen, visited)
        count += self.dfs(x-directions[3][0], y-directions[3][1], grid, seen, visited)
        
        return count
