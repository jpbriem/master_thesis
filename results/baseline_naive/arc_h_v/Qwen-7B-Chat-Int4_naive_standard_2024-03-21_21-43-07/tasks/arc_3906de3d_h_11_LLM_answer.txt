class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0:
            return 0

        start = 0
        max_len = 1

        for i in range(len(s)):
            # check odd length palindromes
            j = i - 1
            while j >= 0 and s[j] == s[i]:
                start = j + 1
                max_len += 2
                j -= 1

            # check even length palindromes
            j = i + 1
            while j < len(s) and s[j] == s[i]:
                start = j - 1
                max_len += 2
                j += 1

        return max_len
