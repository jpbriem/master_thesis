class Solution:
    def longestPalindrome(self, s: str) -> int:
        n = len(s)
        
        if n == 0:
            return 0
        
        start = 0
        max_len = 1
        
        for i in range(n):
            if s[i] != '.':
                continue
            
            j = i + 1
            while j < n and s[j] != '.':
                if s[j - 1] == s[j]:
                    start = i
                    max_len += 2
                    j += 1
                else:
                    break
                    
            j = i - 1
            while j >= 0 and s[j] != '.':
                if s[j + 1] == s[j]:
                    start = i
                    max_len += 2
                    j -= 1
                else:
                    break
                    
        return max_len
