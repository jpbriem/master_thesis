    def convert(self, s):
        res = []
        stack = []
        
        for c in s:
            if c == '.':
                while len(stack) > 0 and stack[-1] != '.':
                    res.append(stack.pop())
                
                if len(stack) > 0:
                    stack.pop()
            
            else:
                stack.append(c)
        
        while len(stack) > 0:
            res.append(stack.pop())
        
        return ''.join(res[::-1])
