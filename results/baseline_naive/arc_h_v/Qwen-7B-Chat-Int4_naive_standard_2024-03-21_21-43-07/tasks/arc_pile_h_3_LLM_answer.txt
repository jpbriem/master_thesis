class Solution:
    def permute(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: List[List[int]]
        """
        if len(nums) == 0 or len(nums[0]) == 0:
            return []
        
        ans = []
        for i in range(len(nums)):
            temp = [nums[i]] + self.permute(nums[:i] + nums[i+1:])
            ans.extend(temp)
            
        return ans
