class Solution:
    def convert(self, s: str) -> List[str]:
        if len(s) == 0 or s[0] != '.':
            return []
        
        stack = [s[0]]
        res = []
        
        for i in range(1, len(s)):
            while stack and s[i] != stack[-1]:
                res.append(stack.pop())
                
            stack.append(s[i])
            
        while stack:
            res.append(stack.pop())
        
        return res[::-1]
