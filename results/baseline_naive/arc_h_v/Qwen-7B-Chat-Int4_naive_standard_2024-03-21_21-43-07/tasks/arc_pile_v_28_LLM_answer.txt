class Solution:
    def removeInvalidParentheses(self, s: str) -> str:
        if len(s) == 0:
            return ''
        
        stack = []
        ans = []
        for char in s:
            if char != ')':
                stack.append(char)
            else:
                while len(stack) > 0 and stack[-1] != '(':
                    stack.pop()
                if len(stack) == 0 or stack[-1] != '(':
                    ans.append(''.join(stack))
                    stack.clear()
                else:
                    stack.pop()
                
        ans.append(''.join(stack))
        
        return ''.join(ans[::-1])
