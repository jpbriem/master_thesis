class Solution:
    def matrixBlockRotations(self, matrix: List[List[str]]) -> List[List[str]]:
        n = len(matrix)
        
        for i in range(n):
            if not any(c == '.' for c in matrix[i]):
                return [['.'] * n] + matrix
        
        for i in range(n):
            for j in range(i+1, n):
                if matrix[i][j] != '.':
                    break

        for i in range(n):
            for j in range(i+1, n):
                if matrix[j][i] != '.':
                    break
                
        start_i = i
        end_i = j - 1
        while start_i < end_i and matrix[start_i][end_i] == '.':
            start_i += 1
            end_i -= 1
        
        start_j = j
        end_j = i - 1
        while start_j < end_j and matrix[end_j][start_j] == '.':
            start_j += 1
            end_j -= 1
            
        matrix = [[matrix[i][k] for k in range(start_i, end_i + 1)] for i in range(start_j, end_j + 1)]
        matrix.reverse()
        for i in range(len(matrix)):
            matrix[i].reverse()

        for i in range(len(matrix)):
            for j in range(len(matrix[0])):
                if matrix[i][j] == '.':
                    matrix[i][j] = '.'
                
        return matrix
