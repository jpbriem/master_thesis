def sort_strings(arr):
    # Convert all characters to lowercase
    arr = [char.lower() for char in arr]
    
    # Create an empty list to store the sorted strings
    sorted_arr = []
    
    # Iterate over each string in the input array
    for i, row in enumerate(arr):
        # Sort the characters in the current string
        row.sort()
        
        # Append the sorted string to the sorted_arr list
        sorted_arr.append(row)
    
    return sorted_arr
