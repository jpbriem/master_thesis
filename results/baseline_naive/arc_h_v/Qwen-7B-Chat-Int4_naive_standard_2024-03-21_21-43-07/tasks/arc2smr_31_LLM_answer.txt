        # If the current substring and its reversed substring are equal, add it to the result

        # Check if there exists any character at index j such that the substring from i to j is equal to its reverse
        # This can be done using dynamic programming
        max_len = 0
        start = None

        for j in range(i, len(input)):
            if input[j] == input[j - i]:
                if j - i > max_len:
                    max_len = j - i
                    start = j - i

        # If there is such a character, add the substring starting from that character to the result
        if start is not None:
            result.append([start, start + max_len])

    return result
