class Solution:
    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        n = len(heights)
        if not heights or n == 0:
            return 0

        max_area = float('-inf')
        start_index = 0
        for i in range(n):
            if heights[i] > max_area:
                max_area = heights[i]
                start_index = i

        end_index = start_index + 1
        while end_index < n and heights[end_index] == max_area:
            end_index += 1

        return (max_area * (end_index - start_index))
