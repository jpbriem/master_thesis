    def connect(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: List[List[str]]
        """
        if not grid or len(grid) != len(grid[0]):
            return grid

        n = len(grid)
        m = len(grid[0])

        for i in range(n):
            for j in range(m):
                if grid[i][j] == '.':
                    continue
                else:
                    break

        directions = [[-1, 0], [0, -1], [1, 0], [0, 1]]

        def dfs(i, j):
            if i < 0 or i >= n or j < 0 or j >= m or grid[i][j] != '.':
                return False
            grid[i][j] = '#'
            for di, dj in directions:
                next_i = i + di
                next_j = j + dj
                if dfs(next_i, next_j):
                    return True
            return False

        result = []
        for i in range(n):
            for j in range(m):
                if grid[i][j] == '.':
                    result.append(grid[i][j])

                    result.append('.')

        return result
