def longest_palindrome(input):
    if len(input) == 0 or len(input[0]) == 0:
        return []
    
    result = []
    for i in range(len(input)):
        # check the current character as a palindrome
        start, end = i, i
        while start >= 0 and input[start] == input[end]:
            result.append(".".join(input[i:end+1]))
            start -= 1
            end += 1
        
        # check the substring starting from the current character
        start, end = i, i + 1
        while start < len(input) and end < len(input) and input[start] == input[end]:
            result.append(".".join(input[i:end+1]))
            start += 1
            end += 1
    
    return result
