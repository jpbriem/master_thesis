class Solution:
    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        if not heights or len(heights) == 0:
            return 0

        n = len(heights)
        dp = [[0] * (n + 1) for _ in range(n + 1)]

        # bottom up approach
        for i in range(1, n + 1):
            for j in range(i - 1, -1, -1):
                if heights[j][i] > heights[j][j]:
                    dp[i][j] = max(dp[i][j], dp[j][i])
                else:
                    dp[i][j] = max(dp[i][j], dp[j][i] + heights[j][i])

        # get the maximum area
        max_area = 0
        for i in range(n + 1):
            max_area = max(max_area, dp[i][n])

        return max_area
