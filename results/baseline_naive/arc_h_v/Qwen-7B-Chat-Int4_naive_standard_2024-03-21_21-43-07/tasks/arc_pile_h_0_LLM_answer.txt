    def convert(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: List[List[str]]
        """

            return grid

        m = len(grid[0])

            for j in range(m):
                if grid[i][j] != '.':
                    continue

                for k in range(i-1, -1, -1):

                        break

                        count += 1

                    if grid[i][l] == '.':
                        break

                        count += 1

        return grid
