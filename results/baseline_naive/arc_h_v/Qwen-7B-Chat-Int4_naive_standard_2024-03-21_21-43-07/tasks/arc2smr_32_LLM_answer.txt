class Solution:
    def matrixBlockOrder(self, matrix: List[List[str]]) -> List[List[str]]:
        directions = [[0, 1], [1, 0], [1, 1], [1, -1]]
        
        def dfs(matrix, start, direction):
            if len(start) == 4:
                return True
            
            for i in range(4):
                row = start[i]
                col = start[3-i]
                
                for d in directions:
                    new_row = row + d[0]
                    new_col = col + d[1]
                    
                    if not (0 <= new_row < len(matrix) and 0 <= new_col < len(matrix[0])):
                        continue
                    
                    if matrix[new_row][new_col] != '.':
                        continue
                    
                    matrix[row][col], matrix[new_row][new_col] = matrix[new_row][new_col], matrix[row][col]
                    
                    if dfs(matrix, [new_row, new_col], direction):
                        return True
                
            return False
        
        res = []
        
        for r in range(len(matrix)):
            for c in range(len(matrix[r])):
                if matrix[r][c] != '.':
                    continue
                
                visited = set()
                
                for di in range(-2, 3):
                    for dj in range(-2, 3):
                        if di**2 + dj**2 != 0:
                            if di != 0 or dj != 0:
                                visited.add((r+di, c+dj))
                            
                if dfs(matrix, [r, c], [0, 1]):
                    res.append([matrix[r][c]])
                elif dfs(matrix, [r, c], [1, 0]):
                    res.append([matrix[r][c]])
                elif dfs(matrix, [r, c], [1, 1]):
                    res.append([matrix[r][c]])
                else:
                    res.append([matrix[r][c]])
                
        return res
