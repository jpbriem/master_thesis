    def longestSubsequence(self, matrix: List[List[str]]) -> int:

        # recursive function to find the length of longest subsequence
        def dfs(i, j):
            
            if i >= n or j >= n:
                return 0
            
            # check if current cell is part of a valid subsequence
            if matrix[i][j] == '.' and dp[i-1][j] + dp[i+1][j] + dp[i][j-1] + dp[i][j+1] > 0:
                
                # extend the subsequence by moving left, right, up, down
                dp[i][j] += max(dp[i-1][j], dp[i+1][j], dp[i][j-1], dp[i][j+1])
                
                # call the recursive function on the four neighboring cells
                return max(dfs(i-1, j), dfs(i+1, j), dfs(i, j-1), dfs(i, j+1))

                # move one step in any direction
                return min(dp[i-1][j], dp[i+1][j], dp[i][j-1], dp[i][j+1]) + 1
                
        # return the length of longest subsequence
        return dfs(0, 0)
