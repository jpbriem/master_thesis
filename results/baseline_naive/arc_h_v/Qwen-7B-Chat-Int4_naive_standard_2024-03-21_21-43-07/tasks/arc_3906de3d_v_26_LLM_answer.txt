    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        heightMap = {heights[i][j]: i for i in range(len(heights)) for j in range(len(heights[0]))}
        
        stack = []

            currentHeight = heights[row]
            
            if not currentHeight or len(currentHeight) == 0:
                continue
            
            maxArea = max(maxArea, self.calculateArea(stack, heightMap, currentHeight))
            
            for i in range(len(currentHeight)-1):
                stack.append(heightMap[currentHeight[i]])

                    stack.pop()
                    
        return maxArea
    
    def calculateArea(self, stack: List[int], heightMap: dict, currentHeight: List[int]) -> int:
        area = 0
        
        while stack and currentHeight[stack[-1]]:

                area += currentHeight[index] * (len(currentHeight) - index)

                break
        
        return area
