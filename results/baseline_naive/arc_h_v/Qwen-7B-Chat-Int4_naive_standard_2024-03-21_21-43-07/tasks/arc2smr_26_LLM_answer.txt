def find_paths(matrix):
    # Define a helper function to check if a cell is valid or not
    def is_valid(x, y):
        return x >= 0 and x < len(matrix) and y >= 0 and y < len(matrix[0]) and matrix[x][y] == '.'

    # Define a queue to store cells to be visited
    queue = [(0, 0)]

    # Initialize an empty list to store all paths
    paths = []

    while queue:
        current_x, current_y = queue.pop(0)

        # If we have reached the end of the path, add it to the list of paths
        if current_x == len(matrix)-1 and current_y == len(matrix[0])-1:
            paths.append(list(zip(*matrix[current_x]))[::-1])

        # Add all neighboring cells to the queue if they are valid
        for dx, dy in ((-1, 0), (1, 0), (0, -1), (0, 1)):
            new_x = current_x + dx
            new_y = current_y + dy
            if is_valid(new_x, new_y):
                queue.append((new_x, new_y))

    return paths
