class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: List[str]
        """
        if len(s) < 2:
            return []
        
        ans = []
        start = 0
        
        for i in range(len(s)):
            if s[i] == s[0]:
                end = i
            else:
                break
            
            ans.append(s[start:end+1])
            
            while start > 0 and s[start-1] != s[0]:
                start -= 1
                
            if start > 0:
                end = start - 1

        return ans
