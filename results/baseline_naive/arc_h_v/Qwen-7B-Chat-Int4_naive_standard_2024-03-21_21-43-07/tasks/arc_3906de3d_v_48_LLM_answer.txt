```python
def longest_palindrome(input):
    words = {''.join(sorted(c)) for c in input}
    result = []
    for i in range(len(input)):
        for j in range(i+1, len(input)+1):
            substring = input[i:j]
            if substring not in words:
                continue
            if len(substring) > max(result):
                result.append(substring)
    return result
```
