class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        
        for s in strs:
            chars = set(s)
            if chars not in d:
                d[chars] = [s]
            else:
                d[chars].append(s)
                
        return list(d.values())
