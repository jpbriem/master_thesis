def transpose(matrix):
    """
    Transposes a matrix.

    Args:
        matrix (list): A list of lists representing a matrix.

    Returns:
        list: A list of lists representing the transposed matrix.
    """
    n = len(matrix)
    m = len(matrix[0])
    
    # Initialize empty result matrix with size (m x n)
    result = [[None] * n for _ in range(m)]
    
    # Iterate over rows and columns to populate result matrix
    for i in range(n):
        for j in range(m):
            result[j][i] = matrix[i][j]
            
    return result
