def longest_common_substring(s):
    n = len(s)
    dp = [[0] * (n + 1) for _ in range(n + 1)]
    max_length = 0
    start_index = 0

    for i in range(1, n + 1):
        for j in range(i, n + 1):
            if s[i - 1] == s[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
                if dp[i][j] > max_length:
                    max_length = dp[i][j]
                    start_index = i - max_length

    return s[start_index : start_index + max_length]
