```python
class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s or len(s) == 0:
            return 0

        n = len(s)
        max_len = 1
        for i in range(n):
            # Check odd length palindromes
            l = self.longestPalindromeUtil(s, i, i + 1)
            if l > max_len:
                max_len = l
                
            # Check even length palindromes
            r = i + (n - i) // 2
            l = self.longestPalindromeUtil(s, i, r)
            if l > max_len:
                max_len = l
        
        return max_len
    
    def longestPalindromeUtil(self, s, left, right):
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
            
        return right - left - 1
```
