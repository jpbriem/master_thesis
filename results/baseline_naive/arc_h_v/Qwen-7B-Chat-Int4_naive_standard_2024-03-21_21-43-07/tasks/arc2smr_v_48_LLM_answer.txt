class Solution:
    def convert(self, matrix):
        """
        :type matrix: List[List[str]]
        :rtype: List[List[str]]
        """
        rows = len(matrix)
        cols = len(matrix[0])
        
        # Convert each character to uppercase
        for i in range(rows):
            for j in range(cols):
                if matrix[i][j] == '.':
                    continue
                else:
                    matrix[i][j] = matrix[i][j].upper()
                    
        # Convert the first row and column to lowercase
        for i in range(cols):
            if matrix[0][i] != '.':
                matrix[0][i] = matrix[0][i].lower()
        for i in range(rows):
            if matrix[i][0] != '.':
                matrix[i][0] = matrix[i][0].lower()
                
        # Check if any characters are not converted correctly
        for i in range(rows):
            for j in range(cols):
                if matrix[i][j] != matrix[i][j].lower() + matrix[i][j].upper():
                    return False
        
        # Convert all remaining characters back to lowercase
        for i in range(1, rows):
            for j in range(1, cols):
                if matrix[i][j] != matrix[i][j].lower():
                    return False
                    
        return matrix
