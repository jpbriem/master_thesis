class Solution:
    def findLargerNumber(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: int
        """
        
        if not nums or len(nums) == 0:
            return None
        
        # check the length of list
        max_len = max(len(num) for num in nums)
        
        # create a dictionary to store each number and its index
        dict_num_index = {num:i for i,num in enumerate(nums)}
        
        # sort the numbers based on their index
        sorted_nums = [nums[i] for i in sorted(dict_num_index.keys())]
        
        # initialize two pointers at the beginning of the sorted list
        left_ptr = right_ptr = 0
        
        larger_number = -float('inf')
        
        while right_ptr < max_len:
            # compare the current elements at the left pointer and right pointer
            current_left = sorted_nums[left_ptr]
            current_right = sorted_nums[right_ptr]
            
            # update the larger number if necessary
            if current_right > current_left:
                larger_number = current_right
            
            # move the pointers towards the middle
            left_ptr += 1
            right_ptr += 1
        
        return larger_number
