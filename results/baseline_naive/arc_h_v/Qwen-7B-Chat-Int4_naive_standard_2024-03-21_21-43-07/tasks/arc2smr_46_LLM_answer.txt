    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 0

        start = 0
        max_length = 1

            if l > 0 and l + 1 >= max_length:
                start = i - (l + 1)
                max_length = l + 1
                
            # Check even length palindromes with center between two characters
            l = self.longestPalindrome(s[i-1:i+3])
            if l > 0 and l >= max_length:
                start = i - l // 2
                max_length = l
            
        return max_length
