def replace_consecutive_chars(grid):
    rows = len(grid)
    cols = len(grid[0])
    
    for i in range(rows):
        for j in range(cols-2):  # only need to check every third character since we're replacing at least two characters
            c = grid[i][j]
            count = 1
            
            while j+count < cols and grid[i][j+count] == c:
                count += 1
                
            if count > 2:
                replacement = None
                
                # find the most common non-consecutive character in the row
                char_counts = {}
                for k in range(cols):
                    if grid[i][k] != c and grid[i][k] not in char_counts:
                        char_counts[grid[i][k]] = 0
                    elif grid[i][k] != c:
                        char_counts[grid[i][k]] += 1
                        
                max_count = max(char_counts.values())
                for k in char_counts:
                    if char_counts[k] == max_count and k != replacement:
                        replacement = k
                    
                # replace the consecutive characters with the chosen replacement
                for k in range(count):
                    grid[i][j+k] = replacement
                    
    return grid
