def count_holes(matrix):
    output = []
    hole_count = 0
    
    # Iterate through rows of the input matrix
    for row in matrix:
        modified_row = []
        
        # Iterate through elements in the row
        for i in range(len(row)):
            # Check if the current element is equal to 'h'
            if row[i] == 'h':
                # Count the number of neighboring elements that are also equal to 'h'
                num_neighbors = sum(1 for j in range(i-1, i+2) if (j >= 0 and j < len(row)) and row[j] == 'h')
                
                # Replace the current element with 'd' and increment the hole count
                if num_neighbors > 1:
                    modified_row.append('d')
                    hole_count += 1
                else:
                    modified_row.append(row[i])
            else:
                modified_row.append(row[i])
        
        output.append(modified_row)
    
    return output, hole_count
