def find_path(grid):
    rows, cols = len(grid), len(grid[0])
    visited = [[False] * cols for _ in range(rows)]
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]

    def dfs(i, j, path):
        if i < 0 or i >= rows or j < 0 or j >= cols or grid[i][j] != 'g' or visited[i][j]:
            return False

        visited[i][j] = True
        path.append((i, j))

        for dx, dy in directions:
            ni, nj = i + dx, j + dy
            if dfs(ni, nj, path):
                return True

        return False

    result = []
    for i in range(rows):
        for j in range(cols):
            if not visited[i][j] and grid[i][j] == 'g':
                path = []
                if dfs(i, j, path):
                    row = [grid[row_index][col_index]
                           for (row_index, col_index) in path]
                    result.append(row)

    return result
