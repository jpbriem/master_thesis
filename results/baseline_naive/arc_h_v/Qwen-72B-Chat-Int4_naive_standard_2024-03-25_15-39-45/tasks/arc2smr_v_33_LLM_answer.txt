def modify_elements(input_array):

        if row.count('e') < 2:
            result.append(row)
            continue
        
        result_row = []
        consecutive_e_count = 0
        start_index = end_index = -1

        for i, char in enumerate(row):
            if char == 'e':
                consecutive_e_count += 1
            elif consecutive_e_count >= 2:
                end_index = i - 1
                break

                start_index = i - consecutive_e_count + 1
