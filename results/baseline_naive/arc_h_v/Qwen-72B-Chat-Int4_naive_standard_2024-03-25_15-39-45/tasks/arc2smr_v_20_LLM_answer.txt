def convert_a_to_g(input_list):
    for i, row in enumerate(input_list):
        has_replaced = False
        for j, cell in enumerate(row):
            if cell == 'a':
                if not has_replaced and (j > 0 and input_list[i][j - 1] == 'e'):

                    has_replaced = True

    return input_list
