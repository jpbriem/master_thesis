To solve this problem, we can use a depth-first search (DFS) approach. This process continues until all 'e' characters have been processed.

           - If any neighbor is not 'a', return False.
         - Mark `c` as visited by adding `(r, c)` tuple to `visited`.
         - Continue the DFS traversal by checking the neighbors of `c` that haven't been visited yet.
3. Return the modified matrix.

def dfs(matrix, r, c):
    rows = len(matrix)
    cols = len(matrix[0])
    
    def is_valid_neighbor(r, c):
        if 0 <= r < rows and 0 <= c < cols and matrix[r][c] == 'e' and (r, c) not in visited:
            return True
        return False

    visited.add((r, c))
    
    # Check neighbors horizontally and vertically
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    for dr, dc in directions:
        nr, nc = r + dr, c + dc
        
        if is_valid_neighbor(nr, nc):
            dfs(matrix, nr, nc)

def replace_e(matrix):

    cols = len(matrix[0])
