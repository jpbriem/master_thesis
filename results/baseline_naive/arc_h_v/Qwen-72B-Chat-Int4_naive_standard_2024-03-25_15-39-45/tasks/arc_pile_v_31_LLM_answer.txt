def replace_elements(input_list):
    output_list = []
    for row in input_list:
        new_row = []
        for i, elem in enumerate(row):
            if elem != 'a':
                count = 0
                for j in range(i+1, len(row)):
                    if row[j] == elem:
                        count += 1
                max_count = max(count, sum(1 for x in input_list if x[i] == elem))
                if count < max_count:
                    elem = 'a'
            new_row.append(elem)
        output_list.append(new_row)
    return output_list
