def replace_b_with_c(arr):
    """
    Replaces all occurrences of 'b' in arr with 'c'.
    
    :param arr: a list of lists containing characters
    :return: a new list of lists where all occurrences of 'b' have been replaced with 'c'
    """
    result = []
    for row in arr:
        temp_row = []
        for cell in row:
            if cell == 'b':
                count = 0
                while True:
                    try:
                        temp_cell = temp_row[count]
                    except IndexError:
                        break
                    if temp_cell != 'b':
                        break
                    else:
                        count += 1
                temp_row.extend(['c'] * (count + 1))
            elif cell == 'a':
                temp_row.append('a')
            else:
                raise ValueError("Invalid character")
        result.append(temp_row)
    return result
