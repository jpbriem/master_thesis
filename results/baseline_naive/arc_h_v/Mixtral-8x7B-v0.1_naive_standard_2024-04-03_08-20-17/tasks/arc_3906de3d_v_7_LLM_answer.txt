def find(matrix):
    # Find first fire in matrix
    x = 0
    y = 0
    while True:
        if len(matrix)-y > 0 and "." == matrix[-y][x]:
            break
        else:
            y += 1
    return {"x": x, "y": -y}

# Create new empty matrix of same size than input one
newMatrix = []
for i in range(len(matrix)):
    line = ["."] * len(matrix[i])
    newMatrix.append(line)

while True:
    posFire = find(matrix)
    print("posFire", posFire["x"], posFire["y"])
    # If no more fires then exit loop
    if posFire["x"] < 0:
        break;
    # Add all g around current position into new matrix
    for j in [-1, 0, +1]:
        try:
            newMatrix[posFire["y"]+j][posFire["x"]] = "g"
        except IndexError:
            pass
    # Remove from original matrix every f found after adding g
    for k in [-1, 0, +1]:
        try:
            del matrix[posFire["y"]+k][posFire["x"]]
        except IndexError:
            pass
print("\n\n")
print("New Matrix:\n")
for l in newMatrix:
    print(*l)
