class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        ans = []
        for i in range(len(nums)):
            if i > 0 and nums[i] == nums[i-1]:
                continue
            for j in range(i+1, len(nums)):
                if nums[j] == nums[j-1]:
                    continue
                left = [x for x in nums[:i] + nums[j:] if x != nums[i] and x != nums[j]]
                right = [nums[i]] + [x for x in nums[:j] if x != nums[i] and x != nums[j]]
                total = sum(left) - nums[i] + nums[j]
                if total == target:
                    ans.append([nums[i], nums[j]])
        
        return ans
