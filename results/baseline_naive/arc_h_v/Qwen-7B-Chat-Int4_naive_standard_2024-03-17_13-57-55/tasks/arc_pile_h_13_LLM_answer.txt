```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0 or len(set(strs[0])) == 1:
            return []
        
        groups = [[] for _ in range(len(set(strs[0])))]
        for s in strs:
            index = set(s).index(min(set(s)))
            groups[index].append(s)
            
        return groups
```
