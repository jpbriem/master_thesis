class Solution:
    def gridAfterFilling(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: List[List[int]]
        """
        m = len(grid)
        n = len(grid[0])
        
        for i in range(m):
            if not grid[i]:
                continue

            for j in range(n):
                if grid[i][j] == 0:
                    if i > 0 and grid[i-1][j] == 1:
                        grid[i][j] = 1
                    elif i < m - 1 and grid[i+1][j] == 1:
                        grid[i][j] = 1
                    else:
                        grid[i][j] = 1 if (i,j) != (m-1,n-1) else 0
                    
        return grid
