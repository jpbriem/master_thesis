```python
class Solution:
    def convert(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: List[List[int]]
        """
        if not nums or len(nums) == 0:
            return []
        
        n = len(nums)
        res = [[] for _ in range(n)]
        for i in range(n):
            for j in range(len(nums[i])):
                val = nums[i][j]
                if val > 0:
                    res[j].append(val - 1)
                    res[j + 1].append(val - 1)
                else:
                    res[j].append(val)
                    res[j + 1].append(val)
            
        return res
```
