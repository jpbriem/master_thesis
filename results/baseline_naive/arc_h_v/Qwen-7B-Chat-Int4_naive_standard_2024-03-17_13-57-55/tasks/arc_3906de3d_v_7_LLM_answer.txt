class Solution:
    def findLargestProduct(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0
        
        n = len(nums)
        
        # 找到最大公约数，因为只有两个元素时才有可能出现最大的乘积
        gcd = self.gcd(nums[0], nums[1])
        
        # 如果gcd为1，则nums中任意两个元素相乘的乘积都是最大可能的
        if gcd == 1:
            return max(nums)
        
        # 否则，需要找到两个因子，使得它们和的最大公约数为gcd，并且这两个因子分别除以gcd后的结果相等
        factors = []
        for i in range(n):
            for j in range(i+1, n):
                if self.gcd(nums[i]//gcd, nums[j]//gcd) == gcd:
                    factors.append((nums[i], nums[j]))
                    
        if not factors:
            return 0
        
        # 对于每个组合，计算它们的乘积并返回最大的一个
        result = 1
        for factor in factors:
            product = factor[0]*factor[1]
            result *= product
            
        return result
    
    def gcd(self, a, b):

            a, b = b, a % b
        return a
