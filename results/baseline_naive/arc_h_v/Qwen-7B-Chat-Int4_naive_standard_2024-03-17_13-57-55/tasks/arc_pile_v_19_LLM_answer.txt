```python
class Solution:
    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        n = len(heights)
        max_area = 0
        
        # dp[i][j] stores the largest area that can be formed by using the top i elements and starting from height j
        dp = [[0 for _ in range(n)] for _ in range(len(heights[0]))]
        
        for j in range(n):
            if heights[j][0] == 0:
                continue
            
            for i in range(1, len(dp)):
                
                if heights[j][i] > heights[j][i-1]:
                    dp[i][j] = dp[i-1][j]
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i-1][j-heights[j][i]] + heights[j][i])
            
            max_area = max(max_area, dp[-1][-1])

        return max_area
```
