class Solution:
    def maxChunksToSorted(self, arr: List[List[int]]) -> int:
        if not arr:
            return 0
        
        res = []
        
        for i in range(len(arr)):
            res.append([arr[i][0]])
            
            for j in range(1, len(arr[i])):
                if arr[i][j] > arr[i][j-1]:
                    res[-1].append(arr[i][j])
                else:
                    res.append([arr[i][j]])
                    
        res.sort()
        
        return len(res)
