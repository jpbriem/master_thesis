class Solution:
    def islandPerimeter(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        
        # dp[i][j] represents the number of cells that are connected to cell (i,j) 
        dp = [[0 for _ in range(n)] for _ in range(m)]
        
        # if we can reach the top left corner from cell (i,j), then we add it's perimeter to the total perimeter
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    continue
                if i == 0 or j == 0 or grid[i][j] == 0:
                    dp[i][j] += 1
                else:
                    dp[i][j] += 4
        
        # calculate the total perimeter
        total_perimeter = 0
        for i in range(m):
            for j in range(n):
                if dp[i][j]:
                    total_perimeter += dp[i][j]
        
        return total_perimeter
