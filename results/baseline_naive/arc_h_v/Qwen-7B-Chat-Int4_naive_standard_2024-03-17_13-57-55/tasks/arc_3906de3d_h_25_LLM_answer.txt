    def canPartition(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        if sum(nums) % 2 != 0:
            return False
        
        def dfs(index, mask):

                return True

            for i in range(1 << len(nums)):
                if not mask or (mask & i):
