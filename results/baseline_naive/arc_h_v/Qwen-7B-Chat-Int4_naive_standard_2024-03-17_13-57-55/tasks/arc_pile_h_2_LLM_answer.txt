class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 1:
            return [nums]

        res = []
        
        def dfs(nums, path):
            for i in range(len(nums)):
                if not path[i] and nums[i]:
                    path[i] = True
                    dfs(nums[:i] + nums[i+1:], path)
                    path[i] = False

        dfs(nums, [])
        for path in res:
            path.reverse()
            res.append(path)

        return res
