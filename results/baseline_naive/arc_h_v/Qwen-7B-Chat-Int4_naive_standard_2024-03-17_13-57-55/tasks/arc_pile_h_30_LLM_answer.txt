class Solution:
    def arrangeBottles(self, bottles):
        """
        :type bottles: List[List[int]]
        :rtype: List[List[int]]
        """
        n = len(bottles)
        for i in range(n):
            if bottles[i][0] == 0 and i != n-1:
                for j in range(i+1, n):
                    if bottles[j][0] > 0:
                        temp = bottles[i]
                        bottles[i] = bottles[j]
                        bottles[j] = temp

        res = []
        for row in bottles:
            if not any([x[0] == 0 for x in row]):
                res.append(row)

        return res
