class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 1:
            return [nums]

        result = []
        
        # generate all permutations of length 1
        for i in range(len(nums)):
            result.append([nums[i]])

        # generate all permutations of length greater than 1
        for length in range(2, len(nums)+1):
            for left in self.permute(nums[:length-1]):
                right = nums[length-1:]
                result.extend(left + right)

        return result
