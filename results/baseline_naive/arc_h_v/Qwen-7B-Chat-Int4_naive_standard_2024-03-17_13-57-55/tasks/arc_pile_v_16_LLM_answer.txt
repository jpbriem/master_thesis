```python
class Solution:
    def canForm(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: bool
        """
        n = len(grid)
        if n == 0 or n == 1:
            return True

        for i in range(n):
            for j in range(n):
                if grid[i][j] != 0 and (i > 0 and grid[i-1][j] == grid[i][j]) \
                        and (j > 0 and grid[i][j-1] == grid[i][j]):
                    return False

        return True
```
