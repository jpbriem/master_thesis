    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0:
            return 0

        result = 0
        left = 0
        right = 0

        while left < len(arr) and right < len(arr):
            # Move the smaller element to the right side of the array
            if arr[left] > arr[right]:
                arr[left], arr[right] = arr[right], arr[left]

            # If the current chunk is sorted, move the left pointer to the next element
            if right - left + 1 >= self.maxSubarraySum(arr[left:right+1]):
                result += 1
                left += 1
            else:
                right += 1
        
        return result
