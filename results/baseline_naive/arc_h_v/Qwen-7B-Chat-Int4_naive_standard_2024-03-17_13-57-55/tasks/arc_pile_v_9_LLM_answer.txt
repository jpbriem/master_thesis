```python
class Solution:
    def minCut(self, graph):
        """
        :type graph: List[List[int]]
        :rtype: int
        """
        
        n = len(graph)
        visited = [False] * n
        
        ans = float('inf')
        stack = [(0, -1)]
        
        while stack:
            node, par = stack.pop()
            
            if visited[node]:
                continue
            
            visited[node] = True
            
            for neighbor in range(n):
                
                if graph[node][neighbor] == 0 or visited[neighbor]:
                    continue
                
                new_par = par + 1 if par != -1 else 0
                
                stack.append((neighbor, new_par))
                
                ans = min(ans, new_par)
        
        return ans
```
