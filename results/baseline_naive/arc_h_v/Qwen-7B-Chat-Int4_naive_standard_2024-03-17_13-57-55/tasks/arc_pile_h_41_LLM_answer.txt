```python
class Solution:
    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        n = len(heights)
        if not heights or n == 0:
            return 0
        
        dp = [[0] * (n + 1) for _ in range(n)]
        
        max_area = 0
        for i in range(1, n):
            for j in range(i):
                height = heights[i]
                area = dp[j][i - 1] * height
                if area > max_area:
                    max_area = area
                    
                dp[i][j] = max(dp[i][j], area)
                
        return max_area
```
