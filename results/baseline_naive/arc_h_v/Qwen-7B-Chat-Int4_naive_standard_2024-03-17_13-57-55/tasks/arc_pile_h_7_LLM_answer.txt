def find_obstacles(matrix):

    dp[0][0] = True
    
    # Iterate over each cell in the matrix
    for i in range(1, n + 1):

            # Check if there is an obstacle at the current position
            if matrix[i - 1][j - 1]:
                dp[i][j] = False
            else:
                # Check if there is an obstacle at the left and top positions
                if matrix[i - 1][j] == 1 or matrix[i][j - 1] == 1:
                    dp[i][j] = False
