class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0 or len(nums) == 1:
            return [[]]

        res = []
        
        def backtrack(start, path=[]):
            if start == len(nums)-1:
                res.append(path)
                return

            for i in range(start+1):
                num = nums[start]
                if num != path[-1]:
                    path.append(num)
                    backtrack(i + 1, path)
                    path.pop()
                
        backtrack(0)

        return res
