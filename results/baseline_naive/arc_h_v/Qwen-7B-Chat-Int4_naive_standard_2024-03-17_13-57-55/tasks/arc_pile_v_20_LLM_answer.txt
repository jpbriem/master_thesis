```python
class Solution:
    def findLargestRectangle(self, heights: List[List[int]]) -> int:
        m = len(heights)
        n = len(heights[0])
        
        dp = [[-1 for _ in range(n)] for _ in range(m)]
        ans = 0
        
        for i in range(1, m):
            for j in range(1, n):
                if heights[i][j] > heights[i - 1][j]:
                    dp[i][j] = max(dp[i][j], dp[i - 1][j] + 1)
                    ans = max(ans, dp[i][j])
        return ans
```
