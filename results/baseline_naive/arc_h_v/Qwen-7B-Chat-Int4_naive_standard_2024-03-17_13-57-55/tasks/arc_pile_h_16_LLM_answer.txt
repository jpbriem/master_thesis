class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[int]]
        """
        groups = {}
        
        for s in strs:
            key = ''.join(sorted(s))
            if key not in groups:
                groups[key] = []
            groups[key].append(len(s))
            
        return [groups[i] for i in sorted(groups)]
