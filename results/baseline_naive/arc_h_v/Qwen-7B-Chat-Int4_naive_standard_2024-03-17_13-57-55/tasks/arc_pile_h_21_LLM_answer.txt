    def findMinMoves(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        directions = [(0, -1), (1, 0), (0, 1), (-1, 0)]
        moves = 0
        
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    queue = []
                    queue.append((i, j))
                    
                    while queue:
                        x, y = queue.pop(0)
                        
                        for dx, dy in directions:
                            nx, ny = x + dx, y + dy

                                grid[x][y], grid[nx][ny] = grid[nx][ny], grid[x][y]
                                queue.append((nx, ny))
                                
                            moves += 1
                    
        return moves
