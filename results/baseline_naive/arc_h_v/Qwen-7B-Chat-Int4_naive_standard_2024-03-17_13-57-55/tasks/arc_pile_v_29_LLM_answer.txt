    if not nums:
        return []

    # Iterate through the input array
    for i in range(len(nums)):
        
        # If the current element is greater than or equal to the previous element,
        # then extend the current subarray
        if nums[i] >= nums[end]:
            end += 1
            
        # Otherwise, reset the current subarray and update its length and sum
        else:
            start = end + 1
            max_len = max(max_len, end - start + 1)
            max_sum = max(max_sum, sum(nums[start:end+1]))
            
    # Return the maximum subarray
    return [nums[start:start+max_len]] * max_len
