```python
class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0 or len(nums) == 1:
            return [[]]

        ans = []
        self.dfs(nums, [], ans)
        return ans

    def dfs(self, nums, path, ans):
        if len(path) == len(nums):
            ans.append(path[:])
        for i in range(len(nums)):
            if nums[i] != 0 and not any(x == nums[i] for x in path):
                path.append(nums[i])
                self.dfs(nums, path, ans)
                path.pop()
```
