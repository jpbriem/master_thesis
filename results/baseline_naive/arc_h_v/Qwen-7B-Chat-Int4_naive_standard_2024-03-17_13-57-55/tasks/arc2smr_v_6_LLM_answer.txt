class Solution:
    def minArea(self, grid: List[List[int]]) -> int:
        if not grid or len(grid) == 0:
            return 0

        m = len(grid)
        n = len(grid[0])

        # dp[i][j] is the minimum area of rectangle with top-left corner at (i, j)
        dp = [[0 for _ in range(n)] for _ in range(m)]

        # If there are no obstacles, just pick any square and make it as big as possible.
        # The area will be (m * n).
        for i in range(m):
            for j in range(n):
                dp[i][j] = m*n

        # For each obstacle, we need to find its largest side and then remove all squares that have this size.
        # We can do this by sweeping from bottom-right to top-left, and keeping track of the largest side seen so far.
        max_side = 0
        for j in reversed(range(n)):
            for i in range(m-1,-1,-1):
                side = abs(grid[i][j])
                if side > max_side:
                    max_side = side
                    for k in range(i+max_side,m):
                        for l in range(j+max_side,n):
                            dp[k][l] += dp[i][j]

        return min(dp[m-1][n-1],dp[n-1][m-1]) - 1
