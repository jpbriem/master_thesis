def max_sum(matrix):
    m = len(matrix)
    n = len(matrix[0])
    
    dp = [[0 for _ in range(n)] for _ in range(m)]
    
    # Base cases
    dp[0][0] = matrix[0][0]
    for i in range(1, m):
        dp[i][0] = dp[i-1][0] + matrix[i][0]
        
    for j in range(1, n):

    # Recursive case
    for i in range(1, m):
        for j in range(1, n):
