class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0 or arr[0] > arr[-1]:
            return 0
        
        n = len(arr)
        
        left = []
        right = []
        
        for i in range(n):
            while left and arr[i] < arr[left[-1]]:
                left.pop()
            
            while right and arr[n - i - 1] <= arr[right[0]]:
                right.pop(0)
                
            left.append(i)
            right.append(n - i - 1)
        
        return min(len(left), len(right))
