class Node(object):
    def __init__(self, val=0, neighbors=None):

class Solution(object):
    def orangesSquaring(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: List[List[str]]
        """
        directions = [(0, -1), (0, 1), (-1, 0), (1, 0)]
        nodes = {node: Node(grid[idx][idy]) for idx, idy in product(*[[x, y] for x, y in enumerate(row) if row[x][y] == 'i' for row in grid]])}
        queue = deque([nodes[(0, 0)]])
        while queue:
            node = queue.popleft()
            for direction in directions:
                new_pos = tuple((n + d) % len(grid[0]) for n, d in zip(node.val, direction))
                if new_pos in nodes and nodes[new_pos].val!= node.val:
                    nodes[new_pos].neighbors += [node]
                    nodes[node.val].neighbors += [nodes[new_pos]]
        return [[node.val for node in sorted(node.neighbors, key=lambda x: x.val)] for node in nodes.values()]
