# Definition for a Node.
class Node(object):
    def __init__(self, val, children=None):
        self.val = val
        self.children = children


class Solution(object):
    def transformToPreorder(self, A):
        """
        :type A: List[List[str]]
        :rtype: List[List[str]]
        """
        
        # 2018/7/4
        # O(n) time complexity
        # O(n) space complexity
        res = []
        stack = [(A[0][0], A)]
        while len(stack) > 0:
            node, parent = stack[-1]
            
            if isinstance(node, str):
                res.append([])
                
                for i in range(len(parent)):
                    if parent[i].startswith('a'):
                        res[-1].extend(['a' for _ in range(int(parent[i]))])
                    
                    elif parent[i].startswith('b'):
                        res[-1].extend(['b' for _ in range(int(parent[i][1:]))])
                    
                    else:
                        res[-1].append(parent[i][1:])
                
                stack.pop()
            else:
                stack.append((node.val, node.children))
                stack.reverse()
        return res
