# Definition for a Node.
class Node(object):
    def __init__(self, val, children=None):
        self.val = val
        self.children = children


class Solution(object):
    def transformToPreorder(self, A):
        """
        :type A: List[List[str]]
        :rtype: List[List[str]]
        """
        
        # 1. Build tree from list of lists
        root = None
        for i in range(len(A)):
            if not root or A[i][0] < root.val:
                root = Node(A[i][0])
            
            if len(root) == 0:
                continue
                
            child = root
            while True:
                if child.val > A[i][0]:
                    break
                else:
                    child = child.children[-1]
                    
            child.add_child(Node(A[i][0]))
        
        # 2. Traverse tree to get pre order traversal
        result = []
        stack = [(root, 0)]
        while stack:
            node, index = stack.pop()
            if index >= len(node.children):
                result.append([node.val] + [x.val for x in reversed(node.children)])
                continue
            
            stack.append((node, index+1))
            stack.append((node.children[index], 0))
        
        return result
