from typing import List, Optional


class Solution:
    def groupAnagrams(self, strs: List[List[str]]) -> List[Optional[List[str]]]:
        """
        1. sort each string in list of strings
        2. compare first character of sorted string with previous one to check if they are anagram or not
            - if yes then append current string to previous one's list
            - else create new list for current string
        3. return all lists
        Time complexity: O(nlogn) where n is number of elements in list of strings
                          (sorting time complexity) + O(m) where m is number of elements in list of strings
                          (comparison time complexity) = O(nlogn+m) = O(nlogn) since m <= n
        Space complexity: O(n) where n is number of elements in list of strings
                           (list creation time complexity)
        """
        result = []  # type: ignore
        for s in strs:
            s.sort()
            if len(result) == 0:
                result.append([])
            elif result[-1][0].startswith(s):
                result[-1].append(s)
            else:
                result.append([s])
        return result
