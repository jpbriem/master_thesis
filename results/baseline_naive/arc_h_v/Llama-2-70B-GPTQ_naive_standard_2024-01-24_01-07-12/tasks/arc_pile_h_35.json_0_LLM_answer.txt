# Definition for a Node.
class Node(object):
    def __init__(self, val, children=None):
        self.val = val
        self.children = children


class Solution(object):
    def transformToPreorder(self, A):
        """
        :type A: List[List[str]]
        :rtype: List[List[str]]
        """
        
        # 2018/7/4
        # O(n) time complexity
        # O(n) space complexity
        if not A or len(A) == 0: return []
        root = Node('')
        stack = [(root, A)]
        while stack:
            node, list_of_lists = stack.pop()
            
            for i in range(len(list_of_lists)):
                child = list_of_lists[i][0]
                
                if child!= '':
                    node.addChild(Node(child))
                    
                else:
                    node.addChild(Node())
                    
                stack.append((node.getChildren()[-1], list_of_lists[i][1:]))
                
        return [[node.val] + sorted([child.val for child in node.getChildren()]) \
               for node in root.getChildren()]
