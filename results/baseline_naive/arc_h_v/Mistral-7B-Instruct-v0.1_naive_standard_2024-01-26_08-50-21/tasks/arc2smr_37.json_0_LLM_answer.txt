function map(array, func) {
    return array.map(func);
}

function transform(arr) {
    var result = [];
    for (var i = 0; i < arr.length; i++) {
        result.push(map(arr[i], function(el) {
            if (el === 0) {
                return el + 1;
            } else {
                return el - 1;
            }
        }));

}
