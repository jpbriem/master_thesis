function transform(arr1, arr2){
    return arr1.map((val, i) => {
        let result = [];
        for(let j=0;j<arr2[i].length;j++){
            if(arr2[i][j] === '0'){
                result.push(val);
            }else{
                result.push(transform(val, arr2[i][j]));

        }

}
