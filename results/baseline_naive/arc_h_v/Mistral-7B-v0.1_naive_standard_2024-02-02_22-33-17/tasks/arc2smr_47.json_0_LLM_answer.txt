    public List<List<String>> replaceWords(List<List<String>> dict, String sentence) {
        // write your code here
        if (dict == null || dict.size() == 0) return new ArrayList<>();
        TrieNode root = buildTrie(dict);
        List<List<String>> res = new ArrayList<>();

            if (!sb.toString().equals("")) {

            }
        }

    }

    private void dfs(TrieNode node, String word, StringBuilder sb) {
        if (node!= null &&!word.isEmpty()) {
            if (node.children[word.charAt(0)-'a'].isEndOfWord) {
                sb.append(word.substring(0, word.indexOf(node.children[word.charAt(0)-'a'].value)+1));
                dfs(node.children[word.charAt(0)-'a'], word.substring(word.indexOf(node.children[word.charAt(0)-'a'].value)+1), sb);
            } else {
                dfs(node.children[word.charAt(0)-'a'], word.substring(1), sb);
            }
        }
    }

    private TrieNode buildTrie(List<List<String>> dict) {
        TrieNode root = new TrieNode('*');
        for (List<String> list : dict) {
            TrieNode cur = root;
            for (String s : list) {
                int index = s.charAt(0)-'a';
                if (cur.children[index] == null) {
                    cur.children[index] = new TrieNode(s.charAt(0));
                }
                cur = cur.children[index];
            }
            cur.isEndOfWord = true;
        }
        return root;
    }
