public class Solution {
    public List<List<String>> removeAnagrams(List<List<String>> arr) {
        if (arr == null || arr.size() == 0) return new ArrayList<>();
        int i = 0;
        while (i < arr.size()) {
            int j = i + 1;
            while (j < arr.size() && checkAnagram(arr.get(i), arr.get(j))) {
                j++;
            }
            if (j > i) {
                arr.remove(i);
            } else {
                i++;
            }
        }
        return arr;
    }

    private boolean checkAnagram(List<String> list1, List<String> list2) {
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < list1.size(); i++) {
            char c = list1.get(i).charAt(0);
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        for (int i = 0; i < list2.size(); i++) {
            char c = list2.get(i).charAt(0);
            if (!map.containsKey(c)) {
                return false;
            }
            map.put(c, map.get(c) - 1);
            if (map.get(c) == 0) {
                map.remove(c);
            }
        }
        return true;
    }
}
