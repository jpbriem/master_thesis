public class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, Set<String> wordList) {
        if (beginWord == null || endWord == null ||!wordList.contains(endWord)) return new ArrayList<>();
        Map<String, Integer> map = new HashMap<>();
        for (String s : wordList) {
            map.put(s, map.getOrDefault(s, 0) + 1);
        }
        Queue<Pair> queue = new LinkedList<>();
        queue.offer(new Pair(beginWord, 1));
        int level = 1;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; ++i) {
                Pair p = queue.poll();
                String cur = p.first;
                int dist = p.second;
                if (cur.equals(endWord)) return getPath(map, queue, cur, dist);
                char[] chars = cur.toCharArray();
                for (int j = 0; j < chars.length; ++j) {
                    char c = chars[j];
                    for (char nc = 'a'; nc <= 'z'; ++nc) {
                        if (nc!= c) {
                            chars[j] = nc;
                            String next = new String(chars);
                            if (map.containsKey(next) && map.get(next) > 0) {
                                queue.offer(new Pair(next, dist + 1));
                                map.put(next, map.get(next) - 1);
                            }
                        }
                    }
                    chars[j] = c;
                }
            }
            level++;
        }
        return new ArrayList<>();
    }

    private List<List<String>> getPath(Map<String, Integer> map, Queue<Pair> queue, String cur, int dist) {
        List<List<String>> res = new ArrayList<>();
        if (dist == 1) {
            res.add(Arrays.asList(cur));
            return res;
        }
        List<List<String>> paths = getPath(map, queue, cur, dist - 1);
        for (List<String> path : paths) {
            path.add(0, cur);
            res.add(path);
        }
        return res;
    }
}
