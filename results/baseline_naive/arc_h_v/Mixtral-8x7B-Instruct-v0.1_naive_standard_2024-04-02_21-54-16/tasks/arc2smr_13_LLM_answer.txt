def add_ghosts(grid, n):
    new_grid = []
    for row in grid:
        new_row = []
        for cell in row:
            new_row.append(cell)
        new_grid.append(new_row)

    # top left corner
    if (n % 4)!= 0 and (n % 4)!= 3:
        new_grid[0][0] = "h"

    # bottom right corner
    if (n % 4)!= 1 and (n % 4)!= 2:
        new_grid[-1][-1] = "h"

    # top right corner
    if (n % 4)!= 0 and (n % 4)!= 1:
        new_grid[0][-1] = "h"

    # bottom left corner
    if (n % 4)!= 2 and (n % 4)!= 3:
        new_grid[-1][0] = "h"

    # horizontal edges
    if (n % 4) == 0 or (n % 4) == 3:
        for i in range(len(new_grid)):
            new_grid[i][0] = "h"
            new_grid[i][-1] = "h"

    # vertical edges
    if (n % 4) == 0 or (n % 4) == 1:
        for i in range(len(new_grid)):
            new_grid[0][i] = "h"
            new_grid[-1][i] = "h"

    return new_grid
