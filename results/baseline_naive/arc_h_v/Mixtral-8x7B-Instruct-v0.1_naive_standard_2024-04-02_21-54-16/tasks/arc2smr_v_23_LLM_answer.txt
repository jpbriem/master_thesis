const replaceDots = (grid) => {
  const nRows = grid.length;
  const nCols = grid[0].length;

  for (let row = 0; row < nRows; ++row) {
    for (let col = 0; col < nCols; ++col) {
      if (grid[row][col] === ".") continue;

      let countNeighbors = 0;
      // Check top, bottom, left and right neighbors
      if (row > 0 && grid[row - 1][col] === ".") ++countNeighbors;
      if (row + 1 < nRows && grid[row + 1][col] === ".") ++countNeighbors;
      if (col > 0 && grid[row][col - 1] === ".") ++countNeighbors;
      if (col + 1 < nCols && grid[row][col + 1] === ".") ++countNeighbors;

      if (grid[row][col] === "d") {
        if (countNeighbors === 1 || countNeighbors === 2) {
          grid[row][col] = ".";
        } else if (countNeighbors === 3) {
          grid[row][col] = "e";
        }
      } else if (grid[row][col] === "e") {
        if (countNeighbors!== 2) {
          grid[row][col] = ".";
        }
      }
    }
  }
};
