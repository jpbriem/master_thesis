public class Solution {
    public String[][] updateBoard(String[][] board, int[] click) {
        if (board[click[0]][click[1]] == "M") return new String[][]{};
        dfs(board, click[0], click[1]);
        return board;
    }

    private void dfs(String[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) return;
        if (board[x][y]!= null &&!board[x][y].equals(".")) return;
        if (board[x][y] == null) {
            int count = 0;
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (i == 0 && j == 0) continue;
                    int a = x + i, b = y + j;
                    if (a < 0 || a >= board.length || b < 0 || b >= board[0].length) continue;
                    if (board[a][b].equals("M")) count++;
                }
            }
            if (count > 0) {
                board[x][y] = "" + count;
                return;
            } else {
                board[x][y] = "";
            }
        }
        dfs(board, x-1, y);
        dfs(board, x+1, y);
        dfs(board, x, y-1);
        dfs(board, x, y+1);
    }
}
