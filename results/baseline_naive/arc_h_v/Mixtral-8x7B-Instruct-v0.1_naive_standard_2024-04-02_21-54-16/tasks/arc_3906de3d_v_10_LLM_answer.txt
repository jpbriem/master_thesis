class Solution {
    public int[][] updateMatrix(int[][] mat) {
        if (mat == null || mat.length == 0 || mat[0].length == 0) return new int[0][0];
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < mat.length; i++) {
            for (int j = 0; j < mat[i].length; j++) {
                if (mat[i][j] == 0) {
                    queue.offer(i * mat[0].length + j);
                } else {
                    mat[i][j] = -1;
                }
            }
        }
        int[] dirs = {-1, 0, 1, 0, -1};
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            int x = cur / mat[0].length;
            int y = cur % mat[0].length;
            for (int k = 0; k < 4; k++) {
                int nx = x + dirs[k];
                int ny = y + dirs[k+1];
                if (nx >= 0 && nx < mat.length && ny >= 0 && ny < mat[0].length && mat[nx][ny] == -1) {
                    mat[nx][ny] = mat[x][y] + 1;
                    queue.offer(nx * mat[0].length + ny);
                }
            }
        }
        return mat;
    }
}

class Solution {
    public int[][] updateMatrix(int[][] mat) {
        if (mat == null || mat.length == 0 || mat[0].length == 0) return new int[0][0];
        int m = mat.length, n = mat[0].length;
        int[][] res = new int[m][n];
        boolean[][] visited = new boolean[m][n];
        Deque<int[]> deque = new ArrayDeque<>();
        // add all the zeroes to the deque
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (mat[i][j] == 0) {
                    deque.addLast(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }
        // bfs
        int[][] dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        while (!deque.isEmpty()) {
            int size = deque.size();
            for (int i = 0; i < size; i++) {
                int[] curr = deque.pollFirst();
                int r = curr[0];
                int c = curr[1];
                for (int[] dir : dirs) {
                    int nextR = r + dir[0];
                    int nextC = c + dir[1];
                    if (nextR >= 0 && nextR < m && nextC >= 0 && nextC < n &&!visited[nextR][nextC]) {
                        res[nextR][nextC] = res[r][c] + 1;
                        deque.addLast(new int[]{nextR, nextC});
                        visited[nextR][nextC] = true;
                    }
                }
            }
        }
        return res;
    }
}
