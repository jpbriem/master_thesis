        }
        for (int j = 0; j < n; ++j) {
            dfs(board, 0, j);
            dfs(board, m - 1, j);
        }
        List<List<Character>> res = new ArrayList<>();
        for (int i = 0; i < m; ++i) {
            List<Character> list = new ArrayList<>();
            for (int j = 0; j < n; ++j) {
                list.add(board[i][j]);
            }
            res.add(list);
        }
        return res;
    }

    private void dfs(char[][] board, int x, int y) {

    }
