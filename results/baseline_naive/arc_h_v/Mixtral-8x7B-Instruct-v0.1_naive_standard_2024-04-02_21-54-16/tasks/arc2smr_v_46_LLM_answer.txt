  const nRows = grid.length;
  const nCols = grid[0].length;

  // Find the first row and column with a diamond
  let startRow, endRow, startCol, endCol;
  for (let i = 0; i < nRows &&!startRow; i++) {
    if (grid[i].includes('d')) {
      startRow = i;
      endRow = i + 1;
    }
  }
  for (let j = 0; j < nCols &&!startCol; j++) {
    if (grid[0][j] === 'd') {
      startCol = j;
      endCol = j + 1;
    }
  }

  // Fill in diamonds to create columns of connected diamonds
  while (endRow <= nRows) {
    for (let col = startCol; col < endCol; col++) {
      for (let row = startRow; row < endRow; row++) {
        grid[row][col] = 'd';
      }
    }

    // Move down one row
    startRow++;
    endRow++;

    // Check if we need to move over one column
    if (grid[startRow - 1][endCol - 1]!== 'd') {
      startCol++;
      endCol++;
    }
  }

  // Replace all remaining dots with circles
  for (let row = 0; row < nRows; row++) {
    for (let col = 0; col < nCols; col++) {
      if (grid[row][col] === '.') {
        grid[row][col] = 'c';
      }
    }
  }
