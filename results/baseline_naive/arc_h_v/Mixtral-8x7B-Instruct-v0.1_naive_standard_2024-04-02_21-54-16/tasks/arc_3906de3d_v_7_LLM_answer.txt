class Solution {
    public List<List<Character>> removeFloodFill(List<List<Character>> image, int sr, int sc, char newColor) {
        if (image == null || image.size() == 0 || image.get(sr).size() == 0) return image;
        dfs(image, sr, sc, image.get(sr).get(sc), newColor);
        return image;
    }

    private void dfs(List<List<Character>> image, int i, int j, Character oldColor, Character newColor) {
        if (i < 0 || i >= image.size() || j < 0 || j >= image.get(i).size()) return;
        if (image.get(i).get(j)!= oldColor) return;
        image.get(i).set(j, newColor);
        dfs(image, i + 1, j, oldColor, newColor);
        dfs(image, i - 1, j, oldColor, newColor);
        dfs(image, i, j + 1, oldColor, newColor);
        dfs(image, i, j - 1, oldColor, newColor);
    }
}

class Solution {
    public List<List<Character>> removeFloodFill(List<List<Character>> image, int sr, int sc, char newColor) {
        if (image == null || image.size() == 0 || image.get(sr).size() == 0) return image;
        Deque<int[]> queue = new ArrayDeque<>();
        queue.offerLast(new int[]{sr, sc});
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int k = 0; k < size; ++k) {
                int[] cur = queue.pollFirst();
                int r = cur[0];
                int c = cur[1];
                if (r < 0 || r >= image.size() || c < 0 || c >= image.get(r).size() || image.get(r).get(c)!= image.get(sr).get(sc)) continue;
                image.get(r).set(c, newColor);
                queue.offerLast(new int[]{r + 1, c});
                queue.offerLast(new int[]{r - 1, c});
                queue.offerLast(new int[]{r, c + 1});
                queue.offerLast(new int[]{r, c - 1});
            }
        }
        return image;
    }
}
