const removeGarbage = (arr) => {
    let newArr = [];
    for(let i=0;i<arr.length;i++){
        if(!arr[i].includes('g')){
            newArr.push(arr[i]);
        } else {
            let indexOfG = arr[i].indexOf('g');
            let tempArr = [...arr[i]];
            while(tempArr.includes('g') && tempArr.includes('<')){
                let indexOfLBracket = tempArr.indexOf('<');
                let indexOfRBracket = tempArr.slice(indexOfLBracket).findIndex((el, idx)=> el === '>');
                tempArr.splice(indexOfLBracket, indexOfRBracket+1);
            }
            tempArr.splice(0, indexOfG);
            newArr.push(tempArr);
        }
    }
    return newArr;
}
