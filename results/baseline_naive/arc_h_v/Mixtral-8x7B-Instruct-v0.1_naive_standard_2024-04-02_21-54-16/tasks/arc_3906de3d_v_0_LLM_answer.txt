const bfs = (grid) => {
    const queue = [[0, 0]]
    grid[0][0] = "v"
    while(queue.length > 0){
        let curr = queue.shift()
        if(curr[0] === grid.length - 1 && curr[1] === grid[0].length - 1) return true
        for(let i = 0; i < 4; i++){
            let x = curr[0] + dx[i]
            let y = curr[1] + dy[i]
            if(x >= 0 && x < grid.length && y >= 0 && y < grid[0].length && grid[x][y]!== "v"){
                if(grid[x][y] === ".") grid[x][y] = "v"
                else if(grid[x][y] === "a") return false
                queue.push([x, y])
            }
        }
    }
}

const removeObstacles = (grid) => {
    let dx = [-1, 0, 1, 0]
    let dy = [0, 1, 0, -1]
    for(let i = 0; i < grid.length; i++)
        for(let j = 0; j < grid[0].length; j++)
            if(grid[i][j] === "a")
                if(!bfs(grid)) return grid.map((e) => e.map(() => "#"))
    return grid
};
