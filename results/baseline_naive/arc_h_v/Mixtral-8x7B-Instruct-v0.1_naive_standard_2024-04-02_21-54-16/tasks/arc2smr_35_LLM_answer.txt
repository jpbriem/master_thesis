class Solution {
    public List<List<String>> solve(char[][] board) {
        if (board == null || board.length == 0) return new ArrayList<>();
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; i++) {
            dfs(board, i, 0);
            dfs(board, i, n - 1);
        }
        for (int j = 0; j < n; j++) {
            dfs(board, 0, j);
            dfs(board, m - 1, j);
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 'P') board[i][j] = 'O';
                else if (board[i][j] == 'o') board[i][j] = 'X';
            }
        }
        return Arrays.asList(Arrays.stream(board).map(x -> Arrays.asList(x)).collect(Collectors.toList()));
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= 'O') return;
        board[x][y] = 'o';
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
}
