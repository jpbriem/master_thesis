class Solution {
    public List<List<String>> updateBoard(List<List<String>> board, int[] click) {
        if (board.get(click[0]).get(click[1]).equals("x")) return board;
        dfs(board, click[0], click[1]);
        return board;
    }

    private void dfs(List<List<String>> board, int x, int y) {
        if (x < 0 || x >= board.size() || y < 0 || y >= board.get(0).size()) return;
        if (!board.get(x).get(y).equals(".")) return;
        int count = getMineCount(board, x, y);
        if (count > 0) {
            board.get(x).set(y, String.valueOf(count));
            return;
        } else {
            board.get(x).set(y, "B");
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (!(i == 0 && j == 0)) {
                        dfs(board, x + i, y + j);
                    }
                }
            }
            if (board.get(x).get(y).equals("B")) {
                board.get(x).set(y, ".");
            }
        }
    }

    private int getMineCount(List<List<String>> board, int x, int y) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if ((i!= 0 || j!= 0) && isMine(board, x + i, y + j)) {
                    count++;
                }
            }
        }
        return count;
    }

    private boolean isMine(List<List<String>> board, int x, int y) {
        if (x < 0 || x >= board.size() || y < 0 || y >= board.get(0).size()) return false;
        return board.get(x).get(y).equals("x");
    }
}
