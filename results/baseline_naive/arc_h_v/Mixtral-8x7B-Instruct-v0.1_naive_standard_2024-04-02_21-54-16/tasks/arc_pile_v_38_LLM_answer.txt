public class Solution {
    public String[][] updateBoard(String[][] board, int[] click) {
        if (board[click[0]][click[1]] == "M") {
            board[click[0]][click[1]] = "X";
            return board;
        }
        dfs(board, click[0], click[1]);
        return board;
    }

    private void dfs(String[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= ".") {
            return;
        }
        int count = getCount(board, x, y);
        if (count > 0) {
            board[x][y] = "" + count;
            return;
        }
        board[x][y] = "";
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                dfs(board, x + i, y + j);
            }
        }
    }

    private int getCount(String[][] board, int x, int y) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                int a = x + i, b = y + j;
                if (a < 0 || a >= board.length || b < 0 || b >= board[0].length || board[a][b]!= "M") continue;
                count++;
            }
        }
        return count;
    }
}
