    private void dfs(List<List<String>> board, int[] click) {
        if (click[0] < 0 || click[0] >= board.size() || click[1] < 0 || click[1] >= board.get(0).size()) return;
        String cell = board.get(click[0]).get(click[1]);
        if (!cell.equals(".") &&!cell.equals("E")) return;
        if (cell.equals("E")) {
            int count = 0;
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (i == 0 && j == 0) continue;
                    int x = click[0] + i, y = click[1] + j;
                    if (x < 0 || x >= board.size() || y < 0 || y >= board.get(0).size()) continue;
                    if (board.get(x).get(y).equals("M")) ++count;
                }
            }

                board.get(click[0]).set(click[1], Integer.toString(count));
                return;

    private void dfs(List<List<String>> board, int r, int c) {

        String cell = board.get(r).get(c);
