public class Solution {
    public String[][] solve(String[][] board) {
        if (board == null || board.length == 0) return new String[0][];
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; ++i) {
            dfs(board, i, 0);
            dfs(board, i, n - 1);
        }
        for (int j = 1; j + 1 < n; ++j) {
            dfs(board, 0, j);
            dfs(board, m - 1, j);
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if ("O".equals(board[i][j])) board[i][j] = "X";
                else if ("#".equals(board[i][j])) board[i][j] = "O";
            }
        }
        return board;
    }

    private void dfs(String[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || "#".equals(board[x][y]) || "X".equals(board[x][y])) return;
        board[x][y] = "#";
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
