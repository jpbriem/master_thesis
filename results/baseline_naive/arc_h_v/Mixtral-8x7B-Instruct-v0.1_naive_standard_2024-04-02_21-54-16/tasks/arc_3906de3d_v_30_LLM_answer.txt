class Solution {
    public List<List<Character>> extendGrid(List<List<Character>> grid) {
        int m = grid.size();
        int n = grid.get(0).size();

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid.get(i).get(j) == 'i') {
                    dfs(grid, i, j);
                }
            }
        }

        return grid;
    }

    private void dfs(List<List<Character>> grid, int x, int y) {
        if (x < 0 || x >= grid.size() || y < 0 || y >= grid.get(0).size()) {
            return;
        }

        if (grid.get(x).get(y)!= 'i') {
            return;
        }

        grid.get(x).set(y, 'e');

        dfs(grid, x + 1, y);
        dfs(grid, x - 1, y);
        dfs(grid, x, y + 1);
        dfs(grid, x, y - 1);
    }
}
