class Solution {
    public List<List<Character>> rotateMatrix(List<List<Character>> matrix) {
        int n = matrix.size();
        for (int i = 0; i < n / 2; ++i) {
            for (int j = i; j < n - i - 1; ++j) {
                char tmp = matrix.get(i).get(j);
                matrix.get(i).set(j, matrix.get(n - j - 1).get(i));
                matrix.get(n - j - 1).set(i, matrix.get(n - i - 1).get(n - j - 1));
                matrix.get(n - i - 1).set(n - j - 1, matrix.get(j).get(n - i - 1));
                matrix.get(j).set(n - i - 1, tmp);
            }
        }
        return matrix;
    }
}
