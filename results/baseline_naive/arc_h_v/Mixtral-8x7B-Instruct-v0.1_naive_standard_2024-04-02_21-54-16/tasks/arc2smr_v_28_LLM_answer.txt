    public char[][] flipAndInvertImage(char[][] A) {
        for (int i = 0; i < A.length; ++i) {
            int j = 0, k = A[i].length - 1;
            while (j <= k) {
                if (A[i][j] == A[i][k]) {
                    A[i][j] ^= 1;
                    A[i][k] ^= 1;
                }
                j++;
                k--;
            }
        }
        return A;
    }
