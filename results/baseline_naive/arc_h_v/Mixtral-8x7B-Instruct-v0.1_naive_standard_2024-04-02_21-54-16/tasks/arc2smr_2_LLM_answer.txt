const fillZigzag = (arr) => {
  const nRows = arr.length;
  if (!nRows ||!arr[0].length) return [];

  let direction = -1, i = 0, j = 0;
  const result = Array(nRows).fill().map(() => Array(arr[0].length).fill('.'));

  while (i < nRows && j < arr[0].length) {
    // move down and right
    while (i + direction >= 0 && i + direction < nRows && j < arr[0].length) {
      if (arr[i][j]!== '.') result[i][j] = arr[i][j];
      i += direction;
      j++;
    }

    // change direction
    direction *= -1;
    i += direction;
  }

  return result;
};
