const replaceGWithE = (grid) => {
  const nRows = grid.length;
  const nCols = grid[0].length;

  for (let row = 0; row < nRows; row++) {
    for (let col = 0; col < nCols; col++) {
      if (grid[row][col] === ".") continue;

      // Check up, down, left and right
      let count = 0;
      if (row > 0 && grid[row - 1][col] === "g") count++;
      if (row + 1 < nRows && grid[row + 1][col] === "g") count++;
      if (col > 0 && grid[row][col - 1] === "g") count++;
      if (col + 1 < nCols && grid[row][col + 1] === "g") count++;

      grid[row][col] = count > 0? "e" : "g";
    }
  }
};
