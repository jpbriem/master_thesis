class Solution {
    public List<List<String>> floodFill(List<List<String>> image, int sr, int sc, String newColor) {
        if (image == null || image.size() == 0 || image.get(sr).size() == 0) return image;
        dfs(image, sr, sc, image.get(sr).get(sc), newColor);
        return image;
    }

    private void dfs(List<List<String>> image, int i, int j, String curColor, String newColor) {
        if (i < 0 || i >= image.size() || j < 0 || j >= image.get(i).size()) return;
        if (!curColor.equals(image.get(i).get(j))) return;
        image.get(i).set(j, newColor);
        dfs(image, i + 1, j, curColor, newColor);
        dfs(image, i - 1, j, curColor, newColor);
        dfs(image, i, j + 1, curColor, newColor);
        dfs(image, i, j - 1, curColor, newColor);
    }
}

class Solution {
    public List<List<String>> floodFill(List<List<String>> image, int sr, int sc, String newColor) {
        if (image == null || image.size() == 0 || image.get(sr).size() == 0) return image;
        Stack<Integer> stack = new Stack<>();
        stack.push(sr * image.get(0).size() + sc);
        while (!stack.isEmpty()) {
            int idx = stack.pop();
            int r = idx / image.get(0).size(), c = idx % image.get(0).size();
            if (image.get(r).get(c).equals(newColor)) continue;
            if (r < 0 || r >= image.size() || c < 0 || c >= image.get(r).size()) continue;
            if (!image.get(r).get(c).equals(image.get(sr).get(sc))) continue;
            image.get(r).set(c, newColor);
            stack.push((r + 1) * image.get(0).size() + c);
            stack.push((r - 1) * image.get(0).size() + c);
            stack.push(r * image.get(0).size() + c + 1);
            stack.push(r * image.get(0).size() + c - 1);
        }
        return image;
    }
}
