class Solution {
    public char[][] fill(char[][] grid) {
        if (grid == null || grid.length == 0) return grid;
        int m = grid.length, n = grid[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j]!= 'F') continue;
                dfs(grid, i, j);
            }
        }
        return grid;
    }

    private void dfs(char[][] grid, int x, int y) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y]!= 'F') return;
        grid[x][y] = 'T';
        dfs(grid, x + 1, y);
        dfs(grid, x - 1, y);
        dfs(grid, x, y + 1);
        dfs(grid, x, y - 1);
    }
}

class Solution {
    public char[][] fill(char[][] grid) {
        if (grid == null || grid.length == 0) return grid;
        int m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (visited[i][j]) continue;
                if (grid[i][j] == 'F') {
                    bfs(grid, visited, i, j);
                } else {
                    visited[i][j] = true;
                }
            }
        }
        return grid;
    }

    private void bfs(char[][] grid, boolean[][] visited, int x, int y) {
        Queue<Integer> queue = new LinkedList<>();
        queue.offer((x << 16) | y);
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            int cx = cur >> 16, cy = cur & ((1 << 16) - 1);
            if (cx < 0 || cx >= grid.length || cy < 0 || cy >= grid[0].length || visited[cx][cy]) continue;
            visited[cx][cy] = true;
            if (grid[cx][cy] == 'F') grid[cx][cy] = 'T';
            queue.offer(((cx + 1) << 16) | cy);
            queue.offer(((cx - 1) << 16) | cy);
            queue.offer((cx << 16) | (cy + 1));
            queue.offer((cx << 16) | (cy - 1));
        }
    }
}
