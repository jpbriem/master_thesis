      x >= board.length ||

      return;
    board[x][y] = "#";

    dfs(x - 1, y);

    for (let j = 0; j < board[i].length; j++) {
      if (board[i][j] === ".") {
        let isDangerous = false;
        // check up and down
        for (let k = i - 1; k <= i + 1; k++) {
