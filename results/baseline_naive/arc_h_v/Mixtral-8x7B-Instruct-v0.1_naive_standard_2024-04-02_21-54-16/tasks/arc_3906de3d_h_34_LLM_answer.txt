function moveDots(arr) {
    for (let i = 0; i < arr.length; i++) {
        let row = arr[i];
        for (let j = 0; j < row.length; j++) {
            if (row[j] === ".") {
                // check left and right of the dot, find first non-dot element
                let l = j - 1;
                while (l >= 0 && row[l] === ".") {
                    l--;
                }
                let r = j + 1;
                while (r < row.length && row[r] === ".") {
                    r++;
                }
                // replace the dot with the value between two non-dots
                if (l < 0 || r >= row.length) {
                    row[j] = ".";
                } else {
                    row[j] = String.fromCharCode(Math.min(row[l].charCodeAt(), row[r].charCodeAt()) + 1);
                }
            }
        }
    }
}
