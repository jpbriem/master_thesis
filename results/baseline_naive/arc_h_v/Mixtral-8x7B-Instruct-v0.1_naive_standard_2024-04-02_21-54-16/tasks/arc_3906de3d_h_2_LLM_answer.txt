const removeInvalid = (arr) => {
    let i = arr.length - 1;
    while(i >= 0){
        if(!isValidRow(arr[i])){
            arr.splice(i, 1);
        }
        i--;
    }
}

const isValidRow = (row) => {
    const firstDotIndex = row.findIndex((el) => el === ".");
    const lastDotIndex = row.lastIndexOf(".");
    return!firstDotIndex ||!lastDotIndex || (firstDotIndex === 0 && lastDotIndex === row.length-1);
}
