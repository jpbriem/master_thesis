def expand(grid):
    for i in range(len(grid)):
        if "e" in grid[i]:
            j = grid[i].index("e")

def fill(grid, x, y):
    if x >= len(grid) or y >= len(grid[x]) or x < 0 or y < 0:
        return
    if grid[x][y]!= ".":
        return
    grid[x][y] = "f"
    fill(grid, x+1, y)

def expandGrid(grid):
    while True:
        expanded = False
        new_grid = []
        for row in grid:
            new_row = list(row)
            expand(new_row)
            new_grid.append(new_row)
        for i in range(len(new_grid)):
            for j in range(len(new_grid[i])):
                if new_grid[i][j] == "e":
                    fill(new_grid, i, j)
                    expanded = True
        if not expanded:
            break
        grid = new_grid
    return grid
