const reorderLogFiles = (logs) => {
  const letterLogs = [];
  const digitLogs = [];

  for (let log of logs) {
    if (isNaN(log.split(' ')[1][0])) {
      letterLogs.push(log);
    } else {
      digitLogs.push(log);
    }
  }

  letterLogs.sort((a, b) => {
    let aContent = a.slice(a.indexOf(' ') + 1);
    let bContent = b.slice(b.indexOf(' ') + 1);

    return aContent > bContent? 1 : -1;
  });

  return [...letterLogs,...digitLogs];
};
