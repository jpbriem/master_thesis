const updateBoard = (board, click) => {
    const [x, y] = click;
    if (board[x][y] === "M") {

        return board;
    } else {
        dfs(board, x, y);
        let count = 0;
        for (let i = -1; i <= 1; i++) {
            for (let j = -1; j <= 1; j++) {
