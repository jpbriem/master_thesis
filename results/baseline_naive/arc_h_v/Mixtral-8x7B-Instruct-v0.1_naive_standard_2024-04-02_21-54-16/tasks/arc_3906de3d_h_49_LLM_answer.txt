  for (let i = 0; i < m; i++) {
    let j = 0;

      j++;
    }

      let k = j;

        if (grid[i][k] === "i" || grid[i][k] === "e") {
          grid[i][j] = "i";

          break;
        }
        k++;
      }
    }
  }
