class Solution {
    public char[][] fill(char[][] image, int x, int y) {
        if (x < 0 || x >= image.length || y < 0 || y >= image[0].length) return new char[0][];
        if (image[x][y] == '0') return new char[0][];

        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{x, y});
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            image[cur[0]][cur[1]] = '0';
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (i!= 0 || j!= 0 && Math.abs(i) + Math.abs(j) > 1) continue;
                    int nx = cur[0] + i, ny = cur[1] + j;
                    if (nx < 0 || nx >= image.length || ny < 0 || ny >= image[0].length) continue;
                    if (image[nx][ny] == '1') {
                        queue.offer(new int[]{nx, ny});
                    }
                }
            }
        }

        return image;
    }

    public void floodFill(char[][] image, int sr, int sc, char newColor) {
        if (sr < 0 || sr >= image.length || sc < 0 || sc >= image[0].length) return;
        if (image[sr][sc] == newColor) return;

        char oldColor = image[sr][sc];
        image = fill(image, sr, sc);
        if (oldColor == '0' && newColor!= '0') {
            image[sr][sc] = newColor;
        }
    }
}
