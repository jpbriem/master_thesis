const removeGarbage = (grid) => {
    const m = grid.length;
    const n = grid[0].length;

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (grid[i][j] === ".") continue;

            let count = 0;
            // check up, down, left and right
            for (let k = -1; k <= 1; k += 2) {
                if (i + k >= 0 && i + k < m && grid[i+k][j]!== ".") {
                    count++;
                }
                if (j + k >= 0 && j + k < n && grid[i][j+k]!== ".") {
                    count++;
                }
            }

            if (count > 1) {
                grid[i][j] = ".";
            }
        }
    }
};
