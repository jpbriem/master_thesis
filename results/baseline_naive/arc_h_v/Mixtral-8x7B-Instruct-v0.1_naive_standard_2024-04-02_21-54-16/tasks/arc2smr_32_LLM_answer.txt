  const m = board.length;
  if (!m) return [];
  const n = board[0].length;
  let res = Array(m).fill().map(() => Array(n).fill('.'));
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (board[i][j] === 'd') {
        dfs(i, j);
      }
    }
  }
  return res;

  function dfs(x, y) {
    if (x < 0 || x >= m || y < 0 || y >= n || board[x][y]!== 'd') return;
    res[x][y] = 'b';
    dfs(x - 1, y);
    dfs(x + 1, y);
    dfs(x, y - 1);
    dfs(x, y + 1);
  }
