class Solution {
    public List<List<String>> solve(List<List<String>> board) {
        if (board == null || board.size() < 3) return board;

        int m = board.size(), n = board.get(0).size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if ((i == 0 || i == m - 1 || j == 0 || j == n - 1) && board.get(i).get(j).equals("h")) {
                    dfs(board, i, j);
                }
            }
        }

        return board;
    }

    private void dfs(List<List<String>> board, int x, int y) {
        if (!isValid(board, x, y)) return;

        String cur = board.get(x).get(y);
        if (!cur.equals("h")) return;

        board.get(x).set(y, ".");
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }

    private boolean isValid(List<List<String>> board, int x, int y) {
        return x >= 0 && x < board.size() && y >= 0 && y < board.get(0).size();
    }
}
