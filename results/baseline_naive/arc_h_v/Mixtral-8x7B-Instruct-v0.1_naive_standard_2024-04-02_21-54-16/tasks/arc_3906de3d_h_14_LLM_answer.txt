public class Solution {
    public char[][] fill(char[][] grid) {
        if (grid == null || grid.length == 0) return new char[0][];
        int m = grid.length, n = grid[0].length;
        for (int i = 0; i < m; ++i) {
            if (grid[i][0]!= 'c') dfs(grid, i, 0);
            if (grid[i][n - 1]!= 'c') dfs(grid, i, n - 1);
        }
        for (int j = 0; j < n; ++j) {
            if (grid[0][j]!= 'c') dfs(grid, 0, j);
            if (grid[m - 1][j]!= 'c') dfs(grid, m - 1, j);
        }
        return grid;
    }

    private void dfs(char[][] grid, int x, int y) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length || grid[x][y] == 'c' || grid[x][y]!= '.' && grid[x][y]!= 'a') return;
        grid[x][y] = 'a';
        dfs(grid, x + 1, y);
        dfs(grid, x - 1, y);
        dfs(grid, x, y + 1);
        dfs(grid, x, y - 1);
    }
}
