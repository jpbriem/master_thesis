class Solution {
    public List<List<Character>> removeGarbage(List<List<Character>> board) {
        int m = board.size(), n = board.get(0).size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n - 1; ++j) {
                if (board.get(i).get(j) == 'g' && board.get(i).get(j + 1) == 'g') {
                    boolean hasB = false, isValid = true;
                    // check up and down
                    for (int k = j - 1; k >= 0 &&!hasB; --k) {
                        if (board.get(i).get(k)!= 'g') {
                            hasB = true;
                        } else if (k > 0 && board.get(i).get(k - 1) == 'g') {
                            isValid = false;
                            break;
                        }
                    }
                    if (!isValid) continue;
                    for (int k = j + 2; k < n &&!hasB; ++k) {
                        if (board.get(i).get(k)!= 'g') {
                            hasB = true;
                        } else if (k < n - 1 && board.get(i).get(k + 1) == 'g') {
                            isValid = false;
                            break;
                        }
                    }
                    if (!isValid) continue;
                    // check left and right
                    for (int k = i - 1; k >= 0 &&!hasB; --k) {
                        if (board.get(k).get(j)!= 'g') {
                            hasB = true;
                        } else if (k > 0 && board.get(k - 1).get(j) == 'g') {
                            isValid = false;
                            break;
                        }
                    }
                    if (!isValid) continue;
                    for (int k = i + 1; k < m &&!hasB; ++k) {
                        if (board.get(k).get(j)!= 'g') {
                            hasB = true;
                        } else if (k < m - 1 && board.get(k + 1).get(j) == 'g') {
                            isValid = false;
                            break;
                        }
                    }
                    if (!isValid) continue;
                    // remove garbage
                    if (hasB) {
                        for (int k = j; k <= j + 1; ++k) {
                            for (int l = i - 1; l <= i + 1; ++l) {
                                if (l >= 0 && l < m && k >= 0 && k < n) {
                                    board.get(l).set(k, '.');
                                }
                            }
                        }
                    } else {
                        for (int k = j; k <= j + 2; ++k) {

                                if (l >= 0 && l < m && k >= 0 && k < n) {
                                    board.get(l).set(k, '.');

                            }

                    }
                }
