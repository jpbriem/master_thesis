    }

    private void dfs(char[][] board, int[] click) {
        if (click[0] < 0 || click[0] >= board.length || click[1] < 0 || click[1] >= board[0].length) return;
        if (board[click[0]][click[1]] == 'E' || board[click[0]][click[1]] == 'X') return;
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                int x = click[0] + i;
                int y = click[1] + j;
                if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) continue;
                if (board[x][y] == 'M') ++count;
            }
        }
        if (count > 0) {
            board[click[0]][click[1]] = (char)(count + '0');
            return;
        } else {
            board[click[0]][click[1]] = 'B';
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (i == 0 && j == 0) continue;
                    int x = click[0] + i;
                    int y = click[1] + j;
                    if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) continue;
                    if (board[x][y]!= 'B' && board[x][y]!= 'X') {
                        dfs(board, new int[]{x, y});
                    }
                }
            }
        }
    }
