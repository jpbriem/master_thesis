  const nRows = grid.length;
  const nCols = grid[0].length;

  for (let i = 0; i < nRows; i++) {
    let row = grid[i];
    for (let j = 0; j < nCols; j++) {
      if (row[j] === ".") continue;

      // check up
      if (i > 0 && grid[i - 1][j]!== "#") {
        row[j] = ".";
      }

      // check down
      if (i + 1 < nRows && grid[i + 1][j]!== "#") {
