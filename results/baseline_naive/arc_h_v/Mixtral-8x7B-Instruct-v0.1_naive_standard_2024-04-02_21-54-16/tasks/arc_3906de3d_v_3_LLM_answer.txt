class Solution {
    public int[][] updateBoard(int[][] board, int[] click) {
        if (board[click[0]][click[1]] == -1) return board;
        dfs(board, click[0], click[1]);
        return board;
    }

    private void dfs(int[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) return;
        if (board[x][y]!= 0 && board[x][y]!= -1) return;

        if (board[x][y] == 0) {
            int count = getCount(board, x, y);
            if (count > 0) {
                board[x][y] = count;
                return;
            } else {
                board[x][y] = -2;
                for (int i = -1; i <= 1; ++i) {
                    for (int j = -1; j <= 1; ++j) {
                        if (!(i == 0 && j == 0)) {
                            dfs(board, x + i, y + j);
                        }
                    }
                }
                board[x][y] = -1;
            }
        } else {
            board[x][y] = -1;
        }
    }

    private int getCount(int[][] board, int x, int y) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if ((i!= 0 || j!= 0) && board[x + i][y + j] == 9) {
                    count++;
                }
            }
        }
        return count;
    }
}
