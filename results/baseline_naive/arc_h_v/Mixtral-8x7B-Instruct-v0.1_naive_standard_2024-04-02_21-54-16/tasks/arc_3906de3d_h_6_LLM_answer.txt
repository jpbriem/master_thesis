const removeDupes = (arr) => {
    let result = [];
    for(let i=0; i<arr.length; i++) {
        if(!result[result.length-1] ||!result[result.length-1].includes(arr[i])) {
            result.push(arr[i]);
        }
    }
    return result;
}

const solve = (board) => {
    const m = board.length, n = board[0].length;
    // iterate through the rows and columns of the matrix
    for(let j=n-1; j>=0; j--) {
        for(let i=m-1; i>=0; i--) {
            // check if current cell is a dot or not
            if(board[i][j] === ".") continue;
            // get all possible directions to move from this cell
            let dirs = [[-1, 0], [0, -1], [1, 0], [0, 1]], count = 0;
            // filter out invalid moves based on boundary conditions
            dirs = dirs.filter((dir) => ((i+dir[0]) >= 0 && (i+dir[0]) < m) && ((j+dir[1]) >= 0 && (j+dir[1]) < n));
            // iterate over valid moves
            for(let k=0; k<dirs.length; k++) {
                let newI = i + dirs[k][0];
                let newJ = j + dirs[k][1];
                // increment count if next cell has same letter as current cell
                if(board[newI][newJ] === board[i][j]) count++;
            }
            // replace current cell with number of similar cells around it
            board[i][j] = String(count);
        }
    }
    // iterate through the rows and columns of the matrix again
    for(let j=n-1; j>=0; j--) {
        for(let i=m-1; i>=0; i--) {
            // check if current cell is a dot or not
            if(board[i][j] === ".") continue;
            // get all possible directions to move from this cell
            let dirs = [[-1, 0], [0, -1], [1, 0], [0, 1]], count = 0;
            // filter out invalid moves based on boundary conditions
            dirs = dirs.filter((dir) => ((i+dir[0]) >= 0 && (i+dir[0]) < m) && ((j+dir[1]) >= 0 && (j+dir[1]) < n));
            // initialize an array to store unique letters in adjacent cells
            let uniqLetters = [];
            // iterate over valid moves
            for(let k=0; k<dirs.length; k++) {
                let newI = i + dirs[k][0];
                let newJ = j + dirs[k][1];
                // push letter into array only once even if there are multiple occurrences
                if(!uniqLetters.includes(board[newI][newJ])) uniqLetters.push(board[newI][newJ]);
            }
            // convert array to string and then replace current cell value
            board[i][j] = uniqLetters.join("");
        }
    }
    // flatten the matrix and pass it to function that removes duplicates
    return removeDupes(board.flat()).map((val) => val===""? ["."] : [val]);
};
