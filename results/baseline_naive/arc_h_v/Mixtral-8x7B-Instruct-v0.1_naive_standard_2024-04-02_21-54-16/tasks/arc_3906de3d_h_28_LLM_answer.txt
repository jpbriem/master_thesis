class Solution {
    public List<List<Character>> rotateMatrix(List<List<Character>> matrix) {
        int n = matrix.size();
        // transpose the matrix
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                char tmp = matrix.get(i).get(j);
                matrix.get(i).set(j, matrix.get(j).get(i));
                matrix.get(j).set(i, tmp);
            }
        }
        // reverse each row
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n / 2; ++j) {
                char tmp = matrix.get(i).get(j);
                matrix.get(i).set(j, matrix.get(i).get(n - j - 1));
                matrix.get(i).set(n - j - 1, tmp);
            }
        }
        return matrix;
    }
