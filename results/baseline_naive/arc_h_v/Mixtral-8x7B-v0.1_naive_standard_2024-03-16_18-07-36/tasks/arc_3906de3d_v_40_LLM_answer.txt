def floodFill(grid):
    # find all starting points where we can start filling from
    starts = []
    for i in range(len(grid)):
        for j in range(len(grid[i])):
            if grid[i][j] == '.' or grid[i][j] == '#':
                continue
            else:
                starts.append((i, j))

    def fill(x, y, color):
        nonlocal grid
        stack = [(x,y)]
        while len(stack)>0:
            x, y = stack[-1]
            del stack[-1]
            if not ((0 <= x < len(grid)) and (0<=y<len(grid[x]))):
                return False
            elif grid[x][y]!= ".":
                return True
            else:
                grid[x][y] = color
                stack += [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]
        return False

    colors = {}
    newGrid = deepcopy(grid)
    for s in starts:
        c = "#" + str(s)[::-1].replace("(","").replace(")", "")
        if c not in colors:
            colors[c] = chr(ord('a')+(len(colors)+1))
        newGrid[s[0]][s[1]] = colors[c]
        filled = fill(s[0]+1, s[1], colors[c])
        if filled:
            print("Filled")
        else:
            print("Not Filled")
    return newGrid
```
