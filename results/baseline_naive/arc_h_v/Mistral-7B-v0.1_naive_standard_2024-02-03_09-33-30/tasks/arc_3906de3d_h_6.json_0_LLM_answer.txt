class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        for (int i=0;i<m;++i){
            for (int j=0;j<n;++j){
                if (mat[i][j] == 0){
                    dfs(mat, i, j);
                }
            }
        }
        return mat;
    }
private:
    void dfs(vector<vector<int>> &mat, int x, int y){
        if (x < 0 || x >= mat.size() || y < 0 || y >= mat[0].size()){
            return;
        }
        if (mat[x][y]!= 0){
            return;
        }
        mat[x][y] = min(min(dfs(mat, x+1, y), dfs(mat, x-1, y)), min(dfs(mat, x, y+1), dfs(mat, x, y-1))) + 1;
    }
