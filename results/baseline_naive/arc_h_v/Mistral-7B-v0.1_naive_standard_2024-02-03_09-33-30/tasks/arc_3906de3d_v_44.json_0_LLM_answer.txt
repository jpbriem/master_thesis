class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n));
        queue<pair<int, int>> q;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!visited[i][j] && mat[i][j]) {
                    q.push({i, j});
                    while (!q.empty()) {
                        auto cur = q.front(); q.pop();
                        for (auto dir : {{0, -1}, {0, 1}, {-1, 0}, {1, 0}}) {
                            int x = cur.first + dir[0];
                            int y = cur.second + dir[1];
                            if (x >= 0 && x < m && y >= 0 && y < n &&!mat[x][y] &&!visited[x][y]) {
                                mat[cur.first][cur.second]++;
                                mat[x][y] = 1;
                                visited[x][y] = true;
                                q.push({x, y});
                            }
                        }
                    }
                }
            }
        }
        return mat;
    }
};
