The idea is that we start from the first row of the matrix, if there is a `0`, then we mark it as visited and continue to search for other `0`. If we find another `0`, we mark it as visited too. Then we move on to the next row.

If we cannot find any more `0` in the current row, we go back to the previous row and check whether there is an unvisited `0` in the previous row. If yes, we mark it as visited and continue to search for other `0`. Otherwise, we go back to the previous row again until we reach the top row.

class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        queue<pair<int, int>> q;
        vector<vector<bool>> vis(m, vector<bool>(n));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!mat[i][j]) {
                    q.push({i, j});
                    vis[i][j] = true;
                }
            }
        }
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            int x = cur.first, y = cur.second;
            for (auto dir : {{-1, 0}, {1, 0}, {0, -1}, {0, 1}}) {
                int newX = x + dir[0];
                int newY = y + dir[1];
                if (newX >= 0 && newX < m && newY >= 0 && newY < n &&!vis[newX][newY] && mat[newX][newY]) {
                    mat[newX][newY] = mat[x][y] + 1;
                    vis[newX][newY] = true;
                    q.push({newX, newY});
                }
            }
        }
        return mat;
    }
};
