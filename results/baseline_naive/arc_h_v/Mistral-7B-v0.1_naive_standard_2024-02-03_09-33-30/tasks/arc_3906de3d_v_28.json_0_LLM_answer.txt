def maxMatrix(matrix):
    # base case
    if not any([any(row) for row in matrix]):
        return []

    # find the first row with zeroes
    row = next((i for i, r in enumerate(matrix) if any(r)), None)

    # if all elements in this row are zeros, then remove this row
    if not any(row):
        return [m[:row] + m[row+1:] for m in matrix]
