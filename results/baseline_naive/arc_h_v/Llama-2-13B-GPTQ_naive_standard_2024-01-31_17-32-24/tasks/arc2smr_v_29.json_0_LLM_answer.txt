class Solution(object):
    def convert(self, s):
        """
        :type s: List[str]
        :rtype: List[List[int]]
        """
        if not s: return []
        n = len(s)
        ans = [[0 for _ in range(n)] for _ in range(n)]
        for i in xrange(n):
            for j in xrange(n):
                if s[i][j] == '#':
                    ans[i][j] = -1
                elif s[i][j] == '.':
                    ans[i][j] = 0
                else:
                    ans[i][j] = 1
        return ans
