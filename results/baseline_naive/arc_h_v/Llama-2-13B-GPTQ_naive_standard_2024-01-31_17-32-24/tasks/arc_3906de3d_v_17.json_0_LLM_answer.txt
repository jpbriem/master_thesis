class Solution:
    def convert(self, s: List[str]) -> List[List[str]]:
        if not s: return []
        
        n = len(s)
        m = len(s[0])
        ans = [[None for _ in range(m)] for _ in range(n)]
        
        for i in range(n):
            for j in range(m):
                if s[i][j].isalpha():
                    ans[i][j] = s[i][j]
                else:
                    ans[i][j] = '#'
            
        for i in range(n):
            for j in range(m):
                if ans[i][j] == '.': continue
                elif ans[i][j] == 'g':
                    ans[i][j] = 'G'
                elif ans[i][j] == 'b':
                    ans[i][j] = 'B'
                else:
                    raise ValueError('Invalid character')
        
        return ans
