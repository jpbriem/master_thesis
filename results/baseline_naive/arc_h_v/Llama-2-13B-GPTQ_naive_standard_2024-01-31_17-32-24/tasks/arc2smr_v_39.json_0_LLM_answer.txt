# Definition for an Intersection Node.
class IntersectionNode(object):
    def __init__(self, x, y, z):
        self.x = x
        self.y = y
        self.z = z

    def __repr__(self):

class Solution(object):
    def intersection(self, nodes):
        """
        :type nodes: List[IntersectionNode]
        :rtype: List[List[int]]
        """
        if not nodes or len(nodes) == 0:
            return []
        
        # find all edges in graph
        edges = set()
        for i in range(len(nodes)):
            for j in range(i+1, len(nodes)):
                edges.add([nodes[i].x, nodes[j].x])
                
        # build a map of node to its neighbors
        neighborMap = {}
        for edge in edges:
            neighborMap[edge[0]] = neighborMap.get(edge[0], []) + [edge[1]]
            neighborMap[edge[1]] = neighborMap.get(edge[1], []) + [edge[0]]
            
        # build a map of node to its predecessor
        predecessorMap = {}
        for key in neighborMap:
            predecessorMap[key] = min(neighborMap[key])
            
        # build a map of node to its successor
        successorMap = {}
        for key in neighborMap:
            successorMap[key] = max(neighborMap[key])
            
        # build a map of node to its distance from source
        distanceMap = {}
        for key in neighborMap:
            distanceMap[key] = distanceMap.get(predecessorMap[key], float('inf')) + 1
            
        # build a map of node to its distance from sink
        sinkDistanceMap = {}
        for key in neighborMap:
            sinkDistanceMap[key] = sinkDistanceMap.get(successorMap[key], float('inf')) + 1

        paths = {}
        for srcKey in neighborMap:
            for dstKey in neighborMap:
                if (srcKey!= dstKey) and (distanceMap[srcKey] < distanceMap[dstKey]):
                    paths[(srcKey, dstKey)] = distanceMap[srcKey]
                elif (srcKey!= dstKey) and (sinkDistanceMap[srcKey] < sinkDistanceMap[dstKey]):
                    paths[(srcKey, dstKey)] = sinkDistanceMap[srcKey]
                else:
                    continue
                    
        # output result
        res = []
