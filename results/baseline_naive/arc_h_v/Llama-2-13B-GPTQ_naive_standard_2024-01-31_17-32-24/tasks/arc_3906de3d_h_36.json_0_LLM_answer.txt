Answer: This is a classic problem in computer science called \em{pattern matching}.  It has been studied extensively for decades.  There are many algorithms that can solve this problem efficiently.  For instance, see https://www.cs.cmu.edu/~rgs/papers/jacm94.pdf

The basic idea is to use a finite state machine (FSM) to recognize patterns.  You start with an FSM that recognizes the empty string.  Then, when you get a new character, you add transitions from the current state to all states reachable from it.  Whenever you find a transition from one state to another, you also add a transition from the first state to the second state.  In other words, you build up a graph of states and transitions.  If there is a path from the initial state to a final state, then your program accepts the input.  Otherwise, it rejects it.

import sys
from collections import defaultdict

def fsm(grammar):
    """Returns a Finite State Machine for the given grammar."""
    # Create a dictionary mapping each state to its successor states.
    succ = {}
    for s in sorted(grammar):
        succ[s] = set()
    for r, s in grammar.items():
        succ[s].add(r)
    return dict(succ)

def dfa(grammar):
    """Returns a Deterministic Finite State Machine for the given grammar."""
    # Create a dictionary mapping each state to its successor states.
    succ = {}
    for s in sorted(grammar):
        succ[s] = set()
    for r, s in grammar.items():
        succ[s].add(r)
    if len(set(succ.values())) > 1:
        raise ValueError("Duplicate successors")
    return {s: succ[s] for s in sorted(succ)}

def nfa(grammar):
    """Returns a Non-Deterministic Finite State Machine for the given grammar."""
    # Create a dictionary mapping each state to its successor states.
    succ = {}
    for s in sorted(grammar):
        succ[s] = set()
    for r, s in grammar.items():
        succ[s].add(r)
    return dict(succ)

def test(nfa):
    """Tests whether the NFA accepts the given input."""
    while True:
        c = raw_input().strip()
        if c == "":
            break
        if c in nfa:
            print "Accepted"
            break
        else:
            print "Rejected"

if __name__ == "__main__":
    grammar = {"S0": ["A", "B"],
              "S1": ["C", "D"],
              "S2": ["E", "F"],
              "S3": ["G", "H"]}
    nfa = nfa(grammar)
    test(nfa)
