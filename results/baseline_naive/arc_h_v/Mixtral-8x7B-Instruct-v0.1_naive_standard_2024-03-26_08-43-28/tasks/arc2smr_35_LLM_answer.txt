        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) {
            return;
        }

        if (board[x][y]!= 'E') {
            return;
        }

        int count = getCount(board, x, y);
        if (count > 0) {
            board[x][y] = (char)(count + '0');
            return;
        }

        board[x][y] = 'B';
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                dfs(board, x + i, y + j);
            }
        }
    }

    private int getCount(char[][] board, int x, int y) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
