vector<vector<char>> replaceAllAWithG(vector<vector<char>> mat) {
    int m = mat.size(), n = mat[0].size();
    vector<pair<int, int>> dirs = {{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
    queue<pair<int, int>> q;
    bool firstA = false;
    for (int i = 0; i < m; ++i) {
        if (!firstA && mat[i][0] == 'a') {
            firstA = true;
            q.push({i, 0});
        }
        if (mat[i][n - 1] == 'a') {
            q.push({i, n - 1});
        }
    }
    for (int j = 0; j < n; ++j) {
        if (!firstA && mat[0][j] == 'a') {
            firstA = true;
            q.push({0, j});
        }
        if (mat[m - 1][j] == 'a') {
            q.push({m - 1, j});
        }
    }
    while (!q.empty()) {
        auto cur = q.front();
        q.pop();
        mat[cur.first][cur.second] = 'g';
        for (auto& dir : dirs) {
            int x = cur.first + dir.first, y = cur.second + dir.second;
            if (x >= 0 && x < m && y >= 0 && y < n && mat[x][y] == 'a') {
                q.push({x, y});
            }
        }
    }
    return mat;
}

void dfs(vector<vector<char>>& mat, int x, int y) {
    if (x < 0 || x >= mat.size() || y < 0 || y >= mat[0].size() || mat[x][y]!= 'a') {
        return;
    }
    mat[x][y] = 'g';
    dfs(mat, x - 1, y);
    dfs(mat, x + 1, y);
    dfs(mat, x, y - 1);
    dfs(mat, x, y + 1);
}
