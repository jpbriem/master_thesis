public class Solution {
    public char[][] replace(char[][] board) {
        if (board == null || board.length == 0 || board[0].length == 0) return board;
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j]!= 'O') continue;
                boolean isBoundary = false;
                // check boundary
                if (i > 0 && board[i - 1][j] == 'O') isBoundary = true;
                if (i + 1 < m && board[i + 1][j] == 'O') isBoundary = true;
                if (j > 0 && board[i][j - 1] == 'O') isBoundary = true;
                if (j + 1 < n && board[i][j + 1] == 'O') isBoundary = true;
                if (!isBoundary) dfs(board, i, j);
            }
        }
        return board;
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= 'O') return;
        board[x][y] = '#';
        dfs(board, x - 1, y);
        dfs(board, x + 1, y);
        dfs(board, x, y - 1);
        dfs(board, x, y + 1);
    }
}
