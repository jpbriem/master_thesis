public class Solution {
    public String[][] replace(String[][] board) {
        if (board == null || board.length == 0) return board;
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j].equals("a")) continue;
                boolean flag = true;
                for (int k = -1; k <= 1; k++) {
                    for (int l = -1; l <= 1; l++) {
                        if (!(k == 0 && l == 0)) {
                            if (i + k >= 0 && i + k < m && j + l >= 0 && j + l < n &&!board[i+k][j+l].equals("a")) {
                                flag = false;
                                break;
                            }
                        }
                    }
                }
                if (!flag) {
                    board[i][j] = "a";
                }
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j].equals("a")) continue;
                char c = board[i][j].charAt(0);
                int count = 0;
                for (int k = -1; k <= 1; k++) {
                    for (int l = -1; l <= 1; l++) {
                        if (!(k == 0 && l == 0)) {
                            if (i + k >= 0 && i + k < m && j + l >= 0 && j + l < n && board[i+k][j+l].equals("a")) {
                                count++;
                            }
                        }
                    }
                }
                if (count > 0) {
                    StringBuilder sb = new StringBuilder();
                    sb.append((char)(c-count));
                    while (sb.length() < board[i][j].length()) {
                        sb.append('a');
                    }
                    board[i][j] = sb.toString();
                }
            }
        }
        return board;
    }
}
