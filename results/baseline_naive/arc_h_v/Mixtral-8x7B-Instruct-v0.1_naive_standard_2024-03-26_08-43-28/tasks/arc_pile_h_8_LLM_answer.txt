public class Solution {
    public int[][] solve(int[][] grid) {
        if (grid == null || grid.length < 1 || grid[0].length < 1) return new int[0][];
        int m = grid.length, n = grid[0].length;
        boolean[] row = new boolean[m], col = new boolean[n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '1') {
                    if (i > 0) row[i - 1] |= row[i];
                    if (j > 0) col[j - 1] |= col[j];
                    if (i + 1 < m) row[i] |= row[i + 1];
                    if (j + 1 < n) col[j] |= col[j + 1];
                    row[i] = true;
                    col[j] = true;
                }
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (row[i] && col[j]) grid[i][j] = '1';
            }
        }
        return grid;
    }
