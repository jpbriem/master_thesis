    int m = board.size(), n = board[0].size();
    vector<pair<int, int>> pos;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (board[i][j] == 'g') {

                for (auto p : {{0, -1}, {-1, 0}, {0, 1}, {1, 0}}) {
                    int x = i + p.first, y = j + p.second;
                    if (x >= 0 && x < m && y >= 0 && y < n && board[x][y]!= 'g') {
                        flag = false;
                        break;
                    }
                }
                if (flag) pos.push_back({i, j});
            }
        }
    }
    for (auto &p : pos) {
