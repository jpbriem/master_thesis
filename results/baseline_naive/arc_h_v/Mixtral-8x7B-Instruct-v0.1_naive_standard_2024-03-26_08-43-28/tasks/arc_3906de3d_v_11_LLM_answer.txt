public class Solution {
    public int[][] solve(int[][] A) {
        if (A == null || A.length == 0 || A[0].length == 0) return new int[0][];
        int m = A.length, n = A[0].length;
        for (int i = 0; i < m; ++i) {
            boolean flag = false;
            for (int j = 0; j < n; ++j) {
                if (A[i][j]!= 'a') {
                    flag = true;
                    break;
                }
            }
            if (!flag) continue;
            for (int j = 0; j < n; ++j) {
                if (A[i][j] == 'a' && isValid(A, i, j)) {
                    dfs(A, i, j);
                }
            }
        }
        return A;
    }

    private void dfs(int[][] A, int x, int y) {
        if (x < 0 || x >= A.length || y < 0 || y >= A[0].length || A[x][y]!= 'a') return;
        A[x][y] = 'j';
        dfs(A, x + 1, y);
        dfs(A, x - 1, y);
        dfs(A, x, y + 1);
        dfs(A, x, y - 1);
    }

    private boolean isValid(int[][] A, int x, int y) {
        if (x > 0 && A[x-1][y] == 'b') return false;
        if (x+1 < A.length && A[x+1][y] == 'b') return false;
        if (y > 0 && A[x][y-1] == 'b') return false;
        if (y+1 < A[0].length && A[x][y+1] == 'b') return false;
        return true;
    }
}
