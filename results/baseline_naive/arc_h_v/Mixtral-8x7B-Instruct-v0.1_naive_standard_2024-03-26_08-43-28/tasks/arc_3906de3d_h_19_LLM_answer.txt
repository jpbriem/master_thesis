vector<vector<char>> replaceDandJ(vector<vector<char>> mat) {
    int m = mat.size(), n = mat[0].size();
    for (int i=0;i<m;i++) {

void dfs(vector<vector<char>>& mat, int x, int y) {
    if (x < 0 || x >= mat.size() || y < 0 || y >= mat[0].size()) return;
    if (mat[x][y]!= 'd' && mat[x][y]!= 'a') return;
    mat[x][y] = 'd';
    dfs(mat, x+1, y);
    dfs(mat, x-1, y);
    dfs(mat, x, y+1);
    dfs(mat, x, y-1);
}

    for (int i=0;i<m;i++) {
        for (int j=0;j<n;j++) {
            if (mat[i][j] == 'd') {
                queue.push_back({i, j});
                while (!queue.empty()) {
                    auto p = queue.front(); queue.erase(queue.begin());
                    if (p.first < 0 || p.first >= m || p.second < 0 || p.second >= n) continue;
                    if (mat[p.first][p.second] == 'a') {
                        mat[p.first][p.second] = 'd';
                    } else if (mat[p.first][p.second] == 'd') {
                        mat[p.first][p.second] = 'j';
                    }
                    queue.push_back({p.first + 1, p.second});
                    queue.push_back({p.first - 1, p.second});
                    queue.push_back({p.first, p.second + 1});
