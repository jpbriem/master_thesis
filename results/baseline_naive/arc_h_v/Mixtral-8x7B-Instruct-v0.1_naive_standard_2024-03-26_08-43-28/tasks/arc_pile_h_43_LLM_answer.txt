public class Solution {
    public String[][] shiftGrid(String[][] grid, int k) {
        if (grid == null || grid.length == 0 || grid[0].length == 0) return new String[0][];
        int m = grid.length;
        int n = grid[0].length;
        for (int i = 0; i < k; i++) {
            List<String> list = new ArrayList<>();
            for (int j = 0; j < m; j++) {
                if (j == 0) {
                    for (int l = n - 1; l >= 0; l--) {
                        list.add(grid[m-1][l]);
                    }
                } else {
                    for (int l = n - 1; l >= 0; l--) {
                        list.add(grid[j-1][l]);
                    }
                }
            }
            grid = new String[m][n];
            for (int j = 0; j < m; j++) {
                for (int l = 0; l < n; l++) {
                    grid[j][l] = list.get(j * n + l);
                }
            }
        }
        return grid;
    }
}
