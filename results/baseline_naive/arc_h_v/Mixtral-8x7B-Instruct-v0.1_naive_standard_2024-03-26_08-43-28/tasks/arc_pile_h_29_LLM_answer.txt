public class Solution {
    public String[][] expand(String[][] matrix) {
        if (matrix == null || matrix.length == 0) return matrix;
        int m = matrix.length, n = matrix[0].length;
        boolean[][] visited = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (!visited[i][j]) {
                    char c = matrix[i][j];
                    dfs(matrix, visited, i, j, c);
                }
            }
        }
        return matrix;
    }

    private void dfs(String[][] matrix, boolean[][] visited, int x, int y, char c) {
        if (x < 0 || x >= matrix.length || y < 0 || y >= matrix[0].length || visited[x][y] || matrix[x][y].charAt(0)!= c) return;
        visited[x][y] = true;
        matrix[x][y] = "" + c;
        dfs(matrix, visited, x - 1, y, c);
        dfs(matrix, visited, x + 1, y, c);
        dfs(matrix, visited, x, y - 1, c);
        dfs(matrix, visited, x, y + 1, c);
    }
}
