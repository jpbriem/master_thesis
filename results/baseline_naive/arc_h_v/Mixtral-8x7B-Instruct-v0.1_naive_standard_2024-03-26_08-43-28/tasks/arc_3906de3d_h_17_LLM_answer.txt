void dfs(vector<vector<char>>& board, int x, int y) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size()) {
        return ;
    }
    if (board[x][y]!= 'h' && board[x][y]!= 'a') {
        return ;

        board[x][y] = 'c';
    }

}

    int m = board.size(), n = board[0].size();
    vector<pair<int, int>> pos;
    for (int i = 0; i < m; ++i) {
        if (board[i][0] == 'h') {
            pos.push_back({i, 0});
        }
    }
    for (auto p : pos) {
        queue<pair<int, int>> q{{p.first, p.second}};
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            if (cur.first < 0 || cur.first >= m || cur.second < 0 || cur.second >= n) continue;
            if (board[cur.first][cur.second]!= 'h' && board[cur.first][cur.second]!= 'a') continue;
            if (board[cur.first][cur.second] == 'a') {
                board[cur.first][cur.second] = 'c';
            }
            q.push({cur.first + 1, cur.second});
            q.push({cur.first - 1, cur.second});

        }
    }

}
