vector<vector<char>> replaceJ(vector<vector<char>>& board) {
    int m = board.size(), n = board[0].size();
    vector<pair<int, int>> jPos;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (board[i][j] == 'J') {
                jPos.push_back({i, j});
            }
        }
    }
    for (auto &p : jPos) {
        bool flag = false;
        // check up
        for (int i = p.first - 1; i >= 0 &&!flag; --i) {
            if (board[i][p.second]!= 'A' && board[i][p.second]!= 'J') {
                flag = true;
            } else if (board[i][p.second] == 'J') {
                break;
            }
        }
        // check down
        for (int i = p.first + 1; i < m &&!flag; ++i) {
            if (board[i][p.second]!= 'A' && board[i][p.second]!= 'J') {
                flag = true;
            } else if (board[i][p.second] == 'J') {
                break;
            }
        }
        // check left
        for (int j = p.second - 1; j >= 0 &&!flag; --j) {
            if (board[p.first][j]!= 'A' && board[p.first][j]!= 'J') {
                flag = true;
            } else if (board[p.first][j] == 'J') {
                break;
            }
        }
        // check right
        for (int j = p.second + 1; j < n &&!flag; ++j) {
            if (board[p.first][j]!= 'A' && board[p.first][j]!= 'J') {
                flag = true;
            } else if (board[p.first][j] == 'J') {
                break;
            }
        }
        if (!flag) {
            continue;
        }
        queue<pair<int, int>> q;
        q.push(p);
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            if (cur.first >= 0 && cur.first < m && cur.second >= 0 && cur.second < n && board[cur.first][cur.second] == 'A') {
                board[cur.first][cur.second] = 'C';
                q.push({cur.first - 1, cur.second});
                q.push({cur.first + 1, cur.second});
                q.push({cur.first, cur.second - 1});
                q.push({cur.first, cur.second + 1});
            }
        }
    }
    return board;
}

void dfs(vector<vector<char>>& board, int x, int y) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size() || board[x][y]!= 'A') {
        return ;
    }
    board[x][y] = 'C';
    dfs(board, x - 1, y);
    dfs(board, x + 1, y);
    dfs(board, x, y - 1);
    dfs(board, x, y + 1);
}
