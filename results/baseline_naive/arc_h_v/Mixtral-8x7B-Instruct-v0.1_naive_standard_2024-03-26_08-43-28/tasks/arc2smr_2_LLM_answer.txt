vector<vector<char>> replace(vector<vector<char>>& board) {
    int m = board.size(), n = board[0].size();
    vector<pair<int, int>> pos;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (board[i][j] == 'b') {
                pos.push_back({i, j});
            }
        }
    }
    for (auto p : pos) {
        dfs(p.first, p.second, board);
    }
    return board;
}
void dfs(int x, int y, vector<vector<char>>& board) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size()) {
        return ;
    }
    if (board[x][y]!= 'b' && board[x][y]!= 'g') {
        return ;
    }
    board[x][y] = 'g';
    dfs(x + 1, y, board);
    dfs(x - 1, y, board);
    dfs(x, y + 1, board);
    dfs(x, y - 1, board);
}

    vector<vector<char>> replace(vector<vector<char>>& board) {
        int m = board.size(), n = board[0].size();
        queue<pair<int, int>> q;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 'b') {
                    q.push({i, j});
                }
            }
        }
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            if (cur.first < 0 || cur.first >= m || cur.second < 0 || cur.second >= n) continue;
            if (board[cur.first][cur.second]!= 'b' && board[cur.first][cur.second]!= 'g') continue;

        }
