void dfs(vector<vector<char>> &board, int x, int y) {
    int m = board.size(), n = board[0].size();
    if (x < 0 || x >= m || y < 0 || y >= n) return;
    if (board[x][y]!= 'E') return;
    int count = 0;
    for (int i = -1; i <= 1; ++i) {
        for (int j = -1; j <= 1; ++j) {
            if (i == 0 && j == 0) continue;
            int a = x + i, b = y + j;
            if (a < 0 || a >= m || b < 0 || b >= n) continue;
            if (board[a][b] == 'M' || board[a][b] == 'X') {
                ++count;
            }
        }
    }
    if (count > 0) {
        board[x][y] = '0' + count;
    } else {
        board[x][y] = 'B';

            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                int a = x + i, b = y + j;
                if (a < 0 || a >= m || b < 0 || b >= n) continue;
                dfs(board, a, b);
            }
        }
    }
}
