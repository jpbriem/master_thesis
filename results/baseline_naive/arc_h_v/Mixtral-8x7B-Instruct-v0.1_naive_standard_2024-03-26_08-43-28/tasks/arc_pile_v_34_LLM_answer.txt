public class Solution {
    public String[][] solve(String[][] board) {
        if (board == null || board.length < 1 || board[0].length < 1) return board;
        int m = board.length, n = board[0].length;
        boolean hasO = false;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == "O") {
                    dfs(board, i, j);
                    hasO = true;
                }
            }
        }
        if (!hasO) return new String[0][];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == "#") board[i][j] = "O";
                else if (board[i][j] == "O") board[i][j] = "X";
            }
        }
        return board;
    }

    private void dfs(String[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= "O") return;
        board[x][y] = "#";
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
}
