public class Solution {
    public String[][] solve(String[][] board) {
        if (board == null || board.length < 3 || board[0].length < 3) return board;
        int m = board.length, n = board[0].length;
        boolean hasJ = false;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 'j') {
                    dfs(board, i, j);
                    hasJ = true;
                }
            }
        }
        if (!hasJ) return new String[0][];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 'a') board[i][j] = 'j';
            }
        }
        return board;
    }

    private void dfs(String[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= 'j') return;
        board[x][y] = 'a';
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
