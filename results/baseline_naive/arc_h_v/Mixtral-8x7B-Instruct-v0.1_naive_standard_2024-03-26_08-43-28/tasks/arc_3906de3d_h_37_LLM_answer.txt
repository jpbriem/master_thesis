class Solution {
public:
    vector<vector<char>> updateBoard(vector<vector<char>>& board, vector<int>& click) {
        int m = board.size(), n = board[0].size();
        function<void(int, int)> dfs = [&](int x, int y) {
            if (x < 0 || x >= m || y < 0 || y >= n) return;
            if (board[x][y] == 'M' || board[x][y] == 'X' || board[x][y] == 'B') return;
            int count = 0;
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (!(i == 0 && j == 0)) {
                        int nx = x + i, ny = y + j;
                        if (nx >= 0 && nx < m && ny >= 0 && ny < n && board[nx][ny] == 'M') {
                            ++count;
                        }
                    }
                }
            }
            if (count > 0) {
                board[x][y] = count + '0';
            } else {
                board[x][y] = 'B';
                for (int i = -1; i <= 1; ++i) {
                    for (int j = -1; j <= 1; ++j) {
                        if (!(i == 0 && j == 0)) {
                            int nx = x + i, ny = y + j;
                            if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                                dfs(nx, ny);
                            }
                        }
                    }
                }
            }
        };
        dfs(click[0], click[1]);
        return board;
    }
};
