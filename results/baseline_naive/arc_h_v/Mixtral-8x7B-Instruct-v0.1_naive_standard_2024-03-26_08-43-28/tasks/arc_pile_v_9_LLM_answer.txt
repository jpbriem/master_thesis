public class Solution {
    public char[][] merge(char[][] intervals) {
        if (intervals == null || intervals.length == 0) return new char[0][];
        Arrays.sort(intervals, new Comparator<char[]>() {
            @Override
            public int compare(char[] a, char[] b) {
                return Integer.compare(a[0], b[0]);
            }
        });
        List<char[]> res = new ArrayList<>();
        for (int i = 0; i < intervals.length; i++) {
            if (res.isEmpty() || res.get(res.size() - 1)[1] < intervals[i][0]) {
                res.add(intervals[i]);
            } else {
                res.get(res.size() - 1)[1] = Math.max(res.get(res.size() - 1)[1], intervals[i][1]);
            }
        }
        return res.toArray(new char[res.size()][]);
    }
}
