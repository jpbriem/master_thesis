public class Solution {
    public String[][] merge(String[][] intervals) {
        if (intervals == null || intervals.length <= 0) return new String[0][];
        Arrays.sort(intervals, new Comparator<String[]>() {
            @Override
            public int compare(String[] o1, String[] o2) {
                return Integer.compare(Integer.parseInt(o1[0]), Integer.parseInt(o2[0]));
            }
        });
        List<String[]> res = new ArrayList<>();
        for (int i = 0; i < intervals.length; i++) {
            String[] cur = intervals[i];
            if (res.size() > 0 &&!cur[4].equals(res.get(res.size() - 1)[4])) { // not the same group
                res.add(cur);
            } else {
                String[] last = res.get(res.size() - 1);
                if (!last[4].equals(cur[4]) && Integer.parseInt(cur[1]) >= Integer.parseInt(last[1])) {
                    last[1] = cur[1];
                }
            }
        }
        return res.toArray(new String[res.size()][]);
    }
}
