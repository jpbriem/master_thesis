void dfs(vector<vector<char>> &image, int i, int j, char oldColor, char newColor) {
    if (i < 0 || i >= image.size() || j < 0 || j >= image[0].size()) return;
    if (image[i][j]!= oldColor) return;
    image[i][j] = newColor;
    dfs(image, i + 1, j, oldColor, newColor);
    dfs(image, i - 1, j, oldColor, newColor);
    dfs(image, i, j + 1, oldColor, newColor);
    dfs(image, i, j - 1, oldColor, newColor);
}

vector<vector<char>> floodFill(vector<vector<char>>& image, int sr, int sc, char newColor) {
    if (image[sr][sc] == newColor) return image;
    stack<pair<int, int>> stk;
    stk.push({sr, sc});
    while (!stk.empty()) {
        auto p = stk.top(); stk.pop();
        int x = p.first, y = p.second;
        if (x < 0 || x >= image.size() || y < 0 || y >= image[0].size() || image[x][y]!= image[sr][sc]) continue;
        image[x][y] = newColor;
        stk.push({x+1, y});
        stk.push({x-1, y});
        stk.push({x, y+1});
        stk.push({x, y-1});
    }
    return image;
}
