public class Solution {
    public char[][] replace(char[][] board) {
        if (board == null || board.length == 0) return board;
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j]!= 'O') continue;
                boolean isEdge = false;
                // check top and bottom edges
                if (i == 0 || i == m - 1) {
                    isEdge = true;
                } else {
                    if (board[i-1][j] == 'X' && board[i+1][j] == 'X') {
                        isEdge = true;
                    }
                }
                // check left and right edges
                if (!isEdge) {
                    if (j == 0 || j == n - 1) {
                        isEdge = true;
                    } else {
                        if (board[i][j-1] == 'X' && board[i][j+1] == 'X') {
                            isEdge = true;
                        }
                    }
                }
                if (isEdge) {
                    board[i][j] = 'D';
                }
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 'D') {
                    dfs(board, i, j);
                }
            }
        }
        return board;
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= 'D') return;
        board[x][y] = 'O';
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
}
