void dfs(vector<vector<char>>& board, int x, int y) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size()) {
        return ;
    }
    if (board[x][y]!= 'J' && board[x][y]!= 'T') {
        return ;
    }
    board[x][y] = 'T';
    dfs(board, x - 1, y);
    dfs(board, x + 1, y);
    dfs(board, x, y - 1);
    dfs(board, x, y + 1);
}

        auto p = stack.back();
