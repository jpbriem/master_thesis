vector<vector<char>> replaceJ(vector<vector<char>>& board) {
    int m = board.size(), n = board[0].size();
    vector<pair<int, int>> jPos;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (board[i][j] == 'j') {
                jPos.push_back({i, j});
            }
        }
    }

    // check the boundary of j's
    bool isBoundary = false;
    for (auto pos : jPos) {
        if ((pos.first == 0 || pos.first == m - 1 || pos.second == 0 || pos.second == n - 1)) {
            isBoundary = true;
            break;
        }
    }

    // replace all a with e in the boundary
    if (isBoundary) {
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 'a') {
                    board[i][j] = 'e';
                }
            }
        }
    } else {
        for (auto pos : jPos) {
            queue<pair<int, int>> q;
            q.push(pos);
            while (!q.empty()) {
                auto cur = q.front();
                q.pop();
                if (cur.first >= 0 && cur.first < m && cur.second >= 0 && cur.second < n && board[cur.first][cur.second]!= 'j') {
                    if (board[cur.first][cur.second] == 'a') {
                        board[cur.first][cur.second] = 'e';
                    }
                    q.push({cur.first + 1, cur.second});
                    q.push({cur.first - 1, cur.second});
                    q.push({cur.first, cur.second + 1});
                    q.push({cur.first, cur.second - 1});
                }
            }
        }
    }

    return board;
}

void dfs(vector<vector<char>> &board, int x, int y) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size() || board[x][y]!= 'j') {
        return ;
    }
    board[x][y] = 'e';
    dfs(board, x+1, y);
    dfs(board, x-1, y);
    dfs(board, x, y+1);
    dfs(board, x, y-1);
}

vector<vector<char>> replaceJ(vector<vector<char>>& board) {
    for (int i = 0; i < board.size(); ++i) {
        for (int j = 0; j < board[0].size(); ++j) {
            if (board[i][j] == 'j') {
                dfs(board, i, j);
            }
        }
    }
    return board;
}
