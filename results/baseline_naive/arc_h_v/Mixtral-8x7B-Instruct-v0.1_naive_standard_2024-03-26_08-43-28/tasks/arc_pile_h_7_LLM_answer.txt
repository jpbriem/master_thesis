public class Solution {
    public String[][] replace(String[][] input) {
        if (input == null || input.length == 0) return new String[0][];
        int m = input.length, n = input[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                char c = input[i][j].charAt(0);
                if (c!= 'a') {
                    boolean flag = true;
                    for (int k = 0; k < m && flag; k++) {
                        if (input[k][j].charAt(0)!= c) flag = false;
                    }
                    for (int l = 0; l < n && flag; l++) {
                        if (input[i][l].charAt(0)!= c) flag = false;
                    }
                    if (flag) {
                        for (int p = 0; p < m; p++) {
                            for (int q = 0; q < n; q++) {
                                if (input[p][q].charAt(0) == c) {
                                    input[p][q] = "" + c;
                                }
                            }
                        }
                    } else {
                        for (int r = 0; r < m; r++) {
                            for (int s = 0; s < n; s++) {
                                if (input[r][s].charAt(0) == c) {
                                    input[r][s] = "" + c + c;
                                }
                            }
                        }
                    }
                }
            }
        }
        return input;
    }
}
