class Solution {
    public List<List<Character>> modify(List<List<Character>> grid) {
        int m = grid.size();
        if (m == 0) return grid;
        int n = grid.get(0).size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid.get(i).get(j)!= 'f') continue;
                dfs(grid, i, j);
            }
        }
        return grid;
    }

    private void dfs(List<List<Character>> grid, int x, int y) {
        if (x < 0 || x >= grid.size() || y < 0 || y >= grid.get(0).size()) return;
        if (grid.get(x).get(y) == 'a' || grid.get(x).get(y) == 'f') {
            grid.get(x).set(y, 'f');
            dfs(grid, x + 1, y);
            dfs(grid, x - 1, y);
            dfs(grid, x, y + 1);
            dfs(grid, x, y - 1);
        }
    }
}

class Solution {
    public List<List<Character>> modify(List<List<Character>> grid) {
        int m = grid.size(), n = grid.get(0).size();
        boolean[][] visited = new boolean[m][n];
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid.get(i).get(j) == 'f') {
                    queue.offer(i * n + j);
                    visited[i][j] = true;
                }
            }
        }
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            int x = cur / n, y = cur % n;
            if (x > 0 &&!visited[x - 1][y] && grid.get(x - 1).get(y)!= 'a') {
                grid.get(x - 1).set(y, 'f');
                visited[x - 1][y] = true;
                queue.offer((x - 1) * n + y);
            }
            if (x + 1 < m &&!visited[x + 1][y] && grid.get(x + 1).get(y)!= 'a') {
                grid.get(x + 1).set(y, 'f');
                visited[x + 1][y] = true;
                queue.offer((x + 1) * n + y);
            }
            if (y > 0 &&!visited[x][y - 1] && grid.get(x).get(y - 1)!= 'a') {
                grid.get(x).set(y - 1, 'f');
                visited[x][y - 1] = true;
                queue.offer(x * n + y - 1);
            }
            if (y + 1 < n &&!visited[x][y + 1] && grid.get(x).get(y + 1)!= 'a') {
                grid.get(x).set(y + 1, 'f');
                visited[x][y + 1] = true;
                queue.offer(x * n + y + 1);
            }
        }
        return grid;
    }
}
