vector<vector<char>> merge(vector<vector<char>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<pair<int, int>> dirs = {{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
    auto dfs = [&](auto self, int i, int j, char c) -> void {
        if (i < 0 || i >= m || j < 0 || j >= n || grid[i][j]!= c) return;
        grid[i][j] = '#';
        for (const auto &dir : dirs) {
            self(self, i + dir.first, j + dir.second, c);
        }
    };
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (grid[i][j] == 'a') continue;
            dfs(dfs, i, j, grid[i][j]);
        }
    }
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (grid[i][j] == 'a' &&!island(grid, i, j)) {
                grid[i][j] = 'b';
            }
        }
    }
    return grid;
}
bool island(vector<vector<char>>& grid, int x, int y) {
    if (x < 0 || x >= grid.size() || y < 0 || y >= grid[0].size()) return false;
    if (grid[x][y] == 'b' || grid[x][y] == '#') return true;
    grid[x][y] = '#';
    bool res = island(grid, x+1, y) && island(grid, x-1, y) && island(grid, x, y+1) && island(grid, x, y-1);
    grid[x][y] = 'b';
    return res;
}
