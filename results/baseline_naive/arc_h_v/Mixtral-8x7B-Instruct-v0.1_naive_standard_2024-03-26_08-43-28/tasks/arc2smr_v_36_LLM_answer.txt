public class Solution {
    public char[][] updateBoard(char[][] board, int[] click) {
        if (board[click[0]][click[1]] == 'M') {
            board[click[0]][click[1]] = 'X';
            return board;
        }

        dfs(board, click[0], click[1]);

        return board;
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) {
            return;
        }

        if (board[x][y]!= 'E' && board[x][y]!= 'B') {
            return;
        }

        int count = getCount(board, x, y);

        if (count > 0) {
            board[x][y] = (char)(count + '0');
            return;
        }

        board[x][y] = 'B';

        dfs(board, x - 1, y);
        dfs(board, x + 1, y);
        dfs(board, x, y - 1);
        dfs(board, x, y + 1);
    }

    private int getCount(char[][] board, int x, int y) {
        int count = 0;

        for (int i = Math.max(0, x - 1); i <= Math.min(x + 1, board.length - 1); ++i) {
            for (int j = Math.max(0, y - 1); j <= Math.min(y + 1, board[0].length - 1); ++j) {
                if (board[i][j] == 'M') {
                    count++;
                }
            }
        }

        return count;
    }
}
