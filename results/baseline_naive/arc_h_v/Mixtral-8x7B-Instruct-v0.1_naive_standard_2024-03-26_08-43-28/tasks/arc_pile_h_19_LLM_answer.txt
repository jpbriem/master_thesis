public class Solution {
    public String[][] transpose(String[][] A) {
        int m = A.length, n = A[0].length;
        String[][] res = new String[n][m];
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                res[j][i] = A[i][j];
            }
        }
        return res;
    }
}
