    for (int i = 0; i < mat.size(); ++i) {
        for (int j = 0; j < mat[i].size(); ++j) {

void dfs(vector<vector<char>>& mat, int x, int y) {
    if (x < 0 || x >= mat.size() || y < 0 || y >= mat[0].size()) {
        return ;
    }
    if (mat[x][y]!= 'd' && mat[x][y]!= 'a') {
        return ;

    dfs(mat, x + 1, y);
    dfs(mat, x - 1, y);

    dfs(mat, x, y - 1);
}
