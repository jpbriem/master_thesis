public class Solution {
    public String[][] extendGrid(String[][] grid) {
        if (grid == null || grid.length == 0) return new String[0][];
        int m = grid.length, n = grid[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j].equals("a")) continue;
                dfs(grid, i, j);
            }
        }
        return grid;
    }

    private void dfs(String[][] grid, int x, int y) {
        if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length ||!grid[x][y].equals("a")) return;
        grid[x][y] = "g";
        dfs(grid, x + 1, y);
        dfs(grid, x - 1, y);
        dfs(grid, x, y + 1);
        dfs(grid, x, y - 1);
    }
}
