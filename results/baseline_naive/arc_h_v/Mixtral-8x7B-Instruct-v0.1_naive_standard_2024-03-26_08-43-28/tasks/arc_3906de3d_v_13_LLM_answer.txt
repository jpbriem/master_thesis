        for (int j = 0; j < n; ++j) {
            if (grid[m - 1][j] == 'a') {
                q.push({m - 1, j});
                grid[m - 1][j] = 'g';
            }
        }
        while (!q.empty()) {
            auto p = q.front();
            q.pop();
            int x = p.first, y = p.second;
            if (x > 0 && grid[x - 1][y] == 'a') {
                q.push({x - 1, y});
