class Solution {
    public List<List<Character>> merge(List<List<Character>> lists) {
        if (lists == null || lists.size() <= 0) return new ArrayList<>();
        int m = lists.size();
        int n = lists.get(0).size();
        List<List<Character>> res = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            if (lists.get(i).size()!= n) throw new IllegalArgumentException("Invalid input");
        }
        char[] arr = new char[m * n];
        boolean flag = false;
        for (int j = 0; j < n; j++) {
            Set<Character> set = new HashSet<>();
            for (int i = 0; i < m; i++) {
                Character c = lists.get(i).get(j);
                if (!set.add(c)) {
                    arr[flag? i - 1 : i] = c;
                    flag = true;
                } else {
                    arr[flag? i : i - 1] = c;
                }
            }
            if (!flag) continue;
            List<Character> list = new ArrayList<>(m);
            for (int k = 0; k < m; k++) {
                list.add(arr[k]);
            }
            res.add(list);
            flag = false;
        }
        if (res.isEmpty()) {
            res.add(lists.get(0));
        }
        return res;
    }
}
