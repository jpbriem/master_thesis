vector<vector<char>> replaceJWithG(vector<vector<char>>& board) {
    int m = board.size(), n = board[0].size();
    for (int i = 0; i < m; ++i) {
        if (board[i][0] == 'j') {
            dfs(board, i, 0);
        }
    }
    return board;
}
void dfs(vector<vector<char>> &board, int x, int y) {
    if (x < 0 || x >= board.size() || y < 0 || y >= board[0].size()) {
        return ;
    }
    if (board[x][y]!= 'j' && board[x][y]!= 'J') {
        return ;
    }
    board[x][y] = 'g';
    dfs(board, x - 1, y);
    dfs(board, x + 1, y);
    dfs(board, x, y - 1);
    dfs(board, x, y + 1);
}

vector<vector<char>> replaceJWithG(vector<vector<char>>& board) {
    bool firstRow = true;
    for (auto &row : board) {
        if (firstRow) {
            firstRow = false;
            for (int j = 0; j < row.size(); ++j) {
                if (row[j] == 'j') {
                    row[j] = 'g';
                }
            }
        } else {
            for (int j = 0; j < row.size() - 1; ++j) {
                if (row[j] == 'j' && row[j+1] == 'j') {
                    row[j] = 'g';
                }
            }
            if (!row.empty() && row.back() == 'j') {
                row.back() = 'g';
            }
        }
    }
    return board;
}
