public class Solution {
    public String[][] replace(String[][] board) {
        if (board == null || board.length == 0) return board;
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j].equals("X")) continue;
                boolean isO = true;
                for (int k = -1; k <= 1; k++) {
                    for (int l = -1; l <= 1; l++) {
                        if (k == 0 && l == 0) continue;
                        int x = i + k, y = j + l;
                        if (x >= 0 && x < m && y >= 0 && y < n &&!board[x][y].equals("X") &&!board[x][y].equals("O")) {
                            isO = false;
                            break;
                        }
                    }
                    if (!isO) break;
                }
                if (isO) board[i][j] = "O";
                else board[i][j] = "X";
            }
        }
        return board;
    }
}
