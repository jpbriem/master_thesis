void dfs(vector<vector<char>> &board, int i, int j) {
    int m = board.size(), n = board[0].size();
    if (i < 0 || i >= m || j < 0 || j >= n) return;
    if (board[i][j]!= 'E') return;
    int count = 0;
    for (int x = -1; x <= 1; ++x) {
        for (int y = -1; y <= 1; ++y) {
            if (x == 0 && y == 0) continue;
            int a = i + x, b = j + y;
            if (a < 0 || a >= m || b < 0 || b >= n) continue;
            if (board[a][b] == 'M' || board[a][b] == 'X') {
                ++count;
            }
        }
    }
    if (count > 0) {
        board[i][j] = '0' + count;
        return;
    } else {
        board[i][j] = 'B';
    }

            int a = i + x, b = j + y;

        }
    }
}
