public class Solution {
    public char[][] merge(char[][] grid) {
        if (grid == null || grid.length == 0) return new char[0][];
        int m = grid.length, n = grid[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j + 1 < n; ) {
                if (grid[i][j]!= '\u0000' && grid[i][j] == grid[i][j+1]) {
                    int k = j + 1;
                    while (k + 1 < n && grid[i][k] == grid[i][k+1]) ++k;
                    grid[i][j] += grid[i][k];
                    for (int l = j+1; l <= k; ++l) grid[i][l] = '\u0000';
                    j = k + 1;
                } else ++j;
            }
        }
        List<List<Character>> list = new ArrayList<>();
        for (int i = 0; i < m; ++i) {
            List<Character> temp = new ArrayList<>();
            for (int j = 0; j < n; ++j) {
                if (grid[i][j]!= '\u0000') temp.add(grid[i][j]);
            }
            if (!temp.isEmpty()) list.add(temp);
        }
        char[][] res = new char[list.size()][];
        for (int i = 0; i < list.size(); ++i) {
            List<Character> cur = list.get(i);
            res[i] = new char[cur.size()];
            for (int j = 0; j < cur.size(); ++j) res[i][j] = cur.get(j);
        }
        return res;
    }
}
