    public List<List<Character>> solve(char[][] board) {
        if (board == null || board.length == 0) return new ArrayList<>();
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; ++i) {
            dfs(board, i, 0);
            dfs(board, i, n - 1);
        }
        for (int j = 0; j < n; ++j) {
            dfs(board, 0, j);
            dfs(board, m - 1, j);
        }
        List<List<Character>> res = new ArrayList<>();
        for (int i = 0; i < m; ++i) {
            List<Character> list = new ArrayList<>();
            for (int j = 0; j < n; ++j) {
                list.add(board[i][j]);
            }
            res.add(list);
        }
        return res;
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length || board[x][y]!= 'a') return;

        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
