class Solution:
    def permute(self, nums: List[str]) -> List[List[str]]:
        if not nums: return []
        
        result = []
        for i in range(len(nums)):
            temp = nums[:i] + nums[i+1:]
            result += self.permute(temp)
            
        return result
    
    def permuteKTimes(self, nums: List[str], k: int) -> List[List[str]]:
        if not nums or k == 0: return []
        
        result = []
        for _ in range(k):
            result += self.permute(nums)
