    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        result = []
        for s in strs:
            if len(s) == 0 or s[0] != s[-1]:
                result.append([s])
            else:
                used_char = set()
                ans = []
                for c in s:
                    if c not in used_char:
                        used_char.add(c)
                        ans.append(c)
                result.append(ans)

        return result
