```python
def find_longest_substring(input):
    if not input:
        return []
    
    start = 0
    longest_length = 0
    
    for i in range(len(input)):
        if input[i] != input[start]:
            start = i + 1
        
        longest_length = max(longest_length, i - start + 1)
        
    return [input[j:j+longest_length] for j in range(start, len(input))]
```
