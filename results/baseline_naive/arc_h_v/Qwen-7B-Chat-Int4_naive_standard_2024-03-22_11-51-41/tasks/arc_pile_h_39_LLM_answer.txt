class Solution:
     def groupAnagrams(self, strs):
         """
         :type strs: List[str]
         :rtype: List[List[str]]
         """
         if len(strs) == 0:
             return []
         
         ans = [[] for _ in range(len(set(strs[0])))]
         for s in strs:
             ans[set(s)].append(s)
             
         return ans
