```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        ans = [[] for _ in range(len(strs))]
        
        hash_map = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key in hash_map:
                hash_map[key].append(s)
            else:
                hash_map[key] = [s]
                
        return ans
```
