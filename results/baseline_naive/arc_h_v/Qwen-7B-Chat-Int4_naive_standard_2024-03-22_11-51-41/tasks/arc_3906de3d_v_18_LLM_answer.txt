def find_common_elements(arrays):
    # Flatten each array into a single list
    flattened_arrays = [item for sublist in arrays for item in sublist]
    
    # Remove duplicates from the flattened array
    unique_elements = set(flattened_arrays)
    
    # Create an empty dictionary to store the counts of each element
    count_dict = {}
    
    # Count the occurrences of each element in the unique elements
    for element in unique_elements:
        if element in count_dict:
            count_dict[element] += 1
        else:
            count_dict[element] = 1
    
    # Find the common elements by selecting those that appear in all arrays
    common_elements = []
    for array in arrays:
        for element in array:
            if element in count_dict and count_dict[element] == len(array):
                common_elements.append(element)
    
    return common_elements
