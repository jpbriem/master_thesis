```python
class Solution:
     def groupAnagrams(self, strs):
         """
         :type strs: List[List[str]]
         :rtype: List[List[str]]
         """
         if len(strs) == 0:
             return []
         
         anagrams = {}
         for s in strs:
             key = ''.join(sorted(s))
             if key not in anagrams:
                 anagrams[key] = [s]
             else:
                 anagrams[key].append(s)
         
         result = []
         for k, v in anagrams.items():
             result.append(v)
             
         return result
```
