def merge_adjacent(arr):
    # Base case: if the array has less than 4 elements, return it as it is
    if len(arr) < 4:
        return arr

    # Initialize an empty list to store merged elements
    merged = []

    # Iterate over each element in the array
    for i in range(len(arr)):
        # Check if the current element is equal to its adjacent elements
        if i > 0 and arr[i] == arr[i-1]:
            continue
        elif i + 1 < len(arr) and arr[i] == arr[i+1]:
            continue
        else:
            # Merge the current element with its adjacent elements
            merged.append(arr[i])
            # Remove the merged element from the original array
            del arr[i]
            # Recursively merge adjacent elements of the remaining array
            merged += merge_adjacent(arr)

    # Return the final merged array
    return merged
