    if len(set(input)) != len(input):
        return []
    
    output = []
    for i in range(len(input)):
        if input[i] == input[0]:
            output.append([i])
        else:
            j = 0
            while input[j] != input[0]:
                j += 1
            output.append([i, j-1])

    return output
