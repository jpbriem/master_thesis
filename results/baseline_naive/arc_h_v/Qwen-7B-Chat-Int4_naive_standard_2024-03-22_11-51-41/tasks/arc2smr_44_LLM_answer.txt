def merge_sorted_arrays(arr):
    # Base case: if the array has only one element, return it
    if len(arr) == 1:
        return arr

    # Find the middle index of the array
    mid = len(arr) // 2

    left_half = arr[:mid]
    right_half = arr[mid:]

    # Recursively merge the two halves
