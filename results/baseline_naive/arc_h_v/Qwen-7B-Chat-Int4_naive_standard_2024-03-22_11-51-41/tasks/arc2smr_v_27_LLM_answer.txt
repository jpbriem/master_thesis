def sort_and_unique(lst):
    # Convert to set to remove duplicates and get unique elements
    unique_elements = set(lst)
    
    # Sort the elements in alphabetical order
    sorted_elements = sorted(unique_elements)
    
    # Create an empty list to store the result
    result = []
    
    # Iterate over the sorted elements and append them to the result list
    for element in sorted_elements:
        while len(result) > 0 and result[-1] < element:
            result.pop()
