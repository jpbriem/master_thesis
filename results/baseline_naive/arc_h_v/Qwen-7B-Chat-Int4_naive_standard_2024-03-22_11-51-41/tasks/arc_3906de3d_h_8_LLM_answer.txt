    n = len(matrix)
    if n == 0 or n == 1:
        return [[matrix[i]] for i in range(n)]
    
    res = []
    stack = [(0, 0)]
    while stack:
        row, col = stack.pop()
        
        if col == n - 1:
            res.append(matrix[row])
            
        else:
            for j in range(col + 1, n):
                if matrix[row][j] != matrix[row][col]:
                    break
                stack.append((row, j))
                
    return res
