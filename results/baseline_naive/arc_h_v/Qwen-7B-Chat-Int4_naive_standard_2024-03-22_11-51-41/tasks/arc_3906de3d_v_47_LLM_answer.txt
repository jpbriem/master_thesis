class Solution:
    def longestPalindrome(self, s):
        if len(s) == 0:
            return []
        
        start = 0
        maxLength = 1
        
        for i in range(len(s)):
            # Check for odd length palindromes with center at current character
            maxLen = 1
            j = i - 1
            while j >= 0 and s[j] == s[i]:
                maxLen += 1
                j -= 1
            
            # Check for even length palindromes with center between two characters
            j = i + 1
            while j < len(s) and s[j] == s[i]:
                maxLen += 1
                j += 1
            
            if maxLen > maxLength:
                maxLength = maxLen
                start = i - (maxLength // 2)
                
        return s[start:start+maxLength]
