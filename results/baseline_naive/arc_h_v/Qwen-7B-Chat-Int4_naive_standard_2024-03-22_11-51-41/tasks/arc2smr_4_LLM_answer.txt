def longest_palindrome(input):
    def helper(s, start, end):
        if len(s) == 0 or start > end:
            return 0

        mid = (start + end) // 2

        # Check for palindromes of odd length
        max_length = 1
        i = start - 1
        while i >= 0 and s[i] == s[mid]:
            max_length += 1
            i -= 1
        
        j = end + 1
        while j < len(s) and s[j] == s[mid]:
            max_length += 1
            j += 1
        
        # Check for palindromes of even length
        max_length = max(max_length, helper(s, start, mid), helper(s, mid + 1, end))

        return max_length

    input_str = ''.join(input)
    n = len(input_str)

    result = []
    for l in range(2, n+1):
        for i in range(n-l+1):
            left = input_str[i:i+l]
            right = input_str[i+l-1:i+1]

            if helper(left, 0, len(left)-1) == l and helper(right, 0, len(right)-1) == l:
                result.append([left, right])

    return result
