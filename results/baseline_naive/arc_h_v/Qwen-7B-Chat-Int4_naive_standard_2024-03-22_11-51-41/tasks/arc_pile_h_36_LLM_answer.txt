class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        from collections import defaultdict

        ans = []
        
        d = defaultdict(list)
        
        for s in strs:
            key = ''.join(sorted(s))
            d[key].append(s)
            
        for k, v in d.items():
            if len(v) > 0:
                ans.append(v)
                
        return ans
