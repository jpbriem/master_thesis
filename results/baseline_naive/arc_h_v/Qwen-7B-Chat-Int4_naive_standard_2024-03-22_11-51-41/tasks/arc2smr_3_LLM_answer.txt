def merge(left, right):
    return left + right

def split(string, delimiter):
    if not string:
        return [], []
    else:
        half = string.split(delimiter)
        return split(half[0], delimiter), split(half[1], delimiter)

def group_strings(strings, delimiter):
    result = []
    for i in range(len(strings)):
        if i == 0 or strings[i] != strings[i-1]:

    return result
