def group_unique_elements(matrix):
    n = len(matrix)
    groups = [[matrix[0][i]] for i in range(n)]
    
    for i in range(1, n):
        for j in range(i+1, n):
            if matrix[i] == matrix[j]:
                continue
            
            common_elements = set(matrix[i]).intersection(set(matrix[j]))
            
            if common_elements:
                groups.append([matrix[i], matrix[j]])
                
    return groups
