```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[List[str]]
        :rtype: List[List[str]]
        """
        anagram = {}
        
        for s in strs:
            key = ''.join(sorted(s))
            if key not in anagram:
                anagram[key] = []
            anagram[key].append(s)
            
        return [anagram[i] for i in sorted(anagram)]
```
