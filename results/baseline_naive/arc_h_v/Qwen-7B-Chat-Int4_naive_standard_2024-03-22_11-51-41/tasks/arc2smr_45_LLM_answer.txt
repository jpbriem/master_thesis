class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagrams = []
        
        for str in strs:
            char_dict = {}
            for ch in str:
                if ch not in char_dict:
                    char_dict[ch] = [ch]
                else:
                    char_dict[ch].append(ch)
            
            if len(char_dict) > 0:
                anagrams.append(sorted(char_dict.values()))
                
        return anagrams
