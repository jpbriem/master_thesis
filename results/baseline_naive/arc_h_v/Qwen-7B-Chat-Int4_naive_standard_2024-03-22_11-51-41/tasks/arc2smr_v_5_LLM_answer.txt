```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs or len(strs) == 0:
            return []
        
        anagrams = {}
        for s in strs:
            key = "".join(sorted(s))
            if key in anagrams:
                anagrams[key].append(s)
            else:
                anagrams[key] = [s]
        
        result = []
        for k, v in anagrams.items():
            result.append(v)
        
        return result
```
