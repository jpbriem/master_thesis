def group_chars(input_list):
    # Base case: if the input list is empty or has only one element, return it as a single group
    if len(input_list) <= 1:
        return input_list

    # Recursive case: find the last non-repeated character in the input list
    last_char = max(set(input_list), key=input_list.count)

    # Create an empty list to store the groups of characters
    groups = []

    # Iterate over the input list and add each character to its corresponding group
    for char in input_list:
        if char == last_char:
            groups[-1].append(char)
        else:
            groups.append([char])

    # Return the list of groups
    return groups
