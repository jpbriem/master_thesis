def remove_duplicates(list1, list2):
    # Base case
    if len(list1) == 0 or len(list2) == 0:
        return list1 + list2

    # Recursive case
    result = []
    i = j = 0
    while i < len(list1) and j < len(list2):
        if list1[i] != list2[j]:
            result.append(list1[i])
            i += 1
        else:
            j += 1

    result += list2[j:]

    return remove_duplicates(result, [])
