class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        if len(nums) == 0 or len(nums) == 1:
            return [nums]
            
        for i in range(len(nums)):
            temp = nums[:i] + nums[i+1:]
            res.extend(self.permute(temp))
            
        return res
