class Solution:
    def findLargestNumber(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: str
        """
        if not nums or len(nums) == 0:
            return "Invalid input"
        
        max_num = None
        for row in nums:
            largest = self.find_largest_number(row)
            if max_num is None or largest > max_num:
                max_num = largest
        
        return max_num
    
    def find_largest_number(self, num_list):
        """
        :type num_list: List[int]
        :rtype: int
        """
        max_num = float('-inf')
        for num in num_list:
            num_str = str(num)
            if len(num_str) < 9 and num_str != "0":
                num_str += "0" * (9 - len(num_str))
            else:
                num_str = num_str[:8] + "." + num_str[8:]
            
            num_str = num_str.replace(".", "")
            
            if int(num_str) > max_num:
                max_num = int(num_str)
                
        return max_num
