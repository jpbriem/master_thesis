class Solution:
    def findLCA(self, roots):
        """
        :type roots: List[List[int]]
        :rtype: TreeNode
        """
        root = {}
        
        for node in roots:
            if not node[0] in root:
                root[node[0]] = []
            root[node[0]].append(node)
        
        stack = [root]
        while stack:
            cur_level = []
            
            for level in stack:
                
                for node in level:
                    for child in node[1:]:
                        if child[0] not in cur_level and child[0] != 0:
                            cur_level.append(child)
                            
            stack = cur_level
        
        lca = None
        path = []
        
        for node in stack:
            path.append(node[0])
            
            if len(path) > 2 and path[-2:] == path[:-2]:
                lca = node[0]
                break
            
        return lca
