```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        d = {}
        for s in strs:
            sorted_s = ''.join(sorted(s))
            if sorted_s in d:
                d[sorted_s].append(s)
            else:
                d[sorted_s] = [s]
                
        result = [[d[i]] for i in sorted(d)]
        return result
```
