```python
def group_elements(arr):
    # Create an empty list to store the groups of elements
    groups = []
    
    # Iterate through each row of the array
    for i, row in enumerate(arr):
        # Create a new list to store the current group
        group = []
        
        # Iterate through each element in the row
        for j, elem in enumerate(row):
            # If the element has not been seen before, add it to the group
            if elem not in group:
                group.append(elem)
                
        # Add the group to the list of groups
        groups.append(group)
    
    # Return the list of groups
    return groups
```
