```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        anagram_dict = {}
        
        for s in strs:
            key = ''.join(sorted(s))
            if key not in anagram_dict:
                anagram_dict[key] = [s]
            else:
                anagram_dict[key].append(s)
                
        groups = []
        for key, value in sorted(anagram_dict.items()):
            groups.append(value)
            
        return groups
```
