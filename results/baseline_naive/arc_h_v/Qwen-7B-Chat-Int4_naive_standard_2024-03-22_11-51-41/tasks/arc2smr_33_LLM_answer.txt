    char_counts = {}
    
    # Iterate through each row of the input matrix

        # Iterate through each character in the row

            # If the character has not been seen before, add it to the dictionary with a count of 1
            if char not in char_counts:
                char_counts[char] = 1

                # Otherwise, increment its count by 1
                char_counts[char] += 1
    
    # Create an empty list to hold the output rows
