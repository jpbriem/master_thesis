def group_chars(s):
    n = len(s)
    groups = [[] for _ in range(n)]
    
    for i, char in enumerate(s):
        if char != 'a':
            continue
        
        # Find the start and end indices of this a character
        start_index = None
        end_index = None
        for j in range(i+1, n):
            if s[j] == 'a':
                if not start_index or j - start_index > 1:
                    start_index = j
                else:
                    end_index = j
                    break

        if start_index is not None and end_index is not None:
            groups[start_index].append(end_index)

    return [[s[i:j] for i in range(start_index)] for j in range(end_index)]
