class Solution:
    def findLadders(self, start: List[str], end: List[str]) -> List[List[int]]:
        graph = {}
        
        for i in range(len(start)):
            if start[i] not in graph:
                graph[start[i]] = []
            
            graph[start[i]].append(i)
                
        queue = [(0, 0)]
        
        while queue:
            d, r = heapq.heappop(queue)
            
            if d == len(end):
                return [r]
            
            for i in range(d, -1, -1):
                if end[i] in graph and (r, i) not in graph[end[i]]:
                    graph[end[i]][(r, i)] = d + 1
                    
                    heapq.heappush(queue, (d + 1, i))
                    
        return []
