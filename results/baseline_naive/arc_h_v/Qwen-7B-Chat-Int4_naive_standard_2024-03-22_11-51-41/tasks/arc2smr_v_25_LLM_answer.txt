def group_same_letters(input):
    # Initialize an empty dictionary to store the groups of same letters
    groups = {}
    
    for row in input:
        # Iterate over each letter in the current row
        for i, letter in enumerate(row):
            # If the current letter has not been seen before, add it as a key to the dictionary with value None
            if letter not in groups:
                groups[letter] = None
                
            # If the current letter has already been seen and its corresponding value is still None, set it to True
            elif groups[letter] is None:
                groups[letter] = True
                
            # Otherwise, if the current letter has already been seen and its corresponding value is True, set it to False
            else:
                groups[letter] = False
    
    # Create a list of lists containing all the groups of same letters
    output = []
    for letter, value in groups.items():

            continue

            output.append([letter])

    return output
