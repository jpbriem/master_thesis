def longest_common_subsequence(input):
    m = len(input)
    n = len(max(input))
    
    dp = [[0] * n for _ in range(m)]
    
    # Fill in the first row and column with 1s
    for i in range(n):
        dp[0][i] = 1
    
    for j in range(1, m):
        for i in range(1, n):
            if input[j-1][i-1] == input[j-1][i]:
                dp[j][i] = dp[j-1][i-1] + 1
            else:
                dp[j][i] = max(dp[j-1][i], dp[j][i-1])
    
    return dp[-1][-1]
