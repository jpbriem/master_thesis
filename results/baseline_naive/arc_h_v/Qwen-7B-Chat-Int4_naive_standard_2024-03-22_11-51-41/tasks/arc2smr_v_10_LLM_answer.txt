```python
def group_elements(arr):
    groups = []
    for i in range(len(arr)):
        if arr[i] not in groups:
            groups.append([arr[i]])
        else:
            groups[groups.index(arr[i])].append(arr[i])
    return [group for group in groups if len(group) > 1]
```
