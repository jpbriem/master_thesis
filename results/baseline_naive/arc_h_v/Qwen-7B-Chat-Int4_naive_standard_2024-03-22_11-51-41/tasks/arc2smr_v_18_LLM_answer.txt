```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        anagrams = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key in anagrams:
                anagrams[key].append(s)
            else:
                anagrams[key] = [s]
                
        result = []
        for k, v in anagrams.items():
            result.append(v)
            
        return result
```
