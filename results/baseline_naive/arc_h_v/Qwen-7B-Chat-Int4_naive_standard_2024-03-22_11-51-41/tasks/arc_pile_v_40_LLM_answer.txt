def find_duplicates(input):
    def is_duplicate(sublist):
        # Check if the second and third elements are equal to the first element
        if len(sublist) < 4 or sublist[1] != sublist[2]:
            return False

        # Compare the fourth element with the last element of the previous sublist
        if len(sublist) > 3 and sublist[3] == sublist[-1]:
            return True

        # Recursively check the remaining sublists
        for i in range(4, len(sublist)):
            if is_duplicate(sublist[i:]):
                return True

        return False

    # Sort the input array in ascending order based on the first element of each sublist
    input.sort()

    # Initialize an empty list to store the duplicates
    output = []

    # Iterate over each sublist
    for sublist in input:
        # Add the sublist to the output list if it contains duplicates
        if is_duplicate(sublist):
            output.append(sublist)

    return output
