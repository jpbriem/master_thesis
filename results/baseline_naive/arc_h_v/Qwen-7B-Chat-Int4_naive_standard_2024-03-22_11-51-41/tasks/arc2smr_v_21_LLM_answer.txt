def longest_common_subsequence(s):
    n = len(s)
    dp = [[0] * (n + 1) for _ in range(n + 1)]

    def helper(i, j):
        # base cases
        if i == 0 or j == 0:
            return 0

        # if characters at current positions match
        if s[i - 1] == s[j - 1]:
            dp[i][j] = helper(i - 1, j - 1) + 1

        # otherwise, take maximum of two options
        else:
            dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

        return dp[i][j]

    # initialize the first row and column as all zeros
    for i in range(1, n + 1):
        dp[i][0] = 0
        dp[0][i] = 0

    # calculate the length of the longest common subsequence
    lcs_length = helper(n, n)

    # create a list to store the result
    result = []

    # start from the bottom-right corner of the matrix
    i, j = n, n
    while i > 0 and j > 0:
        # get the index of the common ancestor
        index = min(dp[i][j], dp[i - 1][j], dp[i][j - 1]) - 1

        # add that character to the result
        result.append(s[index])

        # move up one level
        i -= 1
        j -= 1

    # reverse the resulting string
    result.reverse()

    return "".join(result)
