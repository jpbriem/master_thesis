def group_elements(arr):
    # Create an empty list to store the groups of elements
    groups = []

    # Iterate over each element in the array
    for i in range(len(arr)):
        # If the current element has not been added to any group yet, create a new group and add it
        if len(groups) == 0 or arr[i] != groups[-1][0]:
            groups.append([arr[i]])
        else:
            # Otherwise, find the last group and append the current element to it
            groups[-1].append(arr[i])

    return groups
