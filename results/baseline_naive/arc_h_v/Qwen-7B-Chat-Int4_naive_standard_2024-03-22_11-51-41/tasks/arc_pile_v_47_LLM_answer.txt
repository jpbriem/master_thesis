    def findDuplicate(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: List[List[int]]
        """
        seen = set()
        result = []
        
        for num in nums:
            if num not in seen:
                seen.add(num)
                result.append(num)
                
        return result
