def longest_common_subsequence(A, B):
    m = len(A)
    n = len(B)

    # Create a 2D matrix to store the lengths of common subsequences
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # Fill the first row and column with zeros
    for i in range(1, m + 1):
        dp[i][0] = 0
    for j in range(1, n + 1):
        dp[0][j] = 0

    # Iterate through each element of A and B
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if A[i - 1] == B[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # Initialize an empty list to store the common subsequence
    result = []

    # Start from the bottom right corner of the matrix
    i, j = m, n

    while i > 0 and j > 0:
        # If the current elements are equal, add it to the result and move diagonally up left
        if A[i - 1] == B[j - 1]:
            result.append(A[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] >= dp[i][j - 1]:
            i -= 1
        else:
            j -= 1

    return ''.join(result[::-1])
