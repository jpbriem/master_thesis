def countPaths(matrix):
    n = len(matrix)
    m = len(matrix[0])
    dp = [[0]*m for _ in range(n)]
    dp[0][0] = 1

    for i in range(1, n):
        for j in range(1, m):
            if matrix[i-1][j-1] != matrix[i][j]:
                dp[i][j] = 0
            else:
                dp[i][j] += dp[i-1][j]
                dp[i][j] += dp[i][j-1]
