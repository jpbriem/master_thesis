```python
class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        def backtrack(nums, path=None):
            if not nums:
                res.append(path)
                
            for i in range(len(nums)):
                num = nums[i]
                if path is None:
                    path = [num]
                else:
                    path.append(num)
                    
                backtrack(nums[:i] + nums[i+1:], path)
                
        backtrack(nums)
        return res
```
