class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        dic = {}
        
        for s in strs:
            if len(s) == 0:
                continue
            
            # Convert the string to lowercase and remove any whitespace characters.
            s = ''.join(sorted(s.lower().replace(' ', '')))
            
            # Check if the string is already in the dictionary. If it is not, add it to the list of answers.
            if s not in dic:
                dic[s] = [s]
            else:
                dic[s].append(s)
                
        return ans
