```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        groups = [[] for _ in range(len(strs))]
        
        for i in range(len(strs)):
            hash_table = {}
            for char in strs[i]:
                if char not in hash_table:
                    hash_table[char] = [i]
                else:
                    hash_table[char].append(i)
            
            groups[hash_table[char[0]]].extend(hash_table[char])
        
        return groups
```
