```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0 or len(set(strs[0])) == 0:
            return []
        
        d = {}
        for s in strs:
            t = ''.join(sorted(s))
            if t not in d:
                d[t] = [s]
            else:
                d[t].append(s)
                
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
```
