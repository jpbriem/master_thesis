```python
def group_rows(matrix):
    # Create an empty dictionary to store the groups
    groups = {}

    for i, row in enumerate(matrix):
        # Initialize a new group if necessary
        if i not in groups:
            groups[i] = []

        # Add each element of the current row to its corresponding group
        for j, elem in enumerate(row):
            if j not in groups[i]:
                groups[i].append([])
            groups[i][j].append(elem)

    # Convert the groups into a list of lists
    grouped_matrix = [[row[j] for row in groups[i]] for i, row in enumerate(groups)]

    return grouped_matrix
```
