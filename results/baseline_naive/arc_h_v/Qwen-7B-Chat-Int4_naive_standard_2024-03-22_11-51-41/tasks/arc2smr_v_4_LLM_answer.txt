def longest_common_subsequence(A, B):
    m = len(A)
    n = len(B)

    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if A[i - 1] == B[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    result = []
    index = m

    while index > 0:
        subsequence = [A[index - 1]]
        for i in range(index - 1, 0, -1):
            if dp[i][index - 1] == dp[index][i - 1]:
                subsequence.append(A[i - 1])
                index -= 1
        result.append(subsequence[::-1])

    return result

For each element at position `(i, j)` in the matrix `dp`, we check if the characters at positions `i-1` and `j-1` are equal or not. After filling up the entire matrix `dp`, we can backtrack through it to find the longest common subsequence between the two lists `A` and `B`. We start from the bottom right corner of the matrix `dp` and work our way towards the top left corner.
