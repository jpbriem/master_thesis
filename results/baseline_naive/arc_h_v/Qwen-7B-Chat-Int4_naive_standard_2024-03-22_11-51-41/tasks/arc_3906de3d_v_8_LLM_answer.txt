```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        result = []
        
        for s in strs:
            # convert string to list of characters
            chars = list(s)
            
            # sort the characters and remove duplicates
            sorted_chars = ''.join(sorted(chars))
            
            # check if the sorted character is already in the result
            found = False
            
            for r in result:
                # iterate through each element in the result
                for char in r:
                    # check if the current character matches the sorted character
                    if char == sorted_chars:
                        # add the current element to the result
                        found = True
                        break
                        
                # exit loop if the current element is found
                if found:
                    break
                
            # if the current character is not found in the result,
            # append it as a new element
            if not found:
                result.append([sorted_chars])
                
        return result
```
