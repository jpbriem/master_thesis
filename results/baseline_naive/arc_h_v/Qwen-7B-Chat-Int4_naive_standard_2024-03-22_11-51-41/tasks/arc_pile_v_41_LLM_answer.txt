```python
class Solution:
     def groupAnagrams(self, strs):
         """
         :type strs: List[str]
         :rtype: List[List[str]]
         """
         result = []
         for s in strs:
             if len(s) == 0 or s[0] not in result[0]:
                 result.append([s])
             else:
                 index = [i for i, x in enumerate(result) if x[0] == s[0]][0]
                 result[index].append(s)
         return result
```
