def longest_palindrome(input):
    # Convert input string to lowercase and remove spaces
    input = "".join(c for c in input.lower() if c.isalnum()).replace(" ", "")
    
    # Initialize variables
    start, max_len = 0, 1
    
    # Iterate over each character of the input string
    for i in range(len(input)):
        # If current character is not repeated, update max_len
        if i - max_len + 1 < len(input) and input[i-max_len] == input[i]:
            max_len += 1
        
        # Update start variable if we find a longer palindrome starting from this character
        if i - max_len >= 0 and input[i-max_len:i+1] == input[i-max_len-1:i+2][::-1]:
            start = i - max_len
    
    return " ".join(input[start:start+max_len])
