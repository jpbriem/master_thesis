class Solution:
    def replaceBlanks(self, words: List[List[str]]) -> List[List[str]]:
        """
        思路：
            将所有单词中的空格都替换为'f'，然后对每个单词进行分割。
                如果一个单词中只包含了字母'f'，则说明这个单词是由多个连续的空格组成的；
                否则，就按照单词中出现的顺序，将其中的每个字符串放入新的列表中。
            最后再对每个新的列表进行去重操作，得到最终结果。
            O(n) time complexity. O(n) space complexity.
        :param words: A list of strings.
        :return: A list of strings.
        """
        # 将所有单词中的空格都替换为'f'
        for i in range(len(words)):
            words[i] = "".join([word if word!= "" else "f" for word in words[i].split()])

        # 对每个单词进行分割
        result = []
        for word in words:
            temp = []
            for char in word:
                if char == "f":
                    continue
                elif len(temp) > 0 and temp[-1] == char:
                    pass
                else:
                    temp.append(char)
            result += [[item] for item in set(temp)]
        return result
