from typing import List, Optional


class Solution:
    def groupAnagrams(self, strs: List[List[str]]) -> List[List[str]]:
        """
        1. sort each string in list
        2. compare first char of sorted strings to find anagram groups
            - if they match, add them to a new list
                  - if not, create a new list for this group
        3. return all lists
        :param strs:
        :return:
        """
        # O(n) time | O(n) space - where n is len(strs)
        result = []
        for s in strs:
            sorted_string = "".join(sorted(s))
            found = False
            for i, r in enumerate(result):
                if sorted_string == "".join(sorted(r[0])):
                    result[i].append(s)
                    found = True
                    break
            if not found:
                result.append([s])
        return result
