import collections
from typing import List, Tuple


class Solution:
    def replaceElements(self, arr: List[List[str]]) -> List[List[str]]:
        """
        思路：
            将所有元素分成两组，一组是'a'和'c'，另一组是其他字符。
            a 与 c 的替换规则为：
                - 如果前面是 a 或者 c，那么后面就要变成对应的字符；
                    否则保持原样。
            b 等其他字符的替换规则为：
                - 如果前面是 a 或者 c，那么后面就要变成对应的字符；
                    否则不变。
        :param arr:
        :return:
        """
        # 记录每个位置上出现了多少次 a/c
        count = collections.Counter()
        for row in arr:
            count.update([x for x in row if x not in ('a', 'c')])
        # 计算出每个位置上 a/c 的数量
        count += collections.Counter(['a', 'c']) * len(arr)
        # 根据计算结果进行替换
        for i, row in enumerate(arr):
            for j, char in enumerate(row):
                if char == 'a':
                    arr[i][j] = 'a' if count[(i, j)] > 0 else 'c'
                elif char == 'c':
                    arr[i][j] = 'c' if count[(i, j)] < 4 else 'a'
                elif char!= 'a' and char!= 'c':
                    arr[i][j] = char if count[(i, j)] <= 1 else 'a'
        return arr
