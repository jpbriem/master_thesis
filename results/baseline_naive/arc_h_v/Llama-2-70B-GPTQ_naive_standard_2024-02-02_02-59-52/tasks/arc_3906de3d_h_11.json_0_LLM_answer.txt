class Solution(object):
    def reorderGrid(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: void Do not return anything, modify grid in-place instead.

    def _reorderRow(self, grid, i):

                break
            else:
                j += 1

            if grid[i][k]!= 'a':
                break
            else:
                k -= 1

        for l in reversed(xrange(j, k + 1)):
