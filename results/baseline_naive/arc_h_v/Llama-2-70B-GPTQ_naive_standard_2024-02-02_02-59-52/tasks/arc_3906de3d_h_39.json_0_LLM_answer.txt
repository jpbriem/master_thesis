class Solution:
    def transform(self, arr: List[List[str]]) -> List[Optional[str]]:
        """
        思路：
            将每一行的字符串转换为二进制，然后按照二进制中0和1对应的位置来排序。
            >>> 输入：[[‘a’,’b’,’c’], [‘d’,’e’,’f’],[‘g’,’h’,’i’]]
            >>> 输出：[[‘a’,’d’,’g’], [‘b’,’e’,’h’], [‘c’,’f’,’i’]]

        注意事项：
            如果有两个或者多个相同的元素，则只保留第一次出现的那个。
            >>> 输入：[[‘a’,’b’,’c’], [‘d’,’e’,’f’],[‘g’,’h’,’i’], [‘j’,'k','l']]
            >>> 输出：[[‘a’,’d’,’g’], [‘b’,’e’,’h’], [‘c’,’f’,’i’], [‘j’,'k','l']]

        复杂度分析：
            时间复杂度：O(n)，其中 n 是数组的长度。
                遍历所有的元素，并且对于每一个元素都需要 O(m) 的时间复杂度，其中 m 是该元素的长度。
            空间复杂度：O(n)，其中 n 是数组的长度。
                O(n) 的空间用于存储最终结果。
        """
        # 使用正则表达式将每一行的字符串转换为二进制
        binary = [[bin(ord(x)).replace('0b', '') for x in row] for row in arr]
        # 按照二进制中0和1对应的位置来排序
