from typing import List, Optional


class Solution:
    def groupAnagrams(self, strs: List[List[str]]) -> List[List[str]]:
        """
        1. sort each string in list
        2. compare first character of sorted strings to find anagram groups
            - if they match then add them to a new list
                  - if not then create a new list for that group
        3. return all lists with their original order
        :param strs:
        :return:
        """
        # O(n) time | O(n) space - where n is len(strs)
        result = []
        for s in strs:
            temp = "".join(sorted(s))
            found = False
            for i, r in enumerate(result):
                if temp == "".join(sorted(r)):
                    result[i].append(s)
                    found = True
                    break
            if not found:
                result.append([s])
        return result
