    n = len(matrix)
    m = len(matrix[0])
    
    # Create a list of tuples representing each cell in the matrix,
    # where each tuple contains the row and column indices and the value.
    cells = [(i, j, v) for i in range(n) for j in range(m) if matrix[i][j] != 0]
    
    # Sort the cells based on their values in descending order.
    cells.sort(key=lambda x: x[2], reverse=True)
    
    # Iterate through the sorted cells and fill in zeros to the left and above them.
    for i, j, v in cells:
        for k in range(i+1):
            matrix[k][j] = 0
