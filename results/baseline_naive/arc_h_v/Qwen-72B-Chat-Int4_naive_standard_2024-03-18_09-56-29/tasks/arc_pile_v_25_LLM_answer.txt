```python
def merge_adjacent(grid):
    rows = len(grid)
    cols = len(grid[0])

    for i in range(rows):
        for j in range(cols - 1): # Iterate until one before last column
            if grid[i][j] == grid[i][j+1]:
                merged_value = grid[i][j]
                while j < cols and grid[i][j] == merged_value:
                    j += 1
                j -= 1 # Backtrack to the previous cell

                # Update cells with merged value
                for k in range(i, rows):
                    grid[k][j] = merged_value

    return grid
```
