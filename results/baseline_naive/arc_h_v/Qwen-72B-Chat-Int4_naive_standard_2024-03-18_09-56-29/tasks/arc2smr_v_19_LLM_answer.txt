import heapq

def dijkstra(grid, start, end):
    rows, cols = len(grid), len(grid[0])
    
    # Initialize dictionaries to store distances and parents
    distances = {frozenset({(r, c)}): float('inf') for r in range(rows) for c in range(cols)}
    distances[frozenset({start})] = 0
    
    parent = {}
    
    # Add start node to priority queue
    pq = [(0, frozenset({start}))]

    while pq:
        dist, visited_nodes = heapq.heappop(pq)
        
        if end in visited_nodes:
            break
        
        for dr, dc in [(0, -1), (-1, 0), (0, 1), (1, 0)]:
            r, c = start[0]+dr, start[1]+dc

                if new_node not in distances or dist + 1 < distances[new_node]:

                    heapq.heappush(pq, (distances[new_node], new_node))
