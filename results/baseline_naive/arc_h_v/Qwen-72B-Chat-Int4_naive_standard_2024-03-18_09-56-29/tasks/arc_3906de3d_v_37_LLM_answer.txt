def flood_fill(matrix, row, col):
    if not (0 <= row < len(matrix)) or not (0 <= col < len(matrix[0])):
        return False

    if matrix[row][col] != 9:
        return False

    matrix[row][col] = 6

    for dr, dc in [(0, -1), (-1, 0), (0, 1), (1, 0)]:
        r, c = row + dr, col + dc
        if flood_fill(matrix, r, c):
            continue

    return True


def replace_nines(matrix):
    for i, row in enumerate(matrix):
        for j, _ in enumerate(row):
            if flood_fill(matrix, i, j):
                continue

    return matrix
