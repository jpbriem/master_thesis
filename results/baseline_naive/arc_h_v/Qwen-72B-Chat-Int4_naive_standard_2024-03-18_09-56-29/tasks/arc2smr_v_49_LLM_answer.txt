```python
def update_matrix(matrix):
    # Get the number of rows and columns in the matrix
    n = len(matrix)
    
    # Iterate over each cell in the matrix
    for i in range(n):
        for j in range(n):
            # Calculate the sum of the current row and column
            row_sum = sum(matrix[i])
            col_sum = sum(matrix[k][j] for k in range(n))
            
            # Update the value of the current cell
            matrix[i][j] = row_sum + col_sum
    
    return matrix
```
