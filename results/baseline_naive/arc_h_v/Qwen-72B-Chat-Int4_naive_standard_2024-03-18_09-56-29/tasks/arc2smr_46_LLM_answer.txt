def replace_elements(matrix):
    rows = len(matrix)
    cols = len(matrix[0])
    
    for i in range(rows):
        for j in range(cols):
            if matrix[i][j] == 4:
                # Replace all elements in the same row as 3
                for k in range(cols):
                    matrix[i][k] = 3

                for k in range(rows):
                    matrix[k][j] = 3
