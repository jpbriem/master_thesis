def mark_path(matrix):
    rows = len(matrix)
    cols = len(matrix[0])
    
    for i in range(rows):
        for j in range(cols):
            if matrix[i][j] == 5:
                # Mark cells in the same row as 1
                for k in range(cols):
                    matrix[i][k] = 1
