def count_islands(grid):
    m = len(grid)
    n = len(grid[0])
    visited = set()
    
    def dfs(i, j):
        if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] == 0 or (i,j) in visited:
            return
        
        visited.add((i,j))
        dfs(i+1, j)
        dfs(i-1, j)
        dfs(i, j+1)
        dfs(i, j-1)
        
    counts = []
    for i in range(m):
        for j in range(n):
            if grid[i][j] == 1 and (i,j) not in visited:
                visited.clear()
                dfs(i, j)
                counts.append(len(visited))
                
    return max(counts)
