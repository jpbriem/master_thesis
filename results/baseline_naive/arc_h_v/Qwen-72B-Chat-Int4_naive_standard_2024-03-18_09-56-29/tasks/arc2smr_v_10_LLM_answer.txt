def count_boulders(grid):
    m = len(grid)
    n = len(grid[0])
    
    for i in range(m):
        for j in range(n):
            if grid[i][j] == 2:
                count = 1

                while queue:
                    x,y = queue.pop(0)
                    
                    for dx, dy in [(0,1), (0,-1), (1,0), (-1,0)]:
                        nx, ny = x + dx, y + dy
