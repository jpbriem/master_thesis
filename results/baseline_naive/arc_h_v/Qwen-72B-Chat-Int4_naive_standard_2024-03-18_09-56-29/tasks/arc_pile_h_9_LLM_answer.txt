def move_zeros(matrix):
    non_zero_indices = []
    
    # Find indices of non-zero elements in each row
    for i in range(len(matrix)):
        for j in range(len(matrix[i])):
            if matrix[i][j] != 0:
                non_zero_indices.append((i, j))
                
    # Move all zeros to the end of their respective rows
    for (row_index, col_index) in reversed(non_zero_indices):
        while matrix[row_index].count(0) > 0:
            zero_index = matrix[row_index].index(0)
            matrix[row_index][zero_index], matrix[row_index][col_index] = \
                matrix[row_index][col_index], matrix[row_index][zero_index]
            
    return matrix
