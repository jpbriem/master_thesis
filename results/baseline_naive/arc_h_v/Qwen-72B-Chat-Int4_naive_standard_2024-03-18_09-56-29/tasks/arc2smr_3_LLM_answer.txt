```python
def fill_grid(grid):
    n = len(grid)
    for i in range(n):
        for j in range(n):
            if grid[i][j] == 3:
                k = 1
                while True:
                    x = i + k
                    y = j + k
                    if x >= n or y >= n or grid[x][y] != 3:
                        break
                    k += 1
                x = i - k + 1
                y = j - k + 1
                for dx in range(k):
                    for dy in range(k):
                        grid[x+dx][y+dy] = 1
    return grid
```
