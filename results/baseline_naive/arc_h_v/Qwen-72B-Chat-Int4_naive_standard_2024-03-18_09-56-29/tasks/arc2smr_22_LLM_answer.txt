def connect_squares(grid):
    # Create an empty grid to store the connected squares
    connected_grid = []
    for row in grid:
        connected_row = []
        for cell in row:
            connected_row.append(0)
        connected_grid.append(connected_row)

    # Iterate over each cell in the original grid
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == 0:
                continue

            # Find all cells that are part of this square
            square_cells = set()
            queue = [(i, j)]
            while queue:
                x, y = queue.pop(0)
                if (x < 0 or x >= len(grid) or y < 0 or y >= len(grid[0]) or
                        grid[x][y] != grid[i][j]):
                    continue
                square_cells.add((x, y))
                queue.extend(((x-1, y), (x+1, y), (x, y-1), (x, y+1)))

            # Connect all cells in the square by setting their values in the new grid
            for x, y in square_cells:
                connected_grid[x][y] = max(square_cells, key=lambda p: abs(p[0]-i)+abs(p[1]-j))[1]+1

    return connected_grid
