public class Solution {
    public int[] rearrangeArray(int[] nums) {
        if (nums == null || nums.length < 2) return nums;
        // 奇数个元素，最后一个元素不能移动
        boolean odd = nums.length % 2!= 0;
        int n = nums.length / 2 + (odd? 1 : 0);
        for (int i = 0; i < n; ++i) {
            int j = i * 2 + 1 - (odd && i == n - 1? 1 : 0);
            swap(nums, i, j);
        }
        return nums;
    }

    private void swap(int[] nums, int i, int j) {
        int tmp = nums[j];
        nums[j] = nums[i];
        nums[i] = tmp;
    }
}
