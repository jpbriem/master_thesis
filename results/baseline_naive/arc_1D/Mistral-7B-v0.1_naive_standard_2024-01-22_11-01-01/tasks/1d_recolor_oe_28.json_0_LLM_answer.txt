public class Solution {
    public int[] replaceZeroes(int[] arr) {
        if (arr == null || arr.length < 2) return arr;
        int n = arr.length;
        for (int i = 0; i < n - 1; ++i) {
            if (arr[i] == 0 && arr[i + 1]!= 0) {
                swap(arr, i, i + 1);
                --i;
            } else if (arr[i] == 0 && arr[i + 1] == 0) {
                while (i < n - 1 && arr[i + 1] == 0) {
                    ++i;
                }
                if (i >= n - 1) break;
                swap(arr, i, i + 1);
                --i;
            }
        }
        return arr;
    }

    private void swap(int[] arr, int a, int b) {
        int tmp = arr[a];
        arr[a] = arr[b];
        arr[b] = tmp;
    }
}
