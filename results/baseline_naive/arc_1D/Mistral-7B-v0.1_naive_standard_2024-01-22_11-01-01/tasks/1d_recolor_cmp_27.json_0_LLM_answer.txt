public class Solution {
    public int[] replaceZeroes(int[] arr) {
        if (arr == null || arr.length < 1) return new int[]{};
        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i] == 0 && arr[i + 1]!= 0) {
                swap(arr, i);
                i++;
            }
        }
        return arr;
    }

    private void swap(int[] arr, int index) {

        arr[index + 1] = temp;
    }
