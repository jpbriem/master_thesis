class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        int n = nums.size();
        if (n == 0 || n == 1) return nums;
        // 奇数个元素，最后一个元素不能移动
        bool odd = n % 2!= 0;
        for (int i = 0; i < n - 1; ++i) {
            int j = i + 1;
            while (j < n && nums[j] > nums[i]) ++j;
            swap(nums[i], nums[j]);
        }
        if (!odd) {
            --n;
            for (int i = 0; i < n / 2; ++i) {
                swap(nums[i], nums[n - i - 1]);
            }
        } else {
            for (int i = 0; i < n / 2; ++i) {
                swap(nums[i], nums[n - i - 1]);
            }
        }
        return nums;
    }
