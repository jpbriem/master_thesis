public class Solution {
    public int[] sortArrayByParityII(int[] A) {
        if (A == null || A.length < 2) return A;
        List<Integer> list = new ArrayList<>();
        for (int i : A) {
            if ((i & 1)!= 0) {
                list.add(i);
            }
        }
        for (int i : A) {
            if ((i & 1) == 0) {
                list.add(i);
            }
        }
        return list.stream().mapToInt(x -> x).toArray();
    }
}
