class Solution {
public:
    vector<int> countBits(vector<int>& nums) {
        int n = nums.size();
        vector<int> res;
        for (int i=0;i<n;++i){
            if (nums[i]<=1){
                res.push_back(nums[i]);
            }else{
                int cnt = 0;
                while ((nums[i]&cnt)==0){

                res.push_back(cnt);
            }
        }

    }
