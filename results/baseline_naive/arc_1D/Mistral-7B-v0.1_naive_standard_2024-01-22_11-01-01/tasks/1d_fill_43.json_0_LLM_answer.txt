public class Solution {
    public int[] doubleZeros(int[] arr) {
        if (arr == null || arr.length < 2) return arr;
        int i = 0, j = arr.length - 1;
        while (i < j && arr[i]!= 0) i++;
        while (j > i && arr[j]!= 0) j--;
        for (int k = i + 1; k < j; k++) {
            arr[k] = 0;
        }
        return arr;
    }
