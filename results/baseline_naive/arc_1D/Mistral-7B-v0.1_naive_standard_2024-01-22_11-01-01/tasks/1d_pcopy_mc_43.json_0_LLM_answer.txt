public class Solution {
    public int[] duplicateZeros(int[] arr) {
        if (arr == null || arr.length < 1) return new int[]{};

        for (int i = 0; i < n; i++) {
            if (arr[i] == 0) {
                // 如果当前元素为0，则将后面的元素向前移动一位
                for (int j = n - 1; j > i + 1; j--) {
                    arr[j] = arr[j - 1];

                // 然后将当前元素设置为0

        }
