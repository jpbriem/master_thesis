class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        int n = nums.size();
        if (n == 0 || n == 1) return nums;
        // 奇数个元素，最后一个元素不能移动
        bool odd = n % 2!= 0;
        // 偶数个元素，最后两个元素不能移动
        bool even = n % 2 == 0;
        // 记录每个元素的出现次数
        unordered_map<int, int> cnt;
        for (auto &num : nums) ++cnt[num];
        // 构造新数组
        vector<int> res(n);
        int i = 0;
        while (!odd &&!even) {
            auto it = cnt.begin();
            while (it->second > 0) {
                res[i++] = it->first;
                --it->second;
            }
            ++it;
            while (it!= cnt.end()) {
                res[i++] = it->first;
                --it->second;
            }
            odd = false;
            even = false;
            if (i >= n - 1) break;
            it = cnt.begin();
            while (it->second > 0) {
                ++it;
            }
            if (it == cnt.end() || it->second <= 0) continue;
            else {
                odd = true;
                even = true;
            }
        }
        return res;
    }
};
