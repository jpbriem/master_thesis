public class Solution {
    public int[] duplicateZeros(int[] arr) {
        if (arr == null || arr.length < 2) return arr;
        int n = arr.length - 1;
        for (int i = 0; i <= n; i++) {
            if (i >= n && arr[n]!= 0) break;
            if (arr[i] == 0) {
                for (int j = n; j > i + 1; j--) {
                    arr[j] = arr[j - 1];
                }
                arr[i + 1] = 0;
                n++;
            }
        }
        return arr;
    }
}
