class Solution {
public:
    vector<int> countBits(int num) {
        vector<int> res;
        for (int i = 0; i <= num; ++i){
            int cnt = 0;
            while (i > 0){
                if ((i & 1) == 1) cnt++;
                i >>= 1;

        }
        return res;
