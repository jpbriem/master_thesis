public class Solution {
    public int[] replaceElements(int[] arr) {
        if (arr == null || arr.length <= 1) return arr;
        int max = Integer.MIN_VALUE;
        for (int i = arr.length - 1; i >= 0; --i) {

            arr[i] = max;

        }
