public class Solution {
    public int[] replaceZeroes(int[] A) {
        if (A == null || A.length < 1) return new int[]{};
        for (int i = 0; i < A.length - 1; i++) {
            if (A[i] == 0 && A[i + 1]!= 0) {
                swap(A, i);
                i++;
            }
        }
        return A;
    }

    private void swap(int[] A, int index) {
        int temp = A[index];
        A[index] = A[index + 1];
        A[index + 1] = temp;
    }
}
