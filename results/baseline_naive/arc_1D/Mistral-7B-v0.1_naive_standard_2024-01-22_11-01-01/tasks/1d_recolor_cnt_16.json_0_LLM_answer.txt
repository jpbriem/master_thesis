public class Solution {
    public int[] reorganizeArray(int[] nums) {
        if (nums == null || nums.length < 2) return nums;
        Map<Integer, Integer> map = new HashMap<>();
        for (int i : nums) {
            map.put(i, map.getOrDefault(i, 0) + 1);
        }
        PriorityQueue<Map.Entry<Integer, Integer>> pq = new PriorityQueue<>((a, b) -> a.getValue() - b.getValue());
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            pq.offer(entry);
        }
        List<Integer> list = new ArrayList<>();
        while (!pq.isEmpty()) {
            Map.Entry<Integer, Integer> first = pq.poll();
            if (list.size() > 0 && list.get(list.size() - 1).equals(first.getKey())) {
                continue;
            }
            list.add(first.getKey());
            if (--first.getValue()!= 0) {
                pq.offer(first);
            }
            if (pq.peek().getValue() == 0) {
                break;
            }
            Map.Entry<Integer, Integer> second = pq.poll();
            list.add(second.getKey());
            if (--second.getValue()!= 0) {
                pq.offer(second);
            }
        }
        if (list.size() % 2 == 1) {
            return new int[0];
        } else {
            int[] res = new int[list.size() / 2];
            for (int i = 0; i < res.length; ++i) {
                res[i] = list.get(i * 2);
            }
            return res;
        }
    }
}
