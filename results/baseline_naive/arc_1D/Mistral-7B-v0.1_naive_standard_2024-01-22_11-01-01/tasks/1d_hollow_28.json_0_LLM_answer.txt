class Solution {
public:
    vector<int> replaceZeroes(vector<int>& arr) {
        int n = arr.size();
        for (int i=0;i<n;++i){
            if (arr[i] == 0){
                while (i+1 < n && arr[i+1]==0) ++i;
                if (i+1 >= n || arr[i+1]!= 0) break;
                swap(arr[i], arr[i+1]);
            }
        }
        return arr;
    }
