class Solution(object):
    def reverseWords(self, s: str) -> str:
        """
        :type s: str
        :rtype: str
        """
        
        if not isinstance(s,str):
            return None
            
        words = []
        for i in range(len(s)):
            word = ''
            j = i+1
            while j< len(s):
                char = s[j]
                
                # if it's a space or punctuation mark
                if (char =='') | (char == ',') | (char == ';'):
                    break
                    
                elif char == '"':
                    j+=1
                    continue
                    
                else:
                    word += char
                    j+=1
                    
            words.append(word)
            
        return " ".join(words)
