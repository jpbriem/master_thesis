class Solution(object):
    def reverseWords(self, s: str) -> str:
        """
        :type s: str
        :rtype: str
        """
        
        if not isinstance(s,str):
            return None
            
        words = []
        for i in range(len(s)):
            word = ''
            for j in range(i+1, len(s), 1):
                char = s[j]
                
                if char =='' or char == '\n':
                    break
                    
                else:
                    word += char
                    
            words.append(word)
            
        return " ".join([w for w in reversed(words)]).strip()
