import sys
from typing import List

class Solution(object):
    def reverseWords(self, s: str) -> str:
        """
        :type s: str
        :rtype: str
        """
        
        if not isinstance(s,str):
            return None
            
        words = []
        for i in range(len(s)):
            word = ''
            j = i + 1
            while j < len(s):
                char = s[j]
                
                # 判断是否为空格，如果不是则加入到字符串中
                if char =='' or char == '\n':
                    break
                    
                else:
                    word += char
                    j+=1
                    
            words.append(word)
            
        return " ".join([w for w in reversed(words)]).strip()
