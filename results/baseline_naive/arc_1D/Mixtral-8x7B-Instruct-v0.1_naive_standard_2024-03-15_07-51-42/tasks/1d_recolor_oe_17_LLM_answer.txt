public class Solution {
    public int[] solution(int []arr) {
        int count = 0;
        for (int i=0;i<arr.length-1;i++) {
            if (arr[i] == arr[i+1]) {
                count++;
            } else {
                arr[i + 1 - count] = arr[i];
                count = 0;
            }
        }
        arr[arr.length - count] = arr[arr.length - 1];
        return Arrays.copyOfRange(arr, 0, arr.length - count);
    }
}
