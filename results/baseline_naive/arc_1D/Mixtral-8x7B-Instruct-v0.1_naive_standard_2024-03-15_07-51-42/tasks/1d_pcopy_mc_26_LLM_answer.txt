public class Solution {
    public int[] fill(int[] A) {
        if (A == null || A.length < 2) return A;
        int start = -1, end = -1;
        for (int i = 0; i < A.length; ++i) {
            if (start == -1 && A[i]!= 0) {
                start = i;
            } else if (start!= -1 && A[i] == 0) {
                end = i;
                break;
            }
        }
        if (end == -1) {
            end = A.length;
        }
        for (int i = start + 1; i < end; ++i) {
            if (A[i]!= 0) continue;
            int j = start;
            while (j < i) {
                A[i++] = A[j++];
            }
        }
        for (int i = end; i < A.length; ++i) {
            if (A[i]!= 0) continue;
            int j = end - 1;
            while (j >= start && A[j] == 0) --j;
            A[i] = A[j--];
            while (j >= start) {
                A[++i] = 0;
                --j;
            }
        }
        return A;
    }
}
