public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;
        for (int i = input.length - 1; i > 0; --i) {
            if (input[i]!= 0 && input[i-1] == 0) {
                int temp = input[i];
                input[i] = input[i-1];
                input[i-1] = temp;
            }
        }
        return input;
    }
}
