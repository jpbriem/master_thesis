public class Solution {
    public int[] sortByHeight(int[] a) {
        List<Integer> list = new ArrayList<>();
        for (int i : a) if (i!= 0) list.add(i);
        Collections.sort(list);
        for (int i = 0; i < a.length; ++i) {
            if (a[i] == 0) continue;
            a[i] = list.remove(0);
        }
        return a;
    }
}
