public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element from left and right side of array
        int l = -1, r = -1;
        for (int i = 0; i < input.length && l == -1; ++i) {
            if (input[i]!= 0) l = i;
        }
        for (int j = input.length - 1; j >= 0 && r == -1; --j) {
            if (input[j]!= 0) r = j;
        }

        // no zero elements in this array
        if (l == -1 && r == -1) return input;

        // fill all zeros between two non-zero elements with min value among them
        int min = Math.min(input[l], input[r]);
        for (int k = l + 1; k <= r - 1; ++k) {
            input[k] = min;
        }

        // fill all zeros before or after non-zero elements with same values as those non-zero elements
        if (l > 0) System.arraycopy(input, 0, input, 0, l);
        if (r < input.length - 1) System.arraycopy(input, r, input, r + 1, input.length - r - 1);

        return input;
    }
}
