def move(numbers):
    for i in range(len(numbers)):
        if numbers[i] == 9:
            j = i - 1
            while j >= 0 and numbers[j] == 0:
                j -= 1
            if j!= i - 1:
                numbers[j], numbers[i - 1] = numbers[i - 1], numbers[j]
    return numbers
