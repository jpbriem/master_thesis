def fill(numbers):
    i = 0
    while True:
        if numbers[i] == 0:
            j = i + 1
            while j < len(numbers) and numbers[j]!= 0:
                j += 1
            if j - i > 1:
                for k in range(i+1, j-1):
                    numbers[k] = numbers[i+1]
        i += 1
        if i >= len(numbers):
            break
    return numbers
