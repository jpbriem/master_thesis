function removeZeros(arr) {
    let zeroCount = 0;
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === 0) {
            zeroCount++;
        } else {
            arr[i - zeroCount] = arr[i];
        }
    }
    return arr.slice(0, arr.length - zeroCount);
}
