public class Solution {
    public int[] replaceZero(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first zero index
        int start = -1;
        for (int i = 0; i < input.length; i++) {
            if (input[i] == 0) {
                start = i;
                break;
            }
        }

        // no zero found in array
        if (start == -1) return input;

        // keep track of last one position
        int end = start + 1;
        while (end < input.length && input[end]!= 0) {
            end++;
        }

        // fill all zeros with '6' from start to end-1
        for (int i = start; i < end; i++) {
            input[i] = 6;
        }

        // recursively call function on rest of the array
        replaceZero(Arrays.copyOfRange(input, end, input.length));

        return input;
    }
}
