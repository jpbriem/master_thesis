    public int[] fill(int[] input) {
        if (input == null || input.length < 2) return input;

        for (int i = 0; i < input.length; ++i) {

                start = i;
                continue;
            }
            if (start!= -1 && input[i] == 0) {
                end = i;
                break;
            }
        }
        if (end == -1) end = input.length;
        int val = input[start];
        for (int i = start; i < end; ++i) {
            input[i] = val;
        }
        return input;
    }
