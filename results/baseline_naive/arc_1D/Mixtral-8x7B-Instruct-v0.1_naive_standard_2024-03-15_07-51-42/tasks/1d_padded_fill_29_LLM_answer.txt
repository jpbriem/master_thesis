function mergeRanges(meetings) {
    if (meetings.length === 0) return [];
    meetings = meetings.sort((a, b) => a[0] - b[0]);
    let mergedMeetings = [meetings[0]];
    for (let i = 1; i < meetings.length; i++) {
        let lastMerged = mergedMeetings[mergedMeetings.length - 1];
        if (lastMerged[1] >= meetings[i][0]) {
            lastMerged[1] = Math.max(lastMerged[1], meetings[i][1]);
        } else {
            mergedMeetings.push(meetings[i]);
        }
    }
    return mergedMeetings;
}
