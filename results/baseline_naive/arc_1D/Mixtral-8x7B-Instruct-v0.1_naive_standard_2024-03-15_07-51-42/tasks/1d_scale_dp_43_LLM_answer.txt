public class Solution {
    public int[] moveZeroes(int[] nums) {
        if (nums == null || nums.length <= 1) return nums;
        int left = 0, right = 0;
        while (right < nums.length) {
            if (nums[right]!= 0) {
                swap(nums, left++, right);
            }
            right++;
        }
        return nums;
    }

    private void swap(int[] nums, int i, int j) {
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    }
}
