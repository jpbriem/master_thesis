public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find first non-zero element and its value
        int start = -1;
        for (int i = 0; i < input.length && start == -1; ++i) {
            if (input[i]!= 0) {
                start = i;
                break;
            }
        }

        if (start == -1) return input;

        int val = input[start];
        int end = start + 1;
        while (end < input.length && input[end] == val) ++end;

        // update the subarray from index'start' to 'end'
        for (int i = start; i < end; ++i) {
            input[i] = val;
        }

        // recursively process remaining elements in array
        fill(Arrays.copyOfRange(input, end, input.length));

        return input;
    }
}
