public class Solution {
    public int[] replaceZero(int[] input) {
        if (input == null || input.length <= 1) return input;
        int start = -1;
        for (int i = 0; i < input.length; i++) {
            if (input[i]!= 0 && start == -1) {
                start = i;
            } else if (input[i] == 0 && start!= -1) {
                while (start < i) {
                    input[start++] = 0;
                }
                input[i] = input[--i];
                input[i] = 0;
            }
        }
        return input;
    }
}
