public class Solution {
    public int[] longestOnes(int[] A, int K) {
        if (A == null || A.length <= 0) return new int[0];
        int left = 0;
        for (int right = 0; right < A.length; ++right) {
            if (A[right] == 0) --K;
            while (K < 0 && A[left++] == 0);
            if (K >= 0) System.arraycopy(A, left, A, left - right, right - left + 1);
        }
        return Arrays.copyOfRange(A, 0, A.length - (A.length - left));
    }
}
