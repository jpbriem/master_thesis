public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;
        for (int i = 0; i < input.length - 1; i++) {
            if (input[i]!= 0 && input[i + 1] == 0) {
                int j = i + 1;
                while (j < input.length && input[j] == 0) j++;
                if (j >= input.length) break;
                else {
                    int temp = input[j];
                    for (int k = j; k > i; k--) {
                        input[k] = input[k-1];
                    }
                    input[i] = temp;
                }
            }
        }
        return input;
    }
}
