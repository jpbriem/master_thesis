public class Solution {
    public int[] removeDuplicates(int[] A) {
        if (A == null || A.length <= 1) return A;
        int len = A.length;
        for (int i = 0; i < len - 1; ) {
            if (A[i + 1] == A[i]) {
                for (int j = i + 1; j < len - 1; ++j) {
                    A[j] = A[j + 1];
                }
                --len;
            } else {
                ++i;
            }
        }
        int[] res = new int[len];
        System.arraycopy(A, 0, res, 0, len);
        return res;
    }
}
