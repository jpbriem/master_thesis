    public int[] Fill(int[] nums) {
        if (nums == null || nums.Length < 2) return nums;

        for (int i = 0; i < nums.Length - 1; i++) {
            if (nums[i]!= 0 && nums[i + 1] == 0) {
                int j = i + 1;
                while (j < nums.Length && nums[j] == 0) {
                    nums[j++] = nums[i];
                }
            }
        }

        return nums;
    }
}
