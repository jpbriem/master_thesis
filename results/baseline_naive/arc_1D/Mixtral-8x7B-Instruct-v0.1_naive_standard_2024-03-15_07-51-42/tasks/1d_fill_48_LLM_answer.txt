public class Solution {
    public int[] fill(int[] A) {
        if (A == null || A.length < 2) return A;
        int start = -1, end = -1;
        for (int i = 0; i < A.length; ++i) {
            if (start == -1 && A[i]!= 0) {
                start = i;
            } else if (A[i]!= 0) {
                end = i;
            }
        }
        if (start == -1) return A;
        int len = end - start + 1;
        int[] B = new int[A.length];
        System.arraycopy(A, 0, B, 0, A.length);
        for (int i = 0; i <= len; ++i) {
            B[start + i] = Math.max(B[start + i], len);
        }
        return B;
    }
}
