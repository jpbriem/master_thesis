    public int[] zeroFilledSubarray(int[] nums) {
        int count = 0;
        int totalCount = 0;
        for (int i = 0; i < nums.length; ++i) {
            if (nums[i] == 0) {
                count++;
            } else {
                totalCount += (count * (count + 1)) / 2;

        }

        return new int[]{totalCount};
    }
