    public int[] solution(int []arr) {
        int cnt = 0;
        for (int i=0;i<arr.length-1;i++) {
            if (arr[i] == arr[i+1]) {
                cnt++;
            } else {
                arr[i+1-cnt] = arr[i];
            }
        }
        return Arrays.copyOfRange(arr, 0, arr.length - cnt);
    }
}
