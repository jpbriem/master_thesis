    public int[] fill(int[] input) {

        int start = -1, end = -1;

                }
                input[start] = 0;

            }
        }

            while (start <= end && input[start]!= 0) {
                input[start++] = input[end];
            }
            input[start] = 0;
        }
        // copy the last element to all empty slots at the beginning of array

        for (int i = input.length-1; i >= 0; --i) {

        // move non-zero elements forward by 'count' positions
