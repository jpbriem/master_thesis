public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length < 2) return input;

        // find the first non-zero element from left and right side of array
        int l = -1, r = -1;
        for (int i = 0; i < input.length && l == -1; ++i) {
            if (input[i]!= 0) l = i;
        }
        for (int j = input.length - 1; j >= 0 && r == -1; --j) {
            if (input[j]!= 0) r = j;
        }

        // no zero elements in this array
        if (l == -1 && r == -1) return input;

        // set all elements between l and r to be the value at index l
        for (int k = l + 1; k <= r; ++k) {
            input[k] = input[l];
        }

        // reset values on both sides of the array
        for (int m = 0; m < l; ++m) {
            input[m] = 0;
        }
        for (int n = input.length - 1; n > r; --n) {
            input[n] = 0;
        }

        return input;
    }
}
