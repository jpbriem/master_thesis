def fill(arr):
    for i in range(len(arr)):
        n = 0
        while arr[i - n] == 0:
            n += 1
        if n > 0 and i + n < len(arr) and arr[i + n]:
            arr[i : i + n] = [arr[i + n]] * n
    return arr
