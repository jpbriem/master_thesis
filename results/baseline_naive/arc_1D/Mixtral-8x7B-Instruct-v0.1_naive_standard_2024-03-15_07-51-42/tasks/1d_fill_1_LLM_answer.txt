public class Solution {
    public int[] duplicateZeros(int[] arr) {
        if (arr == null || arr.length < 2) return arr;

        // keep track of the number of zeros to be duplicated
        int count = 0;

        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i] == 0) {
                count++;
            }
        }

        // move all elements from right to left by 'count' positions
        for (int i = arr.length - 1; i >= count; i--) {
            if (i + count < arr.length && arr[i] == 0) {
                arr[i + count] = arr[i];
                count--;
            }
        }

        // fill in the remaining zeroes at the beginning of array
        for (int j = 0; j < count; j++) {
            arr[j] = 0;
        }

        return arr;
    }
}
