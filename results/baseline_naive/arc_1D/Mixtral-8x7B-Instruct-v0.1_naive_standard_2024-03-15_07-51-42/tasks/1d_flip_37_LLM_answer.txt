    private static void moveMaxNumbersToRight(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i] == 0 && arr[i + 1]!= 0) {
                swap(arr, i, findIndexOfFirstNonZeroElement(arr, i + 1));
            } else if (arr[i] > arr[i + 1]) {
                swap(arr, i, i + 1);
            }
        }
    }

    private static int findIndexOfFirstNonZeroElement(int[] arr, int startIndex) {
        while (startIndex < arr.length && arr[startIndex] == 0) {
            startIndex++;
        }
        return startIndex;
    }

    private static void swap(int[] arr, int indexOne, int indexTwo) {
        int temp = arr[indexOne];
        arr[indexOne] = arr[indexTwo];
        arr[indexTwo] = temp;
    }
