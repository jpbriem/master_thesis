public class Solution {
    public int[] fill(int[] A) {
        if (A == null || A.length < 2) return A;
        for (int i = 0; i < A.length - 1; i++) {

                int j = i + 1;
                while (j < A.length && A[j] == 0) j++;
                if (j >= A.length) break;
                if (A[i] == A[j]) continue;
                else swap(A, i, j);
            }
        }
        return A;
    }

    private void swap(int[] A, int i, int j) {
        int temp = A[i];
        A[i] = A[j];
        A[j] = temp;
    }
