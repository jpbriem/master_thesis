public class Solution {
    public int[] fill(int[] A) {
        if (A == null || A.length < 2) return A;
        int start = -1, end = -1;
        for (int i = 0; i < A.length; ++i) {
            if (start == -1 && A[i]!= 0) {
                start = i;
            } else if (start!= -1 && A[i] == 0) {
                end = i - 1;
                break;
            }
        }
        if (end == -1) {
            end = A.length - 1;
        }
        for (int i = start; i <= end; ++i) {
            A[i] = 0;
        }
        for (int i = end + 1; i < A.length; ++i) {
            if (A[i]!= 0) continue;
            int j = i;
            while (j > start && A[j-1] == 0) --j;
            if (j == start) continue;
            int k = j;
            while (k <= i) {
                A[k++] = A[j];
            }
        }
        return A;
    }
}
