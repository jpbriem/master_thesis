def fillZeros(arr):
    i = 0
    while arr[i] == 0:
        i += 1
    j = i
    while j < len(arr):
        if arr[j]!= 0:
            k = j + 1
            while k < len(arr) and arr[k] == 0:
                arr[k] = arr[j]
                k += 1
        j += 1
    return arr
