You can use `numpy` to do this in a vectorized way:

```
import numpy as np

def fill(arr):
    arr = np.array(arr)
    idx = np.where(np.diff(arr))[0]+1
    for i in range(len(idx)):
        if arr[idx[i]] == 0:
            arr[idx[i]:idx[i]+2] = arr[idx[i]-1]
    return list(arr)

If you want to avoid using external libraries, here is an equivalent solution without them:

```
def fill(arr):
    idx = []
    for i in range(1, len(arr)):
        if arr[i]!= arr[i-1]:
            idx += [i-1]
    for i in idx:
        if arr[i] == 0:
            arr[i], arr[i+1] = arr[i-1], arr[i-1]
    return arr

print(fill([0, 6, 0, 0, 0, 0, 0, 0, 0, 0, 0, 6, 0, 0, 0, 0, 0, 0, 0, 0, 0, 6, 0, 0, 0, 0, 0, 0, 0, 0, 0, 6, 0, 0, 0, 0, 0, 0, 0, 0, 0, 6, 0, 0, 0, 0, 0, 0, 0, 0, 0, 6, 0, 0, 0, 0, 0, 0, 0, 0]))
