public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;
        int start = -1, end = -1;
        for (int i = 0; i < input.length; ++i) {
            if (start == -1 && input[i]!= 0) {
                start = i;
                end = i;
            } else if (input[i]!= 0) {
                end = i;
            } else if (start!= -1) {
                for (int j = start; j <= end; ++j) {
                    input[j] = input[i];
                }
                start = -1;
                end = -1;
            }
        }
        if (start!= -1) {
            for (int j = start; j <= end; ++j) {
                input[j] = input[input.length-1];
            }
        }
        return input;
    }
}
