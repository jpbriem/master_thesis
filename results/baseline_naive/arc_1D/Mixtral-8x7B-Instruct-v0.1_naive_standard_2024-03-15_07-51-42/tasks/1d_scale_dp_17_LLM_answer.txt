public class Solution {
    public int[] moveZeroes(int[] nums) {
        if (nums == null || nums.length <= 1) return nums;
        int index = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i]!= 0) {
                swap(nums, index++, i);
            }
        }
        return nums;
    }

    private void swap(int[] nums, int a, int b) {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}
