public class Solution {
    public int[] zeroFilledSubarray(int[] nums) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < nums.length; ) {
            if (nums[i] == 0) {
                int count = 0;
                while (i < nums.length && nums[i++] == 0) {
                    count++;
                }
                list.add(count);
            } else {
                i++;
            }
        }
        int[] res = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            res[i] = list.get(i);
        }
        return res;
    }
}
