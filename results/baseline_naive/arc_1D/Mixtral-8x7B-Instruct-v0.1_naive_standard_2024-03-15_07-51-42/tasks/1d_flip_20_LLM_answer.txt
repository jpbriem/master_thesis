    private static void moveZeroesToEnd(int[] arr) {
        for (int i = 0; i < arr.length - 1; ) {
            if (arr[i] == 0 && arr[i + 1]!= 0) {

    private static void swap(int[] arr, int i, int j) {

        arr[i] = arr[j];
        arr[j] = temp;
    }
