def fillZeros(arr):
    i = 0
    while i < len(arr):
        if arr[i] == 0:
            j = i + 1
            while j < len(arr) and arr[j] == 0:
                j += 1
            if j < len(arr):
                arr[i] = arr[j]
        i += 1
    return arr
