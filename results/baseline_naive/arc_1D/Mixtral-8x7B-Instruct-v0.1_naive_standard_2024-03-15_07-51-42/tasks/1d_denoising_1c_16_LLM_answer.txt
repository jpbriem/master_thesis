public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element from left and right side of array
        int l = -1, r = -1;
        for (int i = 0; i < input.length && l == -1; ++i) {
            if (input[i]!= 0) {
                l = i;
                break;
            }
        }
        for (int j = input.length - 1; j >= 0 && r == -1; --j) {
            if (input[j]!= 0) {
                r = j;
                break;
            }
        }

        // no zero elements in this array
        if (l == -1 && r == -1) return input;

        // set all zeros to be max value between two ends
        int val = Math.max(input[l], input[r]);
        for (int k = 0; k < input.length; ++k) {
            if (input[k] == 0) input[k] = val;
        }

        // reset the boundary values
        if (l > 0) {
            for (int m = 0; m < l; ++m) {
                if (input[m] == val) input[m] = 0;
            }
        }
        if (r + 1 < input.length) {
            for (int n = r + 1; n < input.length; ++n) {
                if (input[n] == val) input[n] = 0;
            }
        }

        return input;
    }
}
