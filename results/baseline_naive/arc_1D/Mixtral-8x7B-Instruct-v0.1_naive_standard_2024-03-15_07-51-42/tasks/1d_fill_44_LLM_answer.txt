public class Solution {
    public int[] duplicateZeros(int[] arr) {
        // edge case
        if (arr == null || arr.length <= 1) return arr;

        // copy the original array to a new one
        int[] result = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            result[i] = arr[i];
        }

        // find the first zero in the original array and start duplicating zeros from there
        int index = -1;
        for (int i = 0; i < arr.length; i++) {
            if (index >= 0 && arr[i]!= 0) {
                index++;
                continue;
            }
            if (arr[i] == 0) {
                index = i;
            }
            if (index > -1 && index + 1 < arr.length) {
                System.arraycopy(result, index + 1, result, index + 2, arr.length - index - 1);
                result[index + 1] = 0;
            }
        }

        return result;
    }
}
