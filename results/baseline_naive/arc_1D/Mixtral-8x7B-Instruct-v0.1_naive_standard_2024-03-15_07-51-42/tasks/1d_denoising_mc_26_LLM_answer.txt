public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element from left and right side of array
        int l = -1, r = -1;
        for (int i = 0; i < input.length && l == -1; ++i) {
            if (input[i]!= 0) l = i;
        }
        for (int j = input.length - 1; j >= 0 && r == -1; --j) {
            if (input[j]!= 0) r = j;
        }

        // no zero elements in this array
        if (l == -1 && r == -1) return input;

        // copy all elements between l and r to a new array
        int len = r - l + 1;
        int[] output = new int[len];
        System.arraycopy(input, l, output, 0, len);

        // replace all elements before l with max value of output
        for (int i = 0; i < l; ++i) {
            input[i] = output[len - 1];
        }

        // replace all elements after r with min value of output
        for (int i = r + 1; i < input.length; ++i) {
            input[i] = output[0];
        }

        return input;
    }
}
