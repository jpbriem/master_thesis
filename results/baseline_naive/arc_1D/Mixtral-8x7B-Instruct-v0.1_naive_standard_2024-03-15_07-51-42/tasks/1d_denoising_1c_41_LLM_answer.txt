public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element and its value
        int start = -1;
        for (int i = 0; i < input.length && start == -1; ++i) {
            if (input[i]!= 0) {
                start = i;
                break;
            }
        }

        if (start == -1) return input;

        int val = input[start];
        boolean isZero = false;

        // iterate through all elements in array
        for (int i = start; i < input.length; ++i) {
            if (input[i] == 0) {
                if (!isZero) {
                    isZero = true;
                    continue;
                } else {
                    input[i] = val;
                }
            } else {
                if (isZero) {
                    isZero = false;
                    val = input[i];
                }
            }
            input[i] = val;
        }

        // set all zeroes before first non-zero to zero
        if (start > 0) {
            for (int i = 0; i < start; ++i) {
                input[i] = 0;
            }
        }

        return input;
    }
}
