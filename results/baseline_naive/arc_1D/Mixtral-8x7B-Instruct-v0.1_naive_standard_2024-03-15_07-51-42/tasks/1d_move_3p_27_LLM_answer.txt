public class Solution {
    public int[] MoveZeroes(int[] nums) {
        if (nums == null || nums.Length <= 1) return nums;

        var left = 0;
        for (var right = 0; right < nums.Length; right++) {
            if (nums[right]!= 0) {
                Swap(ref nums[left], ref nums[right]);
                left++;
            }
        }

        return nums;
    }

    private void Swap<T>(ref T a, ref T b) {
        var temp = a;
        a = b;
        b = temp;
    }
}
