public class Solution {
    public int[] Fill(int[] a) {
        if (a == null || a.Length < 2) return a;
        var start = -1;
        for (var i = 0; i < a.Length; i++) {
            if (start == -1 && a[i]!= 0) {
                start = i;
            } else if (start > -1 && a[i] == 0) {
                FillArray(a, start, i);
                start = -1;
            }
        }
        if (start > -1) FillArray(a, start, a.Length);
        return a;
    }

    private void FillArray(int[] arr, int indexStart, int indexEnd) {
        var value = arr[indexStart];
        while (indexStart < indexEnd) {
            arr[indexStart++] = value;
        }
    }
