    let start = -1;
    for (let i = 0; i < arr.length; i++) {
        if (arr[i]!== 0 && start === -1) {
            start = i;
        } else if (arr[i] === 0 && start > -1) {
            let end = i;
            while (start <= end) {
                arr[start++] = arr[i];
            }
            start = -1;
        }
    }
    return arr;
}
