public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element from left and right side of array
        int l = -1, r = -1;
        for (int i = 0; i < input.length && l == -1; ++i) {
            if (input[i]!= 0) l = i;
        }
        for (int j = input.length - 1; j >= 0 && r == -1; --j) {
            if (input[j]!= 0) r = j;
        }

        // no zero elements in this array
        if (l == -1 && r == -1) return input;

        // fill all zeros with min value between two non-zero elements
        int min = Integer.MAX_VALUE;
        if (l!= -1 && r!= -1) {
            min = Math.min(input[l], input[r]);
        } else if (l!= -1) {
            min = input[l];
        } else {
            min = input[r];
        }

        for (int k = 0; k < input.length; ++k) {
            if (input[k] == 0) input[k] = min;
        }

        return input;
    }
}
