        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i + 1] > arr[i]) continue;
            else {

    private static int findMaxIndex(int[] arr, int start) {

                index = i;
            }
        }
        return index;
    }

    private static void swap(int[] arr, int a, int b) {
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
