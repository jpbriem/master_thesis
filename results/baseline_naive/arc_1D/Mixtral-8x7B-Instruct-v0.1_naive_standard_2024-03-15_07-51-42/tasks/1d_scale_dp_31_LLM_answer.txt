public class Solution {
    public int[] duplicateZeros(int[] arr) {
        if (arr == null || arr.length < 2) return arr;

        for (int i = 0; i < arr.length - 1; i++) {
            // If the current element is zero and there's no space to insert a new one in front of it, move all elements after it by one position to make room for the duplicated zero
            if (arr[i] == 0 && arr[i + 1]!= 0) {
                System.arraycopy(arr, i + 1, arr, i + 2, arr.length - i - 1);
                arr[i + 1] = 0;
            }
        }

        return arr;
    }
}
