public class Solution {
    public int[] fill(int[] input) {
        if (input == null || input.length <= 1) return input;

        // find the first non-zero element and its index
        for (int i = 0; i < input.length && input[i] == 0; ++i);
        int start = i;

        // check whether all elements are zero or not
        boolean isAllZero = true;
        for (int i = start; i < input.length; ++i) {
            if (input[i]!= 0) {
                isAllZero = false;
                break;
            }
        }

        // if all elements are zero then just return it
        if (isAllZero) return new int[]{0};

        // move to the last non-zero element and its index
        for (int i = input.length - 1; i >= 0 && input[i] == 0; --i);
        int end = i + 1;

        // copy from the first non-zero element to the last one
        System.arraycopy(input, start, input, 0, end - start);

        // set remaining elements as zero
        Arrays.fill(input, end, input.length, 0);

        return input;
    }
}
