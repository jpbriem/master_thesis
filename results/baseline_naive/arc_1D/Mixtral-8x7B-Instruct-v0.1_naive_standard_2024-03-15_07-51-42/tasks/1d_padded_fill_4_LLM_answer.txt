def fill(arr):
    i = 0
    while i < len(arr):
        if arr[i]:
            j = i + 1
            while j < len(arr) and not arr[j]:
                j += 1
            for k in range(i+1, j):
                arr[k] = arr[i]
            i = j
        else:
            i += 1
