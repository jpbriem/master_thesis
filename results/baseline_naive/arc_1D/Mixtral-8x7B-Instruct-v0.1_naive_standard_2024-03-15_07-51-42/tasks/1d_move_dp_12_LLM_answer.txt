public class Solution {
    public int[] moveZeroes(int[] nums) {
        if (nums == null || nums.length <= 1) return nums;
        int j = 0; // index of the first zero in array
        for (int i = 0; i < nums.length; i++) {
            if (nums[i]!= 0) {
                swap(nums, i, j);
                j++;
            }
        }
        return nums;
    }

    private void swap(int[] arr, int a, int b) {
        int temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
}
