def remove_vowels(input_string):
    vowels = set('aeiouAEIOU')
    result = []
    last_occurrence = {vowel: -1 for vowel in vowels}
    
    for i, char in enumerate(input_string):
        if char in vowels:
            last_occurrence[char] = max(last_occurrence[char], i)
        
    for i, char in enumerate(input_string):
        if char in vowels and i != last_occurrence[char]:
            continue
        
        result.append(char)
    
    return ''.join(result)
