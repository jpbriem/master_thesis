def convert_to_hallway(input_list):
    output = []
    prev_char = None
    
    for char in input_list:
        if char == '.':
            if prev_char != '.':
                prev_char = '.'
                output.append('.')
            else:
                continue
        elif prev_char == '.':
            num_dots = 0
            while output[-1] == '.':
                num_dots += 1
                output.pop()
            output.extend(['.' for _ in range(num_dots)])
            prev_char = char
        else:
            output.append(char)
            prev_char = char
            
    return output
