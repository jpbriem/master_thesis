```python
def expand_grid(grid):
    rows = len(grid)
    cols = len(grid[0])
    
    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == '.':
                continue
            
            count = 1
            dx, dy = [0, 1], [1, 0]
            
            while True:
                x, y = i + dx[count % 2], j + dy[count % 2]
                
                if not (0 <= x < rows and 0 <= y < cols) or grid[x][y] != '.':
                    break
                
                grid[x][y] = grid[i][j]
                count += 1
                
    return grid
```
