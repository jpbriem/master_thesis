def replace_chars(arr):

    char_counts = {}

        if c.isalpha():
            char_counts[c] = char_counts.get(c, 0) + 1

    # Use list comprehension and zip function to create a new array with replaced characters
    output = ['.'] * len(arr)
    for i, (c1, c2) in enumerate(zip(arr, arr[1:] + [''])):
        if c1.isalpha() and c2.isalpha() and char_counts[c1] > char_counts[c2]:
