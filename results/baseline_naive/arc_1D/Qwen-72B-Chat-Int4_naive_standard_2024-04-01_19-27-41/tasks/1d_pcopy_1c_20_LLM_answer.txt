    output = input_list.copy()
    n = len(output)
    for i in range(n):
        if output[i] == 'g':
            j = i + 1
            while j < n and output[j] != '.':
                if output[j] == 'g':
                    output[j-1] = 'g'
                j += 1
            j = i - 1
            while j >= 0 and output[j] != '.':
                if output[j] == 'g':
                    output[j+1] = 'g'
                j -= 1
    return output
