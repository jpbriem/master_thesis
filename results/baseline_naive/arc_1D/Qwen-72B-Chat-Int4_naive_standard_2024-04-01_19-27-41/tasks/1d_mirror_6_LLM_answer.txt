```python
def rearrange_input(input_list):
    i_index = input_list.index('i')
    b_or_c_indices = [index for index, char in enumerate(input_list) if char == 'b' or char == 'c']
    output = input_list.copy()
    
    # Move all b/c characters after i character to the end of the list
    for index in reversed(b_or_c_indices):
        if index < i_index:
            output.insert(len(output), output.pop(index))
    
    return output
```
