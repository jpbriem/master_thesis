def connect_adjacent_elements(arr):
    n = len(arr)
    output = arr.copy()
    
    # Iterate over each element in the array
    for i in range(n):
        if arr[i] == '.':
            continue
        
        # Check if there are any adjacent elements of the same value
        j = i + 1
        while j < n and arr[j] == arr[i]:
            j += 1
            
        k = i - 1
        while k >= 0 and arr[k] == arr[i]:
            k -= 1
            
        # Connect the adjacent elements by replacing them with the current element
        for l in range(max(0, k+1), min(j, n)):
            output[l] = arr[i]
            
    return output
