def replace_consecutive_duplicates(lst):
    result = []
    count = 0
    
    for i in range(len(lst)):
        if lst[i] != '.':
            count += 1
            
            if count > 4:
                result.append('d')
            else:
                result.append(lst[i])
                
        elif count > 0:
            result.extend(['d'] * (count // 5 + 1))
            count = 0
                
        result.append('.')
    
    return result[:-1]
