```python
def convert_grid(grid):
    for i, char in enumerate(grid):
        if char == 'g':
            j = i + 1
            while j < len(grid) and grid[j] != '.':
                if grid[j] == 'g':
                    grid[j] = 'b'
                elif grid[j] == 'b':
                    grid[j] = 'f'
                else:
                    grid[j] = 'c'
                j += 1

    return grid
```
