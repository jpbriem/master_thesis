def connect_balloons(arr):
    n = len(arr)
    for i in range(n):
        if arr[i] == '.':
            continue
        j = i + 1
        while j < n and arr[j] != arr[i]:
            j += 1
        k = max(i - 1, j)
        while k >= 0 and arr[k] == arr[i]:
            arr[k], arr[k+1] = arr[k+1], arr[k]
            k -= 1
    return arr
