def clean_string(input_list):
    output = []
    
    for i in range(len(input_list)):
        char = input_list[i]
        
        # Check left and right neighbors
        left_neighbors = [input_list[j] for j in range(max(0, i-2), min(i+1, len(input_list))) if j != i]
        right_neighbors = [input_list[j] for j in range(min(len(input_list)-1, i+2), max(0, i-1), -1) if j != i]

        # Count same-type neighbors
        same_type_left = left_neighbors.count(char)
        same_type_right = right_neighbors.count(char)

        # Replace character with dot if not surrounded by enough same-type neighbors
        if (char.isalpha() and (same_type_left < 2 or same_type_right < 2)) or char == '.':
            output.append('.')
        else:
            output.append(char)
    
    return output
