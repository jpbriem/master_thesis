def convert(input_list):
    vowels = {'a': 0, 'e': 1, 'i': 2, 'o': 3, 'u': 4}
    output = []
    
    counts = {}
    prev_vowel_index = -1
    
    for i, char in enumerate(input_list):
        if char.lower() in vowels:
            # Update the count for the previous vowel (if any)
            if prev_vowel_index != -1 and prev_vowel_index < i - 5:
                del counts[input_list[prev_vowel_index]]
            
            # Increment the count for the current vowel
            if char in counts:
                counts[char] += 1
            else:
                counts[char] = 1
            
            # Find the next available vowel with the minimum count
            min_count_vowel = None
            min_count = float('inf')
            for key, value in counts.items():
                if value <= min_count:
                    min_count_vowel = key
                    min_count = value
                    
            # Replace the current vowel with the one found
            output.append(min_count_vowel)
            prev_vowel_index = i
        else:
            output.append(char)
    
    return output
