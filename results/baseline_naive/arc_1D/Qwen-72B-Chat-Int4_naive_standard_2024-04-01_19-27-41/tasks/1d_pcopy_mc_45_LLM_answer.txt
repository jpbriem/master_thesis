```python
def convert_grid(grid):
    # Create an empty output grid with the same dimensions as the input grid
    output = [['.' for _ in range(len(grid[0]))] for _ in range(len(grid))]
    
    # Iterate over each character in the input grid
    for i, row in enumerate(grid):
        for j, char in enumerate(row):
            if char == '.' or char.islower():
                continue
            
            # Find all adjacent characters of the same type (upper- or lowercase)
            adj_chars = []
            for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                ni, nj = i + di, j + dj
                if 0 <= ni < len(grid) and 0 <= nj < len(grid[j]) and grid[ni][nj].isalpha() and grid[ni][nj].istitle() == char.istitle():
                    adj_chars.append((ni, nj))
            
            # Replace the current character with the corresponding lowercase letter,
            # and replace all its adjacent characters with the corresponding uppercase letter
            output[i][j] = char.lower()
            for ni, nj in adj_chars:
                output[ni][nj] = char.upper()
                
    return output
```
