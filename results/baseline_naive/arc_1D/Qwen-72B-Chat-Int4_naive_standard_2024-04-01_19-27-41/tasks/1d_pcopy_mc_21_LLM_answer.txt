def replace_chars(s):
    # Create a dictionary mapping each character to its count in the string
    char_count = {}
    for c in s:
        if c.isalpha():
            char_count[c] = char_count.get(c, 0) + 1
    
    # Replace characters with their counts until all have been replaced
    i = 0
    while i < len(s):
        if s[i].isalpha() and char_count[s[i]] > 1:
            j = i + 1
            while j < len(s) and s[j] == s[i]:
                j += 1
            s = s[:i] + str(j - i) * (j - i) + s[j:]
            i = i + len(str(j - i)) * (j - i)
        else:
            i += 1
    
    return s
