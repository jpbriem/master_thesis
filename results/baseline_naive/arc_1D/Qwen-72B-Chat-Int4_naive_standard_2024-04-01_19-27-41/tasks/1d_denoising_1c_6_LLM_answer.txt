def compress_cave(cave):
    compressed = []
    prev_char = None
    count = 0

    for char in cave:
        if char == prev_char:
            count += 1
        else:
            if prev_char is not None:
                compressed.append(prev_char)
                if count > 1:
                    compressed.append(str(count))
            prev_char = char
            count = 1
    
    compressed.append(prev_char)
    if count > 1:
        compressed.append(str(count))

    return compressed
