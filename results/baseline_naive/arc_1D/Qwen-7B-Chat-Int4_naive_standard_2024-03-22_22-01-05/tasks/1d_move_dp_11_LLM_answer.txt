class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        word_dict = {}
        
        for word in words:
            key = ''.join(sorted(word))
            
            if key not in word_dict:
                word_dict[key] = []
                
            word_dict[key].append(word)
            
        result = []
        
        for key, value in word_dict.items():
            result.append(value)
            
        return result
