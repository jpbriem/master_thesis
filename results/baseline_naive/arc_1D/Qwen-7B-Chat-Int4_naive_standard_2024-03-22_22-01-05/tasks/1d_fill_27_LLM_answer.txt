def longest_substring(s):
    # Initialize variables
    start = 0
    max_len = 0

    for i in range(len(s)):
        if s[i] != s[start]:
            start += 1
            max_len = max(max_len, i - start + 1)

    return s[:start] + s[start+max_len:]
