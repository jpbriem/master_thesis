def longest_substring(input_string):
    seen = set()
    start = 0
    max_length = 0

    for i in range(len(input_string)):
        char = input_string[i]
        if char not in seen:
            seen.add(char)
            current_length = i - start + 1
            if current_length > max_length:
                max_length = current_length
        else:
            start = i + 1

    return input_string[start:start+max_length]
