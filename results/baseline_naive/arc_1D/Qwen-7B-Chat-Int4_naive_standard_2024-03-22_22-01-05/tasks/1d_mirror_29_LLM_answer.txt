```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        d = {}
        for s in strs:
            if s in d:
                d[s].append(s)
            else:
                d[s] = [s]
                
        res = [[] for _ in range(len(d))]
        for k, v in d.items():
            res[v[0]] += v[1:]
            
        return res
```
