    n = len(input)
    dp = [[0] * (n + 1) for _ in range(n + 1)]
    
    # Fill the first row and column with all ones
    for i in range(n + 1):
        dp[i][i] = 1
    
    # Iterate over each pair of indices (i, j), where i < j
    for i in range(1, n + 1):
        for j in range(i, n + 1):

                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
