def group_chars(s):
    # Create an empty dictionary to store the character counts
    char_counts = {}

    # Iterate through the string and update the counts in the dictionary
    for char in s:
        if char in char_counts:
            char_counts[char] += 1
        else:
            char_counts[char] = 1

    # Return the characters grouped by their frequencies
    return [char for char, count in sorted(char_counts.items(), key=lambda item: item[1], reverse=True)]
