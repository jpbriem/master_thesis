def group_same_chars(input):
    # Create an empty dictionary to store the characters and their counts
    char_counts = {}
    
    # Iterate over each character in the input string
    for char in input:
        # If the character is already in the dictionary, increment its count
        if char in char_counts:
            char_counts[char] += 1
        else:
            # Otherwise, add the character to the dictionary with a count of 1
            char_counts[char] = 1
    
    # Initialize an empty list to store the groups of characters
    groups = []
    
    # Iterate over each character in the dictionary
    for char, count in char_counts.items():
        # If there are other characters that have the same count as this character,
        # append them to the current group
        if count > 0:
            groups.append([char])
        
        # Otherwise, create a new group containing just this character
        else:
            groups.append([char])
    
    return groups
