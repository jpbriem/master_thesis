class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagrams = {}
        for s in strs:
            if s not in anagrams:
                anagrams[s] = []
            else:
                anagrams[s].append(s)
        return list(anagrams.values())
