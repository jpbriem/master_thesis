def find_unique_chars(input_list):
    # Create an empty set to store unique characters
    unique_chars = set()

    # Iterate through each character in the input list
    for char in input_list:
        # If the character is already in the set, it means it's a duplicate and we can skip it
        if char in unique_chars:
            continue

        # Otherwise, add the character to the set
        unique_chars.add(char)

    # Return the list of unique characters
    return list(unique_chars)
