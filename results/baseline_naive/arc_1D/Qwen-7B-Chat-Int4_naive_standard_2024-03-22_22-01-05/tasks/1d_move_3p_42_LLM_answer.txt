def find_unique_chars(input_str):
    # Step 1: Count the frequency of each character
    char_counts = {}
    for char in input_str:
        if char in char_counts:
            char_counts[char] += 1
        else:
            char_counts[char] = 1
    
    # Step 2: Sort the characters by their frequency in descending order
    sorted_char_counts = sorted(char_counts.items(), key=lambda x: x[1], reverse=True)
    
    # Step 3: Take the first character from each group of consecutive identical characters
    output_str = ''
    current_group = []
    for item in sorted_char_counts:
        if len(current_group) == 0 or item[0] != current_group[-1]:
            current_group.append(item[0])
        else:
            output_str += current_group.pop()
    
    return output_str
