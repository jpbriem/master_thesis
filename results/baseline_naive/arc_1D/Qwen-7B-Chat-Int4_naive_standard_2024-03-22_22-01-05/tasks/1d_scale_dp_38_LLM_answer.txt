class Solution:
    def longestCommonPrefix(self, strs):
        if not strs or len(strs) == 0:
            return ""
        
        prefix = strs[0]
        for s in strs[1:]:
            while s.startswith(prefix):
                prefix = prefix[:-1]
            if prefix == "":
                break
        return prefix
