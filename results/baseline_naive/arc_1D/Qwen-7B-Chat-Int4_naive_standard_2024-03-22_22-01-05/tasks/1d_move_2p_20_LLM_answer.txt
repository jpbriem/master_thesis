```python
class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        d = {}
        for i in s:
            if len(i) == 0:
                continue
            key = ''.join(sorted(i))
            if key not in d:
                d[key] = []
            d[key].append(i)
        return [d[i] for i in sorted(d)]
```
