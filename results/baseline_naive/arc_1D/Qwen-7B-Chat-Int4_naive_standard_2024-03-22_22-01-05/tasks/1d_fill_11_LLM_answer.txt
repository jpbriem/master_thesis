```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        hash_map = {}
        res = []
        
        for s in strs:
            key = ''.join(sorted(s))
            
            if key not in hash_map:
                hash_map[key] = [s]
            else:
                hash_map[key].append(s)
                
        for key in hash_map:
            res.append(hash_map[key])
            
        return res
```
