class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            chars = list(s)
            sorted_chars = ''.join(sorted(chars))
            if sorted_chars not in d:
                d[sorted_chars] = [s]
            else:
                d[sorted_chars].append(s)
        
        groups = []
        for k, v in d.items():
            groups.append(v)
            
        return groups
