class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        if len(s) == 0:
            return []
        
        ans = [[] for _ in range(len(set(s)))]
        
        for i in range(len(s)):
            ans[set(s[i])].append(s[i])
            
        return ans
