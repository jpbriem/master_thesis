class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        d = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in d:
                d[key] = []
            d[key].append(word)
        
        res = [d[key] for key in d]
        return res
