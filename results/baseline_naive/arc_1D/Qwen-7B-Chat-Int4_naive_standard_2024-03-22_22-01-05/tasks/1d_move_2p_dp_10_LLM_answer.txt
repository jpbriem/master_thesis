class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram_dict = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in anagram_dict:
                anagram_dict[key] = [s]
            else:
                anagram_dict[key].append(s)
        
        return list(anagram_dict.values())
