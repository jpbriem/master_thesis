class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        dic = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in dic:
                dic[key] = []
            dic[key].append(word)
        
        result = [[] for _ in range(len(dic))]
        for key, value in dic.items():
            result[dic[key][0]] += value[1:]
        
        return result
