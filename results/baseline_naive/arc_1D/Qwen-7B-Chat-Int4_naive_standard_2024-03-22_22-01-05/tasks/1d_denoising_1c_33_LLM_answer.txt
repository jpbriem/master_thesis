```
def unique_chars(input_string):
    # Initialize an empty set to store used characters
    used_chars = set()

    # Iterate through the input string
    for char in input_string:
        # If the character has not been used before, add it to the output string and update the set of used characters
        if char not in used_chars:
            used_chars.add(char)
            output_string.append(char)

    return output_string
```
