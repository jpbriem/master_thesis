```python
def group_chars(s):
    char_freq = {}
    output = []
    for char in s:
        if char not in char_freq:
            char_freq[char] = 0
        char_freq[char] += 1

    for char, freq in sorted(char_freq.items()):
        while freq > 0:
            output.append(char)
            freq -= 1

    return output
```
