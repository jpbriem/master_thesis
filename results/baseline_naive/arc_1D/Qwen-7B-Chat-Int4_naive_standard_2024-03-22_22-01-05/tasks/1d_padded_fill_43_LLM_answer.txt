```python
def remove_duplicates(input_string):
    # Create an empty dictionary to store counts
    char_counts = {}

    # Iterate over the input string
    for char in input_string:
        # If the character is already in the dictionary, increment its count
        if char in char_counts:
            char_counts[char] += 1
        else:
            # Otherwise, add the character to the dictionary with a count of 1
            char_counts[char] = 1

    # Initialize an empty list to store the unique characters
    unique_chars = []

    # Iterate over the dictionary
    for char, count in char_counts.items():
        # Add the character to the list if it has a count greater than or equal to 2
        if count >= 2:
            unique_chars.append(char)

    # Join the unique characters into a new string and return it
    output_string = ''.join(unique_chars)
    return output_string
```
