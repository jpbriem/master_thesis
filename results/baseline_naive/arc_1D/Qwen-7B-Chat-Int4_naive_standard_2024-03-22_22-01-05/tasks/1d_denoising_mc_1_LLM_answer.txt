class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        d = {}
        for s in strs:
            if s in d:
                d[s].append(s)
            else:
                d[s] = [s]
                
        res = []
        for k in d.keys():
            v = d[k]
            res.append(v)
            
        return res
