```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        
        for s in strs:
            sorted_s = ''.join(sorted(s))
            if sorted_s not in d:
                d[sorted_s] = []
            d[sorted_s].append(s)
            
        res = [ [] for _ in range(len(strs)) ]
        
        for k, v in d.items():
            res[v[0]] += [v[i] for i in range(1, len(v))]
        
        return res
```
