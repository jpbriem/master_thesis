class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        for s in strs:
            if len(s) == 0 or s not in d:
                d[s] = [s]
            else:
                d[s].append(s)
        return list(d.values())
