```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if len(words) == 0:
            return []
        
        result = [[words[0]]]
        for word in words[1:]:
            is_in_group = False
            for group in result:
                if word == group[-1]:
                    group.append(word)
                    is_in_group = True
                    break

            if not is_in_group:
                result.append([word])

        return result
```
