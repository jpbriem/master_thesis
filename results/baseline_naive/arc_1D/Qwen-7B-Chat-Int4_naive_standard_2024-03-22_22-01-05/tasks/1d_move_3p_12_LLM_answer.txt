```python
def group_chars(s):
    freq = {}
    for char in s:
        if char in freq:
            freq[char] += 1
        else:
            freq[char] = 1

    output = []
    for char, count in sorted(freq.items()):
        while count > 0:
            output.append(char)
            count -= 1

    return output
```
