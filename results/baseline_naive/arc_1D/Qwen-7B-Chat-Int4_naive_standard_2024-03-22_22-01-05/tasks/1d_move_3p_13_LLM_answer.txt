```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        res = []
        
        for s in strs:
            if len(s) == 0:
                continue
            
            key = {}
            
            for i in range(len(s)):
                ch = s[i]
                
                if ch in key:
                    key[ch] += 1
                else:
                    key[ch] = 1
                
            if key not in res:
                res.append([s])
            else:
                index = res.index(key)
                res[index].append(s)
                
        return res
```
