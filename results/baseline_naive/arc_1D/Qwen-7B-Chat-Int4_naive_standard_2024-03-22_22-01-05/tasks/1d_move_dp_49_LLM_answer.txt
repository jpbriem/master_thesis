class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        res = []
        
        anagram_dict = {}
        for i in range(len(s)):
            key = ''.join(sorted(s[i]))
            if key not in anagram_dict:
                anagram_dict[key] = [s[i]]
            else:
                anagram_dict[key].append(s[i])
                
        # Grouping the anagrams into list
        for k in anagram_dict.keys():
            res.append(anagram_dict[k])
            
        return res
