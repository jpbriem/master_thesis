```python
class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        if not s or len(s) < 2:
            return []
        
        d = {}
        result = []
        for i in range(len(s)):
            c = s[i]
            if c in d:
                d[c].append(i)
            else:
                d[c] = [i]
                
        for key, value in d.items():
            result.append([s[i] for i in sorted(value)])
            
        return result
```
