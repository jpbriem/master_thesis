class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if len(words) == 0:
            return []
        
        ans = [[words[0]]]
        word_set = set([words[0]])
        for i in range(1, len(words)):
            if words[i] not in word_set:
                word_set.add(words[i])
                ans.append([words[i]])
            else:
                ans[-1].append(words[i])

        return ans
