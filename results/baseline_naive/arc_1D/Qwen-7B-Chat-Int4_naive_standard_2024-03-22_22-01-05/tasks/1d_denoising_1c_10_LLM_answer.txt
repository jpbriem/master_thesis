    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        d = {}
        for i in range(len(s)):
            key = ''.join(sorted(s[i]))
            if key not in d:
                d[key] = [s[i]]
            else:
                d[key].append(s[i])
        
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
