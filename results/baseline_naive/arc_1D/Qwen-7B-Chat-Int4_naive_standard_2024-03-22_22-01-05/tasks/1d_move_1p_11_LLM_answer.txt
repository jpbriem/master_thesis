class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        d = {}
        for i in s:
            if i not in d:
                d[i] = []
            d[i].append(i)
        
        result = []
        for key, value in d.items():
            if len(value) == 0:
                continue
            result.append(value[0])
            for j in range(1, len(value)):
                result[-1] += value[j]
        
        return result
