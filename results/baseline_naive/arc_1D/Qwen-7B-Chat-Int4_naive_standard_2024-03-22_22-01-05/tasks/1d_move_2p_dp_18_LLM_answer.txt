```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        for s in strs:
            if len(set(s)) == 1:
                ans.append([s])
            else:
                tmp = set()
                for i in range(len(strs)):
                    if s[i] not in tmp and s[i] != strs[i]:
                        tmp.add(s[i])
                if len(tmp) > 0:
                    ans.append(list(tmp))
        
        return ans
```
