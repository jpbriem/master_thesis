class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        dic = {}
        
        for i in s:
            if i not in dic:
                dic[i] = []
            dic[i].append(i)
            
        res = []
        
        for key, value in dic.items():
            if len(value) == 1:
                res.append([value[0]])
            else:
                res.append(sorted(value))
                
        return res
