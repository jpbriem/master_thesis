def find_duplicates(input):
    # Create an empty dictionary to store counts of each character
    count_dict = {}
    
    # Iterate through each character in the input string
    for char in input:
        # If the character is already in the dictionary, increment its count
        if char in count_dict:
            count_dict[char] += 1
        else:
            # Otherwise, add the character to the dictionary with a count of 1
            count_dict[char] = 1
    
    # Initialize two lists to store characters and their counts
    chars_list = []
    counts_list = []
    
    # Iterate through the dictionary
    for key, value in count_dict.items():
        # Append the character to the list if it appears more than once
        if value > 1:
            chars_list.append(key)
            counts_list.append(value)
    
    # Return the list of duplicate characters
    return chars_list
