def longest_common_subsequence(input):
    # Initialize a hash set to store the characters that have been seen
    seen = set()

    # Initialize two pointers for the input string
    i = 0
    j = 0

    # Initialize an empty string to store the longest common subsequence
    result = ""

    while i < len(input) and j < len(input[0]):
        # Add the current character to the hash set if it hasn't been seen
        if input[i] not in seen:
            seen.add(input[i])
            result += input[i]
            i += 1
        else:
            # Remove the current character from the hash set
            seen.remove(input[j])
            # Append the previous character to the result string
            result += input[j]
            j += 1

    # Append any remaining characters from the first string to the result string
    while i < len(input):
        result += input[i]
        i += 1

    return result
