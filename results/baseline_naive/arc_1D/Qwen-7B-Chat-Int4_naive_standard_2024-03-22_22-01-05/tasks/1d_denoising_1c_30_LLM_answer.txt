```python
def remove_duplicates(input):
    char_counts = {}
    for char in input:
        if char not in char_counts:
            char_counts[char] = 0
        char_counts[char] += 1
    
    output = []
    for char in input:
        if char_counts[char] == 1:
            output.append(char)
    
    return output
```
