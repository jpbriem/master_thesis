```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        ans = []
        
        for word in words:
            seen = set()
            
            for char in word:
                if char not in seen:
                    seen.add(char)
                    
                    temp_list = [char]
                    while len(temp_list) != 0:
                        new_word = ''.join(temp_list)
                        temp_list.pop(0)
                        
                        if new_word not in ans and new_word == word:
                            ans.append([new_word])
                        else:
                            if new_word in ans:
                                ans[ans.index(new_word)].append(word)
                            else:
                                ans.append([new_word, word])
                
        return ans
```
