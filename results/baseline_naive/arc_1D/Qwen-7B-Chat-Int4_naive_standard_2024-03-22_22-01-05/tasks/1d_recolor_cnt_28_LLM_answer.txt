class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        groups = [[] for _ in range(len(set(strs)))]
        for s in strs:
            index = [0] * len(set(strs))
            for i in range(len(s)):
                index[set(s)[i]] += 1
            groups[index[0]].append(s)
        
        return groups
