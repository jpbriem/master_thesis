class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        if len(nums) == 0:
            return res
        
        for i in range(len(nums)):
            tmp = [nums[i]] + self.permute(nums[:i] + nums[i+1:])
            res.extend(tmp)
            
        return res
