def count_chars(input_string):
    char_counts = {}
    for char in input_string:
        if char in char_counts:
            char_counts[char] += 1
        else:
            char_counts[char] = 1

    output_string = []
    for char, count in char_counts.items():
        while count > 0:
            output_string.append(char)
            count -= 1

    return output_string
