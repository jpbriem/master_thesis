```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        ans = [[] for _ in range(len(strs))]
        
        hash_map = {}
        
        for i, s in enumerate(strs):
            key = ''.join(sorted(s))
            
            if key in hash_map:
                ans[hash_map[key]].append(s)
                
            else:
                hash_map[key] = i
                
                ans.append([s])
                
        return ans
```
