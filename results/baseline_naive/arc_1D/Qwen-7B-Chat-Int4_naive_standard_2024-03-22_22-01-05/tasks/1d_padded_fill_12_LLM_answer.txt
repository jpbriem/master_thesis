def longest_common_substring(input):
    # Create a hash set to store characters
    chars = set()

    # Initialize variables
    max_len = 0
    start_index = 0

    for i in range(len(input)):
        # Check if the current character has already been seen
        if input[i] in chars:
            # Update the start index
            start_index = max(start_index, i - len(chars))

            # Remove the character from the hash set
            chars.remove(input[i])

        else:
            # Add the character to the hash set
            chars.add(input[i])

    return max_len
