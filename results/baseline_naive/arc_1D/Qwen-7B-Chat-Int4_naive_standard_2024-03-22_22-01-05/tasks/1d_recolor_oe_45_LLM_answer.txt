```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        
        for s in strs:
            key = ''.join(sorted(s))
            
            if key not in d:
                d[key] = [s]
            else:
                d[key].append(s)
                
        res = []
        
        for k in d:
            res.append(d[k])
            
        return res
```
