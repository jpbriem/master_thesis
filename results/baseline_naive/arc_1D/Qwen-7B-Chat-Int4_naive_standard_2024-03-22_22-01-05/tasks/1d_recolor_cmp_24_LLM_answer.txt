class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        if not s:
            return []
        
        d = {}
        for i in range(len(s)):
            if s[i] in d:
                d[s[i]].append(i)
            else:
                d[s[i]] = [i]
                
        res = []
        for k, v in d.items():
            if len(v) == 1:
                res.append([k])
            elif len(set([s[v[0]]])) == 1:
                res[-1].extend(v[1:])
            else:
                res.append(v)
                
        return res
