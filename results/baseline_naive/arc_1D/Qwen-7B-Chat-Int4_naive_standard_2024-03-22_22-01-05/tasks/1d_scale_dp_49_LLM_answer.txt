def find_duplicates(input):
    freq = {}
    for char in input:
        if char in freq:
            freq[char] += 1
        else:
            freq[char] = 1

    duplicates = []
    for char, count in freq.items():
        if count > 1:
            duplicates.append(char)

    return duplicates
