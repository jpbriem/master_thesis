def longest_common_subsequence(input):
    n = len(input)
    dp = [[0] * (n + 1) for _ in range(n + 1)]

    # Fill the first row with all zeros
    for i in range(1, n + 1):
        dp[i][0] = 0

    # Fill the first column with all zeros
    for j in range(1, n + 1):
        dp[0][j] = 0

    # Fill the remaining cells using dynamic programming
    for i in range(1, n + 1):
        for j in range(1, n + 1):
            if input[i - 1] == input[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # Find the length of the longest common subsequence by backtracking from the last cell
    result = []
    i = n
    j = n
    while i > 0 and j > 0:
        if input[i - 1] == input[j - 1]:
            result.append(input[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] >= dp[i][j - 1]:
            i -= 1
        else:
            j -= 1

    return ''.join(result[::-1])
