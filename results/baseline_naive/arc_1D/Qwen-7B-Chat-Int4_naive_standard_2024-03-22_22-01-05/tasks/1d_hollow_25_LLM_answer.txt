```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        ans = [[] for _ in range(len(strs))]
        dic = {}
        
        for i in range(len(strs)):
            key = ''.join(sorted(strs[i]))
            if key not in dic:
                dic[key] = [strs[i]]
            else:
                dic[key].append(strs[i])
                
        for k, v in dic.items():
            ans[k-1].extend(v)
            
        return ans
```
