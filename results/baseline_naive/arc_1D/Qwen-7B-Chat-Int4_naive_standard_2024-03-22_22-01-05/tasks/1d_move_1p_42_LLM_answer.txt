```python
class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        if not s or len(s) == 0:
            return []
        
        anagram_map = {}
        result = []
        
        for word in s:
            key = "".join(sorted(word))
            if key in anagram_map:
                anagram_map[key].append(word)
            else:
                anagram_map[key] = [word]
                
        for key in anagram_map:
            result.append(anagram_map[key])
            
        return result
```
