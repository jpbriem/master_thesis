class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        word_count = {}
        
        for word in words:
            if word not in word_count:
                word_count[word] = []
            word_count[word].append(word)
            
        result = [[] for _ in range(len(word_count))]
        
        for key, value in word_count.items():
            result[value[0]] += value[1:]
        
        return result
