class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram_dict = {}
        for s in strs:
            if s not in anagram_dict:
                anagram_dict[s] = [s]
            else:
                anagram_dict[s].append(s)
        
        result = []
        for k, v in anagram_dict.items():
            result.append(v)
        
        return result
