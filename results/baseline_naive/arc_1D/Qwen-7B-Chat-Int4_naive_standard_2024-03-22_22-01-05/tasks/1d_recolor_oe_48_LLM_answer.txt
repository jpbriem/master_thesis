```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        for s in strs:
            d = {}
            for c in s:
                if c not in d:
                    d[c] = [s]
                else:
                    d[c].append(s)
            ans.append(d.values())
        
        return ans
```
