class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        hash_set = set()
        
        for i in range(len(s)):
            key = "".join(sorted(s[i]))
            
            if key not in hash_set:
                hash_set.add(key)
                
                ans.append([s[i]])
            else:
                temp_list = ans[hash_set.index(key)]
                temp_list.append(s[i])
                
                ans[hash_set.index(key)] = temp_list
        
        return ans
