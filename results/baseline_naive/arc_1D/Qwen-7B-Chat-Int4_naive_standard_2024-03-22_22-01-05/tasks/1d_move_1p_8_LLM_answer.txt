class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        res = []
        
        dic = {}
        for s in strs:
            if len(s) == 0:
                continue
            
            key = "".join(sorted(s))
            
            if key not in dic:
                dic[key] = [s]
            else:
                dic[key].append(s)
                
        for k, v in dic.items():
            res.append(v)
        
        return res
