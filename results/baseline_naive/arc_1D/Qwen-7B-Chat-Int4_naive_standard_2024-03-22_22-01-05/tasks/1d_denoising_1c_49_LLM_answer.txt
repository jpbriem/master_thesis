def group_same_elements(input):
    output = []
    seen = set()
    for element in input:
        if element not in seen:
            seen.add(element)
            output.append(element)
        else:
            count = 0
            while True:
                if element == output[-count]:
                    break
                count += 1
            output.pop(count - 1)
            output.append(element)

    return output
