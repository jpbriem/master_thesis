class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        
        for s in strs:
            hash_table = {}
            for ch in s:
                if ch not in hash_table:
                    hash_table[ch] = [s]
                else:
                    hash_table[ch].append(s)
            
            ans.append(hash_table)
        
        return ans
