```python
def group_strings(input):
    # Create an empty dictionary to store the frequency of each character
    freq = {}
    
    # Iterate through each string in the input list
    for s in input:
        # Convert the string to lowercase and remove any non-alphabetic characters
        s = ''.join(c.lower() for c in s if c.isalpha())
        
        # Count the frequency of each character in the string
        for char in s:
            if char not in freq:
                freq[char] = 0
            freq[char] += 1
    
    # Group the strings based on their frequency of characters
    groups = []
    for key, value in freq.items():
        if value > 1:
            groups.append([s for s in input if s == key])
    
    return groups
```
