def group_characters(input_string):
    # Create an empty dictionary to store the frequency of characters
    char_freq = {}

    # Iterate through the input string
    for char in input_string:
        # If the character is already in the dictionary, increment its count
        if char in char_freq:
            char_freq[char] += 1
        else:
            # Otherwise, add the character to the dictionary with a count of 1
            char_freq[char] = 1

    # Return the dictionary of character frequencies
    return char_freq
```

def group_characters_to_list(input_string):
    # Group the characters by their frequency using the group_characters function
    char_freq = group_characters(input_string)

    # Create an empty list to store the grouped strings
    grouped_strings = []

    # Iterate through the character frequencies
    for char, freq in char_freq.items():
        # If the frequency is greater than or equal to 2, create a new string containing all occurrences of the character
        if freq >= 2:
            grouped_strings.append(''.join([char]*freq))
        # If the frequency is less than 2, just append the character itself
        else:
            grouped_strings.append(char)

    # Return the list of grouped strings
    return grouped_strings
```
