```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        groups = [[] for _ in range(len(words))]
        
        # Use a dictionary to store the frequency of each character
        freq_dict = {}
        for word in words:
            for char in word:
                if char not in freq_dict:
                    freq_dict[char] = 0
                freq_dict[char] += 1
        
        # For each character, add all the words that contain it to its corresponding group
        for key, value in freq_dict.items():
            for i in range(value):
                groups[i].append(key)
                
        return groups
```
