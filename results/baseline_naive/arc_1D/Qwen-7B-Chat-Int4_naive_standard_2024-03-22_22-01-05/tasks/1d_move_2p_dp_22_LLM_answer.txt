def group_letters(input):
    # Create an empty dictionary to store the groups of letters
    groups = {}

    # Iterate through each letter in the input
    for letter in input:
        # If the letter is already a key in the dictionary, add it to the corresponding value
        if letter in groups:
            groups[letter].append(letter)
        else:
            # Otherwise, create a new key-value pair with the letter as the key and an empty list as the value
            groups[letter] = [letter]

    # Convert the dictionary into a list of lists where each inner list represents a group of letters
    output = [[group] for group in groups.values()]

    return output
