class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        dic = {}
        for s in strs:
            if s not in dic:
                dic[s] = [s]
            else:
                dic[s].append(s)
                
        # print(dic)

        for key in dic.keys():
            tmp = dic[key]
            ans.append(tmp)
            
        return ans
