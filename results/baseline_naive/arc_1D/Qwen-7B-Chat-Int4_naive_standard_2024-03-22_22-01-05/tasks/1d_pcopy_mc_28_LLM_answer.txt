```python
def longest_common_subsequence(arr):
    n = len(arr)
    m = max(len(arr[i]) for i in range(n))
    dp = [[0] * (m + 1) for _ in range(n + 1)]
    
    for i in range(1, n + 1):
        for j in range(1, m + 1):
            if arr[i - 1] == arr[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
                
    res = []
    i, j = n, m
    while i > 0 or j > 0:
        if arr[i - 1] == arr[j]:
            res.append(arr[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] >= dp[i][j - 1]:
            i -= 1
        else:
            j -= 1
            
    return ''.join(res[::-1])
```
