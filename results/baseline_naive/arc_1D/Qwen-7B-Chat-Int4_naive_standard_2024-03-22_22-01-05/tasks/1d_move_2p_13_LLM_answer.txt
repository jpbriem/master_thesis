```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        dic = {}
        for s in strs:
            if len(s) == 0:
                continue

            key = ''.join(sorted(s))
            if key not in dic:
                dic[key] = [s]
            else:
                dic[key].append(s)

        result = []
        for k in dic.keys():
            result.append(dic[k])

        return result
```
