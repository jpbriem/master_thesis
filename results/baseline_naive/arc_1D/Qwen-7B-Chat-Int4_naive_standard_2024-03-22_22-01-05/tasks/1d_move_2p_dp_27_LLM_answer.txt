class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0:
            return []
        
        d = {}
        for s in strs:
            sorted_s = ''.join(sorted(s))
            if sorted_s not in d:
                d[sorted_s] = [s]
            else:
                d[sorted_s].append(s)
                
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
