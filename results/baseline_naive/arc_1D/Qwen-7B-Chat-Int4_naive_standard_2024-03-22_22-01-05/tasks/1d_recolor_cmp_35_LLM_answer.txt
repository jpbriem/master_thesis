class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        result = []
        
        dic = {}
        for i in range(len(strs)):
            key = ''.join(sorted(strs[i]))
            if key not in dic.keys():
                dic[key] = [strs[i]]
            else:
                dic[key].append(strs[i])
                
        for key in dic.keys():
            result.append(dic[key])
            
        return result
