def longest_common_subsequence(input):
    # Create an empty hash set to store unique characters
    unique_chars = set()

    # Initialize variables to store the length of the longest common subsequence
    lcs_length = 0

    # Iterate through the input string
    for char in input:
        # Check if the character is already in the hash set
        if char in unique_chars:
            # If it is, skip it
            continue

        # Otherwise, add it to the hash set and append it to the output string
        unique_chars.add(char)
        output.append(char)

        # Update the length of the longest common subsequence
        lcs_length += 1

    return output[:lcs_length]
