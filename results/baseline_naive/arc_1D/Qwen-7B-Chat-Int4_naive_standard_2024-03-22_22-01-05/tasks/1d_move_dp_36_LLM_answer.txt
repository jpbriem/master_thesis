class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagrams = []
        
        for i in range(len(strs)):
            anagram = [strs[i]]
            
            # use set to store all characters of current string
            char_set = set()
            for j in range(i+1, len(strs)):
                if strs[j] not in char_set:
                    char_set.add(strs[j])
                    anagram.append(strs[j])
                else:
                    break
            
            # add the anagram to list and continue with next string
            anagrams.append(anagram)
        
        return anagrams
