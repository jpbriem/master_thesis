```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagram = {}
        
        for s in strs:
            if len(s) == 0:
                continue
            
            sorted_s = "".join(sorted(s))
            
            if sorted_s not in anagram:
                anagram[sorted_s] = [s]
                
            else:
                anagram[sorted_s].append(s)
                
        result = []
        
        for k, v in anagram.items():
            result.append(v)
            
        return result
```
