class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagram_map = {}
        
        for i in range(len(strs)):
            key = ''.join(sorted(strs[i]))
            
            if key not in anagram_map:
                anagram_map[key] = []
                
            anagram_map[key].append(strs[i])
        
        res = []
        
        for k, v in anagram_map.items():
            res.append(v)
        
        return res
