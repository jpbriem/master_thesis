```python
class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        if not s or len(s) < 2:
            return s

        result = ''
        for char in s:
            if char.isalpha():
                while result[-1] == char and result != '':
                    result = result[:-1]
                result += char
        return result
```
