def group_same_chars(s):
    # Create a set to store the characters that have been seen so far
    seen = set()

    # Initialize an empty list to store the output
    output = []

    for char in s:
        if char in seen:
            # If we've already seen this character, append it and its count to the output
            output.append((char, len(seen[char])))
        else:
            # Otherwise, add the character to the set and append it to the output
            seen.add(char)
            output.append(char)

    return output
