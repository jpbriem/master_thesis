```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        
        for s in strs:
            # convert to lower case
            s = s.lower()
            
            # remove spaces and split into list of characters
            chars = [char for char in s.split()]
            
            # sort the list of characters
            sorted_chars = ''.join(sorted(chars))
            
            # check if key already exists in dictionary
            if sorted_chars in d:
                d[sorted_chars].append(s)
            else:
                d[sorted_chars] = [s]
                
        return list(d.values())
```
