def group_chars(input_str):
    freq_dict = {}
    for char in input_str:
        if char in freq_dict:
            freq_dict[char] += 1
        else:
            freq_dict[char] = 1
    
    result = []
    for char, count in freq_dict.items():
        if count > 1:
            result.append(char)
    
    return result
