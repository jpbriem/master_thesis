```python
def longest_common_subsequence(input):
    if not input:
        return []

    result = []
    for i in range(len(input)):
        for j in range(i+1, len(input)+1):
            if input[i] == input[j]:
                result.append(input[i])
            else:
                result[-1] += input[j]

    return ''.join(result)
```
