class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        ans = []
        
        d = {}
        for i in range(len(s)):
            if s[i] not in d:
                d[s[i]] = [i]
            else:
                d[s[i]].append(i)
            
        for k, v in d.items():
            ans.append(v)
        
        return ans
