```python
def remove_duplicates(input):
    # Convert input to a set, which automatically removes duplicates
    output = list(set(input))
    
    # If there are any remaining duplicates in the output set, append them back to the end of the list
    for item in input:
        if item not in output:
            output.append(item)
            
    return output
```
