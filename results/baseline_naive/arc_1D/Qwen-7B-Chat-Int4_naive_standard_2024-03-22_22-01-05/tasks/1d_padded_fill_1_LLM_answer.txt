class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0:
            return 0

        max_length = 1
        start_index = 0
        
        for i in range(len(s)):
            # check odd length palindromes
            left, right = self.findLongestPalindrome(s[i-1:i+2], i)
            if right - left + 1 > max_length:
                max_length = right - left + 1
                start_index = left

            # check even length palindromes
            left, right = self.findLongestPalindrome(s[i-1:i+4], i)
            if right - left + 1 > max_length:
                max_length = right - left + 1
                start_index = left

        return max_length
