class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        result = []
        
        for s in strs:
            if len(set(s)) == 1:
                result.append([s])
            else:
                key = ''.join(sorted(s))
                temp = [x for x in result if x[0] == key]
                result.extend(temp)
                
        return result
