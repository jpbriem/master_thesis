```python
class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        anagrams = {}
        
        for i in range(len(s)):
            key = ''.join(sorted(s[i]))
            
            if key not in anagrams:
                anagrams[key] = [s[i]]
            else:
                anagrams[key].append(s[i])
                
        return list(anagrams.values())
```
