class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s or len(s) <= 1:
            return 0

        start = 0
        max_len = 1
        for i in range(len(s)):
            # check odd length palindromes
            left, right = self.findPalindrome(s, i, i)
            if left >= start and right - left + 1 > max_len:
                start = left
                max_len = right - left + 1

            # check even length palindromes
            left, right = self.findPalindrome(s, i, i+1)
            if left >= start and right - left + 1 > max_len:
                start = left
                max_len = right - left + 1

        return max_len

    def findPalindrome(self, s, l, r):
        while l >= 0 and r < len(s) and s[l] == s[r]:
            l -= 1
            r += 1
        return l + 1, r - 1
