class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        dic = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key not in dic:
                dic[key] = []
            dic[key].append(s)
        
        res = [[] for _ in range(len(dic))]
        for k, v in dic.items():
            res[v[0]] += [v[i] for i in range(1, len(v))]

        return res
