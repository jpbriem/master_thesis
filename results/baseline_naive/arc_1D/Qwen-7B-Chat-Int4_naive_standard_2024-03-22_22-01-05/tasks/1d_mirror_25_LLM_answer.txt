class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        from collections import defaultdict
        
        d = defaultdict(list)
        
        for word in words:
            key = ''.join(sorted(word))
            
            if key not in d:
                d[key].append(word)
            else:
                d[key].append(word)
                
        return [list(d.values())]
