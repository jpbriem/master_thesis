class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        res = []
        seen = set()
        
        for num in nums:
            if num not in seen:
                seen.add(num)
                res.append(num)
                
        return res
