class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        anagram = {}
        for word in s:
            key = ''.join(sorted(word))
            if key not in anagram:
                anagram[key] = [word]
            else:
                anagram[key].append(word)
        
        result = []
        for key in anagram:
            result.append(anagram[key])
            
        return result
