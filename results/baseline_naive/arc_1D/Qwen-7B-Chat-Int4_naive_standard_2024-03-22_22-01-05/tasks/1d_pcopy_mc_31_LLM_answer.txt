```python
class Solution:
    def longestCommonSubsequence(self, nums):
        m = len(nums)
        n = len(nums[0])
        
        dp = [[0 for _ in range(n+1)] for _ in range(m+1)]
        
        for i in range(1,m+1):
            for j in range(1,n+1):
                if nums[i-1] == nums[j-1]:
                    dp[i][j] = dp[i-1][j-1]+1
                else:
                    dp[i][j] = max(dp[i-1][j],dp[i][j-1])

        return dp[m][n]
```
