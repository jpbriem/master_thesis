```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if len(words) == 0:
            return []
        
        # use a set to store the frequency of each character in the first word
        freq = {}
        for char in words[0]:
            freq[char] = freq.get(char, 0) + 1
        
        res = [[]]
        for word in words[1:]:
            temp_freq = {}
            for char in word:
                temp_freq[char] = temp_freq.get(char, 0) + 1
            
            match = True
            for key in freq.keys():
                if temp_freq[key] != freq[key]:
                    match = False
                    break
            
            if match:
                res[-1].append(word)
            
            else:
                res.append([word])
                
        return res
```
