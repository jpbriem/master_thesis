def unique_chars(input_string):
    # Create an empty dictionary to store character frequencies
    char_freq = {}
    
    # Iterate over the input string
    for char in input_string:
        # If the character is already present in the dictionary, increment its frequency count
        if char in char_freq:
            char_freq[char] += 1
        else:
            # Otherwise, add the character to the dictionary with a frequency count of 1
            char_freq[char] = 1
    
    # Return the updated dictionary
    return char_freq
