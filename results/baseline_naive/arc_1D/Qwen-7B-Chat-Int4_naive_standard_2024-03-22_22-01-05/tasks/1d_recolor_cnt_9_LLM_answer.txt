class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        
        for s in strs:
            if s not in d:
                d[s] = []
            
            d[s].append(s)
        
        res = [[] for _ in range(len(d))]
        
        for key, value in d.items():
            res[value[0]].extend(value[1:])
        
        return res
