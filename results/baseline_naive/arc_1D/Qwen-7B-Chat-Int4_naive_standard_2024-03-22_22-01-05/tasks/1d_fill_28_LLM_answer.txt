class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram = {}
        for s in strs:
            k = ''.join(sorted(s))
            if k not in anagram:
                anagram[k] = [s]
            else:
                anagram[k].append(s)
        
        return list(anagram.values())
