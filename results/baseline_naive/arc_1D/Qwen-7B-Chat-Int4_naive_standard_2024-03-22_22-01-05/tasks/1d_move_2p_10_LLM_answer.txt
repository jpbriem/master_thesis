```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if len(words) == 0:
            return []
        
        anagrams = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in anagrams:
                anagrams[key] = [word]
            else:
                anagrams[key].append(word)
        
        result = []
        for key, value in sorted(anagrams.items()):
            result.append(value)
        
        return result
```
