```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        for i in range(len(strs)):
            s = "".join(sorted(strs[i]))
            
            if not any(s == j for j in ans):
                ans.append([strs[i]])
            else:
                flag = False
                for k in range(len(ans)):
                    if all(s == j[k] for j in ans[k]):
                        ans[k].append(strs[i])
                        flag = True
                        break
                if not flag:
                    ans.append([strs[i]])
                    
        return ans
```
