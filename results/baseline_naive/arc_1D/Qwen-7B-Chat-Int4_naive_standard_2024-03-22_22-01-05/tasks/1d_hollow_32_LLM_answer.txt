```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        ans = [[] for _ in range(len(set(strs)))]
        for s in strs:
            i = 0
            while i < len(ans) and set(s).issubset(ans[i]):
                i += 1
                
            ans[i].append(s)
            
        return ans
```
