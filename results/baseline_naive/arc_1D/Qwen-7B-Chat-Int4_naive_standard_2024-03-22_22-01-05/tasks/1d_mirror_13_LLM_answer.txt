```python
def group_chars(input):
    # Initialize an empty output array
    output = []
    
    # Iterate through each character in the input array
    for char in input:
        # Keep track of whether this character has been encountered more than twice
        count = 0
        for c in input:
            if c == char:
                count += 1
        
        # If this character has not been encountered more than twice, add it to the output array
        if count <= 2:
            output.append(char)
    
    return output
```
