class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        res = []
        
        dic = {}
        for word in s:
            if word not in dic:
                dic[word] = [word]
            else:
                dic[word].append(word)
                
        for key in dic.keys():
            res.append(dic[key])
            
        return res
