```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if not words:
            return []
        
        result = [[] for _ in range(len(words))]
        
        d = {}
        
        for word in words:
            key = ''.join(sorted(word))
            
            if key in d:
                result[d[key]].append(word)
                
            else:
                d[key] = len(result)
                result.append([word])
                
        return result
```
