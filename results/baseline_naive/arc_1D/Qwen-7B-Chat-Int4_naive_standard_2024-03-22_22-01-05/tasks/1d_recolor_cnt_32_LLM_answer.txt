    # Create an empty dictionary to store the groups of characters
    groups = {}
    
    for char in input:
        if char not in groups:
            groups[char] = []
        
        groups[char].append(char)
    
    # Convert the dictionary into a list of strings, where each string represents a group
    output = [group for group in groups.values()]
