```python
def find_duplicates(lst):
    # Create a dictionary to store the count of each element
    counts = {}
    for elem in lst:
        if elem not in counts:
            counts[elem] = 0
        counts[elem] += 1
    
    # Create an empty list to store duplicates
    duplicates = []
    
    # Iterate through the elements and their counts
    for elem, count in counts.items():
        if count > 1:
            # If the count is greater than 1, add the element to the duplicates list
            duplicates.append(elem)
    
    return duplicates
```
