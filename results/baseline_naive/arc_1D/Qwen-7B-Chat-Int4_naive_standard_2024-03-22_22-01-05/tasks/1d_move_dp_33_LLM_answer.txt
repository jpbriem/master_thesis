class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        dic = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in dic:
                dic[key] = []
            dic[key].append(word)
        
        res = []
        for k, v in dic.items():
            res.append(v)

        return res
