class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        groups = []
        
        for s in strs:
            if len(set(s)) == 1:
                groups.append([s])
            else:
                key = ''.join(sorted(s))
                if key not in groups[0]:
                    groups.append([])
                groups[0].append(s)
                
        return groups
