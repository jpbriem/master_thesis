class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in d:
                d[key] = [s]
            else:
                d[key].append(s)
        
        result = []
        for k in d:
            v = d[k]
            res = sorted(v)
            result.append(res)
        
        return result
