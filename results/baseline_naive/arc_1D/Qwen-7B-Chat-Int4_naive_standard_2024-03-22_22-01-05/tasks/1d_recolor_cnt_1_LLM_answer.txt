class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        d = {}
        
        for word in words:
            key = ''.join(sorted(word))
            if key not in d:
                d[key] = []
            d[key].append(word)
            
        result = [[]]
        for k in sorted(d):
            result.append(d[k])
            
        return result
