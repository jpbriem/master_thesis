class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        for s in strs:
            if s not in d:
                d[s] = [s]
            else:
                d[s].append(s)
        
        result = []
        for k in d.keys():
            v = d[k]
            result.append(v)
            
        return result
