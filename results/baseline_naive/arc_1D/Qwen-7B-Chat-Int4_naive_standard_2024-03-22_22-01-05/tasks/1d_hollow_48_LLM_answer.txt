    # Convert input to lowercase and split into individual characters
    chars = [char.lower() for char in input]
    
    # Group characters by their frequency using collections.Counter
    groups = collections.Counter(chars).most_common()
    
    # Sort groups by frequency in descending order
    groups.sort(key=lambda x: (-x[1], x[0]))
    
    # Concatenate groups into output string, separated by spaces
    output = "".join([group[0] + " " * (len(group) - 1) for group in groups])
    
    return output.strip()
