class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for s in strs:
            sorted_s = "".join(sorted(s))
            if sorted_s not in dic.keys():
                dic[sorted_s] = []
            dic[sorted_s].append(s)
            
        return list(dic.values())
