class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for s in strs:
            # convert string to lowercase and sort it 
            sorted_s = "".join(sorted(s.lower()))
            
            if sorted_s not in dic.keys():
                dic[sorted_s] = []
            dic[sorted_s].append(s)
                
        res = []
        for k, v in dic.items():
            res.append(v)
            
        return res
