class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        seen = set()
        res = []
        
        for num in nums:
            if num not in seen:
                res.append(num)
                seen.add(num)
                
        return res
