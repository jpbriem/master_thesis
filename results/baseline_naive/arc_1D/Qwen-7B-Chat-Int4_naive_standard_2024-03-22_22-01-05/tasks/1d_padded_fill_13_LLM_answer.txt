def group_chars(input):
    freq = {}
    output = []
    
    for char in input:
        if char in freq:
            freq[char] += 1
        else:
            freq[char] = 1
            
    for char, count in freq.items():
        while count > 0:
            output.append(char)
            count -= 1
    
    return output
