```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0:
            return []
        
        dict = {}
        res = []
        
        for s in strs:
            key = ''.join(sorted(s))
            if key not in dict:
                dict[key] = [s]
            else:
                dict[key].append(s)
                
        for k in dict.keys():
            res.append(dict[k])
            
        return res
```
