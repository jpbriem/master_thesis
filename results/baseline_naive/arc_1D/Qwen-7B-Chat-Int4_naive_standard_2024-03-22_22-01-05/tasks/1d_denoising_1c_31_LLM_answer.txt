class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        from collections import defaultdict
        groups = []
        d = defaultdict(list)
        
        for i in range(len(s)):
            if len(d[s[i]]) == 0:
                d[s[i]].append(i)
            else:
                d[s[i]][-1] += i
        
        for k, v in d.items():
            groups.append(v)

        return groups
