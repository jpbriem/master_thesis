```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0 or len(set(strs[0])) < 2:
            return []
        
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in d:
                d[key] = [s]
            else:
                d[key].append(s)
                
        result = [[d[k]] for k in sorted(d)]
        
        return result
```
