```python
class Solution:
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        if not nums:
            return 0

        ans = 1
        dic = {}
        for i in range(n):
            val = nums[i]

            if val in dic:
                start = max(dic[val] - 1, 0)
                end = min(dic[val] + 1, n-1)

                for j in range(start, end+1):
                    ans = max(ans, end - start + 1)

            dic[val] = i
        return ans
```
