```python
class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        from collections import defaultdict

        ans = []
        
        for word in s:
            if len(word) == 0 or len(set(word)) == 1:
                ans.append([word])
            else:
                d = defaultdict(list)
                for char in word:
                    d[char].append(word)
                
                for key in d.keys():
                    ans.append(d[key])
                    
        return ans
```
