def group_same_letters(input):
    # Create an empty dictionary to store the frequency of each letter
    freq = {}

    # Iterate over the input string and update the frequency of each letter
    for char in input:
        if char.isalpha():
            if char.lower() not in freq:
                freq[char.lower()] = 0
            freq[char.lower()] += 1

    # Initialize a list to store the groups of letters with the same frequency
    groups = []

    # Iterate over the keys of the frequency dictionary and add each key-value pair to the appropriate group
    for key, value in sorted(freq.items()):
        # If there are no groups yet, create a new group
        if len(groups) == 0:
            groups.append([key])
        else:
            # Otherwise, iterate over the existing groups and find the first group that contains the current key
            for group in groups:
                if key in group:
                    break
            # If the current key does not belong to any existing group, append it as a new group
            elif key not in group:
                groups.append([key])

    # Convert the groups back into strings and return them
    output = []
    for group in groups:
        output.append(''.join(group))
    return output
