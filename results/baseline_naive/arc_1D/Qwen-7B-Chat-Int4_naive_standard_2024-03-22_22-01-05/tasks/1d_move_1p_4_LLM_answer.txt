```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        ans = [[] for _ in range(len(words))]
        dic = {}
        
        for word in words:
            key = ''.join(sorted(word))
            if key not in dic:
                dic[key] = []
                ans[dic[key].index(key)] += [word]
            else:
                dic[key].append(word)
                
        return ans
```
