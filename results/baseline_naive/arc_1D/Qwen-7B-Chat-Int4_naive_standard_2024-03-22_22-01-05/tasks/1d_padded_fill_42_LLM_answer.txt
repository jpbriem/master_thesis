class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        
        for s in strs:
            # convert string to list of characters
            chars = [c for c in s]
            
            # sort the characters and get their index
            sorted_chars = ''.join(sorted(chars))
            
            if sorted_chars not in d:
                d[sorted_chars] = []
            d[sorted_chars].append(s)
        
        return list(d.values())
