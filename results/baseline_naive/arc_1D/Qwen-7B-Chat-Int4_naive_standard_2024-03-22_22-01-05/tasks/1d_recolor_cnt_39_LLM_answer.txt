def transform_string(s):
    # Initialize a dictionary to store the frequency of each character
    freq = {}
    
    # Iterate through the string and update the frequency count
    for char in s:
        if char not in freq:
            freq[char] = 0
        freq[char] += 1
    
    # Create an empty output string
    out_str = ""
    
    # Iterate through the updated frequency count and replace characters with their corresponding frequency
    for i in range(len(s)):
        if s[i] in freq:
            out_str += str(freq[s[i]])
        else:
            out_str += s[i]
    
    return out_str
