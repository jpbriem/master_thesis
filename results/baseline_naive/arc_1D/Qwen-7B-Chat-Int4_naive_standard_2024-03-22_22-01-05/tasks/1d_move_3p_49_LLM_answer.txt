    # Create a table to store the lengths of the longest common subsequences
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # Fill in the first row and column of the table
    for i in range(1, m + 1):
        dp[i][0] = 0

    for j in range(1, n + 1):
        dp[0][j] = 0

    # Iterate over the remaining cells of the table
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if input[i - 1] == input[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # The length of the longest common subsequence is stored in dp[m][n]
    return dp[m][n]
