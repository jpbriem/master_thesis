```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        for s in strs:
            if len(set(s)) == 1:
                ans.append([s])
            else:
                temp = set()
                for i in range(len(strs)):
                    if s[i] not in temp and strs[i] != s:
                        temp.add(strs[i][0])
                ans.append(list(temp))
                
        return ans
```
