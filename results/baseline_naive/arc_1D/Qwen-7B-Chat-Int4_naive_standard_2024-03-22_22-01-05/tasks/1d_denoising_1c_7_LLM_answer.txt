class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        res = []
        
        for i in range(len(s)):
            if len(set(s[i])) == 1 and not any(x in set(s[:i]) for x in set(s[i+1:])):
                res.append([s[i]])
                
        return res
