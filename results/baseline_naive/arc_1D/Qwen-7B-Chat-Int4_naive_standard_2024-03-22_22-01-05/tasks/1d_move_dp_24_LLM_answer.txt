```python
class Solution:
    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        d = {}
        
        for word in s:
            sorted_word = "".join(sorted(word))
            if sorted_word not in d:
                d[sorted_word] = []
            d[sorted_word].append(word)
            
        return [d[word] for word in d]
```
