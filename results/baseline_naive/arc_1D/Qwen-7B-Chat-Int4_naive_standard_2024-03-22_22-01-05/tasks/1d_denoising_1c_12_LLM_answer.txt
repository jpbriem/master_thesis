```python
class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        word_to_index = {word: i for i, word in enumerate(words)}
        
        groups = []
        for i in range(len(words)):
            if len(words[i]) == 0 or words[i] not in word_to_index:
                continue
            
            current_group = [words[i]]
            
            while True:
                index = word_to_index[words[i]]
                next_word = words[index + 1]
                
                if next_word == "":
                    break
                
                if next_word in word_to_index and index + 1 != word_to_index[next_word]:
                    current_group.append(next_word)
                    
                else:
                    groups.append(current_group)
                    current_group = [next_word]
                
        return groups
```
