class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if len(strs) == 0:
            return []
        
        dict = {}
        for s in strs:
            key = "".join(sorted(s))
            dict[key] = dict.get(key, []) + [s]
            
        res = [[] for _ in range(len(dict))]
        for k, v in dict.items():
            res[v[0]] += v[1:]
        
        return res
