class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        anagram_dict = {}
        for char in s:
            if char not in anagram_dict:
                anagram_dict[char] = []
            anagram_dict[char].append(s)
        
        result = [[]]
        for key in sorted(anagram_dict.keys()):
            result.append(anagram_dict[key])
        
        return result
