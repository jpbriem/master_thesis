class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs or len(strs) == 0:
            return []
        
        ans = [[strs[0]]]
        for i in range(1, len(strs)):
            tmp = {}
            for j in ans[-1]:
                tmp[j] = tmp.get(j, 0) + 1
            for k in strs[i]:
                if k not in tmp:
                    tmp[k] = 0
                tmp[k] += 1
            newAns = [k for k, v in tmp.items() if v > 0]
            ans.append(newAns)
            
        return ans
