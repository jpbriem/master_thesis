```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram_dict = {}
        
        for string in strs:
            key = ''.join(sorted(string))
            if key not in anagram_dict:
                anagram_dict[key] = []
            anagram_dict[key].append(string)
            
        return [anagram_dict[i] for i in sorted(anagram_dict.keys())]
```
