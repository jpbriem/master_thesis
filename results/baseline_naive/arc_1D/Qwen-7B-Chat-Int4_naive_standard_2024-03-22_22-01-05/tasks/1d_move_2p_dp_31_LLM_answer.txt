class Solution:
    def removeDuplicates(self, s):
        """
        :type s: str
        :rtype: str
        """
        if len(s) == 0 or len(set(s)) == len(s):
            return s
        
        res = []
        for i in range(len(s)):
            if s[i] not in res:
                res.append(s[i])
        
        return ''.join(res)
