class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        dic = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in dic:
                dic[key] = []
            dic[key].append(word)
        
        result = [[] for _ in range(len(dic))]
        for key in dic.keys():
            result[dic[key][0]] += dic[key][1:]
        
        return result
