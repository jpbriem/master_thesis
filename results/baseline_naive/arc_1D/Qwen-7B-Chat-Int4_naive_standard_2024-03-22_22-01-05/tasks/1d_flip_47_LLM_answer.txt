class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        res = []
        
        for i in range(len(s)):
            if len(res) == 0 or res[-1][0] != s[i]:
                res.append([s[i]])
            else:
                res[-1].append(s[i])
                
        return res
