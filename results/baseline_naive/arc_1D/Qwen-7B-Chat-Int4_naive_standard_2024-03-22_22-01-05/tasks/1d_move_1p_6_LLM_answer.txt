    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        word_dict = {}
        for word in words:
            if word not in word_dict:
                word_dict[word] = [word]
            else:
                word_dict[word].append(word)
        
        result = []
        for key, value in word_dict.items():
            if len(value) > 1:
                result.append(sorted(value))
        
        return result
