class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        res = []
        for i in range(len(s)):
            hash_map = {}
            for j in range(i+1, len(s)):
                if s[j] not in hash_map and s[i] == s[j]:
                    hash_map[s[j]] = 1
                else:
                    hash_map[s[j]] += 1

            if hash_map[s[i]]:
                res.append([s[i]])
        
        return res
