class Solution:
     def groupAnagrams(self, words):
         """
         :type words: List[str]
         :rtype: List[List[str]]
         """
         d = {}
         for word in words:
             key = ''.join(sorted(word))
             if key not in d:
                 d[key] = [word]
             else:
                 d[key].append(word)
         return list(d.values())
