class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        if not nums:
            return []
        
        s = set()
        for num in nums:
            if num not in s:
                s.add(num)
                yield num
            else:
                break
