class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for i in range(len(strs)):
            key = "".join(sorted(strs[i]))
            
            if key not in dic.keys():
                dic[key] = []
                
            dic[key].append(strs[i])
        
        result = []
        
        for key in dic.keys():
            result.append(dic[key])
        
        return result
