class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs or len(strs) == 0:
            return []
        
        d = {}
        for s in strs:
            if s not in d:
                d[s] = [s]
            else:
                d[s].append(s)
                
        res = [[] for _ in range(len(d))]
        for key, value in d.items():
            res[value[0]] += value[1:]
            
        return res
