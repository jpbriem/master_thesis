```python
def group_chars(input_list):
    # Create an empty dictionary to store the groups
    groups = {}
    
    # Iterate over each character in the input list
    for char in input_list:
        # Check if the character is already in a group
        if char in groups:
            # If it is, add it to the current group
            groups[char].append(char)
        else:
            # If it isn't, create a new group with the character as its key
            groups[char] = [char]
            
    # Convert the dictionary into a list of lists
    output_list = [[key] + value for key, value in groups.items()]
    
    return output_list
```
