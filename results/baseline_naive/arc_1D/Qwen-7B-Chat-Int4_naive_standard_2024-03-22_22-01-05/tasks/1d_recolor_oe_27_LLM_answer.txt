class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        if len(s) < 2:
            return [s]

        res = []
        
        for i in range(len(s)):
            # 哈希表中查找第一个字符
            ch = s[i]
            if ch not in res[0]:
                res.append([ch])
            else:
                j = res.index(ch)
                while j > -1 and res[j][len(res[j])-1] != ch:
                    j -= 1
                if j == -1:
                    res.append([ch])
                else:
                    res[j].append(ch)

        return res
