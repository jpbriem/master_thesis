class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        groups = {}
        for word in words:
            key = ''.join(sorted(word))
            if key not in groups:
                groups[key] = [word]
            else:
                groups[key].append(word)
        
        return [[key] + list(group) for key, group in groups.items()]
