    def groupAnagrams(self, s: str) -> List[List[str]]:
        ans = []
        
        d = {}
        for c in s:
            if c not in d:
                d[c] = [c]
            else:
                d[c].append(c)
                
        for key, value in d.items():
            ans.append(value)

        return ans
