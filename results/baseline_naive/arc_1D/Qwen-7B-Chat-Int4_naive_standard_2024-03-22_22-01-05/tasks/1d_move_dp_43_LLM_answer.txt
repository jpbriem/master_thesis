class Solution:
    def longestCommonPrefix(self, strs):
        if not strs:
            return ''
        
        # find the first different character in all strings
        min_str = min(strs)
        max_str = max(strs)
        for i in range(len(min_str)):
            if min_str[i] != max_str[i]:
                break
        
        prefix = []
        for i in range(max_str.find(min_str)):
            prefix.append(min_str[i])
            
        return ''.join(prefix)
