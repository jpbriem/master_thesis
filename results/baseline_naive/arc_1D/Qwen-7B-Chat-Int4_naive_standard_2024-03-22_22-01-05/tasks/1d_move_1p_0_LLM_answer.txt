class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        res = []
        i = 0
        while i < len(nums) - 1:
            if nums[i] != nums[i + 1]:
                res.append(nums[i])
            else:
                i += 1
        return res
