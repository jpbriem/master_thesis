class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        d = {}
        
        for word in words:
            key = ''.join(sorted(word))
            
            if key not in d:
                d[key] = [word]
            else:
                d[key].append(word)
                
        result = []
        
        for k, v in d.items():
            result.append(v)
            
        return result
