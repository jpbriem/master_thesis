class Solution:
     def groupAnagrams(self, words):
         """
         :type words: List[str]
         :rtype: List[List[str]]
         """
         d = {}
         for word in words:
             key = ''.join(sorted(word))
             if key not in d:
                 d[key] = []
             d[key].append(word)
         
         result = [d[key] for key in sorted(d)]
         return result
