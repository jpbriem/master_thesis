    def removeDuplicates(self, s: str) -> str:
        if not s:
            return ''
        
        hash_set = set()
        res = []
        for ch in s:
            if ch not in hash_set:
                res.append(ch)
                hash_set.add(ch)
                
        return ''.join(res)
