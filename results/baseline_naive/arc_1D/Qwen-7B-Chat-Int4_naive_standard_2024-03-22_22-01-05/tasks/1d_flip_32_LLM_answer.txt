```python
def longest_common_subsequence(arr):
    n = len(arr)
    dp = [[0] * (n+1) for _ in range(n+1)]
    max_len = 0

    for i in range(1, n+1):
        for j in range(1, n+1):
            if arr[i-1] == arr[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
                max_len = max(max_len, dp[i][j])
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])

    return arr[:max_len]
```
