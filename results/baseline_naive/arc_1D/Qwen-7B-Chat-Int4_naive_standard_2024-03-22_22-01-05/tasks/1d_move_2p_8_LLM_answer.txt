class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[str]
        :rtype: List[str]
        """
        if len(nums) == 0:
            return []
        
        seen = set()
        result = []
        for num in nums:
            if num not in seen:
                seen.add(num)
                result.append(num)
                
        return result
