```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        for s in strs:
            if not s:
                continue
            
            count = {}
            
            for c in s:
                if c in count:
                    count[c] += 1
                else:
                    count[c] = 1
                    
            key = list(count.keys())
            val = list(count.values())
            
            key.sort()
            val.sort(reverse=True)
            
            ans.append(key + val)
        
        return ans
```
