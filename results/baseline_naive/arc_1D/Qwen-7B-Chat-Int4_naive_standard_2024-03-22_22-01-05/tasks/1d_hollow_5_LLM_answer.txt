```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key not in anagram:
                anagram[key] = [s]
            else:
                anagram[key].append(s)
        
        result = []
        for k, v in anagram.items():
            result.append(v)
            
        return result
```
