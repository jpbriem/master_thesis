```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagrams = []
        
        for s in strs:
            key = "".join(sorted(s))
            
            if key not in anagrams:
                anagrams.append([s])
                
            else:
                index = 0
                for i in range(len(anagrams)):
                    if sorted(anagrams[i]) == key:
                        index = i + 1
                        
                anagrams[index].append(s)
        
        return anagrams
```
