        max_num = len(nums) + 1
        min_num = 0
        # 双端队列
        queue_left = []
        queue_right = []
        for i in range(len(nums)):
            if nums[i] == 0:
                queue_left.append(max_num)
                queue_right.append(max_num)
            else:
                queue_left.append(min_num)
                queue_right.append(min_num)
        while queue_left or queue_right:

            right_val = queue_right.pop(-1) if queue_right else None
            if left_val is not None:
                nums[left_val - 1] = left_val
            if right_val is not None:
                nums[-right_val - 1] = right_val
        return nums
