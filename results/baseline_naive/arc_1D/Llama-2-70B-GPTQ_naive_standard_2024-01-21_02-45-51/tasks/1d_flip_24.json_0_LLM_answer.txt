class Solution:
    def transform(self, nums: List[int]) -> List[int]:
        """
        思路：双端队列，从左右两边开始循环执行。
                如果当前元素等于队尾元素，则将其移动到队头；
                否则，将其保持原位置。
                O(n) time complexity.
                O(1) space complexity.
            """
        if not nums or len(nums) == 1: return nums
        q = [] # 双端队列
        for i in range(len(nums)):
            q.append(i)
        while True:
            left, right = q[0], q[-1]
            if nums[left]!= nums[right]: break
            nums[left], nums[right] = nums[right], nums[left]
            del q[0], q[-1]
        for i in range(q.__len__()):

            if nums[j] > nums[q[i]]:
                nums[j], nums[q[i]] = nums[q[i]], nums[j]
        return nums
