class Solution:
    def transform(self, nums: List[int]) -> List[int]:
        """
        思路：双端队列，从头和尾循环执行。
                如果当前元素等于左边的元素，则将右侧元素加入到队尾；
                否则，将左侧元素加入到队尾。
                O(n) time complexity.
                O(n) space complexity.
            -------------------
            注意事项：
                1、不能使用数组来存放结果，因为每次操作都会改变原始数据。
                2、不要在函数内部定义新的变量，这样可以提高运算速度。
        """
        if not nums or len(nums) == 1: return nums
        # 双端队列
        q = []
        for i in range(len(nums)):
            while q and nums[i]!= nums[q[-1]]:
                q.append(nums[i])
                i -= 1
            else:
                q.pop()
                q.append(nums[i])
        return q + list(reversed(q))
