    def replaceElements(self, arr: List[int]) -> List[int]:
        """
        思路：双指针，一个从左往右扫描，一个从右往左扫描。
                如果两者相等，则说明是连续的数字序列；否则不是。
                如果是连续的数字序列，则将后面的元素全部置为前面元素值。
                O(n) time complexity. O(1) space complexity.
        :param arr:
        :return:
        """
        if not arr or len(arr) == 0:
            return []
        left = right = 0
        while True:
            # 当左侧和右侧都没有元素时，跳出循环
            if left >= len(arr)-1 and right < 0:
                break
            elif arr[left]!= arr[right]:
                # 如果左边和右边不相等，那么就不是连续的数字序列了
                for i in range(len(arr)):
                    if arr[i] > arr[left]:
                        arr[i] = arr[left]
                left += 1
                continue
            else:
                # 如果左边和右边相等，那么就是连续的数字序列了
                for i in range(left+1, len(arr)+1):
                    arr[i] = arr[left]
                left += 1
                right -= 1
        return arr
