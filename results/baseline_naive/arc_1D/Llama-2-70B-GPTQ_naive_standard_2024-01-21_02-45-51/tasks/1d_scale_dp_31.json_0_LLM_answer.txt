class Solution:
    def transform(self, nums: List[int]) -> List[int]:
        """
        思路：双指针，一个从左往右扫描，一个从右往左扫描。
            当前元素等于后面的元素时，将其复制到后面；
            当前元素不等于后面的元素时，将其跳过。
            O(n) time complexity.
            O(1) space complexity.
        :param nums:
        :return:

        left = right = 0
        for i in range(len(nums)):
            if nums[i] == nums[right]:

                left += 1
            else:

        return nums[:left]
