class Solution:
    def rotate(self, nums: List[int]) -> None:
        """
        思路：双端队列，从左到右找出第一个不为0的数字，然后将其移动到最前面。
                再从右往左找出第一个不为0的数字，并将其移动到最后面。
                如果两者都是0，则直接返回。
                O(n) time complexity. O(1) space complexity.
        :param nums:
        :return:
        """
        if not nums or len(nums) == 0: return
        # 双端队列
        q = []
        for i in range(len(nums)):
            if nums[i]:
                q.appendleft(i)
        while q:
            left, right = q.pop(), q.pop()
            if left < right:
                nums[right], nums[left] = nums[left], nums[right]
                q.extend([left, right + 1])
            else:
                break
