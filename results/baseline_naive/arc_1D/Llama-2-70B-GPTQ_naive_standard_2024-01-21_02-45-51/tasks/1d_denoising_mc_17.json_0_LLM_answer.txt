import collections
from typing import List


class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        """
        思路：双指针，一个指向前端，另一个指向后端。
            如果两者相等则删除前端元素并移动后端指针；
            否则只需要移动前端指针就可以了。
            O(n) time complexity. O(1) space complexity.
        :param nums:
        :return:
        """
        if not nums or len(nums) == 1:
            return len(nums)

        slow = fast = 0
        for i in range(len(nums)):
            if i!= slow and nums[i] == nums[slow]:
                continue
            else:
                nums[fast], nums[i] = nums[i], nums[fast]
                fast += 1
                slow += 1
        return fast
