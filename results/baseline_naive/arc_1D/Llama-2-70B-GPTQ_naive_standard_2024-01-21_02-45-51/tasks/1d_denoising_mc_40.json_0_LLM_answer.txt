class Solution:
    def replaceElements(self, arr: List[int]) -> List[int]:
        """
        思路：双指针，一个从左往右扫描，一个从右往左扫描。
                如果两者相等，则将其中较小的值设置为目标值；
                否则，将距离更近的那个元素设置为目标值。
                O(n) time complexity. O(1) space complexity.
        :param arr:
        :return:
        """
        left = right = len(arr)-1
        while True:
            if arr[left] == arr[right]:
                arr[left], arr[right] = min(arr[left], arr[right]), max(arr[left], arr[right])
            elif arr[left] < arr[right]:
                arr[left] = max(arr[left], arr[right])
            else:
                arr[right] = max(arr[left], arr[right])
            # 注意这里不能用while(True):break语法，因为Python会认为是无限循环而产生RuntimeError
            if not (left > -1 or right >= 0): break
            left -= 1
            right += 1
        return arr
