class Solution:
    def transform(self, nums: List[int]) -> List[int]:
        """
        思路：双端队列，从头和尾循环执行相同的操作。
            - 如果当前元素等于0，则将其转换为3；

            最后返回结果。
        :param nums:

        # 定义双端队列
        queue = []
        for i in range(len(nums)):
            if nums[i] == 0:
                queue.append(3)
            else:
                queue.append(9)
        while len(queue):
            front = queue.popleft()
            back = queue.pop()
            if front == 3 or back == 3:
                nums[front % len(nums)] = 3
                nums[-back % len(nums)] = 3
            elif front == 9 or back == 9:
                nums[front % len(nums)] = 9
                nums[-back % len(nums)] = 9
        return nums
