class Solution:
    def removeZero(self, nums: List[int]) -> List[int]:
        """
        思路：双指针，一个从前往后扫描，一个从后往前扫描。
            如果两者都是非0的话，那么就将第二个指针所在位置的数字赋值给第一个指针所在位置的数字；
            否则，只需要移动第一个指针即可。
            O(n) time complexity. O(1) space complexity.
        """
        if not nums or len(nums) == 0: return []
        i = j = 0
        while i < len(nums):
            # 当前元素不为0时，将其复制到j处
            if nums[i]!= 0:
                nums[j] = nums[i]
                j += 1
            # 更新i和j
            i += 1
            j -= 1
        return nums[:j + 1]
