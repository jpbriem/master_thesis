class Solution:
    def transform(self, nums: List[int]) -> List[int]:
        """
        思路：双端队列，从左右两边开始循环执行。
                如果当前元素等于0，则将其移动到对应的位置上；
                否则，直接跳过该元素。
                O(n)时间复杂度，O(1)空间复杂度。
        :param nums:
        :return:
        """
        if not nums or len(nums) == 1:
            return nums

        left = right = 0
        for i in range(len(nums)):
            if nums[i]!= 0:
                continue
            while left < right:
                if nums[left] == 0:
                    nums[right], nums[left] = nums[left], nums[right]
                    right += 1
                else:
                    break
                left += 1
            elif left > right:
                nums[right], nums[left - 1] = nums[left - 1], nums[right]
                left -= 1
            else:
                break
        return nums
