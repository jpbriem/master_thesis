        """
        :type grid: List[List[str]]
        :rtype: int
        思路：扫描一行，如果发现岛的话就将其标记为岛，然后再从这个位置开始向四周遍历，直到不是岛或者已经访问过。

        if not grid or len(grid) == 0:
            return 0
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == "1":
                    self.dfs(grid, i, j)
                    count += 1
        return count

    def dfs(self, grid, x, y):
        if x < 0 or x >= len(grid) or y < 0 or y >= len(grid[x]) or grid[x][y]!= "1":
            return
        grid[x][y] = "-"
        self.dfs(grid, x - 1, y)
        self.dfs(grid, x + 1, y)
        self.dfs(grid, x, y - 1)
        self.dfs(grid, x, y + 1)
