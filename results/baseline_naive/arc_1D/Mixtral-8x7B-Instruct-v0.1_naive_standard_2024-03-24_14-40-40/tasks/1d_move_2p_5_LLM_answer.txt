const longestSubstring = (str) => {
    let maxLength = 0;
    let currentChar = str[0];
    let count = 1;
    for(let i=1;i<str.length;i++){
        if(currentChar === str[i]){
            count++;
        } else {
            if(count > maxLength){
                maxLength = count;
            }
            count = 1;
            currentChar = str[i];
        }
    }
    return Math.max(maxLength, count);
}
