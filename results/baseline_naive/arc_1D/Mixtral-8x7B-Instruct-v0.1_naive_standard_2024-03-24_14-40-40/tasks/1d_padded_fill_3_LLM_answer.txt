const groupBy = (arr, fn) => {
    const groups = {};
    for(let val of arr){
        const key = fn(val);
        if(!groups[key]){
            groups[key] = [];
        }
        groups[key].push(val);
    }
    return groups;
}

const replaceWithGrouped = (str, grpFn) => {
    let strArr = [...str];
    let i = 0;
    while(i < strArr.length - 1){
        const currGrp = grpFn([strArr[i], strArr[i + 1]])[0];
        strArr[i] = currGrp;
        strArr.splice(i+1, 1);
    }
    return strArr.join('');
};

const mergeGroups = (grpObj) => {
    let newGrpObj = {};
    Object.keys(grpObj).forEach((key)=>{
        if(!newGrpObj[grpObj[key][0]]){
            newGrpObj[grpObj[key][0]] = [grpObj[key][0]];
        }
        else{
            newGrpObj[grpObj[key][0]].push(...grpObj[key]);
        }
    });
    return newGrpObj;
}

const stringCompressionII = (s) => {
    // edge cases
    if(s === null || s.length <= 1) return s;

    // helper functions
    const getCharCode = (char) => char.charCodeAt();
    const isSameGroup = (arr) => arr[0] === arr[1];

    // main logic
    let compressedStr = replaceWithGrouped(s, groupBy(s, getCharCode));
    compressedStr = replaceWithGrouped(compressedStr, (arr) => {
        if(isSameGroup(arr)){
            return [arr[0]];
        }
        else{
            return mergeGroups(groupBy(arr, getCharCode))[arr[0]];
        }
    })

    return compressedStr;
};
