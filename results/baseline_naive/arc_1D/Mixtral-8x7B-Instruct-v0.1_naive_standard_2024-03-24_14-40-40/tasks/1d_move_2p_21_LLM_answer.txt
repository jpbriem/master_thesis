const mergeSortedArrays = (arr) => {
    let left = 0;
    let right = arr.length - 1;

    while(left < right){
        if(arr[left] === arr[left + 1]){
            for(let i = left + 1; i <= right; i++){
                arr[i-1] = arr[i];
            }
            right--;
        } else {
            left++;
        }
    }

    return arr;
}
