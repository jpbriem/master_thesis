function removeDuplicateLetters(str) {
    let stack = [];
    const freqMap = {};
    for (let char of str) {
        if (!freqMap[char]) {
            freqMap[char] = 0;
        }
        freqMap[char]++;
    }
    for (let char of str) {
        while (stack.length && stack[stack.length - 1] > char && freqMap[stack[stack.length - 1]] > 0) {
            freqMap[stack.pop()]--;
        }
        if (freqMap[char] > 0) {
            stack.push(char);
        }
    }
    return stack.join('');
}
