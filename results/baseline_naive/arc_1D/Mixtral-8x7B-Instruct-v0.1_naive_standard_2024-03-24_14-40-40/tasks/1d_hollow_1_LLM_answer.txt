    if (!Array.isArray(arr)) return "Not an array";

    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i] === toMove && arr[i + 1]!== toMove) {
            swap(arr, i, findNextIndex(arr, toMove, i));
        }
    }

    function findNextIndex(arr, target, currentIndex) {
        for (let i = currentIndex + 1; i < arr.length; i++) {
            if (arr[i] === target) {
                return i;
            }
        }
        return -1; // not found
    }

    return arr;
};
