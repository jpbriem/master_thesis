function sortByHeight(arr) {
    let trees = [];
    for (let i = 0; i < arr.length; i++) {
        if (arr[i]!== -1) {
            trees.push([arr[i], i]);
        }
    }
    trees.sort((a, b) => a[0] - b[0]);
    for (let j = 0; j < trees.length; j++) {
        while (arr[trees[j][1]] === -1) {
            trees[j][1]++;
        }
        arr[trees[j][1]] = trees[j][0];
    }
    return arr;
}
