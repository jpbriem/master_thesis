function swap(arr, i, j) {
  let temp = arr[i];
  arr[i] = arr[j];
  arr[j] = temp;
}

function moveToEnd(arr, target) {
  for (let i = 0; i < arr.length - 1; i++) {
    if (arr[i] === target && arr[i + 1]!== target) {
      swap(arr, i, i + 1);
    }
  }
}

function sortByHeight(arr) {
  moveToEnd(arr, "a");
  moveToEnd(arr, "g");
  moveToEnd(arr, "h");
  moveToEnd(arr, "j");
}
