function groupAnimals(arr) {
    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i]!== arr[i + 1]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);
    return result.map((el, idx) => {
        if (idx === 0) {
            return el;
        } else {
            const firstEl = el[0];
            const secondEl = String.fromCharCode(firstEl.charCodeAt() + 1);
            if (secondEl > el[el.length - 1]) {
                return [el[0],...el.slice(1).sort()]
            } else {
                return [...el.sort()]
            }
        }
    }).flat();
}
