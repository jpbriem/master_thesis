def merge_groups(lst):
    if len(lst) == 0:
        return []

    res = []
    i = 0
    while i < len(lst):
        j = i + 1
        while j < len(lst) and lst[j] == lst[i]:
            j += 1

        grp = lst[i : j]
        res.append(grp[0])
        if len(grp) > 1:
            res[-1] *= len(grp) - 1

        i = j

    return res
