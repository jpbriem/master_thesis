const removeDuplicateLetters = (s) => {
    const stack = [];
    let lastOccurence = {};

    for(let i=0;i<s.length;i++){
        if(!lastOccurence[s[i]]){
            while(stack.length && s[i] < stack[stack.length-1] && lastOccurence[stack[stack.length-1]] > i){
                stack.pop();
            }
            stack.push(s[i]);
            lastOccurence[s[i]] = i;
        }
    }

    return stack.join('');
};
