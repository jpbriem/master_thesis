class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        int len = arr.length;
        for (int i = 0; i < len - 1; i++) {
            if (!arr[i].equals(arr[i + 1])) continue;
            int j = i + 1;
            while (j < len && arr[j].equals(arr[i])) j++;
            // fill the array with char of current element from index i to j-1
            Arrays.fill(arr, i, j, arr[i]);
            i = j - 1;
        }
        return arr;
    }
}

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (i > 0 && arr[i].equals(arr[i - 1])) {
                sb.append(arr[i]);
            } else {
                sb.setLength(0);
                sb.append(arr[i]);
                arr[i] = sb.toString();
            }
        }
        return arr;
    }
