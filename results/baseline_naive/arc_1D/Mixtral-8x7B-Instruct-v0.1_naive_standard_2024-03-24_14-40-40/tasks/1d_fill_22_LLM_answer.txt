const uniq = (arr) => {
    let result = [];
    for(let i=0;i<arr.length;i++){
        if(!result.includes(arr[i])){
            result.push(arr[i]);
        }
    }
    return result;
}

const replaceDuplicate = (arr) => {
    let uniqueArr = uniq(arr);
    let index = 0;
    for(let i=0;i<arr.length;i++){
        arr[i] = uniqueArr[index];
        if(arr[i]!== arr[i+1]){
            index++;
        }
    }
    return arr;
}
