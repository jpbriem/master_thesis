You can use `itertools.groupby()` to group the elements by their value, then iterate over these groups and replace them with a new list that contains either all of the original values or just one if there are more than 5 in a row.

```
from itertools import groupby

def fix(lst):
    return [next(g)[0] for g in map(lambda x: (x[0], list(x[1])),
                                   groupby(lst)) if len(list(x[1])) <= 5)]

print(fix(['a', 'a', 'a', 'a', 'a', 'a', 'g', 'g', 'g', 'g', 'a', 'a', 'a', 'a', 'a']))
# ['a', 'a', 'a', 'a', 'a', 'a', 'a', 'g', 'g', 'g', 'g', 'a', 'a', 'a']
print(fix(['a', 'c', 'c', 'c', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a']))
# ['a', 'a', 'a', 'c', 'c', 'c', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a']
print(fix(['a', 'i', 'i', 'i', 'i', 'i', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a']))
# ['a', 'a', 'a', 'i', 'i', 'i', 'i', 'i', 'a', 'a', 'a', 'a', 'a', 'a', 'a']
