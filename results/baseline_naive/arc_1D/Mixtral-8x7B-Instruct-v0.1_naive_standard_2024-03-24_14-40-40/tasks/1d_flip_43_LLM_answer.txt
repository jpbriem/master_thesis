public class Solution {
    public String[] rearrange(String[] input) {
        if (input == null || input.length <= 1) return input;

        int len = input.length, countA = 0, countB = 0, indexA = -1, indexB = -1;
        for (int i = 0; i < len; ++i) {
            if (input[i].equals("a")) {
                countA++;
                indexA = i;
            } else if (input[i].equals("b")) {
                countB++;
                indexB = i;
            }
        }

        // no b in the array
        if (countB == 0) {
            for (int i = 0; i + 1 < len; ++i) {
                if (input[i].equals("a") && input[i+1].equals("a")) {
                    input[i] = "b";
                    input[indexA--] = "a";
                }
            }
        }
        // no a in the array
        else if (countA == 0) {
            for (int i = 0; i + 1 < len; ++i) {
                if (input[i].equals("b") && input[i+1].equals("b")) {
                    input[i] = "a";
                    input[indexB--] = "b";
                }
            }
        }
        // both a and b exist in the array
        else {
            for (int i = 0; i + 1 < len; ++i) {
                if (input[i].equals("a") && input[i+1].equals("a")) {
                    input[i] = "b";
                    input[indexA--] = "a";
                } else if (input[i].equals("b") && input[i+1].equals("b")) {
                    input[i] = "a";
                    input[indexB--] = "b";
                }
            }
        }

        return input;
    }
}
