class Solution {
    public String[] replaceFs(String[] arr) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i].equals("f")) {
                int j = i - 1;
                while (j >= 0 && arr[j].equals("a")) {
                    j--;
                }
                char c = j == -1? 'e' : 'h';
                arr[i] = "" + c;
            } else {
                arr[i] = "a";
            }
        }
        return arr;
    }
}

class Solution {
    public String[] replaceFs(String[] arr) {
        StringBuilder sb = new StringBuilder();
        boolean isA = true;
        for (String s : arr) {
            if (isA) {
                if (!s.equals("a")) {
                    sb.append('e');
                    isA = false;
                } else {
                    sb.append('a');
                }
            } else {
                if (s.equals("a")) {
                    sb.append('h');
                }
            }
        }
        String str = sb.toString();
        String[] res = new String[str.length()];
        for (int i = 0; i < res.length; i++) {
            res[i] = "" + str.charAt(i);
        }
        return res;
    }
}
