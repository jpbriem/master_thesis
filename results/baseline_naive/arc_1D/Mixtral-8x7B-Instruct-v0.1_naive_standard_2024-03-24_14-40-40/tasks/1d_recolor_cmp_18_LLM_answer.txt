function freqAlphabets(s) {
    let output = [];
    for (let i = s.length - 1; i >= 0; i--) {
        if (s[i] === '#') {
            const num = parseInt(`${s[i-2]}${s[i-1]}`);
            output.unshift(String.fromCharCode(num + 96));
            i -= 2;
        } else if (s[i]!== '.') {
            output.unshift(s[i]);
        }
    }
    return output.join('');
}
