function groupByConsecutiveLetters(arr) {
    let result = [];
    for (let i = 0; i < arr.length; i++) {
        if (!result.length || result[result.length - 1][0]!== arr[i]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    return result;
}

function solution(arr) {
    let groups = groupByConsecutiveLetters(arr);
    for (let i = 0; i < groups.length; i++) {
        if (groups[i].length > 1 && groups[i][0] === groups[i][groups[i].length - 1]) {
            let rep = groups[i][0];
            while (rep === groups[i][0]) {
                rep += '_';
            }
            for (let j = 0; j < groups[i].length; j++) {
                groups[i][j] = replaceAll(groups[i][j], groups[i][0], rep);
            }
        }
    }
    return [].concat.apply([], groups);
}
