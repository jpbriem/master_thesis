function removeAdjacent(str) {
  let arr = str.split('');
  for (let i = 0; i < arr.length - 1; i++) {
    if (arr[i] === arr[i + 1]) {
      arr.splice(i, 2);
      i--;
    }
  }
  return arr.join('') || arr[0];
}
