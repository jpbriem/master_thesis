const replaceAll = (str, find, replace) => {
    return str.replace(new RegExp(find, "g"), replace);
}

const groupByConsecutiveLetters = arr => {
    let result = [];
    for (let i = 0; i < arr.length; i++) {
        if (!result[result.length - 1] || result[result.length - 1][0].charCodeAt()!== arr[i].charCodeAt()) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i])
        }
    }
    return result;
}

const expandGroups = groups => {
    let expanded = []
    for (let g of groups) {
        let letter = g[0];
        let count = g.length;
        expanded.push(...Array(count).fill(letter))
    }
    return expanded;
}
