function sortByHeight(arr) {
    let idx = arr.findIndex((el, i) => el === "h");
    if (idx!== -1) {
        for (let j = idx; j < arr.length; j++) {
            if (arr[j]!== "h") {
                [arr[idx], arr[j]] = [arr[j], arr[idx]];
                idx++;
            }
        }
    }
    return arr;
}
