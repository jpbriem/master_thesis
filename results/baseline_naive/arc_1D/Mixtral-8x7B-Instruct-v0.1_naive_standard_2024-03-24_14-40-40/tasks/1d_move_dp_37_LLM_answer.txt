const moveToFront = (arr) => {
    if (!Array.isArray(arr)) return;
    let firstEl = arr[0];
    for (let i = 0; i < arr.length - 1; i++) {
        arr[i] = arr[i + 1];
    }
    arr[arr.length - 1] = firstEl;
}

const moveToFrontTwice = (arr) => {
    if (!Array.isArray(arr)) return;
    moveToFront(arr);
    moveToFront(arr);
}
