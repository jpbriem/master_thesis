    let maxLength = 0;
    for(let i=0; i<arr.length-1; i++) {
        if(arr[i] === arr[i+1]) continue;
        let count = 1, j = i + 1;
        while(j < arr.length && arr[j]!== arr[i]) {
            count++;
            j++;
        }
        maxLength = Math.max(count, maxLength);
    }
    return maxLength;
}
