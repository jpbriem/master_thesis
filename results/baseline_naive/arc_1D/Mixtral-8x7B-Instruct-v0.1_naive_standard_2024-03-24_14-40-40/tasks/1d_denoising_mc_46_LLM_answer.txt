function fixBricks(bricks) {
    // loop through the array and find a group of different bricks
    for (let i = 0; i < bricks.length - 1; i++) {
        if (bricks[i]!== bricks[i + 1]) {
            // check to see if we can replace all elements in this group with one type of brick
            let leftGroup = [];
            let rightGroup = [];
            while (i >= 0 && bricks[i] === bricks[i - 1]) {
                leftGroup.unshift(bricks[i]);
                i--;
            }
            while (i < bricks.length - 1 && bricks[i] === bricks[i + 1]) {
                rightGroup.push(bricks[i]);
                i++;
            }
            // if both groups are not the same, then there is no way to make them the same
            if (!sameBuckets(leftGroup)) return false;
            if (!sameBuckets(rightGroup)) return false;
            // otherwise, we will replace the whole group with either left or right group
            if (leftGroup.length > rightGroup.length) {
                for (let j = i + 1; j <= i + rightGroup.length; j++) {
                    bricks[j] = bricks[i];
                }
            } else {
                for (let j = i; j >= i - leftGroup.length + 1; j--) {
                    bricks[j] = bricks[i];
                }
            }
        }
    }
    return bricks;
}
