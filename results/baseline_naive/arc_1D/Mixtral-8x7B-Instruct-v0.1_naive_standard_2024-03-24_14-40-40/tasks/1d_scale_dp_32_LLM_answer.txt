    for (let i = 0; i < arr.length; i++) {
        if (arr[i]!== val) {

function replaceAll(arr, findVal, repVal) {
    let idx = arr.indexOf(findVal);
    while (idx > -1) {
        arr[idx] = repVal;
        idx = arr.indexOf(findVal);
    }
}

function fixArray(arr) {
    // Fill the array with first value until we reach a different one
    let startVal = arr[0];
    fill(arr, startVal);

    // Find the last index of the starting value and then replace all values after that point
    let endIdx = arr.lastIndexOf(startVal);
    replaceAll(arr, startVal, arr[endIdx + 1]);

    return arr;
}
