            }
        });

        List<String> result = new ArrayList<>();

class Solution {
    public String[] findLongestChain(String[] pairs) {
        if (pairs == null || pairs.length < 2) return pairs;

        Map<Integer, Integer> map = new HashMap<>();
        List<String> list = new ArrayList<>();

        // sort by first element of each pair
        Arrays.sort(pairs, new Comparator<String[]>() {
            @Override
            public int compare(String[] a, String[] b) {
                return Integer.compare(Integer.parseInt(a[0]), Integer.parseInt(b[0]));
            }
        });

        for (String[] p : pairs) {
            int maxEnd = -1;
            for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
                if (entry.getValue() < Integer.parseInt(p[0])) {
                    maxEnd = Math.max(maxEnd, entry.getKey());
                }
            }
            if (maxEnd!= -1) {
                map.put(maxEnd, Integer.parseInt(p[1]));
            } else {
                map.put(Integer.parseInt(p[0]), Integer.parseInt(p[1]));
                list.add(p[1]);
            }
        }

        return list.toArray(new String[list.size()]);
    }
}
