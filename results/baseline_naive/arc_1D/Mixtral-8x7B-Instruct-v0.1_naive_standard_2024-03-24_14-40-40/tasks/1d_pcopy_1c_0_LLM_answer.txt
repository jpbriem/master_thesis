    const groups = []
    let currentGroup = []

    for (const animal of animals) {
      if (!currentGroup.length) {
        currentGroup.push(animal)
      } else {
        const lastAnimal = currentGroup[currentGroup.length - 1]
        const isSameType = animal.toLowerCase().charCodeAt(0) === lastAnimal.toLowerCase().charCodeAt(0)

        if (isSameType) {
          currentGroup.push(animal)
        } else {
          groups.push(currentGroup)
          currentGroup = [animal]
        }
      }
    }

    groups.push(currentGroup)

    return groups
  }
