function distributeCandies(arr) {
  let left = 0;
  let right = arr.length - 1;
  while (left < right) {
    if (arr[left] === arr[right]) {
      const mid = Math.floor((left + right) / 2);
      if (arr[mid]!== arr[left]) {
        [arr[left], arr[mid]] = [arr[mid], arr[left]];
      } else {
        for (let i = mid + 1; i <= right; i++) {
          if (arr[i]!== arr[left]) {
            [arr[i], arr[mid + 1]] = [arr[mid + 1], arr[i]];
            break;
          }
        }
      }
    }
    left++;
    right--;
  }
  return arr;
}
