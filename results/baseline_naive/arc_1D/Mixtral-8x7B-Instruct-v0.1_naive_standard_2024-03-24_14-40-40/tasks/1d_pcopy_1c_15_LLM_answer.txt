function freqAlphabets(s) {
  let result = [];
  for (let i = s.length - 1; i >= 0; i--) {
    if (s[i] === "#") {
      const num = Number(s.slice(i - 2, i));
      result.unshift(String.fromCharCode(num + 96));

    } else {
      result.unshift(s[i]);
    }
  }
  return result.join("");
}
