function sortByHeight(arr) {
  let idx = arr.findIndex((el, i) => el === -1);
  if (idx!== -1) {
    let left = arr.slice(0, idx).filter(el => el > -1);
    let right = arr.slice(idx + 1).filter(el => el > -1);
    return [...left.sort((a, b) => a - b),...[...arr].splice(idx, 1),...right.sort((a, b) => a - b)];
  } else {
    return arr;
  }
}
