const longestSubstring = (str) => {
    let maxLength = 0;
    let currentChar = str[0];
    let count = 1;
    for(let i=1;i<str.length;i++){
        if(str[i] === currentChar){
            count++;
        } else {
            if(count > maxLength){
                maxLength = count;
            }
            currentChar = str[i];
            count = 1;
        }
    }
    return Math.max(maxLength, count);
}
