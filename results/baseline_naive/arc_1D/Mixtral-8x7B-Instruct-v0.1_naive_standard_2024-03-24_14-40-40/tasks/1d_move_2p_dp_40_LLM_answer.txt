function sortByHeight(arr) {
    let idx = arr.indexOf('h');
    if (idx === -1) return arr;
    let left = [];
    let right = [];
    for (let i = 0; i < arr.length; i++) {
        if (i < idx) left.push(arr[i]);
        else right.push(arr[i]);
    }
    left = left.filter(el => el!== 'h').sort((a, b) => a - b);
    right = right.filter(el => el!== 'h').sort((a, b) => a - b);
    let result = [...left, 'h',...right];
    return result;
}
