    public String[] findLongestChain(String[] words) {
        if (words == null || words.length == 0) return new String[0];
        Arrays.sort(words, (w1, w2) -> Integer.compare(w1.length(), w2.length()));
        List<String> res = new ArrayList<>();
        Map<Integer, Set<Character>> map = new HashMap<>();
        for (int i = 0; i < words.length; i++) {
            int len = words[i].length();
            char c = words[i].charAt(len - 1);
            if (!map.containsKey(len)) {
                map.put(len, new HashSet<>());
            }
            map.get(len).add(c);
        }
        for (int i = 0; i < words.length; i++) {
            int len = words[i].length();
            boolean flag = true;
            for (int j = len - 1; j > 0 && flag; j--) {
                char c = words[i].charAt(j);
                if (map.containsKey(j) &&!map.get(j).contains(c)) {
                    flag = false;
                }
            }
            if (flag) {
                res.add(words[i]);
            }
        }
        return res.toArray(new String[res.size()]);
    }
}

class Solution {
    public String[] findLongestChain(String[] words) {
        if (words == null || words.length == 0) return new String[0];
        Arrays.sort(words, (w1, w2) -> Integer.compare(w1.length(), w2.length()));
        List<String> res = new ArrayList<>();
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < words.length; i++) {
            StringBuilder sb = new StringBuilder();
            stack.push(words[i].charAt(words[i].length() - 1));
            while (!stack.isEmpty()) {
                sb.append(stack.pop());
            }
            if (sb.toString().equals(words[i])) {
                res.add(words[i]);
            } else {
                stack.push(words[i].charAt(words[i].length() - 1));
            }
        }
        return res.toArray(new String[res.size()]);
    }
}
