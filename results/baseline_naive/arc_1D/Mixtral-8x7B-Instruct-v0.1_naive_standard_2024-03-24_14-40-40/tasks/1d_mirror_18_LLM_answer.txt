    let temp = arr[i];

function moveElementToEnd(array, toMove) {
    if (!Array.isArray(array)) return "Not an array";
    if (typeof toMove!== "string" && typeof toMove!== "number") return "toMove is not a string or number";
    for (let i = 0; i < array.length - 1; i++) {
        if (array[i] === toMove && array[i + 1]!== toMove) {
            let lastIndex = findLastIndex(array, toMove);
            swap(array, i, lastIndex);
        }
    }
    return array;
}

function findLastIndex(arr, target) {
    for (let i = arr.length - 1; i >= 0; i--) {
        if (arr[i] === target) return i;
    }
}
