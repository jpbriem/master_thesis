class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        int len = arr.length;
        for (int i = 0; i < len - 1; i++) {
            // find the first different char from index i
            while (i + 1 < len && arr[i].equals(arr[i+1])) i++;

            // check whether there is a same char before current position
            boolean hasSameCharBefore = false;
            for (int j = i-1; j >= 0; j--) {
                if (arr[j].equals(arr[i])) {
                    hasSameCharBefore = true;
                    break;
                }
            }

            // replace all chars between [i, len] with current char or next char
            for (int k = i; k < len; k++) {
                if (!hasSameCharBefore && k > i) {
                    arr[k] = arr[k].equals(arr[i])? arr[i+1] : arr[i];
                } else {
                    arr[k] = arr[k].equals(arr[i])? arr[i] : arr[i+1];
                }
            }
        }

        return arr;
    }
}

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        StringBuilder sb = new StringBuilder();
        for (String s : arr) {
            sb.append(s);
        }

        String str = sb.toString();
        int idx = 0;
        StringBuilder resSb = new StringBuilder();
        while (idx < str.length()) {
            char c = str.charAt(idx);
            int count = 1;
            while (idx + 1 < str.length() && str.charAt(idx + 1) == c) {
                idx++;
                count++;
            }

            if (count > 1) {
                resSb.append(c).append(c);
            } else {
                resSb.append(c);
            }

            idx += count;
        }

        String resStr = resSb.toString();
        String[] result = new String[resStr.length() / 2];
        for (int i = 0; i < result.length; i++) {
            result[i] = resStr.substring(i * 2, i * 2 + 2);
        }

        return result;
    }
}
