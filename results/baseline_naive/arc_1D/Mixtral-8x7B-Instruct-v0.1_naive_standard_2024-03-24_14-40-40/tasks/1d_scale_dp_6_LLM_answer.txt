function removeDuplicateLetters(str) {
    let stack = [];
    const freqMap = {};
    for (let char of str) {
        if (!freqMap[char]) {
            while (stack.length && stack[stack.length - 1] > char && freqMap[stack[stack.length - 1]] >= 1) {
                stack.pop();
            }
            stack.push(char);
            freqMap[char] = 1;
        } else {
            freqMap[char]++;
        }
    }
    return stack.join('');
}
