    let result = [];
    for (let i = 0; i < arr.length; i++) {
        if (!result[result.length - 1] || result[result.length - 1][0]!== arr[i]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    return result.map((el, idx) => el.every(val => val === el[0])? el[0] : el);
}
