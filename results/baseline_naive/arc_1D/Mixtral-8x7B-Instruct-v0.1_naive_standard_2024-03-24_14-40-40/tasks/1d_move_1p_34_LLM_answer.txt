You can use `itertools.groupby()` to group the elements by their value, then iterate over these groups and take the first element of each group if it is an "a", otherwise append all other elements in this group to the result list:

from itertools import groupby

def compress(lst):
    return [next(grp) if grp[0] == 'a' else list(grp) for _, grp in groupby(lst)]

print(compress(['a', 'a', 'a', 'a', 'a', 'a', 'g', 'g', 'g', 'g', 'a', 'a', 'a', 'a', 'a']))
# ['a', 'a', 'a', 'a', 'a', 'a', 'g', 'g', 'g', 'g', 'a', 'a', 'a', 'a', 'a']
print(compress(['a', 'c', 'c', 'c', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a']))
# ['a', 'a', 'c', 'c', 'c', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a', 'a']
