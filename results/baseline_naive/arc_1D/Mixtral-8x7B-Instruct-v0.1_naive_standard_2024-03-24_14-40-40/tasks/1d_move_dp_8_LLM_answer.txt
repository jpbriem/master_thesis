const sortArray = (arr) => {
    let left = [];
    let right = [];
    for(let i=0; i<arr.length; i++){
        if(arr[i] === "a"){
            left.push(arr[i]);
        } else {
            right.unshift(arr[i])
        }
    }
    return [...left,...right];
}
