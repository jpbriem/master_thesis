var arraysEqual = function (arr1, arr2) {
    if (arr1.length!== arr2.length) return false;
    for (let i = 0; i < arr1.length; i++) {
        if (arr1[i]!== arr2[i]) return false;
    }
    return true;
};

function arrayReplace(inputArray) {
    let result = [];
    let currentChar = inputArray[0];
    let count = 1;
    for (let i = 1; i <= inputArray.length; i++) {
        if (i === inputArray.length || inputArray[i - 1]!== inputArray[i]) {
            if (count > 1 && inputArray[i - 1]!== currentChar) {
                let repCount = Math.floor((count + 1) / 2);
                let replacement = Array(repCount + 1).fill(currentChar);
                result = [...result,...replacement,...replaceAll(inputArray.slice(i), currentChar, '')];
            } else {
                result = [...result,...inputArray.slice(i - count, i)];
            }
            if (i < inputArray.length) {
                currentChar = inputArray[i];
                count = 1;
            }
        } else {
            count++;
        }
    }
    return result;
}
