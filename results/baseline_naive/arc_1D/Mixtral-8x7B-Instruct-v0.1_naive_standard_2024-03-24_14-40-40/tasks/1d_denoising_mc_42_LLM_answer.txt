function removeDuplicateLetters(s) {
  const stack = [];
  const countMap = {};
  for (const char of s) {
    countMap[char] = (countMap[char] || 0) + 1;
  }
  let lastNonRepeatingChar = -1;
  for (let i = 0; i < s.length; i++) {
    if (countMap[s[i]] > 0 && lastNonRepeatingChar === -1) {
      lastNonRepeatingChar = i;
    }
    while (stack.length!== 0 && stack[stack.length - 1] > s[i]) {
      if (countMap[stack[stack.length - 1]] > 0) break;
      stack.pop();
    }
    if (lastNonRepeatingChar!== -1 && i >= lastNonRepeatingChar) {
      stack.push(s[lastNonRepeatingChar]);
      countMap[s[lastNonRepeatingChar]]--;
      lastNonRepeatingChar++;
    } else if (i === lastNonRepeatingChar) {
      stack.push(s[i]);
      countMap[s[i]]--;
    }
  }
  return stack.join('');
}
