const removeDuplicateLetters = (s) => {
    const stack = [];
    const countMap = {};
    for (let i = 0; i < s.length; i++) {
        countMap[s[i]] = (countMap[s[i]] || 0) + 1;
    }
    for (let char of s) {
        if (!stack.includes(char)) {
            while (stack.length && stack[stack.length - 1] > char && countMap[stack[stack.length - 1]] > 0) {
                stack.pop();
            }
            if (!stack.includes(char)) {
                stack.push(char);
            }
        }
        countMap[char]--;
    }
    return stack.join('');
};
