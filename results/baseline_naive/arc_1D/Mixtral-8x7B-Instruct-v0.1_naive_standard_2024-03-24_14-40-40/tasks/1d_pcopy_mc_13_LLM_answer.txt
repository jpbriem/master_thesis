function groupAnimals(arr) {
    // sort the array first
    arr.sort();

    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i + 1].charCodeAt()!== arr[i].charCodeAt() + 1) {
            result.push([arr[i]]);
        } else {
            result[result.length - 1].push(arr[i + 1]);
        }
    }

    return [...result, [arr[arr.length - 1]]];
}
