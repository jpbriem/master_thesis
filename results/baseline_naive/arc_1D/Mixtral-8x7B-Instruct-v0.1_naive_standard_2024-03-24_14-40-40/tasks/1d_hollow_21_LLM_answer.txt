def fix(lst):
    i = 0
    in_section = False
    while i < len(lst):
        if lst[i] == 'a':
            if not in_section:
                in_section = True
        else:
            if in_section:
                lst[i] = 'a'
                in_section = False
        i += 1
    return lst
