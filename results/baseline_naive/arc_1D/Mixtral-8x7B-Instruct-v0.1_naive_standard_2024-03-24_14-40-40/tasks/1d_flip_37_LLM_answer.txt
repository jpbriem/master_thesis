function sortByHeight(arr) {
  let idx = arr.findIndex((el, i) => el!== -1); // find index of first non-negative integer
  if (idx === -1) return arr; // if no positive integers, just return the array as is

  const sortedArr = [...arr]; // create a copy to avoid mutating original input
  for (let i = idx; i < sortedArr.length; i++) {
    if (sortedArr[i] === -1) continue; // skip negative numbers

    let j = i - 1;
    while (j >= 0 && sortedArr[j] > sortedArr[i]) {
      [sortedArr[j], sortedArr[j + 1]] = [sortedArr[j + 1], sortedArr[j]];
      j--;
    }
  }
  return sortedArr;
}
