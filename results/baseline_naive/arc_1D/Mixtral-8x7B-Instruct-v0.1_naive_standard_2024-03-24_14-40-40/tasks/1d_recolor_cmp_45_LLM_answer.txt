def fix_list(lst):
    if len(lst) < 2:
        return lst

    i = 0
    while i + 1 < len(lst):
        j = i + 1
        count = 1
        while j < len(lst) and lst[j] == lst[i]:
            j += 1
            count += 1

        # Replace old group with new value
        if count > 4:
            for k in range(i, j - 1):
                lst[k] = 'G'
        elif lst[i] == 'A':
            for k in range(i, j - 1):
                lst[k] = 'A'
        else:
            for k in range(i, j - 1):
                lst[k] = 'I'

        # Add new element to end of list
        lst.append('A')
        i = j

    return lst[:-1]
