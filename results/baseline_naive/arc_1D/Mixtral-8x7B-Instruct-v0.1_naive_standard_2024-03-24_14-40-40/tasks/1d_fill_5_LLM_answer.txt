const compress = (str) => {
    let result = [];
    for(let i = 0; i < str.length; i++) {
        if(!result[result.length - 1] || result[result.length - 1][0]!== str[i]) {
            result.push([str[i], 1]);
        } else {
            result[result.length - 1][1]++;
        }
    }
    return result.map((item) => item[0]).join('') + Array(result.length).fill('').reduce((acc, cur) => acc += [...cur].filter(x => x === '1').length, '')
}
