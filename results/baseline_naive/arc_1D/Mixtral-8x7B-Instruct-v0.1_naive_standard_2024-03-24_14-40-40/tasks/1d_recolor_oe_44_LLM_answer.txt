def convert(arr):
    i = 0
    j = -1
    while True:
        try:
            while arr[i] == 'a':
                i += 1
            if i > j + 2:
                arr[i] = 'g'
            else:
                arr[i] = 'j'
            j = i
            i += 1
        except IndexError:
            break
    return arr
