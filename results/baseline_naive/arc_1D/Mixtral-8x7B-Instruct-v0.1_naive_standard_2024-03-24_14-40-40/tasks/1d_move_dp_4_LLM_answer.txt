function moveAhead(arr) {
  let i = arr.indexOf('a');
  while (i!== -1 && i < arr.length - 1 && arr[i + 1] === 'a') {
    [arr[i], arr[i + 1]] = [arr[i + 1], arr[i]];
    i++;
  }
}

function moveBack(arr) {
  let j = arr.lastIndexOf('a');
  while (j > 0 && j > arr.indexOf('a') && arr[j - 1] === 'a') {
    [arr[j], arr[j - 1]] = [arr[j - 1], arr[j]];
    j--;
  }
}

function arrangeString(str) {
  if (!str || str.length <= 1) return str;
  const arr = [...str].map((el) => el);
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] === 'a') {
      moveAhead(arr.slice(i));
      moveBack(arr.slice(0, i));
    }
  }
  return arr.join('');
}
