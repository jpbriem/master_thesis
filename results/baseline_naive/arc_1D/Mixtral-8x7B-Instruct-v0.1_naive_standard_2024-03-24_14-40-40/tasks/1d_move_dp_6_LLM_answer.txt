const reorderLogFiles = (logs) => {
    let letterLogs = [];
    let digitLogs = [];

    for(let i=0; i<logs.length; i++) {
        if(isNaN(logs[i].split(' ')[1][0])) {
            letterLogs.push(logs[i]);
        } else {
            digitLogs.push(logs[i]);
        }
    }

    letterLogs.sort((logA, logB) => {
        const [identifierA, contentA] = logA.split(' ');
        const [identifierB, contentB] = logB.split(' ');

        return contentA > contentB? 1 : -1;
    });

    return [...letterLogs,...digitLogs];
};
