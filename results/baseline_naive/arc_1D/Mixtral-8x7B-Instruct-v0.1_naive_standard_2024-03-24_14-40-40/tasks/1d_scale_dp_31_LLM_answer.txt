function removeDuplicateLetters(str) {
  const stack = [];
  const freqMap = {};
  for (const char of str) {
    freqMap[char] = (freqMap[char] || 0) + 1;
  }
  let lastUniqueCharIndex = -1;
  for (let i = 0; i < str.length; i++) {
    if (!freqMap[str[i]]) continue;
    while (stack.length && stack[stack.length - 1] > str[i]) {
      const top = stack.pop();
      if (--freqMap[top] === 0) break;
      lastUniqueCharIndex = stack.length;
    }
    stack.push(str[i]);
    if (lastUniqueCharIndex === i - 1) {
      freqMap[str[i]]--;
    }
  }
  return stack.join('');
}
