const replaceWithLetter = (arr) => {
    let freqMap = {};
    for(let i=0;i<arr.length-1;i++){
        if(!freqMap[arr[i]]){
            freqMap[arr[i]] = 1;
        } else {
            freqMap[arr[i]]++;
        }
    }

    let result = [];
    for(let j=0;j<arr.length;j++){
        if(freqMap[arr[j]] === 1){
            result.push('a');
        } else if(freqMap[arr[j]] > 1 && arr[j]!== arr[j+1]){
            result.push('f');
        } else if(freqMap[arr[j]] > 1 && arr[j] === arr[j+1]){
            result.push('c');
        } else if(freqMap[arr[j]] < arr.length - j && arr[j]!== arr[j+1]){
            result.push('j');
        } else {
            result.push('a');
        }
    }
    return result;
}
