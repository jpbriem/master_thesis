const moveHead = (arr) => {
    if (!Array.isArray(arr)) return "Not an array";
    let head = arr[0];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i + 1] === head) continue;
        else arr[i + 1] = head;
        head = arr[i + 1];
    }
    return arr;
}
