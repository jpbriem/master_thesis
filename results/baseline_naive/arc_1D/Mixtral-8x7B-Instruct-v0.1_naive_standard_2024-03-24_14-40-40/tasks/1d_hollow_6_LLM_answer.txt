function freqAlphabets(s) {
  let result = [];
  for (let i = s.length - 1; i >= 0; i--) {
    if (i > 9 && s[i] === "z") {
      result.unshift(String.fromCharCode(84 + parseInt(s.slice(i - 2, i), 10)));
      i -= 2;
    } else if (i > 8 && s[i]!== "#") {
      result.unshift(String.fromCharCode(parseInt(s.slice(i - 2, i), 10) + 55));
      i -= 2;
    } else {
      result.unshift(s[i]);
    }
  }
  return result.join("");
}
