const mergeTwoSortedArrays = (arr1, arr2) => {
    let result = [];
    while(arr1.length && arr2.length){
        if(arr1[0] < arr2[0]){
            result.push(arr1.shift());
        } else {
            result.push(arr2.shift())
        }
    }
    return [...result,...arr1,...arr2];
}

const mergeSort = (arr) => {
    if(arr.length <= 1) return arr;
    const mid = Math.floor(arr.length / 2);
    return mergeTwoSortedArrays(mergeSort(arr.slice(0, mid)), mergeSort(arr.slice(mid)));
}

/*
Time Complexity: O(nlogn), where n is the length of input array
Space Complexity: O(n), for recursive stack and extra space to store merged arrays
