const groupBy = (arr, fn) => {
    const groups = {};
    for(let val of arr){
        const key = fn(val);
        if(!groups[key]){
            groups[key] = [];
        }
        groups[key].push(val);
    }
    return groups;
}

const mergeGroups = (groupA, groupB) => {
    let mergedGroup = [...groupA];
    for(let i=0;i<groupB.length;i++){
        mergedGroup.splice(-1-i, 1,...groupB.slice(i));
    }
    return mergedGroup;
}

const longestConsecutive = function(arr) {
    // edge cases
    if(arr.length === 0 ||!Array.isArray(arr)) return [];

    // convert to object with consecutive values as keys and their lengths as values
    const groupedArr = Object.entries(groupBy(arr, x=>x===arr[0]?0:(x+1)%2)).reduce((acc, [k, v])=>mergeGroups(acc, Array(v.length).fill([parseInt(k)])));

    // find the longest consecutive sequence
    let maxLength = -Infinity;
    let result = [];
    for(let i=0;i<groupedArr.length;i++){
        if(maxLength < groupedArr[i].length){
            maxLength = groupedArr[i].length;
            result = groupedArr[i];
        }
    }

    return result;
};
