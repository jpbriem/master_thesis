function groupAnimals(arr) {
    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i]!== arr[i + 1]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);

    // console.log('result :>> ', result);

    for (let j = 0; j < result.length; j++) {
        const currentGroup = result[j];
        const nextGroup = result[j + 1];
        while (nextGroup && currentGroup[currentGroup.length - 1].charCodeAt() === nextGroup[0].charCodeAt() - 1) {
            currentGroup.push(...nextGroup);
            result.splice(j + 1, 1);
            j--;
            nextGroup = result[j + 1];
        }
    }

    return result.flat();
}
