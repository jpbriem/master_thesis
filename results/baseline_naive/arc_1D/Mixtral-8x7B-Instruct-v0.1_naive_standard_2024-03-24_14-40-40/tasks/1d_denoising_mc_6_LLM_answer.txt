function sameBuckets(arr) {
    let i = 0;
    while (i < arr.length - 1) {
        if (arr[i]!== arr[i + 1]) {
            if (checkBucket(arr, i)) return false;
        }
        i++;
    }
    return true;
}

const checkBucket = (arr, index) => {
    const bucketStart = index;
    const bucketEnd = findBucketEnd(arr, bucketStart);
    for (let j = bucketStart; j <= bucketEnd; j++) {
        if (arr[bucketStart] === arr[j + 1]) return true;
    }
    return false;
};

const findBucketEnd = (arr, startIndex) => {
    let endIndex = startIndex;
    while (endIndex < arr.length && arr[startIndex] === arr[endIndex + 1]) {
        endIndex++;
    }
    return endIndex;
};
