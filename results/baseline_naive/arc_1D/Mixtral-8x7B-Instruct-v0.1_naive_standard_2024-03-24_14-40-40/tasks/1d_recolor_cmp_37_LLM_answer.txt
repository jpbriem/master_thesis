def fix(arr):
    i = 0 #index of last "a"
    j = 0 #current index
    while j < len(arr):
        if arr[i] == 'a':
            i += 1
        elif arr[i] == 'f':
            k = i
            while k <= j:
                arr[k] = 'j'
                k += 1
        j += 1
    return arr
