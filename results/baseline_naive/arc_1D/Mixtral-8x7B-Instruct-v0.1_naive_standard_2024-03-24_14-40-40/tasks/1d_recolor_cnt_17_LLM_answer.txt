function replaceAll(str, find, replace) {
    return str.split(find).join(replace);
}

function letterChanges(str) {
    let vowels = ['a', 'e', 'i', 'o', 'u'];
    for (let i = 0; i < str.length; i++) {
        if (vowels.includes(str[i])) {
            if (str[i] === 'e') {
                str = replaceAll(str, str[i], 'i');
            } else if (str[i] === 'a') {
                str = replaceAll(str, str[i], 'c');
            } else if (str[i] === 'i') {
                str = replaceAll(str, str[i], 'f');
            } else if (str[i] === 'o') {
                str = replaceAll(str, str[i], 'p');
            } else if (str[i] === 'u') {
                str = replaceAll(str, str[i], 'y');
            }
        }
    }
    // convert to uppercase
    str = str.toUpperCase();
    // change the first character of each word to lowercase
    str = str.charAt(0).toLowerCase() + str.slice(1);
    return str;
}
