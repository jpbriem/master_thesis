function groupAnimals(arr) {
    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i]!== arr[i + 1]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);

    // sorting the array of arrays based on alphabetical order
    result.sort((a, b) => a[0].charCodeAt() > b[0].charCodeAt());

    // concatenating all elements in each sub-array to form one string
    return result.map(e => e.join(''));
}
