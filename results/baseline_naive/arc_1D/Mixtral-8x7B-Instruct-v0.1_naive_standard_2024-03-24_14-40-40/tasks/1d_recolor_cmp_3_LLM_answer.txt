function group(arr) {
    let result = [];
    for (let i = 0; i < arr.length; i++) {
        if (!result[result.length - 1] || result[result.length - 1][0]!== arr[i]) {
            result.push([arr[i], 1]);
        } else {
            result[result.length - 1][1]++;
        }
    }
    return result;
}

function replaceG(arr, index) {
    while (index < arr.length && arr[index][0] === 'g') {
        index++;
    }
    return [arr[index][0], Math.max(...arr.slice(index).map((el) => el[1]))];
}

function solution(strArr) {
    let groups = group(strArr);

            let replacement = replaceG(newGroups, i);

}
