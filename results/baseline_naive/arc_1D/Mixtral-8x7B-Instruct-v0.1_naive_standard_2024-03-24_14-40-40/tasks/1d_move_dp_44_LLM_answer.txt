const moveElementToEnd = (arr, toMove) => {
    let left = 0;
    let right = arr.length - 1;

    while(left < right){
        if(arr[right] === toMove && arr[left]!== toMove){
            [arr[left], arr[right]] = [arr[right], arr[left]];
            left++;
            right--;
        } else if(arr[right]!== toMove){
            right--;
        } else{
            right--;
        }
    }

    return arr;
}
