class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        int i = 0;
        while (i < arr.length - 1 && arr[i].equals("a")) {
            i++;
        }

        for (int j = i + 1; j < arr.length; j++) {
            if (!arr[j].equals("a") &&!arr[j].equals("d")) continue;
            if (arr[j].equals("a")) {
                arr[j] = "i";
            } else {
                arr[j] = "g";
            }
        }

        for (int k = i - 1; k >= 0; k--) {
            arr[k] = "a";
        }

        return arr;
    }
}

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        boolean isA = true;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < arr.length; i++) {
            if (isA && arr[i].equals("a")) {
                sb.append("a");
            } else if (isA && arr[i].equals("d")) {
                sb.append("g");
                isA = false;
            } else if (!isA && arr[i].equals("a")) {
                sb.append("i");
            } else if (!isA && arr[i].equals("d")) {
                sb.append("g");
            }
        }

        String str = sb.toString();
        String[] res = new String[str.length()];
        for (int i = 0; i < res.length; i++) {
            res[i] = str.charAt(i) + "";
        }

        return res;
    }
}
