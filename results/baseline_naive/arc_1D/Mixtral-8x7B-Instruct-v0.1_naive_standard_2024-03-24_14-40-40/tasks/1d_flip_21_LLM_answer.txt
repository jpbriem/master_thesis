const sortByHeight = (arr) => {
    const sortedArr = [...arr];
    return sortedArr.sort((a, b) => {
        if(a === b){
            return 0;
        } else if(!isNaN(parseInt(a)) &&!isNaN(parseInt(b))) {
            return parseInt(a) - parseInt(b);
        } else if (!isNaN(parseInt(a))) {
            return -1;
        } else {
            return 1;
        }
    });
}
