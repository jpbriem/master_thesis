function removeDuplicateLetters(str) {
    let stack = [];
    const freqMap = {};
    for (const char of str) {
        if (!freqMap[char]) {
            freqMap[char] = true;
        }
    }
    for (const char of str) {
        while (stack.length && stack[stack.length - 1] > char && freqMap[stack[stack.length - 1]]) {
            stack.pop();
        }
        if (!freqMap[char]) continue;
        stack.push(char);
    }
    return stack.join('');
}
