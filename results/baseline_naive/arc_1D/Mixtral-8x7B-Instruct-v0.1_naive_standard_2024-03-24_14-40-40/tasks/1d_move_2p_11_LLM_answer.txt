def fix(s):
    i = 0
    while i < len(s)-1:
        j = i+1
        while j < len(s) and s[j] == s[i]:
            j += 1
        if j-i > 1 and s[i]!= 'a':
            s[i:j] = ['a']*(j-i)
        i = j
    return s
