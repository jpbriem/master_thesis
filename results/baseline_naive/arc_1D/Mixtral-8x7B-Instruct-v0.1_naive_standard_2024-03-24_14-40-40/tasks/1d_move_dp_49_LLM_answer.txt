function sortArray(arr) {
  let left = [];
  let right = [];
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] === "a") {
      left.push(arr[i]);
    } else if (arr[i]!== "a" && arr[i]!== "i") {
      right.unshift(arr[i]);
    } else if (arr[i] === "i") {
      left.push("i");
    }
  }
  return [...left,...right];
}
