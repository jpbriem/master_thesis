function removeDuplicateLetters(str) {
    let stack = [];
    const freqMap = {};
    for (const char of str) {
        if (!freqMap[char]) {
            while (stack.length && stack[stack.length - 1] > char && freqMap[stack[stack.length - 1]] >= str.lastIndexOf(stack[stack.length - 1])) {
                stack.pop();
            }
            stack.push(char);
            freqMap[char] = true;
        }
    }
    return stack.join('');
}
