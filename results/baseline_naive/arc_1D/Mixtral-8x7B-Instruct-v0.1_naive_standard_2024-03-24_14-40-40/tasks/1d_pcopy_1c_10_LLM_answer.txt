            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);
    return result.map((el, idx) => {
        if (idx === 0 || el[0] > result[idx - 1][0]) {
            return el;
        } else {
            let temp = [...el];
            while (temp[0] <= result[idx - 1][0]) {
                temp.unshift(result[idx - 1].pop());
            }
            return temp;
        }
    });
