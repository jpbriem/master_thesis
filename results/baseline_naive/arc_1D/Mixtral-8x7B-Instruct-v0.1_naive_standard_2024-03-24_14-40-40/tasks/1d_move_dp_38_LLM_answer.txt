function sortByHeight(arr) {
    let idx = arr.findIndex((el, i) => el!== -1); // find the first non-negative number index
    if (idx === -1) return arr;
    for (let i = idx + 1; i < arr.length; i++) {
        if (arr[i] > -1 && arr[idx] < arr[i]) {
            [arr[idx], arr[i]] = [arr[i], arr[idx]];
            idx++;
            while (arr[idx] === -1) idx++;
        } else if (arr[i] === -1 && arr[idx] > -1) {
            [arr[idx], arr[i]] = [arr[i], arr[idx]];
            idx++;
            while (arr[idx] === -1) idx++;
        }
    }
    return arr;
}
