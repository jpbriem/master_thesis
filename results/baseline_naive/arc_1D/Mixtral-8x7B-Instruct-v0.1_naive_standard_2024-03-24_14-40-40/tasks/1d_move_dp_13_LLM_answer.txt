function sortArray(arr) {
    let left = [];
    let right = [];
    for (let i = 0; i < arr.length; i++) {
        if (i === 0 || arr[i]!== arr[i - 1]) {
            left.push(arr[i]);
        } else {
            right.unshift(arr[i]);
        }
    }
    return [...left,...right];
}
