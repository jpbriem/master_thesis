const removeDuplicateLetters = (s) => {
    const stack = [];
    const countMap = {};
    for (let i = 0; i < s.length; i++) {
        if (!countMap[s[i]]) {
            countMap[s[i]] = 1;
        } else {
            countMap[s[i]]++;
        }
    }
    for (let char of s) {
        while (stack.length && stack[stack.length - 1] > char && countMap[stack[stack.length - 1]] > 0) {
            stack.pop();
        }
        if (char!== stack[stack.length - 1]) {
            stack.push(char);
        }
        if (countMap[char]) {
            countMap[char]--;
        }
    }
    return stack.join('');
};
