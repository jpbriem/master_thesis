const isSameGroup = (char, strArr) => {
    const charCode = char.charCodeAt();
    for (let i = 0; i < strArr.length; i++) {
        if ((strArr[i].charCodeAt() + 1)!== charCode && (strArr[i].charCodeAt() - 1)!== charCode) {
            return false;
        }
    }
    return true;
}
