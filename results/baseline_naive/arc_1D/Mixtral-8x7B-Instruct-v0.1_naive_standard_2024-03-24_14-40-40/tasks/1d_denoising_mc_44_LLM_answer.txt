const removeDuplicateLetters = (s) => {
    let lastOccurrence = new Array(26).fill(-1); // a -> 0, b -> 1... z -> 25
    let countMap = new Array(26).fill(0);
    for (let i = 0; i < s.length; i++) {
        countMap[s[i].charCodeAt() - 'a'.charCodeAt()]++;
    }
    let stack = [];
    for (let i = 0; i < s.length; i++) {
        const charCode = s[i].charCodeAt() - 'a'.charCodeAt();
        if (lastOccurrence[charCode] === -1) {
            lastOccurrence[charCode] = i;
        }
        countMap[charCode]--;
        while (stack.length && stack[stack.length - 1] > s[i] && lastOccurrence[stack[stack.length - 1].charCodeAt() - 'a'.charCodeAt()] > i) {
            stack.pop();
        }
        stack.push(s[i]);
    }
    return stack.join('');
