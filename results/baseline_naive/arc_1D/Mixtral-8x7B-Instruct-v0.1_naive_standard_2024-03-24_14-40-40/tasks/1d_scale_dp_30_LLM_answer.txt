    public String[] rearrangeArray(String[] nums) {
        int len = nums.length;
        if (len < 2) return nums;

        List<String> list = new ArrayList<>();
        for (int i = 0, j = 0; i < len && j < len; ) {
            while (i < len && "a".equals(nums[i])) i++;
            while (j < len &&!"a".equals(nums[j])) j++;
            if (i < len) list.add(nums[i++]);
            if (j < len) list.add(nums[j++]);
        }

        return list.toArray(new String[list.size()]);
    }
}

class Solution {
    public String[] rearrangeArray(String[] nums) {
        int len = nums.length;
        if (len < 2) return nums;

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; i++) {
            if ("a".equals(nums[i])) {
                sb.append("a");
            } else {
                for (int j = 0; j < i; j++) {
                    if (!"a".equals(nums[j])) {
                        swap(nums, i, j);
                        break;
                    }
                }
                sb.append("d");
            }
        }

        return sb.toString().split("");
    }

    private void swap(String[] arr, int a, int b) {
        String temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
}
