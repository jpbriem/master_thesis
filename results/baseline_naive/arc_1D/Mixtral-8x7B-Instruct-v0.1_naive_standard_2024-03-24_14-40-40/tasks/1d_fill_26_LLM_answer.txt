function groupAnimals(arr) {
    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i]!== arr[i + 1]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);

    // console.log('result :>> ', result);

    while (result.some((el, idx) => el.join('') > result[idx + 1]?.join(''))) {
        const firstEl = result.shift();
        const lastEl = result.pop();
        result.splice(1, 0, [...firstEl,...lastEl]);
    }

    return result.flatMap(el => el);
}
