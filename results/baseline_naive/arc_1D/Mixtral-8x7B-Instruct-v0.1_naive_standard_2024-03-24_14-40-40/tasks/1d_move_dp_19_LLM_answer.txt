function sortByFrequency(arr) {
    let freq = {}; // frequency object to store the count of each element in array.
    for (let i = 0; i < arr.length; i++) {
        if (!freq[arr[i]]) {
            freq[arr[i]] = 1;
        } else {
            freq[arr[i]]++;
        }
    }

    return arr.sort((a, b) => {
        if (freq[a] === freq[b]) {
            return a > b? -1 : 1;
        } else {
            return freq[a] > freq[b]? -1 : 1;
        }
    });
}
