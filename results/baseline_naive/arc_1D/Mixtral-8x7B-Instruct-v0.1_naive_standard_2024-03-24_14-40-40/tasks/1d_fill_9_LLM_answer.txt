const groupBy = (arr, fn) => {
    const groups = {};
    for(let i=0;i<arr.length;i++) {
        const key = fn(arr[i]);
        if(!groups[key]) groups[key] = [];
        groups[key].push(arr[i]);
    }
    return groups;
}

const replaceWithGroup = (str, arr) => str.split('').map((char) => arr[char][arr[char].length-1]).join('');

const expandGroups = (str) => {
    let currentChar = str[0];
    let result = [currentChar];
    for(let i=1;i<str.length;i++) {
        if(str[i] === currentChar) continue;
        else {
            currentChar = str[i];
            result.push(currentChar);
        }
    }
    return result;
}

const solution = (str) => {
    const charToIndexMap = groupBy(expandGroups(str), x=>x);
    return replaceWithGroup(str, charToIndexMap);
};
