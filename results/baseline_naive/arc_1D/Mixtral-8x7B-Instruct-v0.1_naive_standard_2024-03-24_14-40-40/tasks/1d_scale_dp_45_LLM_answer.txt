function replaceWithLongest(arr) {
    let longest = '';
    for (let i = 0; i < arr.length; i++) {
        if (i === 0 || arr[i]!== arr[i - 1]) {
            const current = arr[i];
            let count = 1;
            while (count + i < arr.length && arr[i + count] === current) {
                count++;
            }
            if (count > longest.length) {
                longest = current.repeat(count);
            }
        }
    }
    return arr.map((el, idx) => el === longest[0]? longest : el);
}
