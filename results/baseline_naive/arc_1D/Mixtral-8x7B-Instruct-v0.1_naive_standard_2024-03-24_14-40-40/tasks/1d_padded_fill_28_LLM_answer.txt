    let result = '';
    for(let i=0;i<s.length;i++){
        if(i+2 < s.length && s[i+2] === '#'){
            result += String.fromCharCode((parseInt(s.slice(i, i+2))-1)+97);
            i+=2;
        } else {
            result += String.fromCharCode((s[i]-1)+97);
        }
    }
    return result;
}
