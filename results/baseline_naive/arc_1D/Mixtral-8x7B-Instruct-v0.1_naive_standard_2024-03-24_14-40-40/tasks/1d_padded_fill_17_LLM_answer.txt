const groupBy = (arr, fn) => {
    const groups = {};
    for(let i=0;i<arr.length;i++) {
        const key = fn(arr[i]);
        if(!groups[key]) groups[key] = [];
        groups[key].push(arr[i]);
    }
    return groups;
}

const mergeAdjacentGroups = (groups) => {
    let result = [];
    Object.keys(groups).forEach((key)=>{
        if(result.length === 0){
            result.push(groups[key][0]);
        } else {
            const lastItemInResult = result[result.length-1];
            const lastItemGroupLength = groups[key].length;
            if(lastItemInResult === groups[key][0]) {
                result[result.length - 1] += groups[key][lastItemGroupLength - 1];
            } else {
                result.push(groups[key][0]);
            }
        }
    });
    return result;
}

const solution = (str) => {
    const arr = str.split('');
    const groupedArr = groupBy(arr, (item) => item);
    return mergeAdjacentGroups(groupedArr);
};
