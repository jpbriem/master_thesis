function fixBrackets(arr) {
    let left = [];
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === "(" || arr[i] === "[") {
            left.push(arr[i]);
        } else if (arr[i] === ")") {
            while (left.pop()!== "(");
        } else if (arr[i] === "]") {
            while (left.pop()!== "[");
        }
    }
    return [...arr].map((el, idx) => el === "?"? left.pop() : el).join("");
}
