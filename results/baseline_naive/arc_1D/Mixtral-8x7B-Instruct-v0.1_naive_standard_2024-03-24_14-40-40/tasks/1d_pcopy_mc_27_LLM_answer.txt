function replaceAll(str, find, replace) {
    return str.split(find).join(replace);
}

var replaceNonCopiedChars = function (s) {
    let arr = s.split('');
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i]!== arr[i + 1]) {
            arr[i + 1] = arr[i];
        }
    }
    return arr.join('');
};
