function sortByFrequency(arr) {
    let freq = {}; // object to store frequency of each element in the array.
    for (let i=0;i<arr.length;i++) {
        if (!freq[arr[i]]) {
            freq[arr[i]] = 1;
        } else {
            freq[arr[i]]++;
        }
    }

    arr.sort((a, b) => {
        if (freq[a] === freq[b]) return a > b? -1 : 1;
        return freq[a] > freq[b]? -1 : 1;
    });
}
