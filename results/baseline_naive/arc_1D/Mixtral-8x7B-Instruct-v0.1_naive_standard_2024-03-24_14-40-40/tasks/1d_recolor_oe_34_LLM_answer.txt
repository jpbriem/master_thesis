def replace(letters):

    prev = None
    i = 0
    while i < len(letters):
        c = letters[i]
        if stack and stack[-1][0] == c:
            # same letter, increase counter
            stack[-1] = (c, stack[-1][1]+1)
        else:
            # different letter or empty stack, push new entry
            stack.append((c, 1))
        if prev!= c:
            # only update sequence if we are changing letter
            j = i - stack[-1][1]
            if j >= 0:
                letters[j:i+1] = [stack[-1][0]] * stack[-1][1]
        i += 1
        prev = c
    return letters
