function groupAnimals(arr) {
    let result = [];
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i + 1]!== arr[i]) {
            result.push([arr[i]])
        } else {
            result[result.length - 1].push(arr[i]);
        }
    }
    result.push([arr[arr.length - 1]]);

    // console.log('result :>> ', result);

    let finalResult = [];
    while (result.length > 0) {
        const firstElm = result.shift();
        let nextElm = null;
        if (result.length > 0) {
            nextElm = result[0];
        }
        if (!nextElm || nextElm[0][0].charCodeAt() === firstElm[0].charCodeAt() + 1) {
            firstElm.push(...nextElm);
            result.shift();
        }
        finalResult.push(firstElm);
    }

    return finalResult.map((elm) => elm.join(''));
}
