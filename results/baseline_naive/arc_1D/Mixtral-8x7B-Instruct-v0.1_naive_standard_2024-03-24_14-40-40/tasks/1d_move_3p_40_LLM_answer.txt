// Solution 1: Two pointers, O(n) time and space.
class Solution {
    public String[] sortPeople(String[] names, int[] heights) {
        // Create a new array to store the sorted people.
        String[] result = new String[names.length];

        // Use two pointers to iterate through the input arrays.
        for (int i = 0, j = names.length - 1; i <= j; ) {
            if (heights[i] < heights[j]) {
                // If the height of person at index `i` is smaller than that of person at index `j`, swap them.
                result[--j] = names[i];
                result[i++] = names[j];
            } else {
                // Otherwise, move forward on both pointers.
                ++i;
            }
        }

        return result;
    }
}
