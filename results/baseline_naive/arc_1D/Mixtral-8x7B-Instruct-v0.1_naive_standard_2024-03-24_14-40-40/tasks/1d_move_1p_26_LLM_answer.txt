def compress(arr):
    n = len(arr)
    res = []
    i = 0 # Last seen index of "a"
    while i < n:
        j = i
        while j < n and arr[j] == arr[i]:
            j += 1
        res.append(arr[i])
        if j - i > 1:
            res.append("x") * (j - i - 1)
        i = j
    return res
