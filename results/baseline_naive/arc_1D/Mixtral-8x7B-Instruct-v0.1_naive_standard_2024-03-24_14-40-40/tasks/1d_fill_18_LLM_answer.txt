var arraysEqual = function (a, b) {
    if (a === b) return true;
    if (a == null || b == null) return false;
    if (a.length!= b.length) return false;

    for (var i = 0; i < a.length; ++i) {
        if (a[i]!== b[i]) return false;
    }
    return true;
};

function arrayReplace(arr) {
    var result = [];
    var lastChar = arr[0];
    result.push(lastChar);

    for (var i = 1; i < arr.length; i++) {
        if (arr[i]!== lastChar) {
            lastChar = arr[i];
            result.push(lastChar);
        }
    }

    return result;
}

function firstNonRepeatingCharacter(inputArray) {
    // your code here
    if (!inputArray || inputArray.length <= 1) {
        return inputArray;
    }

    var replacedArr = arrayReplace(inputArray);
    var output = replaceAll(inputArray.join(''), replacedArr.join(''), '');

    return output.split('').map((val) => {
        return parseInt(val);
    });
}
