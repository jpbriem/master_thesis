def maxSubArraySum(arr):
    n = len(arr)
    curr_sum = arr[0]
    prev_sum = 0
    for i in range(1, n):
        if curr_sum >= prev_sum:
            curr_sum += arr[i]
            prev_sum = curr_sum
        else:
            curr_sum = arr[i]
            prev_sum = curr_sum
    return max(curr_sum, prev_sum)
