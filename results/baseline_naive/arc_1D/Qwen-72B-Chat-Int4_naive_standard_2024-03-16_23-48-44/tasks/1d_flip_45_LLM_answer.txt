def sort_matrix(matrix):
    n = len(matrix)
    for i in range(n):
        max_val = matrix[i][0]
        max_idx = 0
        for j in range(1, n):
            if matrix[i][j] > max_val:
                max_val = matrix[i][j]
                max_idx = j
        # Swap the values at the current position with the previous maximum value
        matrix[i][max_idx], matrix[i][i] = matrix[i][i], matrix[i][max_idx]
