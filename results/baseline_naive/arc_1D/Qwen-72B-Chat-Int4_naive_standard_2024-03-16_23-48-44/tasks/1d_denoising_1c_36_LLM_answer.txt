def remove_surrounded_regions(grid):
    rows = len(grid)
    cols = len(grid[0])

    # Iterate through each cell in the grid

            if grid[i][j] == 0:
                continue

            # Check if the current cell is surrounded by zeros on both sides
            left_zero = False
            right_zero = False

            down_zero = False

            if j > 0 and grid[i][j-1] == 0:
                left_zero = True

                right_zero = True

                up_zero = True
            if i < rows-1 and grid[i+1][j] == 0:
