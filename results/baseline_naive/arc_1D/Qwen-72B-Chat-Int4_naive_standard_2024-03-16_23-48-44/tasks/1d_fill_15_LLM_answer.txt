def expand_zeros(input_list):
    output_list = []
    for i in range(len(input_list)):
        if input_list[i] != 0:
            output_list.append(input_list[i])
            j = i + 1
            while j < len(input_list) and input_list[j] == 0:
                j += 1
            for k in range(i+1, j):
                output_list[k-i] = input_list[i]
    return output_list
