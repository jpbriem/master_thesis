def replace_consecutive_zeros(arr):
    n = len(arr)
    start = end = 0
    
    for i in range(n):
        if arr[i] == 0:
            # Check if there are any consecutive non-zero numbers before the current zero
            if i > start and arr[i-1] != 0:
                # Replace the consecutive non-zero numbers with zeros
                for j in range(start, i):
                    arr[j] = 0
                start = i + 1
        
        elif i > start:
            # Update the end pointer to the current position
            end = i
            
    # If there are any remaining consecutive non-zero numbers at the end of the array,
    # replace them with zeros
