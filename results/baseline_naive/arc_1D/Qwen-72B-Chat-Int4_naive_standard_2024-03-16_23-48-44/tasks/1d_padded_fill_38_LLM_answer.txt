def compress_input(input_list):
    output = []
    for i in range(len(input_list)):
        if input_list[i] != 0:
            count = 1
            j = i + 1
            while j < len(input_list) and input_list[j] == input_list[i]:
                count += 1
                j += 1
            output.extend([input_list[i]] * count)

            output.append(0)
