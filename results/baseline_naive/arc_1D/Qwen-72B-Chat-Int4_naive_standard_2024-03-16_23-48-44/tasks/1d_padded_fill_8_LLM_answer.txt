def compress(input_list):
    output = []
    prev = 0
    
    for i in range(len(input_list)):
        if input_list[i] != 0:
            if prev == 0 or input_list[i] != prev + 1:
                output.append(input_list[i])
                prev = input_list[i]
            else:
                prev += 1
        elif len(output) > 0 and output[-1] == 0:
            output.pop()
    
    return output
