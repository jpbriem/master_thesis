def expand_zeros(input_list):
    output_list = []
    prev_non_zero = False
    
    for i in range(len(input_list)):
        if input_list[i] != 0:
            prev_non_zero = True
        elif prev_non_zero:
            output_list.extend([1] * (i - len(output_list)))
            prev_non_zero = False
            
        output_list.append(input_list[i])
    
    if prev_non_zero:
        output_list.extend([1] * (len(input_list) - len(output_list)))
        
    return output_list
