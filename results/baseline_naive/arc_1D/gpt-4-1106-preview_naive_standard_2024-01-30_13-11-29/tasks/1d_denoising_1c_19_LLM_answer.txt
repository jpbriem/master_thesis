def process_list(input_list):
    output_list = input_list.copy()
    replace_char = 'a'
    count = 0
    for i in range(len(input_list)):
        if input_list[i] != replace_char:
            count += 1
        else:
            if count > 0:
                output_list[i-count:i] = [replace_char] * count
                count = 0
    if count > 0:  # Handle case where the list ends with characters to be replaced
        output_list[-count:] = [replace_char] * count
    return output_list
