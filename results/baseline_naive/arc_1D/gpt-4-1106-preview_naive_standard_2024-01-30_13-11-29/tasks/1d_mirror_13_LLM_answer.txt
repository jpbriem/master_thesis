def transform_sequence(input_list):

    main_char = input_list[0]
    # Find the index of the unique character
    unique_char_index = next((i for i, ch in enumerate(input_list) if ch != main_char), -1)
    # If no unique character is found, or it's at the end, return the input as is
    if unique_char_index == -1 or unique_char_index == len(input_list) - 1:
        return input_list
    # Find the index where the main character sequence resumes
    resume_index = next((i for i, ch in enumerate(input_list[unique_char_index:], start=unique_char_index) if ch == main_char), -1)
    # Split the list into three parts
    part1 = input_list[:unique_char_index]
    part2 = input_list[unique_char_index:resume_index]
