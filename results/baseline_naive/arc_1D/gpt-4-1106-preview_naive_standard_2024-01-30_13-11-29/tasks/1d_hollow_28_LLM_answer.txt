def replace_third_with_a(input_list):
    count_dict = {}
    output_list = []

    for item in input_list:
        if item not in count_dict:
            count_dict[item] = 1
        else:
            count_dict[item] += 1

        if count_dict[item] == 3:
            output_list.append('a')
            count_dict[item] = 0
