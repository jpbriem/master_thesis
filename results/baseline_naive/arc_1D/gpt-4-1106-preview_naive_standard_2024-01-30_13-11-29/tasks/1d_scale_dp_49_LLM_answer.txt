def maximize_runs(input_list):
    if not input_list:  # Handle empty list
        return []

    output_list = []
    current_char = input_list[0]
    current_count = 1
    max_count = 0

    # Iterate through the input list
    for i in range(1, len(input_list)):
        if input_list[i] == current_char:
            # If the same character, increment the count
            current_count += 1
        else:
            # If a different character, update the max_count if necessary and reset the count
            max_count = max(max_count, current_count)
            output_list.extend([current_char] * max_count)
            current_char = input_list[i]
            current_count = 1

    # Update the max_count for the last character
    max_count = max(max_count, current_count)
    # Add the last character with the max_count
    output_list.extend([current_char] * max_count)

    return output_list
