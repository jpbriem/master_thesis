def group_letters(input_list):
    output_list = []
    current_letter = input_list[0]
    count = 0

    for letter in input_list:
        if letter == current_letter:
            # Increment the count for the current letter group
            count += 1
        else:
            # Append the current letter group to the output list
            output_list.extend([current_letter] * count)
            # Start a new letter group
            current_letter = letter
            count = 1

    # Don't forget to append the last letter group
    output_list.extend([current_letter] * count)

    return output_list
