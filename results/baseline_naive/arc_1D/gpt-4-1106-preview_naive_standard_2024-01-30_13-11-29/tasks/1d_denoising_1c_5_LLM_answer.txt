def move_most_frequent_to_end(input_list):
    # Count the frequency of each character except 'a'
    frequency = {}
    for char in input_list:
        if char != 'a':
            frequency[char] = frequency.get(char, 0) + 1

    # Find the character with the highest frequency (except 'a')
    most_frequent_char = None
    max_frequency = 0
    for char, freq in frequency.items():
        if freq > max_frequency:
            most_frequent_char = char
            max_frequency = freq

    # Reorder the list: characters other than the most frequent one come first,
    # followed by all occurrences of the most frequent character
    reordered_list = [char for char in input_list if char != most_frequent_char] + \
                     [most_frequent_char] * max_frequency if most_frequent_char else input_list

    return reordered_list
