def expand_non_a(input_list):
    length = len(input_list)
    result = input_list.copy()  # Make a copy to work on
    for i in range(length):
        if input_list[i] != 'a':
            # Expand the non-'a' character to the left and right
            left = i
            right = i
            while left > 0 and result[left - 1] == 'a':
                left -= 1
            while right < length - 1 and result[right + 1] == 'a':
                right += 1
            for j in range(left, right + 1):
                result[j] = input_list[i]
    return result
