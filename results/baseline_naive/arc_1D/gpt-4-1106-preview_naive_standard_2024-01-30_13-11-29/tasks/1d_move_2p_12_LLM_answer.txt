```python
from collections import Counter

def sort_by_frequency_preserve_order(lst):
    # Count the frequency of each element
    freq = Counter(lst)
    
    # Perform a stable sort based on frequency
    lst.sort(key=lambda x: (-freq[x], lst.index(x)))
    return lst
```
