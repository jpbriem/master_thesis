def replace_h_with_d_or_f(input_list):
    # This function will replace 'h' with 'd' if preceded by 'a', or with 'f' if followed by 'a'
    output = input_list.copy()  # Create a copy of the input list to modify
    length = len(input_list)

    for i in range(length):
        if input_list[i] == 'h':
            # Look ahead for an 'a'
            has_a_after = any(input_list[j] == 'a' for j in range(i + 1, length))
            # Look behind for an 'a'
            has_a_before = any(input_list[j] == 'a' for j in range(0, i))

            if has_a_before:
                output[i] = 'd'
            if has_a_after:
                output[i] = 'f'
            # If 'h' is both preceded and followed by 'a', it will end up as 'f'

    return output
