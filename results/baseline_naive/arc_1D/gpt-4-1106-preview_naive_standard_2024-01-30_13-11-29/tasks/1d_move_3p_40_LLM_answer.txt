def rearrange_characters(input_list):
    from collections import Counter
    
    # Count the frequency of each character
    char_count = Counter(input_list)
    
    # Find the least frequent character
    least_freq_char = min(char_count, key=char_count.get)
    
    # Separate the least frequent character from others
    other_chars = [char for char in input_list if char != least_freq_char]
    least_freq_chars = [char for char in input_list if char == least_freq_char]
    
    # Place the last occurrence of the least frequent character at the beginning
    output_list = [least_freq_chars[-1]] + other_chars
    
    # Place the first occurrence of the least frequent character after the most frequent characters
    output_list += least_freq_chars[:-1]
    
    return output_list
