class Solution {
public:
    string removeDuplicateLetters(string s) {
        vector<int> count(26);
        for (char c : s) ++count[c - 'a'];
        stack<char> st;
        for (char c : s) {
            --count[c - 'a'];
            while (!st.empty() && st.top() > c && count[st.top() - 'a']) st.pop();
            if (count[c - 'a']) st.push(c);
        }
        string res = "";
        while (!st.empty()) {
            res += st.top(), st.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
