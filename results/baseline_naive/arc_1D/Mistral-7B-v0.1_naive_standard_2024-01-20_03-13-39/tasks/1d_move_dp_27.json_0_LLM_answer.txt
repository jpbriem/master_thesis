    def rearrangeBarcodes(self, barcodes):
        """
        :type barcodes: List[int]
        :rtype: List[int]

        d = {}

                d[barcodes[i]] += [i]
            else:
                d[barcodes[i]] = [i]
        res = []
        while True:
