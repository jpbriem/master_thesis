    public String[] rearrangeString(String s) {
        int n = s.length();
        if (n == 0 || n == 1) return new String[]{s};
        char[] chars = s.toCharArray();
        Map<Character, Integer> map = new HashMap<>();

            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        List<Integer> list = new ArrayList<>(map.values());
        Collections.sort(list);
        int max = list.get(list.size() - 1), min = list.get(0);
        int mid = (max + min) / 2;
        int count = 0;
        while (count < n && max > 0) {

                if (list.get(i) <= mid) {
                    chars[count++] = map.keySet().iterator().next();
                    map.remove(map.keySet().iterator().next());
                    --max;
                    break;
                } else {
                    --min;
                }
            }
        }
        while (count < n) {

                if (list.get(i) >= mid) {
                    chars[count++] = map.keySet().iterator().next();
                    map.remove(map.keySet().iterator().next());
                    --min;
                    break;

                }
            }
        }
        return Arrays.stream(chars).collect(Collectors.joining(""));
