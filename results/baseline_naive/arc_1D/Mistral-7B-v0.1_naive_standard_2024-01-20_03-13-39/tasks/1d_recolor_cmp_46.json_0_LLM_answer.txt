public class Solution {
    public String[] reorderLogFiles(String[] logs) {
        List<String> letterList = new ArrayList<>();
        List<String> digitList = new ArrayList<>();
        for (int i=0;i<logs.length;i++) {
            if (logs[i].split(" ")[1].matches("[0-9]+")) {
                digitList.add(logs[i]);
            } else {
                letterList.add(logs[i]);
            }
        }
        Collections.sort(letterList, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                String[] s1 = o1.split(" ");
                String[] s2 = o2.split(" ");
                // sort by second part first
                if (!s1[1].equals(s2[1])) {
                    return s1[1].compareTo(s2[1]);
                } else {
                    // then sort by first part
                    return s1[0].compareTo(s2[0]);
                }
            }
        });
        letterList.addAll(digitList);
        return letterList.toArray(new String[letterList.size()]);
    }
}
