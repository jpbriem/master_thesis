    string removeDuplicates(string s) {
        stack<char> st;
        for (int i = 0; i < s.size(); ++i) {
            if (!st.empty() && st.top() == s[i]) {
                st.pop();
            } else {
                st.push(s[i]);
            }
        }
        string res;
        while(!st.empty()){

        }
        reverse(res.begin(), res.end());
        return res;
    }
