class Solution(object):
    def replaceWords(self, dictionary, sentence):
        """
        :type dictionary: List[str]
        :type sentence: str

        # 2. for each word in sentence, find the longest prefix match in dictionary
        # 3. if no match found, return original word
        # 4. otherwise, return matched word
        # time complexity O(nlogn + n)
        # space complexity O(n)
