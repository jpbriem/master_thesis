class Solution(object):
    def reorganizeString(self, S):
        """
        :type S: str
        :rtype: str
        """
        if len(S) == 0 or len(S) % 2!= 0: return ''
        d = {}
        for c in S:
            d[c] = d.get(c, 0) + 1
        max_count = max([d[k] for k in d])
        if max_count > (len(S)+1)/2: return ''
        res = []

                res += [key]*min(max_count - count, len(S)-len(res))

                break
        return ''.join(res)
