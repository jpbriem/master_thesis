class Solution {
public:
    vector<char> reorganizeString(string S) {
        int n = S.size();
        if (n == 0 || n % 2!= 0) return {};
        unordered_map<char, int> mp;
        for (auto c : S) ++mp[c];
        priority_queue<pair<int, char>> pq;
        for (auto [ch, cnt] : mp) {
            if (cnt > n / 2) return {};
            pq.push({cnt, ch});
        }
        vector<char> ans(n);
        while (!pq.empty()) {
            auto [cnt, ch] = pq.top();
            pq.pop();
            ans[(n - 1) * 2 + cnt] = ch;
            --cnt;
            if (cnt) pq.push({cnt, ch});
        }
        return ans;
    }
};
