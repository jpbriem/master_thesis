public class Solution {
    public String[] removeDuplicateLetters(String s) {
        if (s == null || s.length() <= 1) return new String[]{s};

        int[] count = new int[26]; // count of each char in the string
        for (char c : s.toCharArray()) count[c - 'a']++;

        Stack<Character> stack = new Stack<>();
        boolean[] visited = new boolean[26]; // whether a char has been added to the result or not

        for (int i = 0; i < s.length(); i++) {
            char cur = s.charAt(i);
            count[cur - 'a']--;

            while (!stack.isEmpty() && stack.peek() > cur && count[stack.peek() - 'a'] > 0) {
                char top = stack.pop();
                visited[top - 'a'] = false;
            }

            if (!visited[cur - 'a']) {
                stack.push(cur);
                visited[cur - 'a'] = true;
            }
        }

        String[] res = new String[stack.size()];
        for (int i = res.length - 1; i >= 0; i--) {
            res[i] = "" + stack.get(i);
        }

        return res;
    }
}
