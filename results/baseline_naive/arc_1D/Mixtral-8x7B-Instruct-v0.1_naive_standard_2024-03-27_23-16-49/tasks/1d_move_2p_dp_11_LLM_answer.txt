public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        int left = -1;
        for (int right = 0; right < len; ++right) {
            if (arr[right] == '.' && left >= 0) {
                continue;
            } else if (arr[right] == 'L' || arr[right] == 'R') {
                left = right;
            } else if (left >= 0) {
                if (arr[left] == 'L') {
                    while (left + 1 < right) {
                        arr[++left] = 'L';
                    }
                } else { // arr[left] == 'R'
                    while (left + 1 < right) {
                        arr[--right] = 'R';
                    }
                }
                left = -1;
            }
        }
        return new String(arr);
    }
}
