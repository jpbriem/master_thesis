public class Solution {
    public String[] fullBloomFlowers(String[] flowers, int n) {
        if (flowers == null || flowers.length == 0 || n <= 0) return new String[0];

        // initialize the result array with all dots
        String[] res = new String[n];
        Arrays.fill(res, ".");

        for (int i = 0; i < flowers.length; i++) {
            char c = flowers[i].charAt(0);
            int start = Math.max(0, i - n / 2);
            int end = Math.min(n + i - 1, flowers.length - 1);
            for (int j = start; j <= end; j++) {
                if (flowers[j].charAt(0)!= c) continue;
                res[j - i + n/2] = Character.toString(c);
            }
        }

        return res;
    }
}
