public class RemoveDupFromEnds {
    public static void main(String[] args) {
        String input = "..... b b b b b b b b b b b f..";
        System.out.println("Input: " + input);
        System.out.println("Output: " + removeDupsAtBothEnds(input));
    }

    private static String removeDupsAtBothEnds(String str) {
        int left = 0, right = str.length() - 1;
        while (left < right && str.charAt(left) == '.') {
            left++;
        }
        while (right > left && str.charAt(right) == '.') {
            right--;
        }
        return str.substring(left, right + 1);
    }
}
