    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        while (a > 0 || b > 0 || c > 0) {
            if (a == 0) {
                sb.append('b');
                b--;
            } else if (b == 0) {
                sb.append('c');
                c--;
            } else if (a > b && a > c) {
                sb.append('a');
                a--;
            } else if (b >= c) {
                sb.append('b');
                b--;
            } else {
                sb.append('c');
                c--;
            }
            // check the last two characters in string builder
            if (sb.length() >= 2 && sb.charAt(sb.length() - 1) == sb.charAt(sb.length() - 2)) {
                // remove the second to last character
                sb.deleteCharAt(sb.length() - 2);
                // add one more character from the same group as the first character of the remaining substring
                if (sb.charAt(sb.length() - 1) == 'a') {
                    a--;
                } else if (sb.charAt(sb.length() - 1) == 'b') {
                    b--;
                } else {
                    c--;
                }
            }
        }
        return sb.toString();
    }
