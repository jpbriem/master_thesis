public class Solution {
    public String[] solve(String[] board) {
        if (board == null || board.length < 3 || board[0].length() < 3) return board;
        int m = board.length, n = board[0].length();
        boolean[][] vis = new boolean[m][n];
        for (int i = 0; i < m; ++i) {
            if (board[i].charAt(0) == 'O') dfs(board, vis, i, 0);
            if (board[i].charAt(n - 1) == 'O') dfs(board, vis, i, n - 1);
        }
        for (int j = 0; j < n; ++j) {
            if (board[0].charAt(j) == 'O') dfs(board, vis, 0, j);
            if (board[m - 1].charAt(j) == 'O') dfs(board, vis, m - 1, j);
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!vis[i][j] && board[i].charAt(j) == 'O') board[i] = board[i].substring(0, j) + "X" + board[i].substring(j + 1);
            }
        }
        return board;
    }

    private void dfs(String[] board, boolean[][] vis, int x, int y) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length() || vis[x][y] || board[x].charAt(y)!= 'O') return;
        vis[x][y] = true;
        dfs(board, vis, x - 1, y);
        dfs(board, vis, x + 1, y);
        dfs(board, vis, x, y - 1);
        dfs(board, vis, x, y + 1);
    }
}
