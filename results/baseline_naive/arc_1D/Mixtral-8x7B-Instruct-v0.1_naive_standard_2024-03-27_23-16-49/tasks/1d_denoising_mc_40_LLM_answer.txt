function removeDuplicateLetters(s) {
  const stack = [];
  const lastIndexMap = {};
  for (let i = s.length - 1; i >= 0; i--) {
    if (!lastIndexMap[s[i]]) {
      lastIndexMap[s[i]] = i;
    }
  }
  for (const char of s) {
    while (stack.length && stack[stack.length - 1] > char && lastIndexMap[char] > stack[stack.length - 1]) {
      stack.pop();
    }
    if (stack.indexOf(char) === -1) {
      stack.push(char);
    }
  }
  return stack.join('');
}
