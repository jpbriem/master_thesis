class Solution {
    public String[] rearrange(String[] input) {
        int len = input.length;
        if (len < 3) return input;

        List<Character> list = new ArrayList<>();
        for (int i = 0; i < len; ++i) {
            char c = input[i].charAt(0);
            if (!list.contains(c)) {
                list.add(c);
            } else {
                int index = list.indexOf(c);
                while (index + 1 < list.size() && list.get(index + 1).equals(c)) {
                    index++;
                }
                list.set(index, c);
            }
        }

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; ++i) {
            char c = input[i].charAt(0);
            if (i % 2 == 0) {
                if (list.get(i / 2).equals(c)) {
                    sb.append(c);
                } else {
                    sb.append('.');
                }
            } else {
                if (list.get((i - 1) / 2).equals(c)) {
                    sb.append(c);
                } else {
                    sb.append('.');
                }
            }
        }

        String str = sb.toString();
        String[] output = new String[len];
        for (int i = 0; i < len; ++i) {
            output[i] = Character.toString(str.charAt(i));
        }

        return output;
    }
}
