public class Solution {
    public String[] pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == '.') {
                arr[i] = '0';
            } else {
                arr[i] = dominoes.charAt(i);
            }
        }

        // left to right
        for (int i = 0, j = 0; i < len && j < len; ) {
            while (j < len && arr[j]!= 'L' && arr[j]!= 'R') {
                ++j;
            }
            if (j >= len) break;

            if (arr[i] == '.' && arr[j] == 'L') {
                int k = i + 1;
                while (k < j) {
                    arr[k++] = 'L';
                }
            }

            i = j + 1;
            j += 2;
        }

        // right to left
        for (int i = len - 1, j = len - 1; i > 0 && j > 0; ) {
            while (j > 0 && arr[j]!= 'L' && arr[j]!= 'R') {
                --j;
            }
            if (j <= 0) break;

            if (arr[i] == '.' && arr[j] == 'R') {
                int k = i - 1;
                while (k > j) {
                    arr[k--] = 'R';
                }
            }

            i = j - 1;
            j -= 2;
        }

        for (int i = 0; i < len; ++i) {
            if (arr[i] == '0') {
                arr[i] = '.';
            }
        }

        return Arrays.toString(arr).replaceAll("[\\[\\],]", "").split(", ");
    }
}
