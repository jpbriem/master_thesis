    public String[] move(String[] board) {
        int len = board.length;
        for (int i = 0; i < len - 1; ++i) {
            if (!board[i].equals(".")) continue;

            for (int j = i + 1; j < len &&!found; ++j) {

                    board[i] += board[j];
                    board[j] = ".";
                    found = true;
                }
            }
        }

        }
        return list.toArray(new String[list.size()]);
    }
