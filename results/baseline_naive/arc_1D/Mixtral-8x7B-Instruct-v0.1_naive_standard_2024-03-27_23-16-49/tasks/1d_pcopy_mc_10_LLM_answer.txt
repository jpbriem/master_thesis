public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int len = 0;
        while (len < S.length()) {
            char c = S.charAt(len);
            if (c == '.') break;
            int count = 0;
            while (len < S.length() && S.charAt(len++) == c) count++;
            len--; // back to the last letter of this group
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < count; i++) sb.append(c);
            while (len < S.length() && S.charAt(len) == '.') {
                len++;
                sb.append('.');
            }
            res.add(sb.toString());
            len++;
        }
        if (len == S.length()) return new String[]{S};
        List<String> temp = new ArrayList<>();
        dfs(res, temp, "", 0);
        Collections.sort(temp);
        return temp.toArray(new String[temp.size()]);
    }

    private void dfs(List<String> res, List<String> temp, String cur, int idx) {
        if (idx >= res.size()) {
            temp.add(cur);
            return;
        }
        String s = res.get(idx);
        for (int i = 0; i <= s.length(); i++) {
            dfs(res, temp, cur + s.substring(0, i), idx + 1);
        }
    }
}
