    public String[] expand(String S) {
        List<String> res = new ArrayList<>();

        while (len < S.length()) {
            char c = S.charAt(len);

            int count = 0;
            while (len < S.length() && Character.isDigit(S.charAt(len))) {
                count = count * 10 + (S.charAt(len++) - '0');
            }
            len++;
            for (int i = 0; i < count; i++) {
                res.add("" + c);
            }
        }

                sb.append(s);
            }

    }
