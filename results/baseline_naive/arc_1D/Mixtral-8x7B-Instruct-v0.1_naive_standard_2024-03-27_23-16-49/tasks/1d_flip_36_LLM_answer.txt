    public String reverseLeftWords(String s, int n) {
        if (s == null || s.length() <= 1 || n < 0) return s;
        StringBuilder sb = new StringBuilder();
        for (int i = n; i < s.length(); i++) {
            sb.append(s.charAt(i));
        }

            sb.append(s.charAt(j));
        }
        return sb.toString();
    }
