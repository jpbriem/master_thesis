    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        int len = arr.length;
        for (int i = 0; i < len - 1; i++) {
            char c = arr[i].charAt(arr[i].length() - 1);
            if (c!= '.' && c!= 'g' && c!= 'd' && c!= 'e') continue;

            // find the next different character
            while (++i < len) {
                if (arr[i].charAt(0)!= c) break;
            }

            // replace all elements between two different characters with one of them
            StringBuilder sb = new StringBuilder();
            for (int j = i - 1; j >= 0; j--) {
                sb.append(arr[j]);
            }
            String str = sb.reverse().toString();

                arr[k] = str;
            }
        }

        return arr;
    }

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        Deque<Character> stack = new ArrayDeque<>();
        int idx = 0, n = arr.length;
        while (idx < n) {
            char cur = arr[idx].charAt(arr[idx].length() - 1);
            if (cur == '.' || cur == 'g' || cur == 'd' || cur == 'e') {
                stack.push(cur);
            } else {
                while (!stack.isEmpty()) {
                    char top = stack.pop();
                    for (int i = idx; i > 0; i--) {
                        arr[i] = top + arr[i];
                    }
                }
            }
            idx++;
        }

        while (!stack.isEmpty()) {
            char top = stack.pop();
            for (int i = n - 1; i >= 0; i--) {
                arr[i] = top + arr[i];
            }
        }

        return arr;
    }
