public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int len = 0;
        while (len < S.length()) {
            char c = S.charAt(len);
            if (c == '.') break;
            int count = 0, start = len + 1;
            while (start < S.length() && Character.isDigit(S.charAt(start))) {
                count = count * 10 + (S.charAt(start++) - '0');
            }
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < count; ++i) {
                sb.append(c);
            }
            res.add(sb.toString());
            len += 2 + String.valueOf(count).length();
        }
        if (len == S.length()) {
            res.add("" + S.charAt(len));
        } else {
            String rest = S.substring(len);
            List<String> temp = expand(rest);
            for (String str : temp) {
                res.addAll(Collections.nCopies(count, str));
            }
        }
        return res.toArray(new String[res.size()]);
    }
}
