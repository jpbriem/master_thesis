class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length < 2) return arr;

        int len = arr.length, idx = -1;
        for (int i = 0; i < len; i++) {
            if (arr[i].equals(".")) continue;
            if (idx == -1) idx = i;
            else arr[++idx] = arr[i];
        }

        // check the last element of array is not ".", then copy it to all empty elements behind it
        if (!arr[len-1].equals(".")) {
            char c = arr[len-1].charAt(0);
