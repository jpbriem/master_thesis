public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int len = 0;
        while (len < S.length()) {
            char c = S.charAt(len);
            if (Character.isLetter(c)) {
                res.add("" + c);
            } else if (c == '.') {
                res.add(".");
            } else { // c is a left bracket
                int rightBracketIndex = findRightBracketIndex(S, len);
                int count = Integer.parseInt(S.substring(len + 1, rightBracketIndex));
                StringBuilder sb = new StringBuilder();
                for (int i = 0; i < count; ++i) {
                    sb.append(res.get(res.size() - 1));
                }
                res.remove(res.size() - 1);
                res.add(sb.toString());
                len = rightBracketIndex + 1;
            }
            len++;
        }
        return res.toArray(new String[res.size()]);
    }

    private int findRightBracketIndex(String s, int index) {
        int cnt = 1;
        while (index < s.length() && cnt > 0) {
            if (s.charAt(index++) == '[') {
                cnt++;
            } else if (s.charAt(index - 1) == ']') {
                cnt--;
            }
        }
        return index - 1;
    }
}
