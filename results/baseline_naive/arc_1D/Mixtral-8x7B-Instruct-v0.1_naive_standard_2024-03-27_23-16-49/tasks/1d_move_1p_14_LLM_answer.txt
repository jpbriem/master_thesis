public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        // left and right are the index of last non-dot character to the left and right of current position respectively
        int left = -1, right = -1;
        boolean isLeftForcePush = false;
        for (int i = 0; i < len; ++i) {
            if (arr[i] == '.') continue;
            if (left == -1 && arr[i] == 'L') {
                left = i;
            } else if (left >= 0 && arr[i] == 'R' &&!isLeftForcePush) {
                int j = left + 1;
                while (j <= i) {
                    arr[j++] = 'R';
                }
                left = -1;
            } else if (left >= 0 && arr[i] == 'R' && isLeftForcePush) {
                int j = left + 1;
                while (j <= i) {
                    arr[j++] = 'R';
                }
                left = -1;
                isLeftForcePush = false;
            } else if (left >= 0 && arr[i] == 'L') {
                int j = left + 1;
                while (j <= i) {
                    arr[j++] = 'L';
                }
                left = -1;
                isLeftForcePush = true;
            }
            if (right == -1 && arr[len - i - 1] == 'R') {
                right = len - i - 1;
            } else if (right >= 0 && arr[len - i - 1] == 'L' &&!isLeftForcePush) {
                int j = right - 1;
                while (j >= len - i - 1) {
                    arr[j--] = 'L';
                }
                right = -1;
            } else if (right >= 0 && arr[len - i - 1] == 'L' && isLeftForcePush) {
                int j = right - 1;
                while (j >= len - i - 1) {
                    arr[j--] = 'L';
                }
                right = -1;
                isLeftForcePush = false;
            } else if (right >= 0 && arr[len - i - 1] == 'R') {
                int j = right - 1;
                while (j >= len - i - 1) {
                    arr[j--] = 'R';
                }
                right = -1;
                isLeftForcePush = true;
            }
        }
        return new String(arr);
    }
}
