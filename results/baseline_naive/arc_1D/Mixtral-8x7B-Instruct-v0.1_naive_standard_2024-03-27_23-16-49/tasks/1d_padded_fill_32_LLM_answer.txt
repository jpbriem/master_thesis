class Solution {
    public String[] findWords(String[] words) {
        List<String> list = new ArrayList<>();
        for (int i = 0; i < words.length; ++i) {
            if (isMatch(words[i])) {
                list.add(words[i]);
            }
        }
        return list.toArray(new String[list.size()]);
    }

    private boolean isMatch(String word) {
        int row = -1, col = -1;
        char c = word.charAt(0);
        if ('a' <= c && c <= 'z') {
            row = c / 6 + 1;
            col = c % 6 == 0? 6 : c % 6;
        } else if ('A' <= c && c <= 'Z') {
            row = c / 6 + 1;
            col = c % 6 == 0? 6 : c % 6;
        } else {
            return false;
        }
        for (int j = 1; j < word.length(); ++j) {
            c = word.charAt(j);
            if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) {
                if (((c / 6 + 1)!= row) || ((c % 6 == 0)? (col!= 6) : (col!= c % 6))) {
                    return false;
                }
            }
        }
        return true;
    }
}
