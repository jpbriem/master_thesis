public class Solution {
    public String removeDuplicateLetters(String s) {
        if (s == null || s.length() <= 1) return s;

        int[] count = new int[26]; // count of each char in the string
        for (char c : s.toCharArray()) {
            count[c - 'a']++;
        }

        Stack<Character> stack = new Stack<>();
        boolean[] visited = new boolean[26]; // record whether a char has been added to the result or not

        for (int i = 0; i < s.length(); i++) {
            char cur = s.charAt(i);
            count[cur - 'a']--;

            /*
             * If current character is smaller than top element, and it's unvisited, then pop out all elements that are larger than current one until we find an unvisited one.
             */
            while (!stack.isEmpty() && stack.peek() > cur && count[stack.peek() - 'a'] > 0) {
                char temp = stack.pop();
                visited[temp - 'a'] = false;
            }

            if (!visited[cur - 'a']) {
                stack.push(cur);
                visited[cur - 'a'] = true;
            }
        }

        StringBuilder sb = new StringBuilder();
        for (char c : stack) {
            sb.append(c);
        }

        return sb.toString();
    }
}
