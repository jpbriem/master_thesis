public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        // left and right are the index of last non-dot character to the left and right of current position respectively
        int left = -1, right = -1;
        boolean isLeftForcePush = false;
        for (int i = 0; i < len; ++i) {
            if (arr[i] == '.') continue;
            if (left == -1 && arr[i] == 'l') {
                left = i;
            } else if (left!= -1 && arr[i] == 'r') {
                right = i;
                if (!isLeftForcePush) {
                    for (int j = left + 1; j <= right; ++j) {
                        arr[j] = 'l';
                    }
                }
                left = -1;
                right = -1;
                isLeftForcePush = true;
            } else if (left!= -1 && arr[i] == 'l') {
                right = i;
                if (isLeftForcePush) {
                    for (int j = left + 1; j <= right; ++j) {
                        arr[j] = 'r';
                    }
                }
                left = -1;
                right = -1;
                isLeftForcePush = false;
            }
        }
        if (left!= -1) {
            if (right == -1 || arr[left] == 'l' && arr[right] == 'r') {
                for (int i = left + 1; i < len; ++i) {
                    arr[i] = 'l';
                }
            } else if (right!= -1 && arr[left] == 'r' && arr[right] == 'l') {
                for (int i = left + 1; i < right; ++i) {
                    arr[i] = 'r';
                }
                arr[right] = 'l';
            }
        }
        return new String(arr);
    }
}
