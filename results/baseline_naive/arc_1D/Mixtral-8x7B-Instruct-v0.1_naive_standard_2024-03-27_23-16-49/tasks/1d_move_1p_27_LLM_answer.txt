    public String move(String s) {

        char[] arr = s.toCharArray();

        while (i < arr.length && j < arr.length) {

            }
            j++;
        }

    }

    private void swap(char[] arr, int a, int b) {

        arr[a] = arr[b];
        arr[b] = tmp;
    }
