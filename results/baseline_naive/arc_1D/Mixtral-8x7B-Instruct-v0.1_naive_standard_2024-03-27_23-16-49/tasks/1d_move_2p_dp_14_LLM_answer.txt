function removeChar(str) {
    let arr = str.split('');
    for (let i = 0; i < arr.length - 1; i++) {
        if (arr[i] === '.' && arr[i + 1]!== '.') {
            arr[i] = '';
        } else if (arr[i]!== '.' && arr[i + 1] === '.') {
            arr[i + 1] = '';
        }
    }
    return arr.join('').trim();
}
