public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == '.' || dominoes.charAt(i) == 'R') {
                arr[i] = '.';
            } else {
                arr[i] = 'L';
            }
        }
        for (int i = 0, j = 0; i < len && j < len; ) {
            while (j < len && arr[j]!= 'L') {
                ++j;
            }
            if (j >= len) break;
            int k = j - 1;
            while (k > i && arr[k] == 'L' && arr[k-1] == '.') {
                --k;
            }
            if (arr[k] == 'R') {
                for (int l = j; l <= k + 1; ++l) {
                    arr[l] = 'R';
                }
            }
            i = j;
            ++j;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; ++i) {
            if (arr[i] == 'L') {
                sb.append('L');
            } else if (arr[i] == 'R') {
                sb.append('R');
            } else {
                sb.append('.');
            }
        }
        return sb.toString();
    }
}
