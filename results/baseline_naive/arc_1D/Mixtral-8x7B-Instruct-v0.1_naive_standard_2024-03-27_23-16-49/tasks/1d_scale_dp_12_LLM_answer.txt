class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == '.' || dominoes.charAt(i) == 'R') {
                arr[i] = 'R';
            } else {
                arr[i] = 'L';
            }
        }
        // System.out.println(Arrays.toString(arr));
        for (int i = 0, j = 0; i < len && j < len; ) {
            while (j < len && arr[j]!= 'L') {
                ++j;
            }
            if (j >= len) break;
            int k = j - 1;
            while (k > i && arr[k] == 'R') --k;
            if (k <= i) continue;
            for (int l = i + 1; l < k; ++l) {
                arr[l] = 'R';
            }
            i = k;
        }
        StringBuilder sb = new StringBuilder();
        for (char c : arr) {
            if (c == 'L' || c == 'R') {
                sb.append('.');
            }
            sb.append(c);
        }
        return sb.substring(1);
    }
}

class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length(), left = 0;
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int right = 0; right < len; ++right) {
            if (dominoes.charAt(right) == 'R') {
                while (left < right) {
                    arr[left++] = 'R';
                }
            } else if (dominoes.charAt(right) == 'L') {
                if (left > 0 && arr[left-1] == 'R') {
                    int r = right - 1;
                    while (left < r) {
                        arr[left++] = 'R';
                        arr[r--] = 'L';
                    }
                }
                left = Math.max(left, right+1);
            }
        }
        StringBuilder sb = new StringBuilder();
        for (char c : arr) {
            if (c == 'L' || c == 'R') {
                sb.append('.');
            }
            sb.append(c);
        }
        return sb.substring(1);
    }
}
