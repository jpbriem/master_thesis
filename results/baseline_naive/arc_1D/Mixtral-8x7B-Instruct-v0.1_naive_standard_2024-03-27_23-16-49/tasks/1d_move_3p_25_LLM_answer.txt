public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        // left and right are the index of last non-dot character to the left and right of current position respectively
        int left = -1, right = -1;
        for (int i = 0; i < len; ++i) {
            if (arr[i] == '.' && left >= 0) {
                // If there is a dot on the left side, we need to check whether it should be pushed or not
                if (right > left) {
                    // Push from left to right
                    while (++left < i) {
                        arr[left] = arr[right];
                    }
                } else {
                    // Push from right to left
                    while (--right > i) {
                        arr[right] = arr[left];
                    }
                }
            }
            if (arr[i]!= '.') {
                left = i;
                right = i;
            }
        }
        return new String(arr);
    }
}
