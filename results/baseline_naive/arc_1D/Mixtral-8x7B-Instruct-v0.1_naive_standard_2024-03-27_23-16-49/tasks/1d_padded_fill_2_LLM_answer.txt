public class Solution {
    public String[] findLongestChain(String[] input) {
        if (input == null || input.length <= 0) return new String[0];

        List<List<Integer>> list = new ArrayList<>();
        for (int i = 0; i < input.length; i++) {
            int index = -1;
            for (int j = 0; j < list.size(); j++) {
                if (list.get(j).get(0) > i) {
                    index = j;
                    break;
                }
            }
            if (index!= -1) {
                list.add(index, new ArrayList<>());
            }
            list.get(i).add(i);
            char c = input[i].charAt(0);
            if (c == 'e') {
                list.get(i).add(-1);
            } else {
                list.get(i).add(i + 1);
            }
        }

        StringBuilder sb = new StringBuilder();
        dfs(list, 0, sb);

        String output = sb.toString();
        String[] result = new String[output.length() / 5];
        for (int i = 0; i < result.length; i++) {
            result[i] = output.substring(i * 5, (i + 1) * 5);
        }
        return result;
    }

    private void dfs(List<List<Integer>> list, int idx, StringBuilder sb) {
        while (idx >= 0 && idx < list.size()) {
            List<Integer> cur = list.get(idx);
            if (cur.get(1) == -1) {
                sb.append('e');
            } else {
                sb.append('.');
            }
            int nextIdx = cur.get(1);
            list.remove(idx);
            idx = nextIdx;
        }
    }
}
