def removeDuplicateLetters(s):
    """
    :type s: str
    :rtype: str
    """
    stack = []
    count = collections.Counter(s)
    for c in s:
        if c not in stack:
            while stack and stack[-1] > c and count[stack[-1]] > 0:
                stack.pop()
            stack.append(c)
            count[c] -= 1
    return "".join(stack)
