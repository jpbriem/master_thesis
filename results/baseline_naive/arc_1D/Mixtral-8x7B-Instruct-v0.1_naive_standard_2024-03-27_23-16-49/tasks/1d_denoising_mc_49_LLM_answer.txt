    public String[] replace(String[] arr) {

        int len = arr.length;
        for (int i = 0; i < len; i++) {

            boolean flag = true;

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length <= 0) return new String[0];

        StringBuilder sb = new StringBuilder();
        int count = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i].equals(".")) {
                sb.append('.');
                count++;
            } else {
                if (count > 0) {
                    sb.append(arr[i].charAt(0), count, 1);
                    count = 0;
                }
                sb.append(arr[i]);
            }
        }

        String s = sb.toString();
        return s.split("");
    }
}
