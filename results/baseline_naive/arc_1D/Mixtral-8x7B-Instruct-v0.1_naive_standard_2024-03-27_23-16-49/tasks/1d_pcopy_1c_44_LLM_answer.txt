public class Solution {
    public String[] findWords(String[] words) {
        List<String> result = new ArrayList<>();
        for (String word : words) {
            if (isValidWord(word)) {
                result.add(word);
            }
        }
        return result.toArray(new String[result.size()]);
    }

    private boolean isValidWord(String word) {
        int firstCharCode = Character.toLowerCase(word.charAt(0));
        for (int i = 1; i < word.length(); i++) {
            int charCode = Character.toLowerCase(word.charAt(i));
            if ((firstCharCode == 79 && charCode!= 85 && charCode!= 79) ||
                    (firstCharCode == 67 && charCode!= 67 && charCode!= 71 && charCode!= 84) ||
                    (firstCharCode!= 79 && firstCharCode!= 67)) {
                return false;
            }
        }
        return true;
    }
