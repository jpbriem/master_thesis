    public String[] pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == 'R') {
                while (i + 1 < len && dominoes.charAt(i + 1)!= 'L') {
                    ++i;
                }

                    --i;
                }

class Solution {
    public String[] pushDominoes(String dominoes) {
        StringBuilder sb = new StringBuilder();
        int f = 0, s = 0;
        for (int i = 0; i < dominoes.length(); ++i) {
            switch (dominoes.charAt(i)) {
                case '.':
                    break;
                case 'L':
                    if (f == -1) {
                        sb.append('L');
                    } else {
                        for (int j = s; j > f; --j) {
                            sb.append('.');
                        }
                        sb.append("LL");
                        s = i;
                    }
                    f = i;
                    break;
                default: // R
                    if (f == -1) {
                        sb.append('R');
                    } else {
                        for (int j = f; j <= s; ++j) {
                            sb.append('.');
                        }
                        sb.append("RR");
                        f = i;
                    }
                    s = i;
                    break;
            }
        }
        if (s >= f) {
            for (int i = f; i <= s; ++i) {
                sb.append('.');
            }
        }
        return sb.toString().split("");
    }
}
