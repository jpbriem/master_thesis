public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        boolean leftToRight = true;
        for (int i = 0; i < len; ) {
            // find the first non-dot from left to right
            while (i < len && arr[i] == '.') {
                ++i;
            }
            if (i >= len) break;
            int j = i + 1;
            // find the first dot or end of string from left to right
            while (j < len && arr[j]!= '.') {
                ++j;
            }
            if (leftToRight) {
                // fill all dots between two ends with '-'
                for (int k = i + 1; k < j; ++k) {
                    arr[k] = '-';
                }
                // update the direction based on the sign at the second point
                if (j < len && arr[j] == 'R') {
                    leftToRight = false;
                } else if (j > i + 1 && arr[i] == 'L') {
                    leftToRight = false;
                }
            } else {
                // fill all dots between two ends with '\'
                for (int k = i + 1; k < j; ++k) {
                    arr[k] = '/';
                }
                // update the direction based on the sign at the first point
                if (i > 0 && arr[i - 1] == 'R') {
                    leftToRight = true;
                } else if (j < len && arr[j] == 'L') {
                    leftToRight = true;
                }
            }
            i = j;
        }
        return new String(arr);
    }
}
