    public static void main(String[] args) {
        char[] input = {'.','e', 'e', '.', '.', '.', 'e', 'e', 'e', 'e', 'e', '.', 'e', 'e', 'e', '.', '.', 'e', 'e', 'e', 'e', 'e', '.', 'e', 'e', 'e', 'e', 'e', '.'};

        }
    }

    private static void replaceEwithH(char[] arr) {
        int eCount = 0, hCount = 0;
        // Counting the number of e and h in array
        for (int i=0; i<arr.length; i++) {
            if (arr[i] == 'e') {
                eCount++;
            } else if (arr[i] == 'h') {
                hCount++;
            }
        }

        // If there is no e then we don't need to do anything as output will be same as input
        if (eCount!= 0 && hCount!= 0) {
            // Calculating the position where first e should be replaced with h
            double posOfFirstE = Math.ceil((double)(eCount+hCount)/2);
            // Replacing all e with h till posOfFirstE-1
            for (int i=0; i<posOfFirstE-1; i++) {
                if (arr[i] == 'e') {
                    arr[i] = 'h';
                }
            }
            // Replacing all h with e from posOfFirstE till end
            for (int j=posOfFirstE; j<arr.length; j++) {
                if (arr[j] == 'h') {
                    arr[j] = 'e';
                }
            }
            // Again replacing all e with h after posOfFirstE
            for (int k=posOfFirstE; k<arr.length; k++) {
                if (arr[k] == 'e') {
                    arr[k] = 'h';
                }
            }
        }
    }
}
