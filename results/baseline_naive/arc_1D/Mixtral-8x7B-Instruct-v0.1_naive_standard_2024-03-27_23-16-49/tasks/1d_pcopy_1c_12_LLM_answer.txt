public class Solution {
    public String longestOnes(String A, int B) {
        if (A == null || A.length() <= 0) return "";
        StringBuilder sb = new StringBuilder();
        int countZero = 0;
        for (int i = 0; i < A.length(); i++) {
            char c = A.charAt(i);
            if (c == '0') {
                countZero++;
            }
            if (countZero > B) {
                while (countZero > B && i >= 0) {
                    c = A.charAt(--i);
                    if (c == '0') {
                        countZero--;
                    }
                }
            }
            sb.append(c);
        }
        return sb.toString();
    }
