    public String[] pushDominoes(String dominoes) {
        int len = dominoes.length();

                    ++i;
                }
                if (i < len) {
                    int j = i - 1;
                    while (j >= 0 && arr[j] == 'R') {
                        --j;
                    }
                    if (j >= 0) {
                        int k = j + 1;
                        while (k <= i) {
                            arr[k++] = '0';
                        }
                    }
                }
            } else if (dominoes.charAt(i) == 'L') {
                while (i < len && dominoes.charAt(i)!= 'R') {
