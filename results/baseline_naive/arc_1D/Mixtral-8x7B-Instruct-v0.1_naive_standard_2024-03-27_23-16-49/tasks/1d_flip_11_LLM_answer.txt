public class Solution {
    public String[] move(String[] board) {
        // Write your solution here
        if (board == null || board.length < 1 || board[0].length()!= board.length){
            return new String[]{};
        }
        int len = board.length;
        for (int i = 0; i < len / 2; i++){
            swap(board, i, len - i - 1);
        }
        boolean isChanged = true;
        while (isChanged){
            isChanged = false;
            for (int i = 0; i < len - 1; i++){
                if (board[i].equals(".") &&!board[i + 1].equals(".")){
                    char c = board[i + 1];
                    int j = i + 1;
                    while (!board[j].equals(".") && j < len - 1){
                        board[j] = board[++j];
                    }
                    board[j] = c + "";
                    isChanged = true;
                }
            }
        }
        return board;
    }

    private void swap(String[] arr, int a, int b){
        String temp = arr[a];
        arr[a] = arr[b];
        arr[b] = temp;
    }
}
