public class Solution {
    public String[] solve(String[] board) {
        if (board == null || board.length < 3 || board[0].length() < 3) return board;
        int m = board.length, n = board[0].length();
        boolean[][] visited = new boolean[m][n];
        for (int i = 0; i < m; ++i) {
            if (board[i].charAt(0) == 'a') dfs(board, i, 0, visited);
            if (board[i].charAt(n - 1) == 'a') dfs(board, i, n - 1, visited);
        }
        for (int j = 0; j < n; ++j) {
            if (board[0].charAt(j) == 'a') dfs(board, 0, j, visited);
            if (board[m - 1].charAt(j) == 'a') dfs(board, m - 1, j, visited);
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!visited[i][j] && board[i].charAt(j)!= 'a' && board[i].charAt(j)!= 'd') {
                    board[i] = board[i].substring(0, j) + "d" + board[i].substring(j + 1);
                }
            }
        }
        return board;
    }

    private void dfs(String[] board, int x, int y, boolean[][] visited) {
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length()) return;
        if (visited[x][y]) return;
        if (board[x].charAt(y)!= 'a') return;
        visited[x][y] = true;
        dfs(board, x + 1, y, visited);
        dfs(board, x - 1, y, visited);
        dfs(board, x, y + 1, visited);
        dfs(board, x, y - 1, visited);
    }
}
