class Solution {
    public String[] findLongestChain(String[] words) {
        Arrays.sort(words, new Comparator<String>() {
            @Override
            public int compare(String s1, String s2) {
                return Integer.compare(s1.length(), s2.length());
            }
        });

        List<String> result = new ArrayList<>();
        for (int i = 0; i < words.length; ++i) {
            if (result.isEmpty() ||!canAdd(result.get(result.size() - 1), words[i])) {
                result.add(words[i]);
            }
        }

        return result.toArray(new String[result.size()]);
    }

    private boolean canAdd(String word1, String word2) {
        int len1 = word1.length();
        int len2 = word2.length();

        // If the length difference is more than one, it's impossible to add them together
        if (Math.abs(len1 - len2) > 1) {
            return false;
        }

        // Adding two strings with same length
        if (len1 == len2) {
            return true;
        }

        // Adding a shorter string to a longer one
        if (len1 < len2) {
            for (int i = 0; i < len1; ++i) {
                if (word1.charAt(i)!= word2.charAt(i)) {
                    return false;
                }
            }
            return true;
        }

        // Adding a longer string to a shorter one
        for (int i = 0; i < len2; ++i) {
            if (word1.charAt(i)!= word2.charAt(i)) {
                return false;
            }
        }
        return true;
    }
}
