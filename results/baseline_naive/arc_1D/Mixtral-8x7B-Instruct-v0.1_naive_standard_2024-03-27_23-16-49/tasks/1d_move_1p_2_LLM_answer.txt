function removeDuplicateLetters(str) {
    let stack = [];
    const freq = {};
    for (let i = 0; i < str.length; i++) {
        if (!freq[str[i]]) {
            while (stack.length && stack[stack.length - 1] > str[i] && freq[stack[stack.length - 1]] >= 1) {
                stack.pop();
            }
            stack.push(str[i]);
            freq[str[i]] = 1;
        } else {
            freq[str[i]]++;
        }
    }
    return stack.join('');
}
