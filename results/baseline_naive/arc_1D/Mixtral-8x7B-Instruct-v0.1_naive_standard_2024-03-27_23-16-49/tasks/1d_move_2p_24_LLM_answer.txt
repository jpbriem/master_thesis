public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == '.' || dominoes.charAt(i) == 'R') {
                arr[i] = 'R';
            } else {
                arr[i] = 'L';
            }
        }
        // System.out.println(Arrays.toString(arr));
        for (int i = 0; i + 1 < len && arr[i]!= '\u0000'; ) {
            if (arr[i] == 'R' && arr[i+1] == 'L') {
                int j = i + 1;
                while (j + 1 < len && arr[j+1] == 'L') {
                    j++;
                }
                for (int k = i; k <= j; ++k) {
                    arr[k] = '0';
                }
                i = j + 1;
            } else {
                i += 2;
            }
        }
        // System.out.println(Arrays.toString(arr));
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len; ++i) {
            if (arr[i] == '0') continue;
            sb.append('.');
            if (arr[i] == 'R') {
                int j = i + 1;
                while (j < len && arr[j] == 'R') {
                    sb.append('R');
                    j++;
                }
                if (j < len && arr[j] == '0') {
                    sb.append('L');
                }
                i = j - 1;
            } else {
                sb.append('L');
            }
        }
        return sb.toString();
    }
}
