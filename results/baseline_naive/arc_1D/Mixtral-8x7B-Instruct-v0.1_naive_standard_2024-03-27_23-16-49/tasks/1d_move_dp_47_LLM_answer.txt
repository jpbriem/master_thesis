            const temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        } else if(arr[right]!== toMove){
            right--;
        } else if(arr[left] === toMove){
            left++;
        }
    }

    return arr;
}
