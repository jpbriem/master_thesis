public class Solution {
    public String removeDuplicateLetters(String s) {
        if (s == null || s.length() <= 1) return s;

        int[] count = new int[26]; // record the number of each char in string s
        for (char c : s.toCharArray()) {
            count[c - 'a']++;
        }

        Stack<Character> stack = new Stack<>();
        boolean[] visited = new boolean[26]; // mark whether a char has been added to result or not

        for (int i = 0; i < s.length(); i++) {
            char cur = s.charAt(i);
            count[cur - 'a']--;

            /*
             * If current char is smaller than top element, and it's unvisited, then pop out all elements that are larger than current one
             */
            while (!stack.isEmpty() && stack.peek() > cur && count[stack.peek() - 'a'] > 0) {
                visited[stack.pop() - 'a'] = false;
            }

            if (!visited[cur - 'a']) {
                stack.push(cur);
                visited[cur - 'a'] = true;
            }
        }

        StringBuilder sb = new StringBuilder();
        for (char c : stack) {
            sb.append(c);
        }

        return sb.toString();
    }
}
