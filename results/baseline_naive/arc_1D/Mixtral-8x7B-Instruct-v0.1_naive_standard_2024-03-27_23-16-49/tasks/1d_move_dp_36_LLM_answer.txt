const moveElementToEnd = (arr, toMove) => {
    let left = 0;
    let right = arr.length - 1;

    while(left < right){
        if(arr[right] === toMove && arr[left]!== toMove){
            // swap elements at indexes left and right
            const temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        } else if(arr[right]!== toMove){
            right--;
        } else if(arr[left] === toMove){
            left++;
        }
    }

    return arr;
}
