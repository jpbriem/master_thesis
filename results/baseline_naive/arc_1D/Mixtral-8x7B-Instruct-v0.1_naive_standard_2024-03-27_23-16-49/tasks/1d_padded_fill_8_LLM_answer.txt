class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length < 2) return arr;

        int len = arr.length, start = -1, end = -1;
        for (int i = 0; i < len; i++) {
            if (arr[i].equals(".")) continue;
            else if (start == -1) {
                start = i;
                end = i;
            } else {
                end++;
            }

            // check the pattern of ".." or "...", and then fill it with the same character
            if ((end + 1 < len && arr[end+1].equals(".")) || (end + 2 < len && arr[end+2].equals("."))) {
                while (start <= end) {
                    arr[start] = arr[i];
                    start++;
                }
                end = start-1;
            }
        }

        return arr;
    }
}

class Solution {
    public String[] replace(String[] arr) {
        if (arr == null || arr.length < 2) return arr;

        StringBuilder sb = new StringBuilder();
        char curChar = '\u0000';
        boolean isFilled = false;
        for (String s : arr) {
            if (!s.equals(".") &&!isFilled) {
                curChar = s.charAt(0);
                isFilled = true;
            }

            if (s.equals(".") && isFilled) {
                sb.append(curChar);
            }
        }

        int idx = 0;
        for (String s : arr) {
            if (s.equals(".")) {
                arr[idx++] = Character.toString(curChar);
            } else {
                arr[idx++] = s;
            }
        }

        return arr;
    }
}
