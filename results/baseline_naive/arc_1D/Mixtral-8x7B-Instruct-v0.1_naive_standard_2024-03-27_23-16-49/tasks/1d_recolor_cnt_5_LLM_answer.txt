public class Solution {
    public String[] convert(String[] input) {
        if (input == null || input.length <= 0) return new String[0];
        int len = input.length;
        for (int i = 0; i < len / 2; i++) {
            char temp = input[i].charAt(0);
            if (temp!= '.' && temp!= 'b') continue;
            int j = len - 1 - i;
            while (j > i && (input[j].charAt(0) == '.' || input[j].charAt(0) == 'b')) {
                j--;
            }
            if (j == i) break;
            swap(input, i, j);
        }
        // System.out.println(Arrays.toString(input));
        for (int i = 0; i < len; i++) {
            switch (input[i].charAt(0)) {
                case '.':
                    break;
                case 'b':
                    input[i] = "a";
                    break;
                default:
                    if (i % 2 == 0) {
                        input[i] = "e";
                    } else {
                        input[i] = "i";
                    }
            }
        }
        return input;
    }

    private void swap(String[] arr, int a, int b) {
        String tmp = arr[a];
        arr[a] = arr[b];
        arr[b] = tmp;
    }
}
