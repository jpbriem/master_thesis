public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int n = S.length();
        for (int i = 0; i < n; ) {
            char c = S.charAt(i);
            if (c == '.') break;
            int j = i + 1, count = 0;
            while (j < n && S.charAt(j) == c) {
                ++count;
                ++j;
            }
            // add the first group of chars to result list
            StringBuilder sb = new StringBuilder();
            for (int k = 0; k <= count; ++k) {
                sb.append(c);
            }
            res.add(sb.toString());
            i = j;
        }
        // check remaining string and append it to each substring in result list
        for (int i = 0; i < res.size(); ++i) {
            StringBuilder sb = new StringBuilder(res.get(i));
            for (int j = i; j < res.size(); ++j) {
                sb.append(S.substring(n - (res.size() - j)));
                res.set(j, sb.toString());
            }
        }
        return res.toArray(new String[res.size()]);
    }
}
