public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i) == '.' || dominoes.charAt(i) == 'L') {
                arr[i] = '.';
            } else {
                arr[i] = 'R';
            }
        }
        // System.out.println("arr=" + Arrays.toString(arr));
        for (int i = 0, j = 0; i < len && j < len; ) {
            while (j < len && arr[j]!= 'L') {
                ++j;
            }
            if (j >= len) break;
            int k = j - 1;
            while (k > i && arr[k] == 'R') --k;
            if (k <= i) continue;
            int d = Math.abs(k - j);
            for (int l = 0; l < d / 2; ++l) {
                arr[j - l - 1] = 'L';
                arr[k + l] = 'R';
            }
            if (d % 2 == 1) {
                arr[k] = 'R';
            }
            i = j + 1;
            j += d;
        }
        StringBuilder sb = new StringBuilder();
        for (char c : arr) {
            sb.append(c);
        }
        return sb.toString();
    }
}
