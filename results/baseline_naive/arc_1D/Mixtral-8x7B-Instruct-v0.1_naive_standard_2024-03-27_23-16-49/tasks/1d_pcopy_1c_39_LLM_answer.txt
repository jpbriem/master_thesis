public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        helper(S, 0, "", res);
        return res.toArray(new String[res.size()]);
    }

    private void helper(String s, int idx, String curStr, List<String> res){
        if (idx == s.length()){
            res.add(curStr);
            return;
        }

        char c = s.charAt(idx);
        if (Character.isLetter(c)){
            helper(s, idx + 1, curStr + c, res);
        } else if (c == '.'){
            helper(s, idx + 1, curStr + ".", res);
        } else { // c is number
            int count = 0;
            while (idx < s.length() && Character.isDigit(s.charAt(idx))){
                count = count * 10 + (s.charAt(idx++) - '0');
            }
            for (int i = 0; i < count; i++){
                helper(s, idx, curStr, res);
            }
        }
    }
}
