function solution(str) {
    let count = {};
    let result = '';
    for (let i = 0; i < str.length; i++) {
        if (count[str[i]] && count[str[i]] + 1 <= str.length / 2) {
            count[str[i]]++;
        } else {
            count[str[i]] = 1;
        }
    }
    for (const key in count) {
        result += Array(count[key]).fill(key).join('');
    }
    return result;
}
