    private static void replaceEwithG(char[] arr) {
        int start = 0, end = arr.length - 1;
        while (start < end && arr[start] == '.') {
            start++;
        }
        while (end > start && arr[end] == '.') {
            end--;
        }
        if (start >= end || arr[start]!= 'e' || arr[end]!= 'e') return;
        //replace e with g at the first and last position of group
        arr[start] = 'g';
        arr[end] = 'g';
        //find next group from right side
        while (++end < arr.length && arr[end]!= 'e');
        //find previous group from left side
        while (--start >= 0 && arr[start]!= 'e');
        //if there is no more groups then return
        if (start >= end) return;
        //replace all elements between two groups with a
        for (int i=start+1;i<end;i++) {
            arr[i] = 'a';
        }
    }
