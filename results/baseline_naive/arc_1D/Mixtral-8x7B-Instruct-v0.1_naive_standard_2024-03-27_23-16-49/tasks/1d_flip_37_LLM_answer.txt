public class Solution {
    public String pushDominoes(String dominoes) {
        int len = dominoes.length();
        char[] arr = new char[len];
        Arrays.fill(arr, '.');
        for (int i = 0; i < len; ++i) {
            if (dominoes.charAt(i)!= '.') {
                arr[i] = dominoes.charAt(i);
            }
        }
        boolean leftToRight = true;
        int force = 0;
        for (int i = 0; i < len; ++i) {
            if (arr[i] == '.' && force!= 0) {
                arr[i] = force > 0? 'R' : 'L';
            } else if (arr[i]!= '.') {
                if (leftToRight) {
                    if (force > 0) {
                        arr[i - 1] = 'R';
                    }
                    force = arr[i] == 'L'? -1 : 1;
                } else {
                    if (force < 0) {
                        arr[i - 1] = 'L';
                    }
                    force = arr[i] == 'R'? 1 : -1;
                }
            }
            if (arr[i] == 'R') {
                leftToRight = false;
            } else if (arr[i] == 'L') {
                leftToRight = true;
            }
        }
        StringBuilder sb = new StringBuilder();
        for (char c : arr) {
            if (c == 'R') {
                sb.append('F');
            } else if (c == 'L') {
                sb.append('E');
            } else {
                sb.append('.');
            }
        }
        return sb.toString().replace("E", "L").replace("F", "R");
    }
}
