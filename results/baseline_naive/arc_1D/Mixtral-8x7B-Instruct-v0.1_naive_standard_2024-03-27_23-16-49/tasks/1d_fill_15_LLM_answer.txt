public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        helper(S, 0, "", res);
        return res.toArray(new String[res.size()]);
    }

    private void helper(String s, int idx, String cur, List<String> res){
        if (idx == s.length()){
            res.add(cur);
            return;
        }

        char c = s.charAt(idx);
        if (c!= '.' && c!= '*'){
            helper(s, idx + 1, cur + c, res);
        } else if (c == '.'){
            helper(s, idx + 1, cur + ".", res);
        } else { // c == '*'
            for (int i = 0; i < cur.length(); i++){
                helper(s, idx + 1, cur.substring(0, i) + "*" + cur.substring(i), res);
            }
            helper(s, idx + 1, cur + "*", res);
        }
    }
}
