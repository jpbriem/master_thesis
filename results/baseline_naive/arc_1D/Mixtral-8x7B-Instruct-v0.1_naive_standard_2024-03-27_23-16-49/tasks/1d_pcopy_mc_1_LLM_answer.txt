    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int len = S.length();
        for (int i = 0; i < len; ) {
            char c = S.charAt(i);
            if (Character.isLetter(c)) {

                while (i < len && Character.isLetter(S.charAt(i))) {

                }
