public class Solution {
    public String[] freqAlphabets(String[] s) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length; ++i) {
            if (s[i].equals(".")) continue;
            int len = s[i].length();
            if (len == 1 || s[i].charAt(len - 2)!= '#') {
                sb.append((char)(s[i].charAt(0) - '0' + 'a'));
            } else {
                sb.append((char)((Integer.parseInt(s[i].substring(0, len-2)) * 10 + Integer.parseInt(s[i].substring(len-2))) + 'a' - 1));
            }
        }
        return sb.toString().split("");
    }
}
