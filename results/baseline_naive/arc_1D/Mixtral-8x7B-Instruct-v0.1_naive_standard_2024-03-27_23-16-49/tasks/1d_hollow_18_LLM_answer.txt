public class Solution {
    public String[] expand(String S) {
        List<String> res = new ArrayList<>();
        int len = 0;
        while (len < S.length()) {
            char c = S.charAt(len);
            if (c == '.') break;
            int count = 0;
            while (len < S.length() && S.charAt(len++) == c) count++;
            len--; // back to the last letter of this group
            for (int i = 0; i <= count; i++) {
                StringBuilder sb = new StringBuilder();
                if (i!= 0) sb.append(c, i);
                if (len + 1 < S.length()) {
                    String subS = S.substring(len + 1);
                    sb.append(expand(subS)[0]);
                }
                res.add(sb.toString());
            }
            len += 2; // skip the two separators
        }
        if (len == S.length()) res.add("");
        Collections.sort(res, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                return Integer.compare(o1.length(), o2.length());
            }
        });
        String[] ans = new String[res.size()];
        for (int i = 0; i < res.size(); i++) ans[i] = res.get(i);
        return ans;
    }
}
