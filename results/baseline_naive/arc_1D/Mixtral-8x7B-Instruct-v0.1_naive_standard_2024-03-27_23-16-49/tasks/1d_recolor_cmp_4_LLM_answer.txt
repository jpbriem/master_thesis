public class Solution {
    public String[] findWords(String[] words) {
        List<String> result = new ArrayList<>();
        for (String word : words) {
            if (isValid(word)) {
                result.add(word);
            }
        }
        return result.toArray(new String[result.size()]);
    }

    private boolean isValid(String word) {
        int firstCharCode = Character.toLowerCase(word.charAt(0));
        Set<Character> set = getSetByFirstCharCode(firstCharCode);
        for (int i = 1; i < word.length(); i++) {
            char c = Character.toLowerCase(word.charAt(i));
            if (!set.contains(c)) {
                return false;
            }
        }
        return true;
    }

    private Set<Character> getSetByFirstCharCode(int code) {
        switch (code) {
            case 97: // a
            case 101: // e
            case 105: // i
            case 111: // o
            case 117: // u
                return createSet('a', 'e', 'i', 'o', 'u');
            case 108: // l
            case 116: // t
                return createSet('l', 't');
            default:
                throw new IllegalArgumentException("Invalid input");
        }
    }

    private Set<Character> createSet(char... chars) {
        Set<Character> set = new HashSet<>(chars.length);
        Collections.addAll(set, chars);
        return set;
    }
}
