class Solution:
    def transform(self, arr):
        """
        思路：
            将数组中的元素分成两类，一种是'a'，另外一种是其他字符。
            然后对于每个不同的字符进行处理，如果该字符在原始数组中出现了奇数次，那么就把这些字符全部转换为'a'；
                如果该字符在原始数组中出现了偶数次，那么就把这些字符全部转换为'e'。

        # 统计所有元素出现的次数
        count = collections.Counter(arr)
        for char in count:
            if count[char] % 2 == 0:
                arr = [char if i!= char else "e" for i in arr]
            elif count[char] % 2 == 1:
                arr = [char if i!= char else "a" for i in arr]
        return arr
