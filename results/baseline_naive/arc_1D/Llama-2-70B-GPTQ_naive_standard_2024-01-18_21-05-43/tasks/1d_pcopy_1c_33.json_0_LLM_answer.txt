class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        """
        1. sort each string in ascending order
        2. compare two strings by comparing their sorted versions
            - if they're equal, then add them to a list of anagrams
                - otherwise, create a new list for that word
                  (and add it to the list)
        3. return the list of lists
        :param strs:
        :return:
        """
        # O(nlogn) time | O(n) space
        # n = len(strs)
        # logn = number of times we iterate over the list
        # O(n) space because we have to store all the words in memory
        # at once
        from itertools import groupby
        from operator import itemgetter

        # sort each string in ascending order
        strs = [sorted(word) for word in strs]

        # compare two strings by comparing their sorted versions
        # if they're equal, then add them to a list of anagrams
        # otherwise, create a new list for that word
        # and add it to the list
        result = []

            result += [[list(item)] for item in group]

        return result
