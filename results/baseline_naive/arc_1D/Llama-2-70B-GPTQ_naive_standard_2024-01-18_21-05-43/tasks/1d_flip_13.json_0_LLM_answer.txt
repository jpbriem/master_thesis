class Solution(object):
    def findReplaceString(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: str
        """
        
        # sol 1: O(n) time and space
        # use a dictionary to store the index of each char in S
        d = {}
        for i, c in enumerate(S):
            if c not in d:
                d[c] = [i]
            else:
                d[c].append(i)
                
        # iterate through all chars in T
        res = []
        for t in T:
            
            # check if we have seen this char before
            if t in d:
                # get its indices from S
                inds = d[t]
                # replace it with the next char in S
                res += S[inds[-1]+1:] + S[:inds[-1]]
            else:
                # append the char itself
                res.append(t)
        return "".join(res)
