class Solution:
    def transform(self, arr: List[str]) -> str:
        """
        思路：双指针，一个从左往右扫描，一个从右往左扫描。
            当前元素与后面的元素相等时，将其移动到后面；
            当前元素与前面的元素相等时，将其移动到前面。
            O(n) 时间复杂度，O(1) 空间复杂度。
        """
        left = right = 0
        while True:
            if left >= len(arr):
                break
            elif arr[left] == arr[right]:
                arr[left], arr[right] = arr[right], arr[left]
                left += 1
                right -= 1
            else:
                left += 1
        return "".join(arr)
