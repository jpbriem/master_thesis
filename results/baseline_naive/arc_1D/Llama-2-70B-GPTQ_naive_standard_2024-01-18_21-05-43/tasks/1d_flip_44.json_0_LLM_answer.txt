import collections

class Solution(object):
    def findReplaceString(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: str
        """
        
        # sol 1: O(n) time and space
        # use a dictionary to store the frequency of each character in S
        freq = collections.Counter(S).items()
        for i in range(len(T)):
            if T[i] not in freq or freq[T[i]] == 0:
                return ""
            
            freq[T[i]] -= 1
            if freq[T[i]] < 0:
                return ""
                
        return "".join([x[0]*abs(y)+1 for x, y in freq])
