# Definition for a Node.
class Node(object):
    def __init__(self, val, children=None):
        self.val = val
        self.children = children


class Solution(object):
    def transformToPreorder(self, A):
        """
        :type A: List[str]
        :rtype: str
        """
        
        # 2018/7/4
        # O(n) time complexity
        # O(n) space complexity
        if not A or len(A) == 0: return ''
        root = Node('')
        stack = []
        for i in range(len(A)):
            node = Node(A[i])
            while stack and stack[-1].val < node.val:
                curNode = stack.pop()
                if curNode.children is None:
                    curNode.children = [node]
                else:
                    curNode.children.append(node)
            stack.append(node)
            
        res = []
        while stack:
            node = stack.pop()
            res.insert(0, node.val)
            if node.children:
                for child in reversed(node.children):
                    stack.append(child)
        return "".join(res)
