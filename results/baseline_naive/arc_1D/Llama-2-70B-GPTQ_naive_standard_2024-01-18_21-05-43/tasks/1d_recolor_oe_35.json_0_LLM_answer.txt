        """

                    解决方案：通过判断单词长度来确定该单词是否为大写单词。
            - 根据拼接后的新单词进行替换。

            - time complexity : O(n), where n is the length of `text`.
            - space complexity : O(m), where m is the number of words in `words` list.

        # convert all words to lowercase
        text = text.lower()
        for word in self.__WORDS__:

            matcher = re.finditer(word, text)
            for match in matcher:

                    text = text[:start] + "i" + text[end:]
                else:

                    text = text[:start] + "i" * (len(match.group())) + text[end:]
        return text
