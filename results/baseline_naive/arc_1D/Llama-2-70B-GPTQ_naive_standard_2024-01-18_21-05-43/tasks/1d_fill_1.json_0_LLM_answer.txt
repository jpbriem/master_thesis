from typing import List, Optional


class Solution:
    def transform(self, s: str) -> str:
        """
        思路：滑动窗口，记录每个字符出现的次数。
                如果当前字符与上一个字符相同，则将两者之间的所有字符都变为当前字符；
                否则，只需要将第一个不同的字符变成当前字符即可。

        时间复杂度：O(n)
        空间复杂度：O(1)
        """
        if not s or len(s) == 0: return ""
        result = []
        count = 1
        for i in range(len(s)):
            if i > 0 and s[i]!= s[i - 1]:
                result += [s[i]] * (count + 1)
                count = 1
            else:
                count += 1
        return "".join(result)
