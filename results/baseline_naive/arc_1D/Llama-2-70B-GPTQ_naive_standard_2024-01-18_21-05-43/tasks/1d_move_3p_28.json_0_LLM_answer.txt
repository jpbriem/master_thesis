import collections

class Solution:
    def transform(self, arr: List[str]) -> str:
        """
        思路：
            将数组转换为字典，key 是元素值，value 是该元素出现的次数。
            然后按照 key 从小到大排序，并且对每个 key 进行处理，如果 value > 0，则将其添加到结果中，否则不添加。
            O(n) time complexity.
            O(n) space complexity.
        :param arr:
        :return:
        """
        d = collections.Counter(arr).items()
        d = sorted(d, key=lambda x: x[0], reverse=True)
        res = []

                res += [k] * v
        return "".join(res)
