# Definition for a Node.
class Node(object):
    def __init__(self, val, children=None):
        self.val = val
        self.children = children


class Solution(object):
    def sortString(self, s):
        """
        :type s: str
        :rtype: str
        """
        # 1. build tree from string
        root = Node('')
        stack = [root]
        for c in reversed(s):
            node = stack[-1].children[ord(c) - ord('a')] if len(stack) > 0 else None
            if not node:
                node = Node(c)
                stack[-1].add_child(node)
            stack.append(node)
        # 2. traverse tree to get sorted string
        res = []
        while stack:
            node = stack.pop()
            if node.is_leaf():
                res.insert(0, node.get_char())
            elif node.has_children():
                stack.extend([node.left(), node])
        return "".join(res)
