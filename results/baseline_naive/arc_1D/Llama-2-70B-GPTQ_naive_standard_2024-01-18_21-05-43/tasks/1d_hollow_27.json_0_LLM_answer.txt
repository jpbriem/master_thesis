class Solution:
    def transform(self, arr: List[str]) -> str:
        """
        思路：双指针，一个从左往右扫描，一个从右往左扫描。
            当前元素与后面的元素相等时，将其转换为前面的元素；

            O(n) 时间复杂度，O(1) 空间复杂度。
        """
        left = right = 0
        for i in range(len(arr)):

                arr[left], arr[right] = arr[right], arr[left]

        return "".join(arr)
