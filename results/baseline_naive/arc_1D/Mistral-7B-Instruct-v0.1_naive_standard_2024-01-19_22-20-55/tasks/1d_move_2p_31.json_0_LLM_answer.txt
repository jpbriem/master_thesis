def permute(lst):
    # Count the number of occurrences of each element in lst
    counts = {}
    for elem in lst:
        if elem not in counts:
            counts[elem] = 1
        else:
            counts[elem] += 1
    
    # Create a new list with the elements from lst, repeating the ones with more than 1 count
