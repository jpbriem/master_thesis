class Solution:
     def longestConsecutive(self, nums):
         """
         :type nums: List[int]
         :rtype: int
         """
         ans = 0
         
         num_set = set(nums)
         
         for num in num_set:
             if num == 0:
                 continue
             
             left_num = num - 1
             right_num = num + 1
             
             left_flag = True
             right_flag = True
            
             while left_num >= 0 and left_flag:
                 if left_num not in num_set:
                     break
                 else:
                     left_flag = False
                     
             while right_num < len(num_set) and right_flag:
                 if right_num not in num_set:
                     break
                 else:
                     right_flag = False
                    
             if left_num < right_num:
                 ans += right_num - left_num - 1
             else:
                 ans += right_num - left_num
                
         return ans
