```python
class Solution:
    def minCut(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0:
            return 0

        stack = []
        visited = set()
        ans = 0
        for i in range(len(nums)):
            if nums[i] not in visited:
                stack.append(i)
                while len(stack)>0:
                    v = stack.pop()
                    visited.add(nums[v])
                    ans += 1
                    if nums[v] == nums[i]:
                        break

        return ans
```
