class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            # sort the string to get anagram
            sorted_s = ''.join(sorted(s))
            if sorted_s not in d:
                d[sorted_s] = []
            d[sorted_s].append(s)
        
        result = []
        for k, v in d.items():
            result.append(v)
        
        return result
