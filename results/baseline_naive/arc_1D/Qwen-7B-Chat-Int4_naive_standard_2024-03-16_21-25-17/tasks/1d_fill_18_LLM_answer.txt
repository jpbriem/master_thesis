class Solution:
    def numIsland(self, grid: List[List[int]]) -> int:
        if not grid or len(grid) == 0 or len(grid[0]) == 0:
            return 0
        
        directions = [(0, -1), (1, 0), (-1, 1), (0, 1)]
        
        m, n = len(grid), len(grid[0])
        visited = [[False] * n for _ in range(m)]
        count = 0
        
        def dfs(x, y):
            nonlocal count
            
            if x < 0 or y < 0 or x >= m or y >= n or grid[x][y] == 0 or visited[x][y]:
                return
            
            visited[x][y] = True
            
            count += 1
            
            for dx, dy in directions:
                dfs(x + dx, y + dy)
                
        dfs(0, 0)
        
        return count
