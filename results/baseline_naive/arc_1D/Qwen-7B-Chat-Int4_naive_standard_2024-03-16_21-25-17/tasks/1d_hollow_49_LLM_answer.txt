```python
class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) <= 1:
            return nums[0] 

        count = {}
        for num in nums:
            if num in count:
                count[num] += 1
            else:
                count[num] = 1

        max_count = 0
        result = None
        for k, v in count.items():
            if v > max_count:
                max_count = v
                result = k

        return result
```
