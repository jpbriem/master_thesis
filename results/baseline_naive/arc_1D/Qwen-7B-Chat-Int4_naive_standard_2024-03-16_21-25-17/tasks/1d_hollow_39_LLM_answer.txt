class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        result = max(count.values())
        
        for i in range(len(nums)):
            if count[nums[i]] == result:
                return nums[i]
            else:
                count[nums[i]] -= 1
        
        return None
