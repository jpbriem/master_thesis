class Solution:
    def countCircles(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        
        if n == 1 or n == 2:
            return n
        
        arr = [[i for i in range(n)] for _ in range(n)]
        arr[0][0] = arr[n-1][n-1] = True
        for i in range(1, n-1):
            arr[i][i] = True
        
        left, right = 0, n - 1
        while left < right:
            # move the current circle to the next position on the same row
            if arr[left][right]:
                temp = arr[left].copy()
                arr[left][left:right+1] = arr[right][left:right+1].copy()
                arr[right][left:right+1] = temp.copy()
                
            # move the current circle to the previous position on the same column
            if arr[left][right]:
                temp = arr[left].copy()
                arr[left:right+1][left] = arr[left:right+1][right].copy()
                arr[right][left:right+1][right] = temp.copy()
            
            # move the current circle to the next position on different rows and columns
            if arr[left][right]:
                temp = arr[left].copy()
                arr[left:right+1][left+1:] = arr[left:right+1][left+1:].copy()
                arr[right][left:right+1][:right-left-1] = arr[right][left:right+1][:right-left-1].copy()
                arr[left][left:right+1][left] = temp.copy()
                
            left += 1
            right -= 1
        
        res = sum(arr[i][i] for i in range(n))
        
        return res
