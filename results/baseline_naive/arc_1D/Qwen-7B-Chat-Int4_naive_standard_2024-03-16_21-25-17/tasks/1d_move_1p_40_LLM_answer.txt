```python
class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        
        max_count = max(count.values())
        candidate = None
        
        for num in count.keys():
            if count[num] == max_count and (candidate is None or count[candidate] < count[num]):
                candidate = num
                
        return candidate
```
