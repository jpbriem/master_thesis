class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        left = []
        right = []
        
        for i in range(len(arr)):
            if i == 0 or arr[i] >= arr[i - 1]:
                left.append(i)
            else:
                break
        
        for i in reversed(range(len(arr))):
            if i == len(arr) - 1 or arr[i] <= arr[i + 1]:
                right.append(i)
            else:
                break
                
        # print(left, right)
        
        return min(max(right) - left + 1, len(arr))
