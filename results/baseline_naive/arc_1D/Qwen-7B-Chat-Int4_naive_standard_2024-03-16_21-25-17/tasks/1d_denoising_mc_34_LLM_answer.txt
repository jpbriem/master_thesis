class Solution:
    def largestComponent(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: int
        """
        m = len(matrix)
        n = len(matrix[0])
        
        if m == 0 or n == 0:
            return 0
        
        visited = [[False] * (n+1) for _ in range(m)]
        stack = [(i,j) for i in range(m) for j in range(n) if not visited[i][j]]
        
        component_size = 0
        
        while stack:
            node = stack.pop()
            
            if visited[node[0]][node[1]]:
                continue
            
            visited[node[0]][node[1]] = True
            
            component_size += 1
            
            for x,y in [(x-1,y), (x,y-1), (x+1,y), (x,y+1)]:
                if 0 <= x < m and 0 <= y < n and matrix[x][y] != 0 and not visited[x][y]:
                    stack.append((x,y))
                    
        return component_size
