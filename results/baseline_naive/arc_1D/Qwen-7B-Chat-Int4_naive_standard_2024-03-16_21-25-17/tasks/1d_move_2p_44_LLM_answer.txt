class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0:
            return 0
        
        l = 0
        r = len(arr) - 1

        res = 0
        while(l < r):
            temp_l = arr[l]
            temp_r = arr[r]
            
            while(l < r and arr[l] <= temp_l):
                l += 1
            
            while(l < r and arr[r] >= temp_r):
                r -= 1
                
            if l < r:
                arr[l], arr[r] = arr[r], arr[l]
                
                res += 1
            else:
                break
        
        return res
