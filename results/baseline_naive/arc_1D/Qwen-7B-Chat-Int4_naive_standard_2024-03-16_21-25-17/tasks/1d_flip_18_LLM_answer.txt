class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        
        # create a dictionary to store the index of each element in the array
        dic = {}
        for i in range(n):
            if arr[i] not in dic:
                dic[arr[i]] = []
            dic[arr[i]].append(i)

        # sort the dictionary by key
        sorted_keys = sorted(dic.keys())
        chunks = []

        # iterate over the sorted keys and add elements from the dictionary to the chunks
        for key in sorted_keys:
            chunk_start = dic[key][0]
            chunk_end = dic[key][-1]
            
            while chunk_start < chunk_end:
                chunks.append(arr[chunk_start])
                chunk_start += 1

        return len(chunks)
