class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            k = "".join(sorted(s))
            if k not in d:
                d[k] = []
            d[k].append(s)
        
        res = [[] for _ in range(len(d))]
        for key, values in d.items():
            res[d[key]].extend(values)
            
        return res
