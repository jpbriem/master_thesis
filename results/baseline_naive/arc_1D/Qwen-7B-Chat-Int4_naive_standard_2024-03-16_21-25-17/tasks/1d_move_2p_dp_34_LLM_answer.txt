```python
class Solution:
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        res = []
        if not nums:
            return 0
        
        for num in nums:
            flag = False
            start = num - 1
            end = num + 1
            
            while start >= 0 and end < len(nums) and nums[start] == nums[end]:
                if not flag:
                    res.append(num)
                else:
                    break
                start -= 1
                end += 1
                flag = True
                
        return max(res)
```
