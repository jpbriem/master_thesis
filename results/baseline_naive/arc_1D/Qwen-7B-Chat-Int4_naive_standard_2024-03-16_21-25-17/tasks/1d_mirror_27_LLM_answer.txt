    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        for s in strs:
            if s not in dic:
                dic[s] = []
            dic[s].append(s)
        
        res = [[] for _ in range(len(dic))]
        for k, v in dic.items():
            res[v.index(k)].extend(v)
            
        return res
