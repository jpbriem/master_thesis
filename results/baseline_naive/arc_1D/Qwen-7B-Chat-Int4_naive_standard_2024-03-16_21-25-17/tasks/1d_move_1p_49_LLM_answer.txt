class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        
        left = right = 0
        res = 0

        while right < len(nums) - 1:
            mid = (left + right) // 2

            if nums[mid] > nums[right]:
                res += 1
                left = mid + 1
            else:
                right = mid + 1

        return res + 1
