```python
class Solution:
    def longestSubsequence(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        
        # dp[i][j] stores the length of the longest subsequence ending at index i and having prefix sum j.
        dp = [[0]*n for _ in range(n)]
        
        # If we have seen a number before, we can simply add it to the current subsequence if its prefix sum is less than or equal to the previous one.
        for i in range(1, n):
            for j in range(i):
                if nums[j] <= nums[i]:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1]) + 1
        
        return max(dp[n-1])
```
