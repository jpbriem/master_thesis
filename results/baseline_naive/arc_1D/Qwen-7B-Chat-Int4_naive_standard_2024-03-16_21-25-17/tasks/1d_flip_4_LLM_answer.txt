class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        if not nums:
            return []
        
        seen = set()
        result = []
        for num in nums:
            if num not in seen:
                seen.add(num)
                result.append(num)
                
        return result
