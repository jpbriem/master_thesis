class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0:
            return [[]]
        
        res = []
        self.dfs(nums, [], res)
        
        return res
    
    def dfs(self, nums, path, res):
        for i in range(len(nums)):
            if not nums[i]:
                continue
            
            new_path = path + [nums[i]]
            
            # try to add current element to existing paths
            for subpath in res:
                if nums[i] in subpath:
                    break
                
            else:
                # add this path as a new solution
                res.append(new_path)
                
            # remove the current element from the list so that we can try other elements
            nums[i] = False
