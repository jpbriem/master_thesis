```python
class Solution:
    def longestSubsequence(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        dp = [[-1 for _ in range(n)] for _ in range(n)]
        
        for i in range(n):
            dp[i][i] = nums[i]
            
        for l in range(2, n+1):
            for i in range(n-l+1):
                j = i+l-1
                
                if dp[j][j-1] != -1 and dp[j][j-1] > nums[i]:
                    dp[i][j] = dp[j][j-1]
                
                else:
                    dp[i][j] = max(dp[i][j], nums[i] + dp[i+1][j])
                    
        return dp[0][-1]
```
