```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[int]]
        """
        anagram_dict = {}
        for i in range(len(strs)):
            str_key = ''.join(sorted(strs[i]))
            if str_key not in anagram_dict:
                anagram_dict[str_key] = []
            anagram_dict[str_key].append(i)
        
        result = [[] for _ in range(len(strs))]
        for key, values in sorted(anagram_dict.items()):
            index = values[0]
            result[index].extend(values[1:])
            
        return result
```
