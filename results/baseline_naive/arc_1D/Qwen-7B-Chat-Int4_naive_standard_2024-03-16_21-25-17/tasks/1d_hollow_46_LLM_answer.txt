```python
class Solution:
    def countLarger(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0

        max_val = max(nums)
        num_set = set()
        res = 0
        
        for num in nums:
            if num > max_val or num == max_val:
                res += 1
            else:
                num_set.add(num)

        return len(num_set) - res
```
