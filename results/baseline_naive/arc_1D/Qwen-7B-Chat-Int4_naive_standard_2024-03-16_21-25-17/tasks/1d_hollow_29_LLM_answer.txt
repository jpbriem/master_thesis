class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 1:
            return 0
        
        # If the array is already sorted then we can return 1
        if arr[0] < arr[-1]:
            return 1
        
        i = j = 0
        while i < len(arr):
            while j < len(arr) and arr[j] <= arr[i]:
                j += 1
            
            if j - i > 1:
                break
            else:
                i += 1
                
        return j - i + 1
