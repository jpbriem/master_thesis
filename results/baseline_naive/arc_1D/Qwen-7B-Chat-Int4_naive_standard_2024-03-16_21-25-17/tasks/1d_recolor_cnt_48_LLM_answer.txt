class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0:
            return 0
        
        result = 0
        left = right = 0
        
        while right < len(arr):
            while left < right and arr[left] > arr[right]:
                left += 1
            
            while left < right and arr[right] >= arr[left]:
                right -= 1
            
            result += 1
            
            # If the element is not in its sorted position then move it to the beginning of array
            if arr[left] != arr[right]:
                arr[left], arr[right] = arr[right], arr[left]
                
        return result
