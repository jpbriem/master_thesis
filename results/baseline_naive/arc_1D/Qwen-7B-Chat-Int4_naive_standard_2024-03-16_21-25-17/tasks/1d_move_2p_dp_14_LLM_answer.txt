class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for i in range(len(strs)):
            if strs[i] not in dic.keys():
                dic[strs[i]] = []
            
            dic[strs[i]].append(i)
        
        res = [[] for _ in range(len(dic))]
        
        for key in dic.keys():
            res[dic[key][0]].extend(res[dic[key][1]])
        
        return res
