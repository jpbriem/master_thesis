class Solution:
    def removeDuplicates(self, nums: List[int]) -> List[int]:
        if not nums:
            return []
        
        count = {}
        res = []
        for num in nums:
            if num not in count:
                count[num] = 1
                res.append(num)
            else:
                count[num] += 1
        
        return res
