```python
class Solution:
    def largestRectangleArea(self, height: List[int]) -> int:
        n = len(height)
        if n == 0 or n == 1:
            return 0
        
        left = right = max_value = 0

        for i in range(n):
            while left < i and height[left] >= height[i]:
                left += 1
                
            while right > i and height[right-1] <= height[i]:
                right -= 1
                
            if left < i and right > i:
                max_value = max(max_value, (right - left + 1) * height[i])
                
        return max_value
```
