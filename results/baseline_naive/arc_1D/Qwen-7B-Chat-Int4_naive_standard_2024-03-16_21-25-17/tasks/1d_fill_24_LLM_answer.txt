class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = [[] for _ in range(len(strs))]
        
        d = {}
        for s in strs:
            if len(s) == 0:
                continue
            
            key = ''.join(sorted(s))
            
            if key not in d:
                d[key] = []
                
            d[key].append(s)
        
        for k in d.keys():
            ans[d[k][0]] += d[k][1:]
        
        return ans
