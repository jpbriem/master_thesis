class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        
        max_count = max(count.values())
        result = None
        
        for key in count.keys():
            if count[key] == max_count:
                result = key
                
        return result
