class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 0:
            return 0

        left = right = i = 0
        while i < len(arr):
            while right < len(arr) and arr[right] <= arr[i]:
                right += 1
            if left < right:
                temp = arr[left:right+1]
                j = left + 1
                while j < right and arr[j] > temp[0]:
                    j += 1
                for k in range(left, j):
                    arr[k], arr[j] = arr[j], arr[k]
                left = j
            else:
                break
            i += 1
        
        return (right - left)
