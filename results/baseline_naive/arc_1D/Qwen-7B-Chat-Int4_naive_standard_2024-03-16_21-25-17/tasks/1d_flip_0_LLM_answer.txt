class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0:
            return 0
        
        # Create a dictionary to store the frequency of each element in arr
        freq_dict = {}
        
        for num in arr:
            if num not in freq_dict:
                freq_dict[num] = 0
            freq_dict[num] += 1
        
        # Find the maximum frequency that is less than or equal to n/2
        max_freq = min(2, sum(freq_dict.values()) // len(arr))
        
        # Count the number of elements with frequency greater than or equal to max_freq
        count = sum([freq_dict[i] >= max_freq for i in freq_dict])
        
        return count
