class Solution:
    def numRabbits(self, answers: List[int]) -> int:
        ans = []
        for i in range(len(answers)):
            if answers[i] == 0:
                continue
            else:
                ans.append(i)
        
        ans.sort()
        
        count = 0
        prev = -1
        for i in range(len(ans)):
            if prev != ans[i]:
                count += (ans[i] - prev) * 2
            prev = ans[i]
            
        return count
