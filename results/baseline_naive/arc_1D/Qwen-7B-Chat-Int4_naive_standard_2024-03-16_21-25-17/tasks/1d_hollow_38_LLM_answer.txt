class Solution:
    def removeElements(self, nums: List[int]) -> List[int]:
        count = {i: 0 for i in range(1, len(nums)+1)}
        
        for num in nums:
            count[num] += 1
            
        res = []
        for i in range(len(nums)):
            if count[nums[i]] == 0:
                continue
            else:
                count[nums[i]] -= 1
                res.append(nums[i])
                
        return res
