class Solution:
    def largestRectangleArea(self, self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        if not height or len(height) == 0:
            return 0

        max_area = 0
        left = 0
        right = 0
        for i in range(len(height)):
            while height[i] > 0 and (left < right or right == 0):
                if height[left] <= height[right]:
                    left += 1
                else:
                    right -= 1
            if height[i] > 0:
                max_area = max(max_area, (right - left + 1) * height[i])
        
        return max_area
