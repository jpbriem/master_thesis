```python
class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        if n == 0 or n == 1:
            return 0

        left, right = 0, n-1
        res = 0
        
        while left < right:
            mid = (left + right) // 2
            if arr[mid] <= arr[right]:
                res += 1
                left = mid + 1
            else:
                right = mid
        return res
```
