```python
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        n = len(heights)
        
        if n == 0 or n == 1:
            return 0
        
        stack = []
        res = 0
        
        for i in range(n):
            while len(stack) > 0 and heights[stack[-1]] >= heights[i]:
                top = stack.pop()
                left = max(0, i - stack[-1])
                right = min(i + 1, n - 1)
                
                res += (right - left + 1) * heights[top]
            
            stack.append(i)
        
        return res
```
