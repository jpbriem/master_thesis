class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        groups = [[] for _ in range(len(set(strs)))]
        
        for s in strs:
            index = [str.index(s) for str in groups]
            groups[index].append(s)
            
        return groups
