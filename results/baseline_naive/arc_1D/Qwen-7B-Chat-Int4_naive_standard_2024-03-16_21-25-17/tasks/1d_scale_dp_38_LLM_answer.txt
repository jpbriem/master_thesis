```python
class Solution:
    def findLargestRectangle(self, height: List[int]) -> int:
        if not height or len(height) == 0:
            return 0
        
        max_area = 0
        stack = []
        
        for i in range(len(height)):
            while stack and height[stack[-1]] >= height[i]:
                top = stack.pop()
                left = i - height[top] + 1
                
                area = (left + 1) * (height[i])
                
                if area > max_area:
                    max_area = area
                    
            stack.append(i)
            
        return max_area
```
