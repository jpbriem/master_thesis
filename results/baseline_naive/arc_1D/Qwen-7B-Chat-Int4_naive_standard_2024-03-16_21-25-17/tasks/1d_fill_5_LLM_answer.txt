            # move the left pointer forward.
            while arr[left] <= arr[right]:
                left += 1

            # Otherwise, move the right pointer backward until we find an element that is greater than or equal to the current maximum value.
            while arr[left] > arr[right]:

            # Update the result count whenever a new maximum value is found.
            result += right - left + 1

        return result

- Whenever a new maximum value is found, we update the result count by adding the number of elements between the left and right pointers (i.e., `right - left + 1`).
- Finally, we return the result count.
