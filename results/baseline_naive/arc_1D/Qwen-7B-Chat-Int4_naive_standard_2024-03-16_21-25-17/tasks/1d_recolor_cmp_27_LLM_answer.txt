class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagram = collections.defaultdict(list)
        
        for s in strs:
            key = "".join(sorted(s))
            anagram[key].append(s)

        result = []
        for k, v in sorted(anagram.items()):
            if len(v) == 1:
                result.append([v[0]])
            else:
                result.append(sorted(v))

        return result
