class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) <= 1:
            return 0
        
        count = 1
        prev = arr[0]
        
        for i in range(1, len(arr)):
            if arr[i] < prev:
                count += 1
                prev = arr[i]
            
        return count
