```python
class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        if len(arr) == 1 or arr[0] != 0:
            return 1
        
        n = len(arr)
        
        left, right = 0, n-1
        
        while left < right:
            
            mid = (left + right) // 2
            
            if arr[mid] <= arr[left]:
                left = mid + 1
                
            else:
                right = mid
                
        return left + 1
```
