```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        d = {}
        for s in strs:
            key = ''.join(sorted(s))
            d[key] = d.get(key, []) + [s]
            
        res = [[] for _ in range(len(d))]
        for k, v in d.items():
            res[v[0]] += v[1:]
            
        return res
```
