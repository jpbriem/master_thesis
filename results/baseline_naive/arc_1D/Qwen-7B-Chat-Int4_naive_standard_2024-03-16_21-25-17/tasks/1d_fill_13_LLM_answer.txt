    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        left = right = -1
        res = 0
        
        while right < len(nums)-1 and nums[right] > nums[left]:
            left += 1
            
        for i in range(left+1, len(nums)):
            if nums[i] <= nums[right]:
                right -= 1

                res += 1
                
        return res
