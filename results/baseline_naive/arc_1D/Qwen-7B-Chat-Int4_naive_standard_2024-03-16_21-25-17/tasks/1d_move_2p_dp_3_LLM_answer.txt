class Solution:
    def groupAnagrams(self, s: str) -> List[List[str]]:
        if len(s) == 0:
            return []
        
        anagrams = {}
        for char in s:
            if char not in anagrams:
                anagrams[char] = [char]
            else:
                anagrams[char].append(char)
                
        result = []
        for key in sorted(anagrams.keys()):
            values = list(sorted(anagrams[key]))
            result.append(values)
            
        return result
