```python
class Solution:
    def longestSubarray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        
        left = right = max_val = 0
        for i in range(len(nums)):
            val = nums[i]
            
            while right < len(nums) and nums[right] == val:
                right += 1
            
            if right - left > max_val:
                max_val = right - left
                
            if val != 0:
                left = i + 1
                
        return max_val
```
