class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        for s in strs:
            # sort the string to get its hash value
            sorted_s = ''.join(sorted(s))
            if sorted_s not in dic:
                dic[sorted_s] = []
            dic[sorted_s].append(s)
        
        return [[dic[i]] for i in sorted(dic)]
