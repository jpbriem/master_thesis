def majorityElement(nums):
    count = {}
    for num in nums:
        if num not in count:
            count[num] = 1
        else:
            count[num] += 1
    
    max_count = max(count.values())
    major_element = [num for num, count in count.items() if count == max_count]
    
    return major_element
