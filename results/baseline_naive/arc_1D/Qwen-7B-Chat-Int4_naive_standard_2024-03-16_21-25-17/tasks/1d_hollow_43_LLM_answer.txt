class Solution:
    def findLargestCircle(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0:
            return 0
        
        nums.sort(reverse=True)
        
        left = 0
        right = len(nums)-1
        largest_circle = -1

        while left < right:
            mid = (left + right) // 2
            
            if nums[mid] > 0 and nums[mid] <= nums[left]:
                left = mid+1
            else:
                right = mid-1
                
        for i in range(left, len(nums)):
            if nums[i] != 0:
                largest_circle = max(largest_circle, i)

        return largest_circle
