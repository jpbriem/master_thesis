class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0 or arr[0] > arr[-1]:
            return 0
        
        left = []
        right = []
        
        for i in range(len(arr)):
            if arr[i] < arr[0]:
                left.append(i)
            else:
                break
                
        for i in reversed(range(len(arr))):
            if arr[i] > arr[-1]:
                right.append(i)
            else:
                break
                
        l = len(left)
        r = len(right)
        
        ans = min(l,r)
        
        left_max = arr[left[0]]
        right_min = arr[right[0]]
        
        while ans > 0 and (left_max < right_min or right_min < left_max):
            mid = (right[r-1]-left[l-1]+1)//2 + left[l-1]
            
            if arr[mid] >= left_max:
                ans -= 1
                left_max = arr[mid+1]
                
            elif arr[mid] <= right_min:
                ans -= 1
                right_min = arr[mid-1]
                
            else:
                ans -= 2
                left_max = arr[mid+1]
                right_min = arr[mid-1]
                
        return ans
