```python
class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        if n == 0 or n == 1:
            return 0

        prefix_sum = [0] * (n + 1)
        for i in range(1, n+1):
            prefix_sum[i] = prefix_sum[i-1] + nums[i-1]
        
        left = 0
        right = n - 1
        
        result = 0
        while left < right:
            count = 0
            current_sum = prefix_sum[right] - prefix_sum[left]
            while left < right and current_sum >= nums[left]:
                count += 1
                current_sum -= nums[left]
                left += 1
            result += count
            right -= 1
        
        return result
```
