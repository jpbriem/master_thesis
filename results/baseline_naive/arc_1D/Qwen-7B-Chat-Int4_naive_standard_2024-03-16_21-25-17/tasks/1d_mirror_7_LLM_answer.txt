class Solution:
    def numIdenticalDigits(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        res = []
        
        for i in range(len(nums)):
            if nums[i] == 0:
                res.append(i)
            
        count = 0
        while len(res) > 0 and count < 9:
            index = res.pop(0)
            count += 1
            
            for j in range(index+1, len(nums)):
                if nums[j] != 0:
                    break
                else:
                    res.append(j)
                    
        return count
