```python
class Solution:
    def largestNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: str
        """
        res = ''
        
        for i in range(len(nums)):
            if not nums[i]:
                continue
            
            s = str(nums[i])
            
            while len(res) < len(s):
                res += '0'
                
            while len(res) > len(s):
                res = res[:-1]
                
            res = int(res) + int(s)
            
        return res
```
