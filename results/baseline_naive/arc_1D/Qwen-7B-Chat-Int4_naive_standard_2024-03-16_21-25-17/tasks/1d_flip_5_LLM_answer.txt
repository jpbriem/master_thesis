class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        if not heights or len(heights) == 0:
            return 0
        
        # Sort the heights in descending order
        heights.sort(reverse=True)
        
        # Initialize two variables to store the maximum area and current area
        max_area = 0
        curr_area = 0
        
        # Iterate through each height in the list
        for i in range(len(heights)):
            # If the current height is greater than the previous height,
            # then it means that we can continue building a rectangle with this height
            if heights[i] > heights[i-1]:
                # Calculate the new width of the rectangle by subtracting one from the previous width
                new_width = heights[i-1] - 1
                
                # Add the new width to the current area
                curr_area += new_width
                
            # Otherwise, it means that we need to start a new rectangle with this height
            else:
                # Update the current area to the sum of the previous width and the current height
                curr_area += heights[i]
                
                # Reset the maximum area to the current area
                max_area = max(max_area, curr_area)
                
                # Reset the current area to zero
                curr_area = 0
                
        # Return the maximum area
        return max_area
