class Solution:
    def dominantIndex(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return -1
        
        left = right = 0
        while left < len(nums) and nums[left] == nums[right]:
            left += 1
            right += 1
            
        pivot = nums[left]
        
        i = j = left + 1
        while i < len(nums):
            while nums[j] > pivot:
                j += 1
                
            if nums[j] <= pivot:
                nums[i], nums[j] = nums[j], nums[i]
                i += 1
            else:
                break
        
        return left + 1
