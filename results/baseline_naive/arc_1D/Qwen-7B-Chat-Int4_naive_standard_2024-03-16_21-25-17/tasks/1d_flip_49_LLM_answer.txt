```python
class Solution:
     def maxChunksToSorted(self, nums):
         """
         :type nums: List[int]
         :rtype: int
         """
         if not nums:
             return 0

         left = right = -1
         res = 0
         while right < len(nums) and left >= 0:
             # 找到第一个大于等于left的数
             for i in range(left + 1, right + 1):
                 if nums[i] > left:
                     break
             else:
                 # 左右指针都向右移动，直到找到一个小于等于right的数
                 left += 1
             else:
                 # 右指针不动，左指针向右移动
                 right += 1
             res += 1
         
         return res
```
