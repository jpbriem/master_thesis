```python
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        n = len(heights)
        if n == 0 or heights[0] <= 0:
            return 0
        
        # dp[i][j] stores the maximum area that can be obtained by drawing a rectangle with height i and starting from index j
        dp = [[0 for _ in range(n)] for _ in range(n)]
        
        # initialize the first row of the dp array to be the height of the first element
        for i in range(1, n):
            dp[i][0] = heights[i-1]
            
        # iterate over the remaining rows and update the dp array accordingly
        for i in range(1, n):
            for j in range(i+1, n):
                max_height = min(dp[j-1][i], heights[j])
                dp[i][j] = max(max_height * (n - j), dp[j-1][i])
                
        return dp[-1][-1]
```
