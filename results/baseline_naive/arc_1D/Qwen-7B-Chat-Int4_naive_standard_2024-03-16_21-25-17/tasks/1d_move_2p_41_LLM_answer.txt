```python
class Solution:
    def largestNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: str
        """
        if len(nums) == 0 or len(set(nums)) < 2:
            return "0"
        
        res = []
        for i in range(len(nums)):
            res.append(str(nums[i]))
            
        while True:
            has_zero = False
            for i in range(len(res)):
                if res[i] == '0':
                    has_zero = True
                    break
            
            if not has_zero:
                break
            
            res = [str(int(x)-1) for x in res]
        
        return ''.join(res)
```
