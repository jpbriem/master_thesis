```python
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        if not heights or len(heights) == 0:
            return 0

        stack = []
        leftmost = float('inf')
        max_area = 0
        
        for height in heights:
            while stack and heights[stack[-1]] >= height:
                top = stack.pop()
                right = leftmost + heights[top]
                
                area = right - leftmost
                if area > max_area:
                    max_area = area
                    
            stack.append(height)
            
            leftmost = min(leftmost, height)

        return max_area
```
