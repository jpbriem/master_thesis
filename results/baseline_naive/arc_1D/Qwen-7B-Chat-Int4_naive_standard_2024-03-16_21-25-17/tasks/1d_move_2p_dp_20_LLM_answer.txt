class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        def dfs(nums, path):
            if len(path) == len(nums):
                res.append(path)
            
            for i in range(len(nums)):
                if nums[i] != 0 and not path.count(nums[i]):
                    path.append(nums[i])
                    dfs(nums[:i] + nums[i+1:], path)

        dfs(nums, [])
        return res
