class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        n = len(arr)
        
        # Initialize a queue to store the indices of elements that are greater than their neighbors.
        q = collections.deque()
        
        # Initialize an array to store the number of elements in each chunk. The first element will be used as the index of the current chunk.
        count = [1] * (n + 1)
        
        for i in range(1, n):
            if arr[i] > arr[i - 1]:
                q.append(i)
            
            while q and arr[q[0]] < arr[i]:
                count[arr[q.pop(0)]] -= 1
                
        return sum(count[:n])
