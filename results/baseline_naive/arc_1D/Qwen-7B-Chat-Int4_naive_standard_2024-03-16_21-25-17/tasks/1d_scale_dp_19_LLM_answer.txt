class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 1 or arr[0] == 0:
            return 1
        
        stack = []
        
        for i in range(len(arr)):
            while stack and arr[i] >= stack[-1][0]:
                stack.pop()
            
            stack.append((arr[i], i))
        
        res = 1
        index = 1
        
        while index < len(stack):
            val, _ = stack[index]
            if val > arr[index]:
                break
            
            index += 1
        
        return res + index - 1
