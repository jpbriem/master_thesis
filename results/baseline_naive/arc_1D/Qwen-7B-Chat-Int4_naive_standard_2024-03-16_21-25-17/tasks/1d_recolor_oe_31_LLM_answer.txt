class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        d = {}
        
        for word in words:
            key = ''.join(sorted(word))
            if key not in d:
                d[key] = []
            d[key].append(word)
            
        ans = []
        for k in d.keys():
            ans.append(d[k])
        
        return ans
