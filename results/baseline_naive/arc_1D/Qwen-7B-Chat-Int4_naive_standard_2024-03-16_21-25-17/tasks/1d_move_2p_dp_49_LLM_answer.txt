class Solution:
    def findMissing(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0 or sum(nums) != len(nums)*len(nums[0]):
            return -1
        
        count = {}
        
        for num in nums:
            count[num] = count.get(num, 0) + 1
            
        missing_num = sum(count.values()) - len(nums)
        return missing_num
