class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 1:
            return 0

        res = float('inf')
        
        left = right = 0
        
        while left < len(arr):
            mid = (left + right) // 2
            
            # check the element at mid is less than all elements in range [left, mid]
            if arr[mid] > arr[left]:
                res = min(res, mid - left)
                
            else:
                right = mid
                
        return res
