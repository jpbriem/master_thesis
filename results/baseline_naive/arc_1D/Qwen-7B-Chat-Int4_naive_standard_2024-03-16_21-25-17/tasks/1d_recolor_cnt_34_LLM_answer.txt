```python
class Solution:
    def dominantIndex(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return -1
        
        count = {}
        
        for num in nums:
            if num not in count:
                count[num] = 1
            else:
                count[num] += 1
                
        index = max(count.keys(), key=count.get)
        while nums[index] == index and index > 0:
            index -= 1
            
        return index + 1
```
