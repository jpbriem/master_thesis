    def maxChunksToSorted(self, nums: List[int]) -> int:
        n = len(nums)
        
        if n == 0 or n == 1:
            return 1
        
        stack = []
        for i in range(n):
            while stack and stack[-1] > nums[i]:
                stack.pop()
            
            stack.append(nums[i])
            
        res = 1
        last = nums[0]
        for i in range(1, n):
            if nums[i] < last:
                res += 1
            else:
                last = nums[i]
                
        return res
