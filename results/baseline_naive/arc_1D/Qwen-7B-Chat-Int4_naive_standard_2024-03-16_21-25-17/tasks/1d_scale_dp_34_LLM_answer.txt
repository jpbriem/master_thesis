class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        counts = collections.Counter(nums)
        max_count = max(counts.values())
        
        result = None
        for num, count in counts.items():
            if count == max_count and result is None or count > max_count//2 + max_count%2:
                result = num
        
        return result
