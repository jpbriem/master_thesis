```python
class Solution:
    def findLargestRectangle(self, heights: List[int]) -> int:
        if not heights:
            return 0

        n = len(heights)
        max_area = 0
        leftmost = -1
        
        for i in range(n):
            while (leftmost + 1 < i and heights[leftmost] >= heights[i]):
                leftmost += 1
                
            rightmost = i
            while (rightmost > leftmost and heights[rightmost-1] >= heights[i]):
                rightmost -= 1
            
            area = max(0, rightmost-leftmost+1) * heights[i]
            max_area = max(max_area, area)

        return max_area
```
