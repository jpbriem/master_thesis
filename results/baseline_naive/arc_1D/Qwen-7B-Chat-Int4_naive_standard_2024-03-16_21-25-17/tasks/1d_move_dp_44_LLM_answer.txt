class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = {}
        
        for num in nums:
            if num not in count:
                count[num] = 1
            else:
                count[num] += 1
        
        max_count = max(count.values())
        res = []
        for key, value in count.items():
            if value == max_count:
                res.append(key)
        
        return res[0]
