```python
class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        if n == 1 or n == 2:
            return 1
        
        left = []
        right = []
        
        for i in range(n-1):
            if arr[i] > arr[i+1]:
                left.append(i)
            
        for i in reversed(range(n)):
            if arr[i] < arr[i-1]:
                right.append(i)
                
        ans = min(len(left), len(right))
        
        while len(left) != 0 and len(right) != 0:
            if arr[left[0]] <= arr[right[0]]:
                left.pop(0)
            else:
                right.pop(0)
            ans += 1
                
        return ans
```
