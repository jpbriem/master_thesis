    def arrangeBags(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0:
            return []
        
        res = [[] for _ in range(len(nums))]
        stack = []
        for i, num in enumerate(nums):
            while stack and stack[-1][1] < num:
                stack.pop()
            if not stack or stack[-1][1] > num:
                stack.append([num, i])

                res[stack[-1][0]].append(i)
