    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr:
            return 0
        
        arr = sorted(arr)
        
        left = right = 0
        result = 0

        while left < len(arr) and right < len(arr):
            if arr[left] == arr[right]:
                left += 1
                right += 1
                result += 1
                
            elif arr[left] > arr[right]:
                left += 1

                right += 1
                
        return result
