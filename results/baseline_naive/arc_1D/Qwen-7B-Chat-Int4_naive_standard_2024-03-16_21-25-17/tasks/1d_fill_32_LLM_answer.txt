class Solution:
    def maxChunksToSorted(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        left = right = 0
        for num in nums:
            while left < len(nums) and nums[left] <= num:
                left += 1
            
            while right < len(nums) and nums[right] >= num:
                right += 1
                
            if left == right:
                return left + 1
                
        return right - left
