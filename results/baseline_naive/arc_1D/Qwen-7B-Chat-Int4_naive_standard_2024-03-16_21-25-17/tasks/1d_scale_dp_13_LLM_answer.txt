class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0 or len(arr) == 1 :
            return 0
        
        # Create a list of all the elements in sorted order
        sorted_arr = []
        for i in range(len(arr)):
            sorted_arr.append(max(sorted_arr + [arr[i]]))
            
        # Use binary search to find the index of the largest element that is smaller than or equal to each element in the input array
        res = []
        left, right = 0, len(arr)-1
        while left <= right:
            mid = (left + right) // 2
            if sorted_arr[mid] >= arr[mid]:
                res.append(mid)
                left = mid + 1
            else:
                right = mid - 1
                
        return len(res)
