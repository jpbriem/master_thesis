```python
class Solution:
    def numSub(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        if n == 0 or n == 1:
            return n
        
        res = [n] + [0 for _ in range(n)]
        
        for i in range(1, n):
            if nums[i-1] < nums[i]:
                res[i] += res[i-1]
                
        for i in reversed(range(n)):
            if nums[n-i-1] > nums[i]:
                res[i] += res[n-i-1]
                
        return res[-1]
```
