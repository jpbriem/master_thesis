class Solution:
    def findLargestValue(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0:
            return 0

        stack = []
        result = 0
        for num in nums:
            while len(stack) > 0 and stack[-1] < num:
                stack.pop()
            stack.append(num)
            if len(stack) > result:
                result = len(stack)

        return result
