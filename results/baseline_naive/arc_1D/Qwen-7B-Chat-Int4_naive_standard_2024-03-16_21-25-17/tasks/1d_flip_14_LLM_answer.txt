class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        n = len(arr)
        
        if n == 0 or n == 1:
            return 1
        
        prefix_sum = [arr[0]]
        for i in range(1, n):
            prefix_sum.append(prefix_sum[i-1] + arr[i])
            
        # find the maximum length of a contiguous subarray that is sorted
        start = -1
        end = start + 1
        max_length = 1
        
        while end < n:
            if prefix_sum[end] <= prefix_sum[start]:
                start += 1
            else:
                end += 1
            
            max_length = max(max_length, end-start+1)
        
        return max_length
