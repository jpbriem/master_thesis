    def rotate(self, nums):
        n = len(nums)
        
        if n == 1 or n == 0:
            return nums
        
        mid = n // 2
        left = nums[:mid]
        right = nums[mid:]
        
        for i in range(mid):
            nums[i], nums[-i-1] = nums[-i-1], nums[i]
            
        nums[mid:], nums[:mid] = right, left
        
        return nums
