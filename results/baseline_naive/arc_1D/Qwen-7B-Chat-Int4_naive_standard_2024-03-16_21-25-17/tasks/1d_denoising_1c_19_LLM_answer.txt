def maxArray(nums):
    if not nums:
        return []
    
    # initialize pointers
    left = 0
    right = len(nums) - 1
    
    # traverse the array using two pointers
    while left < right:
        # swap elements if necessary
        if nums[left] <= nums[right]:
            nums[left], nums[right] = nums[right], nums[left]
            
        # increment pointers
        left += 1
        right -= 1
        
    # return the resulting array
    return nums
