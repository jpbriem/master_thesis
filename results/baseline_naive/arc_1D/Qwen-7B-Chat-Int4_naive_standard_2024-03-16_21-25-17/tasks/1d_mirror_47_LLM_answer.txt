class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        n = len(arr)
        
        # Check if the array is already sorted in ascending order
        for i in range(1, n):
            if arr[i] < arr[i-1]:
                return 0
        
        # Create a dictionary to store the frequency of each element in the array
        freq_dict = {}
        for num in arr:
            freq_dict[num] = freq_dict.get(num, 0) + 1
        
        # Count the number of elements that need to be moved to make the array sorted
        count = 0
        for num in arr:
            if freq_dict[num] > 0:
                count += 1
                
        return count
