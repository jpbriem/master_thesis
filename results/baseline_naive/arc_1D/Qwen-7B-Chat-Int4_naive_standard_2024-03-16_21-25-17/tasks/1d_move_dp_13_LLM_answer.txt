    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) <= 1:
            return 1
        
        result = 1
        left = right = 0
        while right < len(arr):
            if arr[right] > arr[left]:
                arr[left], arr[right] = arr[right], arr[left]
            else:
                result += 1
            right += 1
            if right == len(arr):
                break
            elif arr[left] + arr[right] >= arr[left+1]:
                left += 1
            else:
                right += 1
                
        return result
