class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in d:
                d[key] = []
            d[key].append(s)
        
        res = [[] for _ in range(len(d))]
        for k, v in d.items():
            res[v.index(k)].extend(v)
            
        return res
