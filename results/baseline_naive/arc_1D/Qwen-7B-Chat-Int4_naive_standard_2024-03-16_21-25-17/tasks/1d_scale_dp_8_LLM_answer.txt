```python
class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        #dp[i] = maximum number of chunks with elements less than i in sorted order
        dp = [1 for _ in range(len(arr))]
        
        for i in range(1, len(arr)):
            for j in range(i):
                if arr[j] <= arr[i]:
                    dp[i] = max(dp[i], dp[j]+1)
                
        return dp[-1]
```
