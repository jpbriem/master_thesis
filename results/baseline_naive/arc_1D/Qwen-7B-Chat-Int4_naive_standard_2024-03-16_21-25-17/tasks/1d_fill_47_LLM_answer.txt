class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        i = j = k = 0
        
        while(i < len(arr)):
            if arr[i] == arr[j]:
                j += 1
                
            elif arr[i] > arr[j]:
                # Move elements from j to i
                for _ in range(j - i + 1):
                    arr[k] = arr[j]
                    j -= 1
                    k += 1
                
            else:
                # Move elements from i to j
                for _ in range(i - j + 1):
                    arr[k] = arr[i]
                    i -= 1
                    k += 1
                    
            i += 1
        
        return k
