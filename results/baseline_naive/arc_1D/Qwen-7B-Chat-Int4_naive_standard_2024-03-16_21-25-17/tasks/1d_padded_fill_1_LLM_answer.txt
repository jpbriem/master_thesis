        :rtype: int
        """
        
        # use a sliding window to keep track of the current state
        left = right = 0
        result = []
        
        while left < len(arr) and right < len(arr):
            if arr[left] > arr[right]:
                temp = arr[left]
                arr[left] = arr[right]
                arr[right] = temp
                right += 1
