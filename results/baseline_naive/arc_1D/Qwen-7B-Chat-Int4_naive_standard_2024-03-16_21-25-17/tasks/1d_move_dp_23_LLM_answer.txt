    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[int]]
        """
        anagram_map = collections.defaultdict(list)
        for i in range(len(strs)):
            if strs[i] not in anagram_map:
                anagram_map[strs[i]] = []
            anagram_map[strs[i]].append(i)

        result = []
        for key, value in anagram_map.items():
            group = sorted(value)
            result.append(group)
        return result
