class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        max_count = max(count.values())
        
        result = []
        for num in nums:
            if count[num] == max_count:
                result.append(num)
                
        return result[0]
