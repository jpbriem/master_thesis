    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for s in strs:
            if s not in dic:
                dic[s] = []
            dic[s].append(s)
            
        res = [[] for _ in range(len(dic))]
        for key, val in dic.items():
            res[val[0]].extend(val[1:])
        
        return res
