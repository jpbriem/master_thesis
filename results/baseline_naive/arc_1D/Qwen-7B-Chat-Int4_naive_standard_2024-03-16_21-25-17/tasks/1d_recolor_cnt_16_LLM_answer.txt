class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        
        if n == 0 or n == 1:
            return 0
        
        i = j = k = 0

        while i < n and arr[i] > arr[j]:
            j += 1
            
        for _ in range(n-j):
            arr[k], arr[j] = arr[j], arr[k]
            j += 1
        
        l = j
        r = n-1
        count = 1
        
        while l <= r:
            while l < r and arr[l] > arr[r]:
                l += 1
                
            while l < r and arr[l] < arr[r]:
                r -= 1
                
            count += min(r-l+1, n-count)

        return count
