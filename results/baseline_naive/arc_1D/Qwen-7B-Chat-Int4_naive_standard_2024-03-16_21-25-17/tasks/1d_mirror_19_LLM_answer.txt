```python
class Solution:
    def arrangeCups(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        n = len(nums)
        if n == 0:
            return []
        
        cups = [-1] * (n + 1)
        for i in range(n):
            cups[nums[i]] = i
        
        res = [cups[0]]
        left = cups[0]
        right = cups[n]
        
        while True:
            while left < right and nums[left] > nums[right]:
                left += 1
                
            while left < right and nums[right] > nums[left]:
                right -= 1
            
            if left >= right:
                break
            
            cup = nums[left]
            res.append(cup)
            
            left += 1
            right -= 1
            
        return res[::-1]
```
