class Solution:
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        
        ans = []
        
        for i in range(len(nums)):
            if nums[i] == 0:
                continue
            
            l = r = i + 1
            while l < len(nums) and nums[l] == nums[r]:
                l += 1
                r -= 1
            
            ans.append(max(l - i, r - i))
            
        return max(ans)
