class Solution:
    def countNumbers(self, nums):
        if not nums or len(nums) == 0:
            return []
        
        # 使用滑动窗口的方法，记录当前的子数组中最大的数字和最小的数字
        max_num = min_num = nums[0]
        result = []
        for num in nums[1:]:
            if num > max_num:
                max_num = num
            elif num < min_num:
                min_num = num
            
            # 如果当前的子数组中的最大数字和最小数字之差大于等于3，则将当前子数组从结果集中删除
            if max_num - min_num >= 3:
                result.pop()
            
            # 将当前的子数组添加到结果集中
            result.append(num)
            
        return result
