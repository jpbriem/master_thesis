    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0 or arr[0] > arr[-1]:
            return 0
        
        left = right = 0
        res = 0
        while left < len(arr) and right < len(arr)-1:
            if arr[left] <= arr[right]:
                left += 1

            res += 1
