class Solution:
    def largestRectangleArea(self, height: List[int]) -> int:
        m = max(height)
        stack = []
        res = 0
        
        for i in range(len(height)):
            while stack and height[stack[-1]] < height[i]:
                h = height[stack.pop()]
                w = (i - stack[-1] - 1) * h
                res = max(res, w)

            if height[i] == m:
                stack.append(i)

        return res
