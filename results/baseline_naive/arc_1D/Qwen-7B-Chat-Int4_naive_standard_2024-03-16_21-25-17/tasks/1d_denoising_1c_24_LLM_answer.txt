class Solution:
    def countNumbers(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        if n == 0:
            return 0

        # convert to set for O(1) lookup time
        num_set = set(nums)

        # create a map of numbers and their frequency in the set
        num_freq_map = {num: freq for num, freq in zip(range(n), [len(num_set & {num})] * n)}

        # sum up all frequencies that are not divisible by 4
        total_num_freq = sum(num_freq_map[num] for num in num_freq_map if num % 4 != 0)

        # add back the number of 0s at the end
        total_num_freq += n - (n % 4)

        return total_num_freq
