class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        anagrams = [[] for _ in range(len(set(strs[0])))]
        for s in strs:
            hash_map = {}
            for char in s:
                hash_map[char] = hash_map.get(char, 0) + 1
            anagrams[hash_map[s[0]]].append(s)
            
        return anagrams
