class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 0:
            return 0
        
        arr.sort()
        
        chunks = []
        for i in range(len(arr)):
            chunk = arr[i:i+1]
            if chunk != [] and chunk[0] == 0:
                continue
            
            while True:
                # check if the current chunk is sorted
                if all([arr[j] <= arr[j-1] for j in range(1, len(chunk))]):
                    break
                
                # remove the last element of the chunk
                chunks.append(chunk.pop())
                
        return len(chunks)
