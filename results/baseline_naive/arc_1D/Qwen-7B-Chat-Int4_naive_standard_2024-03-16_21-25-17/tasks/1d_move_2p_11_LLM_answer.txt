class Solution:
    def maxChunksToSorted(self, nums: List[int]) -> int:
        if not nums or len(nums) == 0:
            return 0
        
        res = 0
        for i in range(len(nums)):
            j = i - 1
            while j >= 0 and nums[j] > nums[i]:
                nums[j], nums[i] = nums[i], nums[j]
                res += 1
                j -= 1

        return res + 1
