    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        def dfs(nums, start=0):
            if len(nums) == 0:
                res.append(nums)
            else:
                for i in range(start, len(nums)):
                    temp = nums[:start] + nums[i:] + nums[start:i+1]
