```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dic = {}
        
        for s in strs:
            # Sort the string to get anagram
            sorted_s = "".join(sorted(s))
            
            if sorted_s not in dic.keys():
                dic[sorted_s] = []
                
            dic[sorted_s].append(s)
        
        result = []
        
        for k in dic.keys():
            result.append(dic[k])
        
        return result
```
