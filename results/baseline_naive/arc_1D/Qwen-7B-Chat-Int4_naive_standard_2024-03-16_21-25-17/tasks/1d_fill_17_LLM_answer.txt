class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        
        if not nums:
            return 0
        
        n = len(nums)
        
        prefix_sum = []
        for i in range(n):
            prefix_sum.append(nums[i])
            
        prefix_sum.sort()
        
        result = 0
        
        i = 0
        while i < n:
            j = i + 1
            while j < n and nums[j] == prefix_sum[i]:
                j += 1
            
            result += j - i
            
            i = j
            
        return result
