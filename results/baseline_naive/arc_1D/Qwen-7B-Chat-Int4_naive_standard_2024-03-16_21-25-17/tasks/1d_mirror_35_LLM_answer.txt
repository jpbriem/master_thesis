class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        dp = [[0 for _ in range(n+1)] for _ in range(m+1)]
        
        for i in range(1,n+1):
            dp[1][i] = 1
        
        for j in range(2,m+1):
            for i in range(1,n+1):
                dp[j][i] += dp[j-1][i]
                
        return dp[m][n]
