class Solution:
    def arrangeWays(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        # dp[i] = number of ways to rearrange the first i elements with length i
        # base cases:
        #   - if there are only one element left in the array, then there is only one way to rearrange it.
        #   - if all elements are same, then there are (n-1)! ways to rearrange them. 
        #   - if two adjacent elements are different, then we can swap them and get a new arrangement.
        dp = [1 for _ in range(len(nums))]
        
        for i in range(1, len(nums)):
            # check if two adjacent elements are same or not
            flag = False
            for j in range(i-1, -1, -1):
                if nums[j] == nums[i]:
                    flag = True
                    break

            # calculate number of ways to rearrange last i elements with length i
            # if two adjacent elements are same, then we cannot swap them
            if not flag:
                dp[i] *= dp[i-1]
            else:
                dp[i] += dp[i-1]
                
        return dp[-1]
