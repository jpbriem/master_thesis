class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr:
            return 0

        stack = []
        for i in range(len(arr)):
            while len(stack) > 0 and arr[i] >= arr[stack[-1]]:
                stack.pop()
            stack.append(i)

        result = 0
        start = 0
        end = 0
        
        while start < len(arr):
            while end - start + 1 == len(arr) or arr[end] <= arr[start]:
                end -= 1
            result += 1
            start = end + 1
            
        return result
