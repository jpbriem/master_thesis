class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 1:
            return 1
        
        s = sorted(set(arr))
        
        res = 0
        for i in range(len(arr)):
            while arr[i] != s[0]:
                s.pop(0)
            
            if not s:
                break
            
            res += 1
            
        return res
