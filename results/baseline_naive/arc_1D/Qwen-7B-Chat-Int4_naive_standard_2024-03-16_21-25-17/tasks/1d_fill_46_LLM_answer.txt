class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0
        
        n = len(nums)
        
        res = []
        i = j = 0
        while i < n and j < n:
            # check whether current chunk is sorted
            if nums[i] <= nums[j]:
                res.append(nums[i])
                i += 1
            else:
                res.append(nums[j])
                j += 1
            
        # add remaining elements to the result
        res.extend(nums[i:])
        
        return len(res)
