class Solution:
    def majorityElement(self, nums):
        count = {}
        
        for num in nums:
            if num not in count:
                count[num] = 1
            else:
                count[num] += 1
                
        max_count = max(count.values())
        result = []
        
        for num, c in count.items():
            if c == max_count:
                result.append(num)
                
        return result
