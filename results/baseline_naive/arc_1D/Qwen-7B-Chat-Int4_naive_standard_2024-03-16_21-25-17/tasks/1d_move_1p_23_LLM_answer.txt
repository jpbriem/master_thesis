```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        dic = {}
        for s in strs:
            temp = ''.join(sorted(s))
            if temp not in dic:
                dic[temp] = []
            dic[temp].append(s)
            
        for k in dic.keys():
            dic[k].sort()
            ans.append(dic[k])
            
        return ans
```
