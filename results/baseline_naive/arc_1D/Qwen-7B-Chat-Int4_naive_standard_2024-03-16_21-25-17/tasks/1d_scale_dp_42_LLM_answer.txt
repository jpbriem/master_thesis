class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 0:
            return 0
        
        # Check if the array is already sorted in ascending order
        if all(x < y for x, y in zip(arr, arr[1:])):
            return 1
        
        left = right = 0
        res = 0
        while True:
            if left >= len(arr):
                break
            
            while right < len(arr) and arr[right] <= arr[left]:
                right += 1
                
            if right - left > 1:
                chunk_size = right - left + 1
                res += chunk_size
                
            left += 1
        return res
