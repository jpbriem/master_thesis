```python
class Solution:
    def numSub(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        
        n = len(nums)
        
        dp = [[0 for _ in range(n)] for _ in range(n)]
        for i in range(n):
            dp[i][i] = 1

        for i in range(1, n):
            for j in range(i):
                if nums[j] == nums[i]:
                    continue
                
                max_len = 0
                for k in range(j+1, i+1):
                    if nums[k] != nums[j]:
                        break
                    
                    max_len += 1
                    
                if max_len > 0:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-max_len]) + 1
                    
        return sum(dp[-1])
```
