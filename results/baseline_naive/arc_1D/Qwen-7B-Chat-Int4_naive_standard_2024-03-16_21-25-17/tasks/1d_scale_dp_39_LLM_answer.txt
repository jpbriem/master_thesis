```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key not in d:
                d[key] = []
            d[key].append(s)
        
        ans = [[] for _ in range(len(d))]
        for k, v in d.items():
            ans[v.index(k)].append(k)
            
        return ans
```
