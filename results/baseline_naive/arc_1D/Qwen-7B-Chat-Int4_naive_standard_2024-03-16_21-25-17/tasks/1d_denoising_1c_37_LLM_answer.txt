    def groupAnagrams(self, words: List[str]) -> List[List[str]]:

        d = {}
        for word in words:
            sorted_word = ''.join(sorted(word))
            if sorted_word in d:
                d[sorted_word].append(word)
            else:
                d[sorted_word] = [word]

        for key, value in d.items():

        return groups
