```python
class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0
        
        start = max(0, 1 - (len(s)-1)//2)
        end = min(len(s), start + 1)
        
        while start < end and s[start] == s[end]:
            start += 1
            end -= 1
            
        return end - start + 1
```
