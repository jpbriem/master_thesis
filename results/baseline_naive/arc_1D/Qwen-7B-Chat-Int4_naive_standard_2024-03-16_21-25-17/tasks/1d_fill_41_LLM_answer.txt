    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        def dfs(nums):
            if len(nums) == 1:
                res.append(nums)
            
            for i in range(len(nums)):
                num = nums[i]
                
                if i != 0 and num == nums[i-1]:
                    continue
                
                nums[i], nums[i+1] = nums[i+1], nums[i]
                dfs(nums[1:])

        dfs(nums)
        
        return res
