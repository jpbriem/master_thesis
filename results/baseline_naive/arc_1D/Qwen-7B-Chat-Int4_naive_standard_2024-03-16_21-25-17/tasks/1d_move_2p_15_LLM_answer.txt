class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0 or len(arr) == 1:
            return 0
        
        i = j = k = 0

        while i < len(arr):
            if arr[i] != 0 and arr[j] == 0:
                arr[j], arr[k] = arr[k], arr[j]
                k += 1
            
            elif arr[i] == 0 and arr[j] != 0:
                arr[j], arr[k] = arr[k], arr[j]
                k += 1
            
            else:
                i += 1
                
            if i - j >= 2:
                break

        return i - j + 1
