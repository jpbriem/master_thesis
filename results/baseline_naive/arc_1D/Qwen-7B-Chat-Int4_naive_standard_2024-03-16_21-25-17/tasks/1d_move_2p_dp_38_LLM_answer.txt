```python
class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr or len(arr) == 1:
            return 1
        
        n = len(arr)
        
        # sort the array
        arr.sort()
        
        result = 1
        i = 0
        while i < n - 1:
            j = i + 1
            count = 1
            while j < n and arr[j] <= arr[i]:
                count += 1
                j += 1
            
            if count > result:
                result = count
                
            i = j
    
        return result
```
