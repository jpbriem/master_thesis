```python
def largest_increasing_subarray(nums):
    if not nums:
        return []
    
    max_len = 1
    start = 0
    
    for i in range(1, len(nums)):
        if nums[i] > nums[start]:
            start = i
            max_len += 1
    
    return nums[start:start+max_len]
```
