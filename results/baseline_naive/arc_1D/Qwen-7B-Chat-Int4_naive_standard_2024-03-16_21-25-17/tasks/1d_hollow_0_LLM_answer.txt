class Solution:
    def minOperations(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        ans = 0
        
        for i in range(len(nums)):
            if nums[i] == 0:
                continue

            j = i + 1
            while j < len(nums) and nums[j] != 0:
                if nums[i] > nums[j]:
                    nums[i], nums[j] = nums[j], nums[i]
                    ans += 1
                j += 1
            
        return ans
