def permute(nums):
    n = len(nums)
    # Base cases
    if n == 0 or n == 1:
        return nums

    # Two pointers approach
    left = 0
    right = n - 1

    while left < right:
        # Swap elements at left and right pointers if necessary
        if nums[left] != nums[right]:
            nums[left], nums[right] = nums[right], nums[left]

        # Move left pointer towards right
        left += 1
