class Solution:
    def longestConsecutive(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0
        
        d = {}
        
        for num in nums:
            if num in d:
                d[num] += 1
            else:
                d[num] = 1
                
        max_length = 0
        for num in d:
            if d[num] == len(nums):
                max_length = max(max_length, num)
                
        return max_length
