class Solution:
    def rearrangeArray(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        if len(nums) == 0 or len(nums) == 1:
            return nums
        
        res = []
        
        stack = [nums[0]]
        for i in range(1, len(nums)):
            while stack and nums[i] > nums[stack[-1]]:
                res.append(stack.pop())
            stack.append(nums[i])
            
        while stack:
            res.append(stack.pop())
            
        return res
