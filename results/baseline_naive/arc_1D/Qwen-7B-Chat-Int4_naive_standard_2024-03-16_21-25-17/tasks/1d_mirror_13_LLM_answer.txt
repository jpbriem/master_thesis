class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0:
            return 0
        
        n = len(arr)
        
        chunks = []
        
        # sort the array in increasing order
        arr.sort()
        
        for i in range(n):
            while len(chunks) > 0 and arr[i] >= chunks[-1][0]:
                chunks.pop()
            
            chunks.append((arr[i], i))
        
        return len(chunks)
