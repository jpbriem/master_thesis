class Solution:
    def majorityElement(self, nums):
        if len(nums) < 2:
            return None
        
        count = {}
        
        for num in nums:
            if num not in count:
                count[num] = 1
            else:
                count[num] += 1
                
        max_count = max(count.values())
        count_to_remove = {num: count[num] - max_count for num in count}
        result = []
        
        for num in nums:
            if count_to_remove.get(num, 0) == 0:
                result.append(num)
                
        return result
