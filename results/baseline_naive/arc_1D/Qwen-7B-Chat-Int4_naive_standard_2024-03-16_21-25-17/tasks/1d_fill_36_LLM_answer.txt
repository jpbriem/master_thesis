```python
class Solution:
    def longestIncreasingPath(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        if not n:
            return 0
        
        stack, path = [], []
        
        for i in range(n):
            while stack and nums[i] >= nums[stack[-1]]:
                path.append(stack.pop())
            
            stack.append(i)
            path.append(i)
            
        return len(path) - 1
```
