    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 1:
            return 0
        
        left = right = 0
        result = 0

        while right < len(nums):
            if nums[left] <= nums[right]:
                left += 1
            else:
                result += 1
                right += 1
                
        return result
