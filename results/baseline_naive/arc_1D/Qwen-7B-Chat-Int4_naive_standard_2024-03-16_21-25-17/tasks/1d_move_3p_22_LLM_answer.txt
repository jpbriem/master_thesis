```python
class Solution:
    def minChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0 or len(set(arr)) < 2:
            return 0
        
        n = len(arr)
        count = 0
        
        i = 0
        while i < n-1:
            j = i+1
            while j < n and arr[i] > arr[j]:
                j += 1
                
            if j-i >= 2:
                count += 1
                i = j
            
            else:
                i += 1
        
        return count
```
