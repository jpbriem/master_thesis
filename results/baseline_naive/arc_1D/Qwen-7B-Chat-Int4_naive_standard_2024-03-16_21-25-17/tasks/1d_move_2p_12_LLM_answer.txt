```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[int]]
        """
        d = {}
        for s in strs:
            k = ''.join(sorted(s))
            if k not in d:
                d[k] = []
            d[k].append(len(s))

        res = [[] for _ in range(len(d))]
        for key, value in d.items():
            res[value[0]].extend([i+value[1] for i in range(value[0], len(strs))])
        
        return res
```
