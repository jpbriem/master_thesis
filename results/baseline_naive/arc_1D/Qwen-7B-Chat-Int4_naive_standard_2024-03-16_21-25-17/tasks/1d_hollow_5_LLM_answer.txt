class Solution:
    def removeInvalidParentheses(self, s: str) -> int:
        if not s or len(s) == 0:
            return 0

        stack = []
        count = 0
        for c in s:
            if c == '(':
                stack.append(c)
            elif c == ')':
                while len(stack) > 0 and stack[-1] != '(':
                    stack.pop()
                if len(stack) > 0:
                    stack.pop()
                else:
                    return -1
            else:
                count += 1

        return count
