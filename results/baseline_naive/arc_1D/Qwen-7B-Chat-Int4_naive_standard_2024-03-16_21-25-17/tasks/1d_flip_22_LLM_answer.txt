class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in d:
                d[key] = []
            d[key].append(s)
        
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
