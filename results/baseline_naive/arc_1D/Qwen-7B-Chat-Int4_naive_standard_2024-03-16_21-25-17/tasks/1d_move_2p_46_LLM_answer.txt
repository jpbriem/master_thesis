class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if len(arr) == 0:
            return 0
        
        n = len(arr)
        
        i = j = k = 0
        while(i < n):
            if arr[i] > arr[j]:
                arr[k], arr[j] = arr[j], arr[k]
                j += 1
            else:
                arr[k], arr[i] = arr[i], arr[k]
                i += 1
                
            k += 1
            
        return k - 1
