```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key in d:
                d[key].append(s)
            else:
                d[key] = [s]
                
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
```
