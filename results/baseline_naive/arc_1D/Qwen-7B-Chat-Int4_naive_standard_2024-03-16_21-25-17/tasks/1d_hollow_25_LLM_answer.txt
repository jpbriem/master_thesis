```python
class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        if not arr or len(arr) == 0:
            return 0
        
        left = right = 0
        while left < len(arr):
            mid = (left + right) // 2
            if arr[mid] <= arr[left]:
                left = mid + 1
            else:
                right = mid

        return right - left + 1
```
