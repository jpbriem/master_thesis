class Solution:
    def maxChunksToSorted(self, arr: List[int]) -> int:
        if not arr:
            return 0

        i = j = k = 0
        while i < len(arr):
            while j < len(arr) and arr[j] <= arr[i]:
                j += 1
            
            if j - i > 1:
                count = (j - i - 1)
                for _ in range(count):
                    arr[k], arr[i] = arr[i], arr[k]
                    i += 1
                    k += 1
            else:
                break
        
        return j - i
