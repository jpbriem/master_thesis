class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        if len(count) == 1:
            return count.most_common(1)[0][0]
        
        count_dict = {}
        for k in count.keys():
            count_dict[k] = count[k]
        
        result = []
        max_count = -1
        for i in range(len(nums)):
            if nums[i] not in count_dict:
                continue
            
            count_dict[nums[i]] -= 1
            if count_dict[nums[i]] < 0:
                del count_dict[nums[i]]
            
            if count_dict[nums[i]] > max_count:
                max_count = count_dict[nums[i]]
                result.append(nums[i])
                
        return result[0]
