```python
class Solution:
    def maxChunksToSorted(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return 1
        
        prefix_sum = []
        for i in range(n):
            prefix_sum.append(nums[i])
        
        prefix_sum.sort()
        
        left = 0
        right = 0
        res = 1
        
        while left < n and right < n:
            while right < n and nums[left] <= nums[right]:
                right += 1
                
            if right - left > 0:
                res += 1
            
            left += 1
        
        return res
```
