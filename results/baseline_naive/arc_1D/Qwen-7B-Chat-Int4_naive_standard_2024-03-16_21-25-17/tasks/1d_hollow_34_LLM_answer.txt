class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        
        if n == 0 or n == 1:
            return 0
        
        l = 0
        r = n - 1
        
        while l < r:
            mid = (l + r) // 2
            if arr[mid] > arr[l]:
                l = mid + 1
            else:
                r = mid
                
        return l
