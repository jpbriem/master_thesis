    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        res = []
        i = j = k = 0
        
        while i < len(arr) and j < len(arr)-1:
            if arr[i] <= arr[j+1]:
                res.append(arr[k])
