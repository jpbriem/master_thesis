    n = len(arr)
    start = 0
    end = 0
    max_val = arr[0]

        if arr[end] >= max_val:
            # Update the maximum value
            max_val = arr[end]
            
            # Move the end pointer forward
            end += 1
            
        else:
            # Add the elements to the output array starting from the end of the current window
            output.append(arr[start:end])
            
            # Move the start pointer back to the previous position
            start -= 1

    output.append(arr[start:])
    
    return output
