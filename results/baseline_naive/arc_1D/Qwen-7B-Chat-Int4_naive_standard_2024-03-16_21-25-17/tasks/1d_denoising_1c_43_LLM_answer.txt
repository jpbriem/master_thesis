    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if len(words) == 0:
            return []
        
        ans = [[] for _ in range(len(set(words)))]
        for word in words:
            hash_map = {}
            for char in word:

            ans[hash_map].append(word)
