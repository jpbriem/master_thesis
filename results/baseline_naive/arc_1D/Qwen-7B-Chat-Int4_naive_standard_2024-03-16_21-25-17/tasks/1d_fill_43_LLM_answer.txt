class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        if not arr:
            return 0
        
        n = len(arr)
        
        stack = []
        for i in range(n):
            while stack and arr[i] >= arr[stack[-1]]:
                stack.pop()
            
            stack.append(i)
        
        result = 0
        left = right = stack.pop()
        while left < right:
            result += 1
            
            mid = (left + right) // 2
            while mid > left and arr[mid - 1] <= arr[left]:
                left += 1
                
            while mid < right and arr[right] <= arr[mid]:
                right -= 1
                
        return result
