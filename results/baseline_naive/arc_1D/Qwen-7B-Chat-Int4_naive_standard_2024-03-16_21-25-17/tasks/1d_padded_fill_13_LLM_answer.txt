def max_sum(arr):
    n = len(arr)
    m = len(arr[0])
    
    # Initialize an empty output array with all zeros
    out = [[0] * m for _ in range(n)]
    
    # Iterate over each row of the input
    for i in range(n):
        # Check if there are any 7s in the current row
        count_7 = sum(1 for j in arr[i] if arr[i][j] == 7)
        
        # Set the corresponding cells in the output array to 7 or 0 based on the count of 7s
        for j in range(m):
            if count_7 > 0:
                out[i][j] = 7

    return out
