    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        counts = collections.defaultdict(int)
        
        for num in nums:
            if num == 0:
                continue
            counts[num] += 1
        
        count_dict = {}
        for k, v in counts.items():
            if v > len(nums) // 2:
                count_dict[k] = v
        
        return max(count_dict.values())
