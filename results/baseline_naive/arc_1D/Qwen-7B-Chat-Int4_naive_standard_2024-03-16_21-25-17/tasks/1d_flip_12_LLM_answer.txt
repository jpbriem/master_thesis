```python
class Solution:
    def arrangeCans(self, cans):
        """
        :type cans: List[int]
        :rtype: List[int]
        """
        if not cans:
            return []
        
        can = cans[0]
        ans = []
        while can > 0:
            ans.append(can)
            for i in range(len(cans)):
                if cans[i] == can - 1:
                    cans[i] -= 1
                    break
            else:
                raise ValueError("Invalid input")
            can -= 1
        
        return ans[::-1]
```
