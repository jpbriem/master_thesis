```python
class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        
        if len(arr) == 0:
            return 0
        
        left = 0
        right = len(arr)-1
        res = 0
        while left < right:
            mid = (left+right)//2
            
            if arr[mid] <= arr[left]:
                left = mid + 1
                res += 1
                
            else:
                right = mid - 1
                res += 1
                
        return res
```
