class Solution:
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        counts = {}
        
        for num in nums:
            if num not in counts:
                counts[num] = 1
            else:
                counts[num] += 1
        
        max_count = 0
        result = None
        
        for num, count in counts.items():
            if count > max_count:
                max_count = count
                result = num
                
        return result
