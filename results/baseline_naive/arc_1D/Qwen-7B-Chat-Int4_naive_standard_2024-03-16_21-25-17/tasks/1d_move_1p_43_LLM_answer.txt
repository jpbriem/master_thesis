class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = [[] for _ in range(len(strs))]
        for i, s in enumerate(strs):
            hash_s = {}
            for char in s:
                if char not in hash_s:
                    hash_s[char] = []
                hash_s[char].append(i)
            ans[hash_s['']].append(s)

        return ans
