    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """

            return 0

        # find the index of largest element
        i = 0
        while i < len(arr) and arr[i] <= arr[i+1]:
            i += 1
        
        # initialize result to be number of elements before largest element
        res = i

        # loop through remaining elements and update result
        for j in range(i+1, len(arr)):
            if arr[j] > arr[j-1]:
                res += 1
                
        return res
