    def numSub(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0
        
        count = {}
        
        for i in range(len(nums)):
            if nums[i] in count:
                count[nums[i]] += 1
            else:
                count[nums[i]] = 1
        
        res = 0
        
        for key, value in count.items():
            res += (value*(value-1))//2
        
        return res
