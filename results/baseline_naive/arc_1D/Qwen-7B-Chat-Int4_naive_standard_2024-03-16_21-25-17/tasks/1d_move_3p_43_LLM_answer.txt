class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0:
            return [[]]
        
        res = []
        self.permuteHelper(nums, 0, [], res)
        return res

    def permuteHelper(self, nums, start, path, res):
        if start == len(nums):
            res.append(path[:])
            return
        
        for i in range(start, len(nums)):
            if nums[i] != 0 and not any(x == nums[i] for x in path):
                path.insert(0, nums[i])
                self.permuteHelper(nums, i + 1, path, res)
                path.pop()
