class Solution:
    def longestConsecutive(self, nums):
        if not nums:
            return []
        
        counts = {}
        for num in nums:
            if num in counts:
                counts[num] += 1
            else:
                counts[num] = 1
        
        max_length = 0
        start = None
        result = []
        
        for num, count in counts.items():
            if count == 1:
                continue
            
            length = count - 1
            while length > 0 and nums[start + length] == num:
                length -= 1
                
            if length > max_length:
                max_length = length
                start = num
        
        for i in range(max_length):
            result.append(nums[i])
            
        return result
