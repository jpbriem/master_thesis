```python
class Solution:
    def findLargestRegion(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0

        n = len(nums)
        stack = []
        start = -1
        max_size = 0
        
        for i in range(n):
            while stack and nums[i] > nums[stack[-1]]:
                size = 1 + (i - start - 1) // (n - stack.pop() - 1)
                if size > max_size:
                    max_size = size
                start = stack.pop()
            stack.append(i)

        return max_size
```
