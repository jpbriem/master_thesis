```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        groups = []
        
        for s in strs:
            # create a set of characters to check if it exists in the other strings
            char_set = set(s)
            
            # find all the indices of this character set in the other strings
            indices = [i for i, c in enumerate(strs) if set(c).issubset(char_set)]
            
            # add this string and its indices to the group list
            group = [s] + indices
            groups.append(group)
        
        return groups
```
