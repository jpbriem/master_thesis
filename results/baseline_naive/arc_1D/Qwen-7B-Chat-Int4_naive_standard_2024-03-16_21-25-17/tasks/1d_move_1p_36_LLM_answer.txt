class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        left = right = 0
        res = []
        
        while left < len(arr) and right < len(arr):
            if arr[left] <= arr[right]:
                res.append(arr[left])
                left += 1
            else:
                res.append(arr[right])
                right += 1
                
        return len(res)
