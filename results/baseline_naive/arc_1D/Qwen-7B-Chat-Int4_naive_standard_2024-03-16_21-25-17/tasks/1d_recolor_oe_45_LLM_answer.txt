class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        n = len(arr)
        
        # check if array is sorted
        for i in range(1,n):
            if arr[i] < arr[i-1]:
                return -1
        
        # convert to list of set
        s = set()
        for num in arr:
            s.add(num)
            
        # convert back to list and sort
        arr = sorted(list(s))
        
        # count the number of unique elements in the array
        unique_count = len(set(arr))
        
        # calculate the maximum chunks
        chunk_size = (n+unique_count-1)//unique_count
        
        return chunk_size
