class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        if not nums:
            return []
        
        i = j = k = 0
        
        while i < len(nums) and j < len(nums):
            while nums[j] == nums[i]:
                j += 1
                
            if nums[j] != nums[i]:
                nums[k], nums[j] = nums[j], nums[k]
                
            i += 1
            j += 1
            
        return nums[:k+1]
