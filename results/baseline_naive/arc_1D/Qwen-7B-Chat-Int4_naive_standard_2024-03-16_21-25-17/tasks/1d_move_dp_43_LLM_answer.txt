class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        if len(nums) == 0:
            return res

        stack = [nums[0]]

        while stack:
            temp = []
            for i in range(len(stack)):
                temp.append(stack[i])
                if i < len(stack)-1 and stack[i] + stack[i+1] > nums[-1]:
                    break
            else:
                res.append(temp)

            for i in range(len(stack)):
                if stack[i] != nums[0]:
                    stack.insert(i, nums[stack.index(nums[0])])

        return res
