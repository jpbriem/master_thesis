class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        n = len(nums)
        
        # Sort the list in ascending order
        nums.sort()
        
        # Initialize two pointers, one at the beginning of the list and another at the end
        left = 0
        right = n - 1
        
        # Keep track of the maximum number of chunks that can be formed
        count = 0
        
        while left < right:
            # If the element on the left is less than or equal to the element on the right, move the left pointer forward
            if nums[left] <= nums[right]:
                left += 1
            else:
                # Otherwise, find the index of the next occurrence of the current element on the left side
                i = left + 1
                while i < right and nums[i] == nums[left]:
                    i += 1
                
                # Move the right pointer back to this index
                right = i - 1
                
                # Add a chunk of size (right - left + 1) to the count
                count += right - left + 1
                
        return count
