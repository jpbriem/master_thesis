class Solution:
    def maxChunksToSorted(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        left = right = 0
        res = []
        
        while right < len(arr):
            if arr[left] <= arr[right]:
                res.append(right - left + 1)
                left += 1
            else:
                res.append(left - right)
                right += 1
        
        return sum(res)
