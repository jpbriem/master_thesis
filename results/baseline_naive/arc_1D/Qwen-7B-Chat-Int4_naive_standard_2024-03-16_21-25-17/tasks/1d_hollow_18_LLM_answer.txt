class Solution:
    def numIslands(self, grid: List[List[int]]) -> int:
        if not grid or len(grid) != len(grid[0]):
            return 0

        directions = [(0, -1), (1, 0), (-1, 1), (0, 1)]
        
        queue = collections.deque([(i, j) for i in range(len(grid)) for j in range(len(grid[i]))])
        visited = set()
        count = 0
        
        while queue:
            row, col = queue.popleft()

            # check if the cell is already visited
            if (row, col) in visited:
                continue
            
            # mark the current cell as visited
            visited.add((row, col))
            
            # check all adjacent cells
            for direction in directions:
                new_row = row + direction[0]
                new_col = col + direction[1]
                
                if 0 <= new_row < len(grid) and 0 <= new_col < len(grid[row]) and grid[new_row][new_col] == 0:
                    queue.append((new_row, new_col))

            # check if this cell contains a water element
            if grid[row][col] == 1:
                count += 1
                
        return count
