class Solution:
     def longestPalindrome(self, s):
         """
         :type s: str
         :rtype: int
         """

         n = len(s)

             while left >= 0 and right < n and s[left] == s[right]:
                 left -= 1
                 right += 1
             max_len = max(max_len, right - left + 1)
