class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 0:
            return []
        
        res = [[]]
        for i in range(len(nums)):
            num = nums[i]
            if num not in res[0]:
                res.append([num])
                
            for sub_res in res:
                for j in range(i+1, len(nums)):
                    if nums[j] != num and nums[j] not in sub_res:
                        sub_res.append(nums[j])
                    
        return res
