class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        if not heights or len(heights) == 0:
            return 0

        n = len(heights)
        stack = []
        res = 0
        for i in range(n):
            while stack and heights[i] >= heights[stack[-1]]:
                h = heights[stack.pop()]
                w = i - stack[-1]
                res += h * w
            stack.append(i)

        return res
