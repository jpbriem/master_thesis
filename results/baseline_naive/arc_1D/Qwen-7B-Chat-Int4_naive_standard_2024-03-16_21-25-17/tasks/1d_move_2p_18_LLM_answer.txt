class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            key = "".join(sorted(s))
            if key not in d:
                d[key] = []
            d[key].append(s)
        
        res = [[] for _ in range(len(d))]
        for k in d:
            res[d[k]].append(k)
        
        return res
