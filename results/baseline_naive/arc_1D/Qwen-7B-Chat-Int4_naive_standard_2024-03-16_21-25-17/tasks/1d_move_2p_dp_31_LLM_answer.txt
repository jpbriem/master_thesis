class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if len(nums) == 1:
            return [[nums[0]]]
        
        res = []
        for i in range(len(nums)):
            temp = nums[:i] + nums[i+1:]
            self.permute(temp)
            
            # Insert the current element at the beginning of each permutation
            for p in res:
                p.insert(0, nums[i])
                
            res.append(p)

        return res
