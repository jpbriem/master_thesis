def convert_to_one_hot(input):
    # Convert each element to its index in the one-hot encoding
    encoded = {i: input.count(i) for i in set(input)}

    # Create an empty array with zeros as values
    output = []

    # Add the one-hot encoding of each element to the output array
    for i in range(len(encoded)):
        if encoded[i] > 0:
            output.append([encoded[i], 0] * (len(input) - 1))

    return output
