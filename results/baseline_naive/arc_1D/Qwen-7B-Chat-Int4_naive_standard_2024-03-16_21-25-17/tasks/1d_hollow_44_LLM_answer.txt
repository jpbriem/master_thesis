class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if len(nums) == 0:
            return 0
        
        result = 1
        left = right = 0

        while left < len(nums) and right < len(nums):
            while left < len(nums) and nums[left] > nums[right]:
                left += 1
            
            while right < len(nums) and nums[right] <= nums[left]:
                right += 1
                
            if left >= right:
                break
            else:
                result += 1
                
        return result
