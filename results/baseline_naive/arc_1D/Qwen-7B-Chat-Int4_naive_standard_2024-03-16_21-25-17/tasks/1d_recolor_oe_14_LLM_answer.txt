class Solution:
    def maxChunksToSorted(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return 0

        res = []
        cur = []
        for num in nums:
            while len(cur) > 0 and cur[-1] < num:
                res.append(len(cur))
                cur.pop()
            cur.append(num)
        
        return max(res) + 1
