    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        hash_map = {}
        result = []
        
        for string in strs:
            key = ''.join(sorted(string))
            if key not in hash_map:
                hash_map[key] = []

            result.append(v)
