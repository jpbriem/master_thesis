    def dominantIndex(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return -1
        
        max_num = nums[0]
        max_index = 0

        for i in range(1, len(nums)):
            if nums[i] > max_num:
                max_num = nums[i]
                max_index = i

        return max_index
