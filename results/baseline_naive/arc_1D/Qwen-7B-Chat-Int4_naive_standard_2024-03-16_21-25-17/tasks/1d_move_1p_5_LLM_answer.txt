class Solution:
     def maxChunksToSorted(self, arr):
         """
         :type arr: List[int]
         :rtype: int
         """
         
         if len(arr) == 1 or arr[0] == arr[-1]:
             return 1
        
         stack = []
         n = len(arr)
         i = 0
         while i < n and arr[i] > 0:
             stack.append(i)
             i += 1
         for j in range(n-1, -1, -1):
             if arr[j] <= 0:
                 break
             stack.append(j)
         result = 0
         while stack:
             result += 1
             top = stack.pop()
             while top >= 0 and arr[top] < arr[top+1]:
                 top -= 1
             i = top + 1
             while i < n and arr[i] > 0:
                 stack.append(i)
                 i += 1
         return result
