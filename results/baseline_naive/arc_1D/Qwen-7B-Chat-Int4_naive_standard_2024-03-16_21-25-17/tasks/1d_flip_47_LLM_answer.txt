class Solution:
    def dominantIndex(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums:
            return -1
        
        max_val = max(nums)
        
        for i in range(len(nums)):
            if nums[i] == max_val:
                return i
                
        return len(nums) - 1
