class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        for s in strs:
            if len(set(s)) == 1:
                ans.append([s])
            else:
                ind = {}
                for i in range(len(s)):
                    c = s[i]
                    if c not in ind:
                        ind[c] = []
                    ind[c].append(i)
                ans.extend(ind.values())
        return ans
