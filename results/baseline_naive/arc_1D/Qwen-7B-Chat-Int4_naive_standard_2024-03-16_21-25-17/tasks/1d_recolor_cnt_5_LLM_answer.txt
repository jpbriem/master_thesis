class Solution:
    def numIdenticalPairs(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        count = collections.Counter(nums)
        res = []
        for key in count.keys():
            if key == 0 or count[key] == len(nums) - 1:
                continue
            res.append(count[key])
        
        return sum(res)
