def process_input(input_list):
    output_list = []
    prev_char = ''
    count = 0
    
    for char in input_list:
        if char != prev_char or count == 4:
            output_list.append(char)
            prev_char = char
            count = 1
        else:
            count += 1
            
    return output_list
