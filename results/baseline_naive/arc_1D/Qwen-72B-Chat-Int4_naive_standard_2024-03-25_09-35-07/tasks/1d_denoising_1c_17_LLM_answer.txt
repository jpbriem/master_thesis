def compress(input_list):
    output_list = []
    prev_char = ''
    count = 0
    
    for char in input_list:
        if char != prev_char:
            if count >= 9:
                break
            
            output_list.append(char)
            count = 1
        else:
            count += 1
            
        prev_char = char
        
    return output_list
