def compress_chars(lst):
    prev_char = lst[0]
    count = 1
    
    result = [prev_char]
    
    for i in range(1, len(lst)):
        curr_char = lst[i]
        
        if curr_char == prev_char:
            count += 1
        else:
            result.append(str(count))
            result.append(curr_char)
            prev_char = curr_char
            count = 1
            
    result.append(str(count))
    
    return result
