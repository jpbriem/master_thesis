```python
def compress(input):
    result = []
    current_char = input[0]
    current_count = 1

    for i in range(1, len(input)):
        if input[i] == current_char:
            current_count += 1
        else:
            result.append(current_char)
            result.extend([str(current_count)] * (current_count > 1))
            current_char = input[i]
            current_count = 1

    result.append(current_char)
    result.extend([str(current_count)] * (current_count > 1))

    return result
```
