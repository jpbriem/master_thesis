    result = []
    current_char = s[0]
    count = 1
    
    for i in range(1, len(s)):
        if s[i] == current_char:
            count += 1
        else:
            result.append(current_char)
            result.extend([str(count)] * (count > 1))
            current_char = s[i]
            count = 1
            
    result.append(current_char)
    result.extend([str(count)] * (count > 1))
    
    return ''.join(result)
