def longest_streak(arr):
    result = []
    current_streak = 0

    for i in arr:
        if len(result) == 0 or result[-1] != i:
            result.append(i)
            current_streak = 1
        else:
            current_streak += 1
            if current_streak >= 5:
                break
    
    return result
