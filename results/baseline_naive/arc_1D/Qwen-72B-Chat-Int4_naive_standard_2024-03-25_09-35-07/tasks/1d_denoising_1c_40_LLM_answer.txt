def compress_input(input_list):
    if not input_list:
        return []
    
    result = [input_list[0]]
    count = 1
    
    for i in range(1, len(input_list)):
        if input_list[i] == input_list[i - 1]:
            count += 1
        else:
            result.append(count)
            result.append(input_list[i])
            count = 1
            
    result.append(count)
    result.append(input_list[-1])
    
    return result
