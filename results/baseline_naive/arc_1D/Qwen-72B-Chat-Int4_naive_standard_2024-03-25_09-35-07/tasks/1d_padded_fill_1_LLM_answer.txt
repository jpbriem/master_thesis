def compress_chars(lst):
    if not lst:
        return []
    
    prev_char = lst[0]
    count = 1
    
    result = [prev_char]
    
    for char in lst[1:]:
        if char != prev_char:
            result.append(str(count))
            result.append(char)
            count = 1
        else:
            count += 1
            
    result.append(str(count))
    
    return result
