def remove_consecutive_chars(input_list):
    result = []
    prev_char = ''
    char_count = 0
    
    for char in input_list:
        if char == prev_char:
            char_count += 1
        else:
            if char_count > 3 and prev_char != '':
                result.extend([prev_char] * (char_count - 3))
            prev_char = char
            char_count = 1
            
    if char_count > 3 and prev_char != '':
        result.extend([prev_char] * (char_count - 3))
    
    return result
