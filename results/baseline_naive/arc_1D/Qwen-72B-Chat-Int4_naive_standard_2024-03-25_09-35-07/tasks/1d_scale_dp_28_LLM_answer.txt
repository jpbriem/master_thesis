def compress_chars(lst):
    if not lst:
        return []
    
    prev_char = lst[0]
    freq_count = 1
    
    result = [prev_char]
    
    for i in range(1, len(lst)):
        curr_char = lst[i]
        
        if curr_char == prev_char:
            freq_count += 1
        else:
            result.append(str(freq_count))
            result.append(curr_char)
            freq_count = 1
            
        prev_char = curr_char
        
    # Append last character and its frequency count
    result.append(str(freq_count))
    
    return result
