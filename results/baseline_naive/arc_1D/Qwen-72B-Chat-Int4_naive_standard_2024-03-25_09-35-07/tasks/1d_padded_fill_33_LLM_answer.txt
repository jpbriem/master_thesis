def compress_chars(lst):
    if not lst:
        return []

    prev_char = lst[0]
    count = 1

    result = [prev_char]

    for char in lst[1:]:
        if char == prev_char:
            count += 1
        else:
            result.append(prev_char)
            result.extend([str(count)] * (count > 1))
            prev_char = char
            count = 1

    result.append(prev_char)
    result.extend([str(count)] * (count > 1))

    return result
