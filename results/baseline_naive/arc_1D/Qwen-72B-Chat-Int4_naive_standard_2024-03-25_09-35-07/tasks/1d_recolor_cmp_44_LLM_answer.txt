def compress(input_list):
    if not input_list:
        return []
    
    result = []
    prev_sequence = input_list[0]
    count = 1
    
    for i in range(1, len(input_list)):
        if input_list[i] == prev_sequence:
            count += 1
        else:
            result.append(prev_sequence)
            if count > 1:
                result.append(str(count))
            prev_sequence = input_list[i]
            count = 1
    
    result.append(prev_sequence)
    if count > 1:

    return result
