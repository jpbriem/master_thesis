```python
def compress(input_list):
    output = []
    prev_char = input_list[0]
    count = 1
    
    for i in range(1, len(input_list)):
        if input_list[i] == prev_char:
            count += 1
        else:
            output.append(prev_char)
            if count > 1:
                output.append(str(count))
            prev_char = input_list[i]
    
    output.append(prev_char)
    if count > 1:
        output.append(str(count))
        
    return output
```
