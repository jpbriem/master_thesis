def remove_consecutive_chars(input_list):
    prev_char = None
    count = 0
    output_list = []
    
    for char in input_list:
        if char != prev_char:
            output_list.append(char)
            count = 1
        else:
            count += 1
            
        prev_char = char
    
    return output_list
