```python
def compress(input_list):
    result = []
    prev_char = ""
    count = 0

    for char in input_list:
        if char != prev_char:
            if count > 1 or len(result) == 0:
                result.append(prev_char)
                result.append(str(count))
            count = 1
        else:
            count += 1
        
        prev_char = char
    
    # Append the last character and its count to the result
    if count > 1 or len(result) == 0:
        result.append(prev_char)
        result.append(str(count))

    return result
```
