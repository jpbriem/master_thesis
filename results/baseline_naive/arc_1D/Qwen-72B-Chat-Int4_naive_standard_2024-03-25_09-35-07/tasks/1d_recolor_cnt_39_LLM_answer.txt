```python
def encode_string(s):
    n = len(s)
    out = []
    i = j = 0
    
    while j < n:
        while j + i < n and s[j] == s[j + i]:
            i += 1
        
        # Append the encoded letter to the output list
        if i >= 6:
            out.append(chr(ord('a') + (i // 5)))
        else:
            out.append(s[j])
        
        # Update the indices for the next iteration
        j += i + 1
        i = 0
    
    return ''.join(out)
```
