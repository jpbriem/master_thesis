def convert_string(input_list):
    output_list = []
    prev_char = None
    counts = {}
    
    for char in input_list:
        if char != prev_char or not output_list:
            output_list.append(char)
            counts[char] = 1
        else:
            counts[char] += 1
            
        if counts[char] > 5:
            top_chars = sorted(counts.items(), key=lambda x: x[1], reverse=True)[:2]
            second_most_frequent_char = [c for c, _ in top_chars][1]
            output_list[-1] = second_most_frequent_char
        
        prev_char = char
    
    return output_list
