Answer: Your issue is that you're pushing `7` onto the `output` array when you encounter a `0` in the `input` array, instead of pushing the \em{same value} as the `input` array.

You can fix this by changing the condition inside your `if` statement to simply check if the current `input` element is not `0`:

\begin{code}
function transform(arr) {
    let output = [];
    for (let i = 0; i < arr.length; i++) {
        if (arr[i]!== 0) {
            output.push(arr[i]);
        } else {
            output.push(7);
        }
    }
    return output;
}

console.log(transform([0, 0, 5, 5, 5, 0, 0, 0, 0, 7, 0, 0, 0, 0, 0, 0, 0, 0]));
\end{code}
