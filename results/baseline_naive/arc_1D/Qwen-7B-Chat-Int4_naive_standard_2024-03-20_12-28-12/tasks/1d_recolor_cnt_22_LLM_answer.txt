class Solution:
    def longestPalindrome(self, s: str) -> int:
        n = len(s)
        dp = [[False] * (n + 1) for _ in range(n + 1)]
        
        start = end = 0
        
        # check odd length palindromes
        for i in range(1, n):
            if s[i - 1] == s[i]:
                dp[i][i] = True
                start = max(start, i - 2)
                end = i
                
        # check even length palindromes
        for k in range(4, n + 1, 2):
            l = k // 2
            r = k // 2
            
            while l >= 1 and r < n and s[l - 1] == s[r]:
                dp[l][r] = True
                start = max(start, l - 2)
                end = r
                
                l -= 1
                r += 1
        
        return end - start + 1
