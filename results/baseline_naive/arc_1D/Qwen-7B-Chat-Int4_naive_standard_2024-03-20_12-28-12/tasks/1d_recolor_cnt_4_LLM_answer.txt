def longest_palindrome(input_string):
    # Convert all characters to lowercase
    input_string = input_string.lower()
    
    # Create an empty list to store the palindromes
    palindromes = []
    
    # Iterate over each character in the string
    for i in range(len(input_string)):
        # Check if the current substring is a palindrome
        start = i
        while start >= 0 and input_string[start] == input_string[i]:
            start -= 1
        end = i + 1
        while end < len(input_string) and input_string[end] == input_string[i]:
            end += 1
        
        # If the current substring is a palindrome, add it to the list
        if start - i > 0 and end - i > 0:
            palindromes.append(input_string[start:i+1])
            
    # Return the longest palindrome found
    return max(palindromes, key=len)
