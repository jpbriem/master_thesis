class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s or len(s)<=1:
            return 0
        
        max_len = 1
        start = 0

        for i in range(len(s)):
            # Check odd length palindromes with center at i
            l = self.helper(s, i, i)
            if l>max_len:
                max_len = l
                start = i - (l-1)//2

            # Check even length palindromes with center before i
            l = self.helper(s, i-1, i+1)
            if l>max_len:
                max_len = l
                start = i - (l-2)//2

        return max_len

    def helper(self, s, left, right):
        while left>=0 and right<len(s) and s[left]==s[right]:
            left -= 1
            right += 1
        return right-left-1
