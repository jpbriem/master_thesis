class Solution:
    def groupAnagrams(self, s):
        """
        :type s: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        for word in s:
            key = ''.join(sorted(word))
            if key not in d:
                d[key] = [word]
            else:
                d[key].append(word)
        
        return list(d.values())
