class Solution:
    def longestPalindrome(self, s):
        if not s:
            return 0
        
        max_len = 0
        start = 0
        for i in range(len(s)):
            # Check palindrome from current character to its last occurrence
            j = i - 1
            while j >= 0 and s[j] == s[i]:
                j -= 1
            # If length of palindrome is greater than the previous one, update it
            if len(s[j+1:i+1]) > max_len:
                max_len = len(s[j+1:i+1])
                start = j + 1

        return max_len
