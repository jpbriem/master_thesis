    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        def backtrack(nums, path):
            if not nums and len(path) == len(nums):
                res.append(path)
                
            for i in range(len(nums)):
                if i != 0 and nums[i] == nums[i-1]:
                    continue
                
                if i == len(nums)-1 or nums[i+1] != nums[0]:
                    path.insert(0,nums[i])
                    backtrack(nums[:i]+nums[i+2:],path)
                    path.pop(0)
                    
        backtrack(nums, [])
        return res
