def longest_palindrome(input_string):
    # convert input string to lowercase and remove spaces
    input_string = input_string.lower().replace(" ", "")
    
    # initialize variables for tracking the start and end indices of the current palindrome
    start_index = 0
    end_index = 0
    
    # loop through each character in the input string
    for i in range(len(input_string)):
        # if the current character is the same as the previous character, update the end index
        if i > 0 and input_string[i] == input_string[i-1]:
            end_index += 1
        
        # check if the length of the current palindrome is greater than that of the previous one
        if len(input_string[start_index:end_index+1]) > len(input_string[end_index]):
            start_index = i
            
        # if the current character is different from the previous character, update the start index
        else:
            start_index = i + 1
    
    # return the longest palindrome found
    return input_string[start_index:end_index+1]
