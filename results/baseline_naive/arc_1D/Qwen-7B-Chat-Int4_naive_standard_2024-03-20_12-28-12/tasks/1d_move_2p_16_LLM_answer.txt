    def removeDuplicates(self, nums):
        """
        :type nums: List[str]
        :rtype: List[str]
        """
        if not nums:
            return []
        
        result = [nums[0]]
        for num in nums[1:]:
            if num != result[-1]:
                result.append(num)
                
        return result
