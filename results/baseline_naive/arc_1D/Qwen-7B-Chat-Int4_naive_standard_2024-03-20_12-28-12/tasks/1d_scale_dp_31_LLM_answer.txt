class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        if not words:
            return []
        
        anagram_dict = {}
        for word in words:
            key = ''.join(sorted(word))
            if key in anagram_dict:
                anagram_dict[key].append(word)
            else:
                anagram_dict[key] = [word]
                
        result = []
        for key, value in anagram_dict.items():
            result.append(value)
            
        return result
