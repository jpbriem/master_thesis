def longest_palindrome(input_string):
    if not input_string:
        return []
    
    max_length = 0
    start_index = None
    
    for i, char in enumerate(input_string):
        # Check if current character is part of a palindrome
        length = 1
        j = i - 1
        while j >= 0 and input_string[j] == input_string[i]:
            length += 1
            j -= 1
        
        # If current character is part of a longer palindrome than previously found,
        # update maximum length and starting index
        if length > max_length:
            max_length = length
            start_index = i - (length - 1)
        
    # Return the longest palindrome substring
    return input_string[start_index:start_index + max_length]
