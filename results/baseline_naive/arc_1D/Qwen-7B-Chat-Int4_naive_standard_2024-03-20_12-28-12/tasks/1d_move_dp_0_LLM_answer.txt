class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            if len(s) == 0:
                continue
            key = ''.join(sorted(s))
            if key not in d:
                d[key] = []
            d[key].append(s)
        
        result = [d[key] for key in d]
        return result
