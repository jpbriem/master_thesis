class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        ans = []
        
        dic = {}
        for s in strs:
            if s not in dic:
                dic[s] = [s]
            else:
                dic[s].append(s)
                
        for k in dic.keys():
            ans.append(dic[k])
            
        return ans
