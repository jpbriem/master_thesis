class Solution:
    def convert(self, nums):
        """
        :type nums: List[str]
        :rtype: List[str]
        """
        if len(nums) == 0 or not nums[0].isalpha():
            return nums

        res = []
        prev_char = None
        for char in nums:
            if prev_char != char and prev_char is not None:
                res.append(prev_char)
            prev_char = char
        res.append(prev_char)

        return res
