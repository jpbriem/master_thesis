class Solution:
    def removeKdigits(self, S: str, k: int) -> str:
        if len(S) <= k:
            return S
        
        res = []
        
        while True:
            if k == 0 or len(res) + len(S[:k]) >= len(S):
                break
            
            for i in range(k):
                res.append(S[i])
            
            S = S[k:]
    
        return ''.join(res)
