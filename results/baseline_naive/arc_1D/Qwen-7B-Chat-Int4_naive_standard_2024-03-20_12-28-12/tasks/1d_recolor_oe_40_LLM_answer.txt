class Solution:
     def groupAnagrams(self, strs):
         """
         :type strs: List[str]
         :rtype: List[List[str]]
         """
         d = {}
         for s in strs:
             # sort the string
             s = ''.join(sorted(s))
             if s not in d:
                 d[s] = []
             d[s].append(s)
         
         return list(d.values())
