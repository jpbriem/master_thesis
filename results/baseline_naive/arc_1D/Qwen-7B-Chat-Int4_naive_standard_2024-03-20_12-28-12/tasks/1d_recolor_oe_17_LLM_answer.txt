class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) == 0:
            return 0

        # initialize the maximum length of palindrome and start index of palindrome
        max_len = 1
        start_index = 0

        # iterate over each character in string
        for i in range(len(s)):
            # check if current character is same as previous character
            if i > 0 and s[i] == s[i-1]:
                max_len += 2
                start_index = i - 1
            else:
                # check if current character is not same as previous character
                if s[i] != s[i-1]:
                    max_len = 1
                    start_index = i

        # return the length of longest palindrome found
        return max_len
