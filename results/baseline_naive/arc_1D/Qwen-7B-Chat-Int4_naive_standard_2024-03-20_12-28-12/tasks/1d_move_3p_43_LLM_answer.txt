class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        count = {}
        result = []
        
        for char in s:
            if char not in count:
                count[char] = 0
            count[char] += 1
            
            if count[char] == 1:
                result.append(char)
                
        return ''.join(result)
