def longest_palindrome(input_string):
    if len(input_string) == 0:
        return []
    
    start = 0
    max_length = 1
    
    # Iterate over each character in the string
    for i, char in enumerate(input_string):
        
        # Check if there are two adjacent characters that form a palindrome
        length = 2
        j = i - length
        while j >= 0 and input_string[j] == input_string[i]:
            j -= length
            length += 1
        
        # If we find a palindrome of length greater than the current maximum,
        # update the start index and the maximum length
        if length > max_length:
            max_length = length
            start = i - (length - 1)
            
    # Return the longest palindrome found
    return input_string[start:start + max_length]
