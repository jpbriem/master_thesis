```python
class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0

        max_length = 1
        
        for i in range(len(s)):
            start = i - 1
            end = i + 1
            
            while start >= 0 and end < len(s):
                if s[start] == s[end]:
                    length = end - start + 1
                    
                    if length > max_length:
                        max_length = length
                    
                    start -= 1
                    end += 1
                
                else:
                    break
        
        return max_length
```
