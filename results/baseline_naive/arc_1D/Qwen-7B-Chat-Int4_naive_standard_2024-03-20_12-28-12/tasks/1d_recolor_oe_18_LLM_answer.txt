    def groupAnagrams(self, s: List[str]) -> List[List[str]]:
        d = {}
        
        for i in s:
            key = ''.join(sorted(i))
            if key not in d:
                d[key] = [i]
            else:
                d[key].append(i)
                
        return list(d.values())
