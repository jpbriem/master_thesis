class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0
        
        max_len = 1
        for i in range(len(s)):
            start = i - 1
            end = i + 1
            while start >= 0 and end < len(s):
                if s[start] == s[end]:
                    max_len = max(max_len, end - start + 1)
                    start -= 1
                    end += 1
                else:
                    break
        
        return max_len
