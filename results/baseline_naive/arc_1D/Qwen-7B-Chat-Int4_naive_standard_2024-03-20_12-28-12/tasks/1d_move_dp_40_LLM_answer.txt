class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0

        # check for single character palindrome
        if len(s) == 1:
            return 1
        
        start = 0
        max_len = 1

        for i in range(len(s)):
            if i - max_len >= 0 and s[i-max_len] == s[i]:
                start = max_len + 1
            else:
                max_len = i - start + 1

        return max_len
