def longest_palindrome(input):
    # convert input to lowercase and remove spaces
    input = ''.join(c for c in input.lower() if c.isalnum()).replace(' ', '')
    
    # initialize variables
    max_len = 0
    start_index = None
    
    # iterate over each character in the input string
    for i, char in enumerate(input):
        # check if current character is the same as the previous one
        if i > 0 and input[i] == input[i-1]:
            # update length of palindrome if necessary
            max_len = max(max_len, i+1)
            
            # update starting index if necessary
            start_index = i
            
        else:
            # reset length and starting index if characters are different
            max_len = 1
            start_index = i
            
    # return longest palindrome
    return input[start_index:start_index + max_len]
