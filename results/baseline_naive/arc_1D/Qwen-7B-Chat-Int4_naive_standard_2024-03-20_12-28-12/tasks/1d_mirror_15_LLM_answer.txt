class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = []
        
        for i in range(len(strs)):
            s = set()
            
            for j in range(i+1, len(strs)):
                if sorted(list(strs[i])) == sorted(list(strs[j])):
                    s.add(sorted(list(strs[j])))
                
            ans.append(list(s))
            
        return ans
