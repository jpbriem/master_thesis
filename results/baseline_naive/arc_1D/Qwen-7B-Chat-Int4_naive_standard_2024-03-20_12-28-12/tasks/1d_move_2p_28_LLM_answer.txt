def longest_palindrome(input_string):
    # Initialize variables
    max_length = 0
    start_index = None

    for i in range(len(input_string)):
        # Check if current character is different from previous one
        if i > 0 and input_string[i] != input_string[i-1]:
            # Update maximum length and start index if necessary
            if max_length < len(input_string) - i + 1:
                max_length = len(input_string) - i + 1
                start_index = i - max_length + 1

    # Append a dot at the end of the output string if needed
    if start_index == None or start_index >= len(input_string) - max_length + 1:
        return [".", "."]
    else:
        return input_string[start_index:start_index+max_length]
