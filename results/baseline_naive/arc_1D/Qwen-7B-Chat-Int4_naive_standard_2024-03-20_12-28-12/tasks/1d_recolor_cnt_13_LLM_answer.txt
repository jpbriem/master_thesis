    def longestPalindrome(self, s: str) -> int:
        n = len(s)

            return 0

    def findLongestPalindromeUtil(self, s: str, left: int, right: int) -> int:
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
        
        return right - left - 1
