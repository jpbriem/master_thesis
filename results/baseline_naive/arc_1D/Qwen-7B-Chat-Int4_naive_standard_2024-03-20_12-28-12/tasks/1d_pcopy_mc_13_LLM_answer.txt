    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:

        for s in strs:
            key = ''.join(sorted(s))
            if key not in anagram_map:
                anagram_map[key] = []
            anagram_map[key].append(s)
        
        result = []

                result.append(v)
        
        return result
