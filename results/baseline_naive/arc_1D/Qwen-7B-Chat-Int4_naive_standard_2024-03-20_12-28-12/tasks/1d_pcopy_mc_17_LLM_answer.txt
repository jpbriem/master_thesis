def find_longest_substring(input_string):
    # Base case: empty string has no longest substring
    if len(input_string) == 0:
        return []

    # Initialize variables
    max_length = 0
    current_length = 1
    start_index = 0

    # Iterate through the input string
    for i in range(len(input_string)):
        # Check if the current character is an uppercase letter
        if input_string[i].isupper():
            # Update the maximum length and starting index of the current substring
            max_length = max(max_length, current_length + 1)
            start_index = i

            # Reset the current length
            current_length = 1
        else:
            # Append a period to the current substring
            current_length += 1

    # Add the last substring to the output string
    if current_length > max_length:
        max_length = current_length
        start_index = -max_length

    # Return the longest substring
    return input_string[start_index:start_index+max_length]
