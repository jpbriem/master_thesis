class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        res = []
        
        for i in range(len(strs)):
            if not res:
                res.append([])
            
            res[i] += [strs[i]]
            
            for j in range(i+1, len(strs)):
                if sorted(res[i], key=str.lower) == sorted(strs[j], key=str.lower):
                    res[i].append(strs[j])
                    
        return res
