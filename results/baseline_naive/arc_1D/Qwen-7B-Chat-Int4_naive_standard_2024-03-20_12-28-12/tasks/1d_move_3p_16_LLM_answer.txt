class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[str]
        :rtype: List[str]
        """
        if not nums:
            return []
        
        result = [nums[0]]
        for i in range(1, len(nums)):
            if nums[i] != result[-1]:
                result.append(nums[i])
                
        return result
