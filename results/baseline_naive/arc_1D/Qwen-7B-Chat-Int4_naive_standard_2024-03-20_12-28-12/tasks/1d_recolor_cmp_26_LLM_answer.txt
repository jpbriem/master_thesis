    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        
        if len(nums) == 0:
            return res

        while stack:
            
            temp = []
            for i in range(len(stack)):
                temp.append(stack[i])
                
            res.append(temp)
            
            # pop the last element of the stack and add it to all other elements 
            for i in range(1, len(stack)):
                stack[i-1] += stack[i]

            stack.pop()
        
        return res
