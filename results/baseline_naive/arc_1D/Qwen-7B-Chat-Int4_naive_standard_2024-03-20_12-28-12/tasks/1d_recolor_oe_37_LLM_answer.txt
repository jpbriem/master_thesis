class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0:
            return 0
        
        start = 0
        maxLength = 1
        
        for i in range(len(s)):
            # Check odd length palindromes
            l = r = i
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                l -= 1
                r += 1
            else:
                start = max(start, l + 1)
                maxLength = max(maxLength, r - l + 1)

            # Check even length palindromes
            l = r = i
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                l -= 1
                r += 1
            else:
                start = max(start, l + 1)
                maxLength = max(maxLength, r - l + 1)
        
        return maxLength
