```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram_map = {}
        for i in range(len(strs)):
            key = ''.join(sorted(strs[i]))
            if key not in anagram_map:
                anagram_map[key] = []
            anagram_map[key].append(strs[i])
        
        res = []
        for key in anagram_map.keys():
            res.append(anagram_map[key])
        
        return res
```
