class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s:
            return 0
        
        ans = 1
        start = 0
        max_len = 1
        
        for i in range(len(s)):
            # Check from left and right
            left = i - 1
            while left >= 0 and s[left] == s[i]:
                left -= 1
                max_len += 1
                
            right = i + 1
            while right < len(s) and s[right] == s[i]:
                right += 1
                max_len += 1
                
            # If length of palindrome is greater than current maximum
            # update the answer
            if max_len > ans:
                ans = max_len
                start = i - (max_len // 2)
        
        return ans
