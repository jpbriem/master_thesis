def replace_g(input):
    output = []
    g_count = 0
    c_count = 0

    for char in input:
        if char == '.':
            if not g_count:
                c_count += 1
            else:
                g_count -= 1
            continue
        elif char == 'g':
            if not g_count:
                g_count += 1
            else:
                g_count -= 1
        else:
            c_count += 1

        output.append(char)

    return ''.join(output)
