def longest_substring_vowels(s):
    vowels = "aeiouAEIOU"
    max_len = 0
    start_index = 0

    for i in range(len(s)):
        # Check if the current character is a vowel
        if s[i] in vowels:
            # Update the maximum length and start index
            max_len = max(max_len, i - start_index + 1)
            start_index = i

    return s[start_index-max_len:start_index]
