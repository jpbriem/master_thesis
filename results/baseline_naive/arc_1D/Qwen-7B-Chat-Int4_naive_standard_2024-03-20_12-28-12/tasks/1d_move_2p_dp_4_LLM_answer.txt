class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s:
            return 0

        n = len(s)
        dp = [[False for _ in range(n)] for _ in range(n)]
        
        ans = 0
        start = 0
        
        # odd length palindromes starting from the center
        for i in range(n-1,-1,-1):
            j = i
            while j >= 0 and s[j] == s[i]:
                dp[i][j] = True
                j -= 1
                
            if dp[i][i+1]:
                start = i + 1
                ans += 2
                break
                
        # even length palindromes
        for i in range(n):
            j = i
            while j < n and s[j] == s[i]:
                dp[i][j] = True
                j += 1
                
            if dp[i][i+1]:
                start = i + 1
                ans += 2
                break
                
        return ans
