class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        anagram_dict = {}
        for s in strs:
            if len(s) == 0:
                continue
            key = ''.join(sorted(s))
            if key not in anagram_dict:
                anagram_dict[key] = [s]
            else:
                anagram_dict[key].append(s)
        
        res = []
        for k, v in sorted(anagram_dict.items()):
            res.append(v)
        
        return res
