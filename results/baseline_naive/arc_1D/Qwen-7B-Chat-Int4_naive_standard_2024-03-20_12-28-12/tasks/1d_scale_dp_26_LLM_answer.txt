class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0

        start = 0
        max_length = 1

        for i in range(len(s)):
            # Check odd length palindromes
            left = right = i
            while left >= 0 and right < len(s):
                if s[left] != s[right]:
                    break
                left -= 1
                right += 1
                max_length = max(max_length, right - left + 1)

            # Check even length palindromes
            left = right = i
            while left >= 0 and right < len(s):
                if s[left] != s[right]:
                    break
                left -= 2
                right += 2
                max_length = max(max_length, right - left + 1)

        return max_length
