class Solution:
    def removeInvalidParentheses(self, s: str) -> str:
        stack = []
        
        for i in range(len(s)):
            if s[i] == '(' or s[i] == ')':
                while stack and (s[stack[-1]] != '(' and s[stack[-1]] != ')'):
                    stack.pop()
                
                if s[i] == '(':
                    stack.append(i)
                else:
                    stack.pop()
            
        res = ''
        for i in range(len(s)):
            if s[i] == '(':
                res += s[i]
            elif s[i] == ')':
                while stack and s[stack[-1]] != '(':
                    res += s[stack.pop()]
                res += s[i]

                res += s[i]
                
        return res
