def longest_palindrome(input):
    n = len(input)
    start, end = 0, 0
    for i in range(n):
        if input[i] == '.':
            continue
        j = i + 1
        while j < n and input[j] != '.':
            j += 1
        if j - i > end - start:
            start = i
            end = j
    return ''.join(input[start:end+1])
