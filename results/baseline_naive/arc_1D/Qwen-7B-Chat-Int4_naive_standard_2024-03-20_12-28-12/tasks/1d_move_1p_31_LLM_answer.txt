class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        
        start, end = 0, 0
        for i in range(n):
            if s[i] == '.':
                continue
            
            j = max(0, i - 1)
            while j >= 0 and s[j] == s[i]:
                j -= 1
                
            if j < i:
                start = j + 1
                end = i
            else:
                end = i
        
        return end - start + 1
