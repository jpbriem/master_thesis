def longest_palindrome(input_string):
    if not input_string:
        return []
    
    n = len(input_string)
    start = 0
    max_length = 1
    
    for i in range(n):
        # check odd length palindromes
        l, r = i - 1, i + 1
        while l >= 0 and r < n and input_string[l] == input_string[r]:
            if r - l + 1 > max_length:
                start = l
                max_length = r - l + 1
            l -= 1
            r += 1
        
        # check even length palindromes
        l, r = i - 2, i + 2
        while l >= 0 and r < n and input_string[l] == input_string[r]:
            if r - l + 2 > max_length:
                start = l
                max_length = r - l + 2
            l -= 2
            r += 2
    
    return [input_string[start:start+max_length]]
