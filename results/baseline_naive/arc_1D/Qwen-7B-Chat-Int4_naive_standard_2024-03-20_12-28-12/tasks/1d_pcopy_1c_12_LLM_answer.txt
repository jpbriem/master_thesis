def longest_palindrome(input_string):
    # Convert input string to a list of characters
    char_list = list(input_string)
    
    # Initialize variables for storing start and end indices of longest palindrome
    max_start_index = 0
    max_end_index = 0
    
    # Loop through each character in the list
    for i in range(len(char_list)):
        # If current character is not an alphabet, move on to next character
        if not char_list[i].isalpha():
            continue
        
        # Initialize variables for storing start and end indices of current palindrome
        start_index = i - 1
        end_index = i + 1
        
        # Check for palindromes starting from current character
        while start_index >= 0 and end_index < len(char_list) and char_list[start_index] == char_list[end_index]:
            # Update maximum length and indices of current palindrome
            max_length = max(max_length, end_index - start_index + 1)
            
            # Move start index back one step and end index forward one step
            start_index -= 1
            end_index += 1
            
        # If current character is part of a longer palindrome, update the maximum length
        if start_index >= 0 and end_index < len(char_list) and char_list[start_index] == char_list[end_index]:
            max_length = max(max_length, end_index - start_index + 1)
        
    # Return the longest palindrome found
    return ''.join(char_list[max_start_index : max_end_index + 1])
