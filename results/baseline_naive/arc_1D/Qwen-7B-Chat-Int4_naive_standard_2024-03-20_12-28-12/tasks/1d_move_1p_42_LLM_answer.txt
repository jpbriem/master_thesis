```python
def longest_palindrome(input):
    if len(input) == 0:
        return []
    
    result = []
    for i in range(len(input)):
        if i < len(input)-1 and input[i] == input[i+1]:
            j = i + 2
            while j < len(input) and input[j] == input[j-1]:
                j += 1
            result.append('.'.join([input[i:j]]))
        else:
            result.append(input[i])
    
    return result
```
