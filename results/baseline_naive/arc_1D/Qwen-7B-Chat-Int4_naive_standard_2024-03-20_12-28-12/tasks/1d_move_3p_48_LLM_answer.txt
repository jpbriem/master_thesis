def remove_dots(input_string):
    # Initialize an empty list to store the output strings
    output_strings = []
    
    # Iterate over each character in the input string
    for i, char in enumerate(input_string):
        if char == '.':
            # If the current character is a dot, skip it
            continue
            
        # Otherwise, add the current character to the output string
        output_strings.append(char)
        
    # Join the characters in the output string into a single string using dots as separators
    return ''.join(output_strings).replace('.', '')
