class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 1

        start = 0
        max_len = 1
        
        for i in range(len(s)):
            # check odd length palindromes starting from the current character
            l = self.longestPalindromeUtil(s, i, i)
            
            # check even length palindromes starting from the current character and previous characters
            r = self.longestPalindromeUtil(s, i+1, i+1)
            
            # take the maximum of two palindromes
            max_len = max(max_len, l+r)

        return max_len

    def longestPalindromeUtil(self, s: str, left: int, right: int):
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
            
        return right - left - 1
