    if len(input_string) == 0:
        return input_string

    start = 0
    max_length = 1

    for i in range(len(input_string)):
        # check odd length palindromes starting from current character
        end = i
        while end < len(input_string) and input_string[end] == input_string[i]:
            end += 1
        if end - i > max_length:
            start = i
            max_length = end - i

        # check even length palindromes ending at current character
        start = i
        while start >= 0 and input_string[start] == input_string[i]:
            start -= 1
        if start + i > max_length:
            start = i
            max_length = start + i

    return input_string[start:start+max_length]
