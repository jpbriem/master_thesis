def convert_to_indices(input_list):
    # Create a dictionary to map characters to indices
    char_indices = {char: i for i, char in enumerate(set(input_list))}

    # Iterate over the input list and replace characters with their indices
    output_list = [char_indices[char] if char in char_indices else '.' for char in input_list]

    return output_list
