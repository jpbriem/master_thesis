class Solution:
    def longestPalindrome(self, s):
        if not s:
            return []
        
        n = len(s)
        start = 0
        max_length = 1
        
        for i in range(n):
            # check odd length palindromes
            j = i - 1
            while j >= 0 and s[j] == s[i]:
                start = j + 1
                max_length += 2
                j -= 1
                
            # check even length palindromes
            k = i + 1
            while k < n and s[k] == s[i]:
                start = i
                max_length += 2
                k += 1
                
        return [s[start:start+max_length]]
