class Solution:
    def convert(self, s):
        """
        :type s: str
        :rtype: str
        """
        if len(s) == 0 or s[0] != '.':
            return s

        result = []
        prev_char = None
        for char in s:
            if char != prev_char:
                if prev_char is not None and prev_char != '.':
                    result.append(prev_char)
                prev_char = char
            else:
                prev_char = char

        if prev_char is not None and prev_char != '.':
            result.append(prev_char)

        return ''.join(result)
