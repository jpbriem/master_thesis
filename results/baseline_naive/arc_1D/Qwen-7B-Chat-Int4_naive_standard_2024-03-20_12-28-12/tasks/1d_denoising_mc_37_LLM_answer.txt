def longest_palindrome(input):
    # Convert input to lowercase and remove all non-alphanumeric characters
    input = ''.join(c for c in input.lower() if c.isalnum()).replace('.', '').replace('i', '')

    # Initialize variables
    start = 0
    max_length = 1

    # Iterate over each character in the input
    for i, char in enumerate(input):
        # Check for odd-length palindromes starting from the current position
        length = 1
        while (i + length) < len(input) and input[i:i+length] == input[i:i+length][::-1]:
            length += 1

        # Update the maximum palindrome length found so far
        if length > max_length:
            max_length = length
            start = i - length // 2

    # Return the longest palindrome substring
    return input[start:start+max_length]
