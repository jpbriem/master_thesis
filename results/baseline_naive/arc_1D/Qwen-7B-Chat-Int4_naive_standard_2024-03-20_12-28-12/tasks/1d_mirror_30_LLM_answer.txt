class Solution:
    def longestPalindrome(self, s: str) -> int:
        n = len(s)
        
        if n < 2:
            return n
        
        start = end = 0
        for i in range(n):
            if s[i] == s[i+1]:
                end += 1
            else:
                start = i + 1
                break
        
        for i in range(n - 1, -1, -1):
            if s[i] == s[i-1]:
                end -= 1
            else:
                start = i - 1
                break
        
        max_len = min(start + end, n - start + 1)
        return max_len
