class Solution:
    def maxChunksToSorted(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        res = []
        
        for num in nums:
            res.append([num])
            
        i = 0
        while i < len(res):
            j = i + 1
            while j < len(res) and res[j][0] <= res[i][0]:
                res[j].extend(res[i])
                j += 1
            
            i = j
        
        return len(res)
