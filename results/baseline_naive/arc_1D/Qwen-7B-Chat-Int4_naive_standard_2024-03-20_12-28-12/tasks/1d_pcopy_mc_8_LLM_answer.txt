class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) < 2:
            return 0
        
        max_len = 1
        for i in range(len(s)):
            # Check odd length palindromes with center at current character
            left = right = i
            while left >= 0 and right < len(s) and s[left] == s[right]:
                max_len = max(max_len, right - left + 1)
                left -= 1
                right += 1
            
            # Check even length palindromes with center between two characters
            left = right = i
            while left >= 0 and right < len(s) and s[left] == s[right]:
                max_len = max(max_len, right - left + 2)
                left -= 1
                right += 1
                
        return max_len
