    def findTheDifference(self, nums: List[str]) -> str:
        if len(nums) < 2:
            return None
        
        for i in range(len(nums)):
            for j in range(i+1, len(nums)):
                diff = set(nums[i]) ^ set(nums[j])

                    return '.'.join(sorted(diff))
        
        return None
