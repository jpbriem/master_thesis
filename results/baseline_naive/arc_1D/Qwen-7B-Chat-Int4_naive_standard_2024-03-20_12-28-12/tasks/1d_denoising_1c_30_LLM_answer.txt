class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 0
        
        start = 0
        maxLength = 1
        
        for i in range(len(s)):
            # check odd length palindromes
            l = i - 1
            r = i + 1
            
            while l >= 0 and r < len(s) and s[l] == s[r]:
                if r - l > maxLength:
                    start = l
                    maxLength = r - l + 1
                    
                l -= 1
                r += 1
            
            # check even length palindromes
            l = i - 2
            r = i + 2
            
            while l >= 0 and r < len(s) and s[l] == s[r]:
                if r - l > maxLength:
                    start = l
                    maxLength = r - l + 1
                    
                l -= 2
                r += 2
                
        return maxLength
