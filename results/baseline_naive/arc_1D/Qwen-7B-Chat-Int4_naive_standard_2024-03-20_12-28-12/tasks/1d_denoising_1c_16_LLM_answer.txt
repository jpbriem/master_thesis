def longest_palindrome(input_string):
    # convert string to lowercase and remove all non-alphanumeric characters
    input_string = ''.join(c for c in input_string.lower() if c.isalnum())
    
    # initialize variables
    max_length = 0
    start_index = None
    
    # loop through each character in the string
    for i in range(len(input_string)):
        # check for palindromes of length 1
        if input_string[i:i+1] == input_string[i:i+1][::-1]:
            current_length = 1
        else:
            current_length = 0
        
        # update maximum palindrome length found so far
        if current_length > max_length:
            max_length = current_length
            start_index = i - max_length + 1
            
    # return the longest palindrome
    return input_string[start_index:start_index+max_length]
