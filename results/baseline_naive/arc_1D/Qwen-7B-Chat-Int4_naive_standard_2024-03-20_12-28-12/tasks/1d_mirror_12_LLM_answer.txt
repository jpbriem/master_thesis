class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s:
            return 0

        start = end = -1
        for i in range(len(s)):
            if s[i] == s[end+1]:
                end += 1
            else:
                start = end + 1
                while start >= 0 and s[start] != s[end+1]:
                    start -= 1
                end = start + 1

        return max(end - start + 1, self.longestPalindrome(s[:start+1]))
