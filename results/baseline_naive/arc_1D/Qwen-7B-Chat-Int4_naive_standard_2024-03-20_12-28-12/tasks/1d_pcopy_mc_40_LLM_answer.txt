def longest_palindrome(input_string):

        while i - start >= 0 and input_string[i] == input_string[start]:
            start += 1
        # if we have found such a character, find the last character that is not equal to itself on its right side
        while end < len(input_string) and input_string[end] == input_string[-start-1]:
            end += 1
        # update the maximum palindrome length and the starting and ending indices
        if end - start > max_length:

            end_index = end

    # return the longest palindrome
