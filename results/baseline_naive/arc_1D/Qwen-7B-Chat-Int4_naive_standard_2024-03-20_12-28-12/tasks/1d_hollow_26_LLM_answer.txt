class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0

        # check for single character palindromes
        n = len(s)
        ans = 1
        for i in range(n):
            if s[i] == s[n-i-1]:
                ans += 2
        return ans
        
        # use dynamic programming to find the maximum palindrome length
        start = 0
        max_len = 1
        for end in range(len(s)):
            while start <= end and s[start] != s[end]:
                start += 1
            max_len = max(max_len, end - start + 1)

        return max_len
