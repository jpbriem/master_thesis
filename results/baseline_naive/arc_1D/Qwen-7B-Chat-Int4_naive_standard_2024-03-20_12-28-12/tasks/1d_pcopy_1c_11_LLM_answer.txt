class Solution:
    def longestSubstring(self, s):
        if not s or len(s) == 0:
            return 0

        left = right = 0
        result = 0

        for i in range(len(s)):
            if s[i] != '.' and (right > left):
                right = max(right, i + 1)
            else:
                left = i + 1
                result = max(result, right - left)

        return result
