class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        d = {}
        for s in strs:
            chars = [char for char in s]
            key = ''.join(chars)
            if key not in d:
                d[key] = []
            d[key].append(s)
            
        res = []
        for k, v in d.items():
            res.append(v)
            
        return res
