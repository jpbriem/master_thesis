class Solution:
    def longestSubsequence(self, nums):
        """
        :type nums: List[str]
        :rtype: int
        """
        n = len(nums)
        
        dp = [0] * (n + 1) 
        for i in range(n+1): 
            if(i == 0 or i == 1): 
                dp[i] = 1
            else: 
                dp[i] = max(dp[i-1], dp[i-2]) + 1
        
        return dp[n]
