def remove_consecutive_dots(input):
    i = 0
    j = len(input) - 1

    while i < j:
        if input[i] == '.':
            output.append(input[i])
            i += 1

            output.append(input[j])
            j -= 1
        else:

            i += 1
            output.append(input[j])
