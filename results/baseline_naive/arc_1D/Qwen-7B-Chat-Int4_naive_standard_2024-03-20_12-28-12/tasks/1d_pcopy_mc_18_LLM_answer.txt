class Solution:
    def longestSubsequence(self, s: str) -> int:
        n = len(s)
        dp = [0] * (n + 1)
        for i in range(n):
            if i == 0 or s[i - 1] != s[i]:
                dp[i + 1] = max(dp[i], 1)
            else:
                dp[i + 1] += dp[i]
        
        return dp[-1]
