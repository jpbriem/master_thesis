class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        res = [[] for _ in range(len(set(strs[0])))]
        for s in strs:
            res[set(s)].append(s)
            
        return res
