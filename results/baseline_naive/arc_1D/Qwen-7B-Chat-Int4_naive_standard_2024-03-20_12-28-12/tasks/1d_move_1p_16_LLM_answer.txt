class Solution:
    def numIsland(self, grid: List[List[int]]) -> int:
        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        
        def dfs(x, y):
            if x < 0 or y < 0 or x >= len(grid) or y >= len(grid[0]) or grid[x][y] == 0:
                return False
            
            grid[x][y] = True
            for dx, dy in directions:
                dfs(x + dx, y + dy)
            
            return True
        
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j]:
                    count += dfs(i, j)
        
        return count
