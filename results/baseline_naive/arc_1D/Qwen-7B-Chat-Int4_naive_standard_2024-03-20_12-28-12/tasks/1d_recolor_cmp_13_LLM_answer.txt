class Solution:
    def groupAnagrams(self, words: List[str]) -> List[List[str]]:
        word_map = {}
        
        for word in words:
            sorted_word = ''.join(sorted(word))
            
            if sorted_word not in word_map:
                word_map[sorted_word] = []
                
            word_map[sorted_word].append(word)
            
        return list(word_map.values())
