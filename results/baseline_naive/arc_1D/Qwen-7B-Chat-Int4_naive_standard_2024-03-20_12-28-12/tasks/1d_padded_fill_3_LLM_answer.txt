```python
class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0

        start = 0
        max_length = 1
        
        for i in range(len(s)):
            # check odd length palindromes
            if i - max_length >= 0 and s[i-max_length-1:i+1] == s[i-max_length-1:i+1][::-1]:
                max_length += 2
            else:
                # check even length palindromes
                if i - max_length >= 0 and i + max_length < len(s) and s[i-max_length-1:i+max_length+1] == s[i-max_length-1:i+max_length+1][::-1]:
                    max_length += 2
            
            if max_length > start:
                start = i - max_length
                
        return start + 1
```
