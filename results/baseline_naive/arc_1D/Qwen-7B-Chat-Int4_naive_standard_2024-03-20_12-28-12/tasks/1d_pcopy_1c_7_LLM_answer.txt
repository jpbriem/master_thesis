class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) == 0:
            return 0

        n = len(s)
        start, maxLen = 0, 1

        for i in range(n):
            # check odd length palindromes
            l, r = i, i
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
                maxLen = max(maxLen, r - l + 1)

            # check even length palindromes
            l, r = i, i+1
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
                maxLen = max(maxLen, r - l + 1)

            if maxLen > 1:
                start = i - (maxLen // 2)
        
        return maxLen
