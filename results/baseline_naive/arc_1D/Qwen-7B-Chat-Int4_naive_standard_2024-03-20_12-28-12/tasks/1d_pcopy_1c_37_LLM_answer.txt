def longest_palindrome(input_string):
    # Convert the input string to lowercase and remove all non-alphanumeric characters
    cleaned_input = ''.join(c.lower() for c in input_string if c.isalnum())
    
    # Initialize a dictionary to store the length of each palindrome found so far
    lengths = {}
    
    # Iterate over all possible substrings of the cleaned input string
    for i in range(len(cleaned_input)):
        for j in range(i+1, len(cleaned_input)+1):
            substring = cleaned_input[i:j]
            
            # Check if the substring is a palindrome
            if substring == substring[::-1]:
                # If it is, update the length in the dictionary
                if substring not in lengths or lengths[substring] < j-i-1:
                    lengths[substring] = j-i-1
    
    # Return the longest palindrome found
    return max(lengths, key=lengths.get)
