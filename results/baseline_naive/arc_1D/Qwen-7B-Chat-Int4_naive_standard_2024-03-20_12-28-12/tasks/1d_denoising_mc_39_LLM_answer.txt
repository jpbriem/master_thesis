    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        
        start = 0
        max_len = 1
        
        for i in range(n):
            # check odd length palindromes
            l = i - 1
            r = i + 1
            
            while l >= 0 and r < n and s[l] == s[r]:
                if r - l > max_len:
                    start = l + 1
                    max_len = r - l + 1
                    
                l -= 1

            # check even length palindromes
