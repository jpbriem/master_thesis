class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 0

        start = 0
        max_len = 1
        
        for i in range(len(s)):
            # check odd length palindromes starting from current character
            l = i - 1
            r = i + 1
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                if r - l > max_len:
                    start = l + 1
                    max_len = r - l
                l -= 1
                r += 1
                
            # check even length palindromes centered at current character
            l = i - 1
            r = i + 1
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                if r - l > max_len:
                    start = l + 1
                    max_len = r - l
                l -= 1
                r += 1
            
            # check palindrome with length equal to the current character
            if i == 0 or i == len(s) - 1:
                if s[i] == s[i-1] and s[i] == s[i+1]:
                    max_len = 3
                    
        return max_len
