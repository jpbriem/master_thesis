class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        res = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key in res:
                res[key].append(s)
            else:
                res[key] = [s]
                
        return list(res.values())
