class Solution:
    def longestSubstring(self, s):
        if not s:
            return 0
        
        start = 0
        max_len = 0
        for i in range(len(s)):
            while (start < len(s) and s[start] == s[i]):
                start += 1
                
            max_len = max(max_len, i - start + 1)
        
        return max_len
