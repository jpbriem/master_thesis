class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 0

        max_len = 1
        start = 0
        for i in range(len(s)):
            # check odd length palindromes
            j = i - 1
            while j >= 0 and s[j] == s[i]:
                j -= 1
                max_len += 2
            # check even length palindromes
            k = i + 1
            while k < len(s) and s[k] == s[i]:
                k += 1
                max_len += 2

            if max_len > start + 1:
                start = i - (max_len - 1) // 2
        return max_len
