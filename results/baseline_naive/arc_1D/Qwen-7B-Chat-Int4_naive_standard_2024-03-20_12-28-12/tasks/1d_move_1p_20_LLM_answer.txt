    n = len(input)

        start = i - 1
        while start >= 0 and input[start] == input[i]:
            start -= 1
            max_len = max(max_len, start + 2)

            max_len = max(max_len, end - 2)

    return [input[j:j+max_len] for j in range(0, n, max_len)]
