class Solution:
    def longestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s or len(s) == 0:
            return 0

        result = 0
        left, right = 0, 0
        for i in range(len(s)):
            while right < len(s) and s[right] != s[i]:
                right += 1
            result = max(result, right - i)
            left = right + 1
        
        return result
