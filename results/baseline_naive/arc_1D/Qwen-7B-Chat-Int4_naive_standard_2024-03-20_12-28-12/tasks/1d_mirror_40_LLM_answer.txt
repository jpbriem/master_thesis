```python
def replace_characters_with_letters(input_list):
    output_list = []
    for char in input_list:
        if char.isalpha():
            index = ord(char) - 97  # Convert to ASCII code, subtract 97 to get index in alphabet
            output_list.append(chr(index))
        else:
            output_list.append(char)
    return output_list
```
