class Solution:
    def longestSubstring(self, s: str) -> int:
        if not s:
            return 0

        start = 0
        max_length = 0
        
        for i in range(len(s)):
            # check if the current character is different from the previous one
            if s[i] != s[start]:
                start += 1
                
            # update the maximum length
            max_length = max(max_length, i - start + 1)
        
        return max_length
