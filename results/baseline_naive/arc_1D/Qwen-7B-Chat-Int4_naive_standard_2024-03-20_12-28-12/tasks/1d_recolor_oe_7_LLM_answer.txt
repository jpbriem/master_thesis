class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs or len(strs) == 0:
            return []
        
        res = [[] for _ in range(len(set(strs[0])))]
        for s in strs:
            c_set = set(s)
            for i, item in enumerate(res):
                if c_set == set(item):
                    item.append(s)
                    break
            else:
                res.append([s])
        
        return res
