class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) < 2:
            return 0
        
        start = max(0, -1)
        end = max(-1, 0)
        
        for i in range(len(s)):
            if s[i] == s[start]:
                start += 1
                
            else:
                start = i - 1
        
        for i in range(len(s)-1, -1, -1):
            if s[i] == s[end]:
                end -= 1
                
            else:
                end = i + 1
        
        return max(end - start + 1, 0)
