class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 1
        
        max_length = 1
        for i in range(len(s)):
            start_index = i - 1
            end_index = i + 1
            
            while start_index >= 0 and end_index < len(s):
                # Check if the substring starting from start_index and ending at end_index is a palindrome
                sub_string = s[start_index:end_index+1]
                
                if sub_string == sub_string[::-1]:
                    max_length = max(max_length, end_index - start_index + 1)
                    
                    # Move towards the center of the current palindrome
                    start_index -= 1
                    end_index += 1
                
                else:
                    break
                    
        return max_length
