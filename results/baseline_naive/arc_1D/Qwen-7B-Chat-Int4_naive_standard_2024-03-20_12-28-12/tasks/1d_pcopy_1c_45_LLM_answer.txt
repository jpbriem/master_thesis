def longest_increasing_subsequence(s):
    n = len(s)
    if n == 0:
        return []

    result = [s[0]]
    i = 1

    while i < n:
        if s[i] > s[i-1]:
            result.append(s[i])
        else:
            if result[-1] <= s[i]:
                result.append(result[-1])
            else:
                result = result[:-1]
        i += 1

    return result
