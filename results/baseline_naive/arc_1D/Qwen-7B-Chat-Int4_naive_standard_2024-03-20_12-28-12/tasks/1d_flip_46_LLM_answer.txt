def convert(input_string):
    # Iterate through the input string
    for i in range(len(input_string)):
        # Check if the current character is an uppercase letter
        if input_string[i].isupper():
            # Replace all lowercase letters with their corresponding uppercase letters
            input_string = input_string.replace(input_string[i], input_string[i].upper())
        elif input_string[i].islower():
            # Replace all uppercase letters with their corresponding lowercase letters
            input_string = input_string.replace(input_string[i], input_string[i].lower())

    return input_string
