class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        if not strs:
            return []
        
        ans = [[] for _ in range(len(set(strs[0])))]
        for s in strs:
            ind = [ord(c) - ord('a') for c in s]
            ans[ind].append(s)
            
        return ans
