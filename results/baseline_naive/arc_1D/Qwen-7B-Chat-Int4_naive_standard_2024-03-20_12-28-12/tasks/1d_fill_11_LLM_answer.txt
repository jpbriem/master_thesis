class Solution:
    def longestSubsequence(self, nums):
        """
        :type nums: List[str]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0

        n = len(nums)
        dp = [1] * (n + 1)

        for i in range(n - 1, -1, -1):
            for j in range(i + 1, n + 1):
                if nums[i] != nums[j]:
                    dp[j] = max(dp[j], dp[i] + 1)
        
        return dp[-1]
