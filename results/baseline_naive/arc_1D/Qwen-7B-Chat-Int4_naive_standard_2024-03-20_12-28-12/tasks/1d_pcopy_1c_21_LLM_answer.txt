class Solution:
    def findLargestRectangle(self, heights: List[int]) -> int:
        n = len(heights)
        stack = []
        res = 0
        
        for i in range(n):
            while stack and heights[stack[-1]] < heights[i]:
                h = heights[stack.pop()]
                w = (i - stack[-1] - 1) * h
                res = max(res, w + h)

            stack.append(i)

        return res
