```python
class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) == 0 or len(s) == 1:
            return 1

        start = 0
        max_len = 1
        
        for i in range(len(s)):
            if s[i] != s[0]:
                start = i + 1
                break
            
            if i - max_len >= 0 and s[i-max_len-1:i+1] == s[i-max_len-1:i+1][::-1]:
                max_len += 2
                
        return max_len
```
