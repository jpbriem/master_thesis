class Solution:
    def longestPalindrome(self, s):
        if not s or len(s) == 0:
            return []
        
        n = len(s)
        dp = [[False for _ in range(n)] for _ in range(n)]
        start = 0
        
        for i in range(n-1,-1,-1):
            for j in range(i+1,n):
                # If the characters at indices (i,j) are same, update the maximum length palindromic substring starting from index i and ending at index j
                if s[i] == s[j]:
                    dp[i][j] = True
                    start = max(start, i)
                
                # If the characters at indices (i,j) are different but they form a palindrome when combined with the character at index j-1, update the maximum length palindromic substring starting from index i and ending at index j
                elif dp[i+1][j-1]:
                    dp[i][j] = True
                    start = max(start, i)

        result = [s[start:start+len(dp[0][-1])]]
        
        for i in range(1,len(dp)):
            for j in range(len(dp[i])):
                if dp[i-1][j] and dp[i][j]:
                    result.append(s[start+i:j])
                    
        return result[::-1]
