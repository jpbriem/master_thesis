class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:
            return 0

        n = len(s)
        start = 0
        max_len = 1
        
        for i in range(n):
            l = self.helper(s[i:i+5], start, i-1)
            if l > max_len:
                max_len = l
                start = i - max_len + 1

        return max_len
    
    def helper(self, s, left, right):
        if left == right:
            return 1
        
        while left >= 0 and right < len(s) and s[left] == s[right]:
            left -= 1
            right += 1
            
        return right-left
