class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s or len(s)<=1:
            return 0

        start = end = max_len = 0
        for i in range(len(s)):
            # check even length palindromes
            l = r = i
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                l -= 1
                r += 1
            max_len = max(max_len, r - l + 1)

            # check odd length palindromes
            l = r = i
            while l >= 0 and r < len(s):
                if s[l] != s[r]:
                    break
                l -= 1
                r += 1
                max_len = max(max_len, r - l + 2)
