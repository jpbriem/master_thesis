class Solution:
    def longestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if not s:
            return 0
        
        ans = max(len(s[i:i+4]) for i in range(0, len(s), 4))
        
        # check the last character is same as previous one
        prev = ''
        count = 0
        for c in s:
            if c == prev:
                count += 1
            else:
                ans = max(ans, count)
                prev = c
                count = 1
                
        ans = max(ans, count)
        
        return ans
