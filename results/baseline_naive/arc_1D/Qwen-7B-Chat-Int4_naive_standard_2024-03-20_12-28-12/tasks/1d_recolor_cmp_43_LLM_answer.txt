def rotate_bombs(input):
    # initialize the output string as an empty list
    output = []

    # iterate through the input string
    for char in input:
        # if the character is a black square, add it to the output string
        if char == 'b':
            output.append(char)

        # if the character is not a black square, check if there are at least three consecutive black squares in the output string so far
        else:
            count = 0
            for i in range(len(output)-1):
                if output[i] == 'b' and output[i+1] == 'b' and output[i+2] == 'b':
                    count += 1
            # if there are at least three consecutive black squares in the output string, replace them with an 'f' character
            if count >= 3:
                output.pop()
                output.insert(i-1, 'f')
            # otherwise, simply append the character to the output string
            else:
                output.append(char)

    return ''.join(output)
