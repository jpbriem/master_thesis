```python
class Solution:
    def longestPalindrome(self, s: str) -> int:
        n = len(s)
        
        # base cases
        if n == 0 or n == 1:
            return 0
        
        start, end = 0, 0
        for i in range(n):
            if s[i] != '.' and s[i] != 'i':
                continue
            
            j = i + 1
            while j < n and s[j] != '.' and s[j] != 'i' and s[j] == s[i]:
                j += 1
                
            if j - i > end - start:
                start = i
                end = j
    
        return end - start
```
