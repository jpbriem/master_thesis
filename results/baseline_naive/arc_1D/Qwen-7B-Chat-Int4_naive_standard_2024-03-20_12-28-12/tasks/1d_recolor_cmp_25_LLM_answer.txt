```python
def convert_g_to_i(s):
    i = 0
    while i < len(s) - 1:
        if s[i] == '.' and s[i+1] == '.':
            s[i], s[i+1] = s[i+1], s[i]
            i += 2
        else:
            i += 1
    return ''.join(s)
```
