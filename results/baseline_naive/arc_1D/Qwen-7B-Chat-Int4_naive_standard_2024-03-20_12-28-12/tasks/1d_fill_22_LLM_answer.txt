def kmp(pattern):
    n = len(pattern)
    next = [-1] * n

    # Compute the next array
    i = j = 0
    while i < n and j < n:
        if pattern[j] == pattern[i]:
            i += 1
            j += 1
        else:
            if j != 0:
                j = next[j - 1]
            else:
                i += 1

    # Build the kmp table
    for i in range(1, n):
        next[i] = next[i - 1]

    return next
