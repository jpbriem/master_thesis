class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0
        
        start = 0
        max_length = 1
        
        for i in range(len(s)):
            # check odd length palindromes
            left, right = i - 1, i + 1
            while left >= 0 and right < len(s):
                if s[left] != s[right]:
                    break
                else:
                    if right - left > max_length:
                        start = left + 1
                        max_length = right - left + 1
                    left -= 1
                    right += 1
            
            # check even length palindromes
            left, right = i - 2, i + 2
            while left >= 0 and right < len(s):
                if s[left] != s[right]:
                    break
                else:
                    if right - left > max_length:
                        start = left + 1
                        max_length = right - left + 1
                    left -= 2
                    right += 2
            
        return max_length
