class Solution:
    def longestSubsequence(self, s):
        """
        :type s: str
        :rtype: int
        """
        n = len(s)
        dp = [[0] * (n+1) for _ in range(n+1)]
        
        for i in range(1, n+1):
            if s[i-1] == '.':
                dp[i][i] = 1
            else:
                dp[i][i] = 2
        
        for i in range(1, n+1):
            for j in range(i+1, n+1):
                
                if s[j-1] != '.':
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])
                    
                elif s[j-1] == s[i-1]:
                    dp[i][j] = dp[i-1][j] + 1
                    
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1]) + 1
                
        return dp[n][n]
