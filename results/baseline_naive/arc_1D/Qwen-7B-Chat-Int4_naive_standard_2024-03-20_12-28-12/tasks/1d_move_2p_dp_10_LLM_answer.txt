def longest_palindrome(input):
    input_str = ''.join(input)
    max_length = 0

    for i in range(len(input_str)):
        # check if current character is repeated and if so, find its length
        start_index = i - len(input_str[i]) + 1
        end_index = i + 1
        while start_index >= 0 and end_index < len(input_str) and input_str[start_index] == input_str[end_index]:
            start_index -= 1
            end_index += 1
        if end_index - start_index > max_length:
            max_length = end_index - start_index
    
    return [input_str[i:i+max_length] for i in range(0, len(input_str), max_length)]
