def longest_palindrome(input):
    if not input:
        return []

    # convert to lowercase and remove spaces
    input = ''.join(c for c in input.lower() if c.isalnum()).replace(' ', '')

    # initialize variables
    start, end = 0, 0
    max_len = 0

    # loop through each character of the string
    for i in range(len(input)):
        # check for palindromes starting from this character
        length = 1
        j = i
        while j >= 0 and j < len(input) and input[j] == input[i]:
            j -= 1
            length += 1

        # update maximum palindrome length if necessary
        if length > max_len:
            start = i - (length // 2)
            end = i + (length // 2)

        # check for palindromes starting from the next character
        length = 1
        k = i + 1
        while k < len(input) and input[k] == input[i]:
            k += 1
            length += 1

        # update maximum palindrome length if necessary
        if length > max_len:
            start = i - (length // 2)
            end = i + (length // 2)

    # extract and return the longest palindrome
    return input[start:end+1]
