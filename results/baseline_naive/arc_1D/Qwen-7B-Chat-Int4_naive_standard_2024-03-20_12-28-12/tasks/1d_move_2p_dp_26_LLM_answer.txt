class Solution:
    def removeHangingPunctuation(self, s: str) -> str:
        result = []
        for c in s:
            if c not in {'.', ',', ';'}:
                result.append(c)
            else:
                while True:
                    if len(result) > 0 and result[-1] != '.':
                        result.pop()
                    elif len(result) == 0 or result[-1] == '.':
                        break
        return ''.join(result)
