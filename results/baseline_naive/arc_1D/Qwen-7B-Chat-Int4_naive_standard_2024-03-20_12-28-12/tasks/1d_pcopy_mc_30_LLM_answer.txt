class Solution:
    def longestPalindrome(self, s):
        if len(s) < 2:
            return []
        
        start = 0
        max_len = 1
        
        for i in range(len(s)):
            # check palindrome from center of odd length string
            left, right = i - 1, i + 1
            while left >= 0 and right < len(s) and s[left] == s[right]:
                if (right - left + 1 > max_len):
                    start = left
                    max_len = right - left + 1
                    
                left -= 1
                right += 1
            
            # check palindrome from both ends of even length string
            left, right = i - 2, i + 2
            while left >= 0 and right < len(s) and s[left] == s[right]:
                if (right - left + 1 > max_len):
                    start = left
                    max_len = right - left + 1
                    
                left -= 2
                right += 2
                
        return [s[start:start+max_len]]
