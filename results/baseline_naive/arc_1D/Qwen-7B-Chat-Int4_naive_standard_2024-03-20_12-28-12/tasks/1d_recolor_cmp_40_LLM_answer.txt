class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0
        
        ans = 1
        start = 0
        for i in range(len(s)):
            if s[i] == s[i+1]:
                start += 2
            else:
                max_len = 1 + self.longestPalindrome(s[start:i])
                if max_len > ans:
                    ans = max_len

        return ans
