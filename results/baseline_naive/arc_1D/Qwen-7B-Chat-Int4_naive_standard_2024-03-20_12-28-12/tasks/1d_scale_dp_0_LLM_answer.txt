class Solution:
    def longestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) == 0 or len(set(s)) <= 1:
            return 0
        
        l = r = 0
        for i in range(len(s)):
            while (l < r and not set(s[l:r]).issubset(set(s[l+1:i]))):
                r -= 1
            l += 1
            
            while (r < len(s) and not set(s[r:l-1]).issubset(set(s[r:i]))):
                l -= 1
                
            max_len = r - l + 1
        
        return max_len
