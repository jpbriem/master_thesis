class Solution:
    def longestPalindrome(self, s: str) -> int:
        if len(s) < 2:
            return 0

        start = max(0, 1 - i for i in range(len(s)) if s[i] != s[-i-1])
        end = min(len(s), 1 + i for i in range(len(s)) if s[i] == s[-i-1])

        length = end - start + 1
        if length > 1 and s[start] == s[end]:
            return length

        for i in range(start+1, end):
            if s[i] != s[i-1]:
                break
            length += 1
        
        return length
