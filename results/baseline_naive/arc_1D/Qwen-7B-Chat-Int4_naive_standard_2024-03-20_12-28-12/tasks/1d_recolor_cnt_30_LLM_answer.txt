class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        if not strs:
            return []
        
        result = {}
        for s in strs:
            key = ''.join(sorted(s))
            if key in result:
                result[key].append(s)
            else:
                result[key] = [s]
                
        return list(result.values())
