class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) < 2:
            return 0
        
        n = len(s)
        
        start = end = -1
        ans = 0

        for i in range(n):
            if i <= end and s[i] == s[end]:
                end += 1
                
            elif i > start + 1 and s[start:i+1] == s[i-1:end-i+1][::-1]:
                end = i - 1
                start = i - 2
                
            else:
                start = i
            
        return max(ans, end - start + 1)
