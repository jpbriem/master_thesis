class Solution:
    def longestSubsequence(self, s: str) -> int:
        if not s or len(s) == 0:
            return 0
        
        n = len(s)
        
        dp = [[0] * (n+1) for _ in range(n+1)]
        
        for i in range(1,n+1):
            dp[i][i] = 1
        
        for l in range(2,n+1):
            for i in range(l-1):
                j = i + l - 1
                
                if s[j] == s[i]:
                    dp[i][j] = dp[i+1][j-1]
                    
                else:
                    if dp[i+1][j] > dp[i][j-1]:
                        dp[i][j] = dp[i+1][j]
                        
                    else:
                        dp[i][j] = dp[i][j-1]

        ans = 0

        for i in range(1,n+1):
            ans = max(ans,dp[1][i])

        return ans
