    dp = [[0] * (len(input) + 1) for _ in range(len(input) + 1)]

            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # Find the maximum length of the common subsequence
    result = dp[-1][-1]

    output = []
    index = result - 1
