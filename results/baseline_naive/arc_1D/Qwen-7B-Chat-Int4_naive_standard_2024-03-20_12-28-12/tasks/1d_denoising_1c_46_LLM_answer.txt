def longest_palindrome(input):
    if not input:
        return []
    
    start = 0
    max_length = 1
    
    for i, char in enumerate(input):
        # check for a palindrome starting at this index
        left = right = i
        while left >= 0 and right < len(input) and input[left] == input[right]:
            if right - left + 1 > max_length:
                start = left
                max_length = right - left + 1
            left -= 1
            right += 1
        
    # find all palindromes of length max_length
    result = [input[i:i+max_length] for i in range(start, start+max_length)]
    
    return result
