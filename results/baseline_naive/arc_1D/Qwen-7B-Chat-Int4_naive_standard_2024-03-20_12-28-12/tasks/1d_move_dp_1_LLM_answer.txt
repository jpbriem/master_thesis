class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        if not s:
            return ''
        
        seen = set()
        res = []
        for c in s:
            if c.isalpha():
                if c.lower() not in seen:
                    res.append(c)
                    seen.add(c.lower())
                    
        return ''.join(res)
