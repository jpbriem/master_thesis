    output_grid = [row[:] for row in grid]

    def should_hollow(r, c):

        # Check all four directions to ensure that the character is surrounded by the same character
        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nr, nc = r + dr, c + dc
