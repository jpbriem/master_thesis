def longest_palindrome_substring(s):
    n = len(s)
    if n == 0:
        return []
    
    start, end = 0, 0
    max_len = 1
    
    for i in range(n):
        # Check odd length palindromes centered at current character
        l = i - (i % 2) + 1
        r = i + (i % 2)
        if s[i] == s[l-1] and s[i] == s[r-1]:
            if r-l+1 > max_len:
                start, end = l, r
                
        # Check even length palindromes centered at current character
        l = i - (i % 2)
        r = i + (i % 2) + 1
        if s[i] == s[l-1] and s[i] == s[r-1]:
            if r-l+1 > max_len:
                start, end = l, r
                
    res = [s[start:end+1]]
    while True:
        start -= 1
        if start < 0 or start >= n:
            break
        
        end += 1
        if end >= n:
            break
        
        # Check if the substring starting from `start` is a palindrome
        if s[start] == s[end] and s[start+1] == s[end-1]:
            res.append(s[start:end+1])
        
    return res
