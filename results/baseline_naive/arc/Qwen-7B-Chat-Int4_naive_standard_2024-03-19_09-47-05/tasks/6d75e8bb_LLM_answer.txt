def longest_palindrome_substring(s):
    if not s:
        return []
    
    max_len = 0
    start = None
    
    for i in range(len(s)):
        # check odd length palindromes starting from current character
        l, r = i, i
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > max_len:
                max_len = r - l + 1
                start = l
            l -= 1
            r += 1
        
        # check even length palindromes centered at current character
        l, r = i, i+1
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > max_len:
                max_len = r - l + 1
                start = l
            l -= 1
            r += 1
        
        # check even length palindromes starting from current character and moving towards left
        l, r = i, i
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > max_len:
                max_len = r - l + 1
                start = l
            l -= 1
            r += 1
        
        # check even length palindromes starting from current character and moving towards right
        l, r = i, i+1
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if r - l + 1 > max_len:
                max_len = r - l + 1
                start = l
            l -= 1
            r += 1
    
    result = [s[start:start+max_len]]
    for i in range(max_len-1, -1, -1):
        if start+i < len(s) and s[start+i] == s[start-i]:
            result.append(s[start+i])
    
    return ''.join(result)
