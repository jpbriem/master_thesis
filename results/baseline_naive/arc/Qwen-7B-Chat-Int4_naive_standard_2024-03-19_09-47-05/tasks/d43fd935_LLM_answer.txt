class Solution:
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        n = len(s)
        
        if n == 0:
            return 0
        
        max_len = 1
        
        for i in range(n):
            for j in range(i+1, n+1):
                substr = s[i:j]
                
                # check palindrome length
                start = 0
                while start < len(substr) and substr[start] == substr[-(start+1)] :
                    start += 1
                
                # check palindrome
                end = start + 1
                while end < len(substr) and substr[end] == substr[start]:
                    end += 1
                    
                if end - start > max_len:
                    max_len = end - start
        
        return max_len
