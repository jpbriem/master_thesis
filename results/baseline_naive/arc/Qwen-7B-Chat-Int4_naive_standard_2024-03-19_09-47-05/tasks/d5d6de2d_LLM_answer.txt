def bubbleSort(arr):
    n = len(arr)
    
    # Traverse through all array elements
    for i in range(n):

        # Last i elements are already sorted
        for j in range(0, n-i-1):

            # Swap if the element found is greater than the next element
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
