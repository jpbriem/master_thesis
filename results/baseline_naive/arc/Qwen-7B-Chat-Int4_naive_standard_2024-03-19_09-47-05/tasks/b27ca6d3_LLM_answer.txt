    if n < 3:
        return s

    start, max_len = 0, 1
    for i in range(n):
        # Check odd length palindromes starting from current character
        if i - max_len >= 0 and s[i-max_len-1:i+1] == s[i-max_len-1:i+1][::-1]:

            max_len += 2

    return s[start:start+max_len]
