    def longestPalindrome(self, s):
        if len(s) == 0:
            return []
        
        start = max_len = 1
        
        for i in range(len(s)):
            if i - max_len >= 0 and s[i-max_len-1:i+1] == s[i-max_len-1:i+1][::-1]:
                start = i - max_len
                max_len += 2
            elif i - max_len >= 0 and s[i-max_len:i+1] == s[i-max_len:i+1][::1]:
                start = i - max_len
                
        res = [s[start:start + max_len]]
        while True:
            if not any([i == j for i, j in zip(res[-1], s)]):
                break
            else:
                end = res[-1].find(s)
                if end != -1:
                    res.append(s[end:end+len(res[-1])])
                
        return res[::-1]
