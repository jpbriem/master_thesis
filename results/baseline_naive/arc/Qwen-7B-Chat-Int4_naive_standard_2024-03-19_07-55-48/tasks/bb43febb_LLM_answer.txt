def numWays(matrix):
    n = len(matrix)
    m = len(matrix[0])
    k = max([len(row) for row in matrix])

    dp = [[[[0]*((k+1)**2)]*(m+1)*(n+1) for _ in range(k+1)] for _ in range(m+1)]
    dp[0][0][0] = 1

    for i in range(1, n+1):
        for j in range(1, m+1):
            for l in range(1, k+1):
                for matrix_val in range(0, k+1):
                    # Check if placing matrix_val is valid
                    if matrix[i-1][j-1][l-1] == matrix_val:
                        continue
                    
                    # Recursively update dp[i][j][l]
                    dp[i][j][l] += dp[i-1][j][l]

                    # Update dp[i][j][matrix_val]
