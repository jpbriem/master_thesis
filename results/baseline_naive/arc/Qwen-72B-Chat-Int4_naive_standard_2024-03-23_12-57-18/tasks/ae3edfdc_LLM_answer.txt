def replace_characters(grid):
    n = len(grid)
    m = len(grid[0])
    
    for i in range(n):
        for j in range(m-1, -1, -1):
            if grid[i][j] != 'a':
                for k in range(j+1, m):
                    grid[i][k] = grid[i][j]
                    
    return grid
