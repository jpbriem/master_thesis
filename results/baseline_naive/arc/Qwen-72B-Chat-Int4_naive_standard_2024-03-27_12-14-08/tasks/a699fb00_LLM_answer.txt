def find_paths(grid):
    m, n = len(grid), len(grid[0])
    directions = [(0, 1), (1, 0)]
    
    def dfs(i, j, path):
        if i < 0 or j < 0 or i >= m or j >= n or grid[i][j] != '.' or path.count('a') == 2:
            return
        
        grid[i][j] = 'b'
        for di, dj in directions:
            ni, nj = i + di, j + dj
            dfs(ni, nj, path + 'a')
            
    paths = set()
    for i in range(m):
        for j in range(n):
            if grid[i][j] == 'a':
                dfs(i, j, '')
                paths.add(tuple(grid))
                
    return list(paths)
