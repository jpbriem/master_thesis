def min_path_sum(grid):
    m = len(grid)
    n = len(grid[0])

    # Create a new grid to store the minimum path sums
    dp = [[float('inf')] * n for _ in range(m)]

    # Set the first row and column of the dp grid to the input grid values
    for i in range(m):
        for j in range(n):
            if i == 0 or j == 0:
                dp[i][j] = grid[i][j]

    # Compute the minimum path sum from (0, 0) to each cell using dynamic programming
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = min(dp[i-1][j], dp[i][j-1]) + grid[i][j]
    
    # Find the last non-zero element in the bottom-right corner of the dp grid
    i, j = m-1, n-1
    while i >= 0 and j >= 0 and dp[i][j] == float('inf'):
        i -= 1
        j -= 1
    
    # Return the minimum path sum from (0, 0) to the found cell plus the remaining distance to the bottom-right corner
    return dp[i+1][j+1] - grid[i][j]
