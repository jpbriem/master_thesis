def count_and_replace(matrix):
    counts = {}
    for row in matrix:
        for num in row:
            if num not in counts:
                counts[num] = 0
            counts[num] += 1

    for i in range(len(matrix)):
        for j in range(len(matrix[i])):
            if matrix[i][j] != 9 and counts[matrix[i][j]] > 1:
                matrix[i][j] -= 1
                
    return matrix
