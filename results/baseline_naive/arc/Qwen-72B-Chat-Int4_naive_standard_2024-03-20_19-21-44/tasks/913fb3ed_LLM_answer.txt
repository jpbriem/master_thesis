```python
def spread_numbers(grid):
    rows = len(grid)
    cols = len(grid[0])
    
    for i in range(rows):
        for j in range(cols):
            if grid[i][j] != 0 and (i == 0 or grid[i-1][j] == 0) and (j == 0 or grid[i][j-1] == 0):
                num = grid[i][j]
                queue = [(i, j)]
                
                while queue:
                    x, y = queue.pop(0)
                    
                    if x >= rows or y >= cols or grid[x][y] != 0:
                        continue
                    
                    grid[x][y] = num
                    
                    if x < rows - 1 and grid[x+1][y] == 0:
                        queue.append((x+1, y))
                    
                    if y < cols - 1 and grid[x][y+1] == 0:
                        queue.append((x, y+1))

    return grid
```

We repeat this process until the queue is empty.
