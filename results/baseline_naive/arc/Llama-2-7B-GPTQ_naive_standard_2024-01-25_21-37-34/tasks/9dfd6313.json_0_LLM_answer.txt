class Solution(object):
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        """

        nums = sorted(nums)

            countDict[nums[i]] += 1

                    print("{} is repeated".format(newRes[i]))
                    return None
                    
        return res
