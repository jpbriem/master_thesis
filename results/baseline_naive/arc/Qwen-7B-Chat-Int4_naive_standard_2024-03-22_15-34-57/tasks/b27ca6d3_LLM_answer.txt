    result = []
    for i in range(len(arr)):
        if arr[i] == 'a':
            if len(result) > 0 and result[-1][0] != 'a':
                result.append(['a'])
            result.append([arr[i]])
        else:
            result[-1].append(arr[i])
    return result

def convert_to_one_hot(arr, max_val=3):
    one_hot_arr = np.zeros((len(arr), max_val))
    for i in range(len(arr)):
        one_hot_arr[i][int(arr[i])] = 1
    return one_hot_arr
