def group_lists(input_list):
    groups = []
    for sublist in input_list:
        if len(set(sublist)) == 1:
            groups.append([sublist[0]])
        else:
            # If there are two or more unique elements, create a new group
            group = [sublist[0]]
            for i in range(1, len(sublist)):
                if sublist[i] != group[-1]:
                    group.append(sublist[i])
            groups.append(group)
    return groups
