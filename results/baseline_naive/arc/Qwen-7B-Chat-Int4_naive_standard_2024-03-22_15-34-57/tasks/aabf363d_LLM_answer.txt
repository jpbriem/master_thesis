class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if not nums:
            return []
        
        res = []
        nums.sort()
        self.permuteHelper(nums, 0, len(nums)-1, res)
        return res
    
    def permuteHelper(self, nums, start, end, res):
        if start == end:
            res.append([nums[i] for i in range(start, end+1)])
        else:
            for i in range(start, end+1):
                temp = nums[start:i]
                self.permuteHelper(nums, start+1, end, res)
                temp.reverse()
                self.permuteHelper(nums, start, end-1, res)
                nums[start:end+1] = temp
