```python
class Solution:
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        if not nums:
            return []
        
        res = [[nums[0]]]
        
        for i in range(1, len(nums)):
            # add current number to the beginning of each element in the previous result
            for j in range(len(res) - 1, -1, -1):
                res[j].insert(0, nums[i])
            
            # append current number to the end of each element in the previous result
            for j in range(len(res)):
                res.append(nums[i] + res[j][len(res[j]) - 1])
                
        return res
```
