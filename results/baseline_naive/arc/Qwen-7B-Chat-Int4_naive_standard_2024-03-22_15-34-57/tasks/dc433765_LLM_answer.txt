```python
class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        d = {}
        
        for s in strs:
            if len(s) == 0:
                continue
            
            # sort the string and convert to tuple
            sorted_s = ''.join(sorted(s))
            
            # add tuple as key into dictionary
            if sorted_s not in d:
                d[sorted_s] = []
                
            # append string to list of tuples with same key
            d[sorted_s].append(s)
        
        result = []
        for k, v in d.items():
            result.append(v)
        
        return result
```
