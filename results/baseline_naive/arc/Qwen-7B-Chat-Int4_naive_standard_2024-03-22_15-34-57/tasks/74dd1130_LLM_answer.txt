class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        d = {}
        for s in strs:
            k = "".join(sorted(s))
            if k not in d:
                d[k] = []
            d[k].append(s)
        ans = []
        for v in d.values():
            ans.append(v[0])
            for i in range(1, len(v)):
                ans.append(v[i])
        return ans
