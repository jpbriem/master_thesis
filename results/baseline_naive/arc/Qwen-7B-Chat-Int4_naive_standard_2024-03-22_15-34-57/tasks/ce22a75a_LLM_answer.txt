class Solution:
    def findLadders(self, start: List[str], end: List[str]) -> List[List[int]]:
        if len(start) != len(end):
            return []
        
        d = {}
        for i in range(len(start)):
            d[start[i]] = []
            
        queue = collections.deque([(0, 0)])
        while queue:
            dist, node = queue.popleft()
            if node[0] == len(start):
                break
            
            for i in range(97, 123): # a-z
                if node[1] + chr(i - 97) not in d or (node[0]+1, node[1]+chr(i-97)) not in d:
                    continue
                
                new_node = [node[0]+1, node[1]+chr(i-97)]
                
                if new_node[0] < len(start) and start[new_node[0]] == end[end[0]-new_node[0]]:
                    queue.append((dist+1, new_node))
                    
                if new_node[1] < len(start) and start[new_node[1]] == end[end[1]-new_node[1]]:
                    queue.append((dist+1, new_node))
                    
        res = []
        while queue:
            dist, node = queue.popleft()
            res.append(node)
            
        return res[::-1]
