class Solution:
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        result = []
        
        for i in range(len(strs)):
            anagram_set = set()
            
            for j in range(i+1, len(strs)):
                if sorted(strs[i]) == sorted(strs[j]):
                    anagram_set.add(sorted(strs[j]))
                
            result.append(list(anagram_set))
        
        return result
