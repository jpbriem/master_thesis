The idea is that we start from the first `5`, then find all its neighbors which have value `0`. Then mark them as visited and add their neighbors into queue. Repeat until there's no more unvisited node in the graph.

class Solution {
public:
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int newColor) {
        if (image[sr][sc] == newColor) return image;
        queue<pair<int, int>> q;
        q.push({sr, sc});
        while (!q.empty()) {
            auto cur = q.front(); q.pop();
            for (auto &p : getNeighbors(cur)) {
                if (image[p.first][p.second]!= newColor && image[p.first][p.second]!= 0) continue;
                image[p.first][p.second] = newColor;
                q.push(p);
            }
        }
        return image;
    }
private:
    vector<pair<int, int>> getNeighbors(const pair<int, int> &pos) const {
        vector<pair<int, int>> res;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                res.emplace_back(pos.first + i, pos.second + j);
            }
        }
        return res;
    }
};
