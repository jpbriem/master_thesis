class Solution {
public:
    vector<vector<int>> rotateGrid(vector<vector<int>>& grid) {
        int n = grid.size();
        for (int i=0; i < n/2; ++i){
            for (int j=i; j < n - i - 1; ++j){
                swap(grid[i][j], grid[n-j-1][n-i-1]);
                swap(grid[i][j], grid[n-i-1][j]);
                swap(grid[i][j], grid[j][n-i-1]);
            }
        }
        return grid;
    }
};
