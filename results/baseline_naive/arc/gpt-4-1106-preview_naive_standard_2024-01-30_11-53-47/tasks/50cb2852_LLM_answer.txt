def fill_inner(matrix):
    rows, cols = len(matrix), len(matrix[0])

    def dfs(x, y, target):
        if x < 0 or x >= rows or y < 0 or y >= cols or matrix[x][y] != target:
            return
        matrix[x][y] = 'i'
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        for dx, dy in directions:
            dfs(x + dx, y + dy, target)

            if matrix[x][y] == matrix[x - 1][y] == matrix[x + 1][y] == matrix[x][y - 1] == matrix[x][y + 1]:
                dfs(x, y, matrix[x][y])
