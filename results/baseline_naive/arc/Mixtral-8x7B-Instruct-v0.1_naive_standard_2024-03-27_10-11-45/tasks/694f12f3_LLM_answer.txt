function solution(board) {
    const dx = [-1, 1, 0, 0];
    const dy = [0, 0, -1, 1];

    function dfs(x, y) {
        if (board[x][y] === "x" || board[x][y] === ".") return false;
        if (board[x][y] === "a") return true;

        board[x][y] = "a";
        for (let i = 0; i < 4; i++) {
            const nx = x + dx[i];
            const ny = y + dy[i];
            if (nx >= 0 && nx < board.length && ny >= 0 && ny < board[x].length) {
                if (dfs(nx, ny)) return true;
            }
        }
        board[x][y] = ".";
        return false;
    }

    for (let i = 0; i < board.length; i++) {
        for (let j = 0; j < board[i].length; j++) {
            if (board[i][j] === "d") {
                if (dfs(i, j)) continue;
                else board[i][j] = ".";
            }
        }
    }

    return board;
}
