function bfs(grid) {
    let queue = [[0, 0]],
        dirs = [[-1, 0], [0, -1], [1, 0], [0, 1]],
        m = grid.length,
        n = grid[0].length;

    while (queue.length > 0) {
        const cur = queue.shift();
        if (grid[cur[0]] && grid[cur[0]][cur[1]] === ".") continue;

        for (const dir of dirs) {
            const x = cur[0] + dir[0];
            const y = cur[1] + dir[1];
            if (x < 0 || x >= m || y < 0 || y >= n) continue;
            if (!grid[x]) continue;
            if (grid[x][y]!== ".") {
                grid[x][y] = grid[cur[0]][cur[1]] === "W"? "D" : "S";
                queue.push([x, y]);
            }
        }
    }
}

function numIslands2(grid) {
    if (!grid.length) return [];

    const res = [],
        m = grid.length,
        n = grid[0].length;

    function dfs(i, j) {
        if (i < 0 || i >= m || j < 0 || j >= n ||!grid[i] || grid[i][j] === "D") return;

        grid[i][j] = "D";
        dfs(i - 1, j);
        dfs(i + 1, j);
        dfs(i, j - 1);
        dfs(i, j + 1);
    }

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (grid[i][j] === ".") {
                bfs(grid);
                res.push([]);
                for (let k = 0; k < m; k++) {
                    res[res.length - 1].push("");
                    for (let l = 0; l < n; l++) {
                        if (grid[k][l] === "S") {
                            dfs(k, l);
                            res[res.length - 1][res[res.length - 1].length - 1] += "S";
                        } else if (grid[k][l] === "D") {
                            res[res.length - 1][res[res.length - 1].length - 1] += "D";
                        } else {
                            res[res.length - 1][res[res.length - 1].length - 1] += ".";
                        }
                    }
                }
            }
        }
    }

    return res;
}
