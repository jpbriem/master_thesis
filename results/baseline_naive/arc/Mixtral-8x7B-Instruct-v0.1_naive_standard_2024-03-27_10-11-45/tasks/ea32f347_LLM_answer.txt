Here's how I would do this:

def getPaths(grid):
    def dfs(x, y):
        if x == len(grid) or y == len(grid[0]) or grid[x][y] == "#":
            return []
        elif x == 0 and y == 0:
            return [[grid[x][y]]]
        else:
            res = []
            up = dfs(x - 1, y)
            down = dfs(x + 1, y)
            left = dfs(x, y - 1)
            right = dfs(x, y + 1)
            for u in up:
                temp = [grid[x][y]] + u
                res.append(temp)
            for d in down:
                temp = [grid[x][y]] + d
                res.append(temp)
            for l in left:
                temp = [grid[x][y]] + l
                res.append(temp)
            for r in right:
                temp = [grid[x][y]] + r
                res.append(temp)
            return res

    paths = dfs(len(grid) - 1, len(grid[0]) - 1)
    newGrid = [["."] * len(grid[0]) for _ in range(len(grid))]
    for p in paths:
        i = 0
        while i < len(p) and p[i]!= ".":
            j = 0
            while j < len(newGrid) and newGrid[j][i]!= "e":
                j += 1
            if j < len(newGrid):
                newGrid[j][i] = "a"
            i += 1
    return newGrid
