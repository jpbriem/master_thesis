class Solution {
    public List<List<Character>> updateBoard(char[][] board, int[] click) {
        if (board[click[0]][click[1]] == 'M') return new ArrayList<>(Arrays.asList(board));
        dfs(board, click);
        return new ArrayList<>(Arrays.asList(board));
    }

    private void dfs(char[][] board, int[] pos) {
        int x = pos[0];
        int y = pos[1];
        if (x < 0 || x >= board.length || y < 0 || y >= board[0].length) return;
        if (board[x][y]!= 'E' && board[x][y]!= 'B') return;
        int count = getCount(board, x, y);
        if (count > 0) {
            board[x][y] = (char)(count + '0');
            return;
        } else {
            board[x][y] = 'B';
            for (int i = -1; i <= 1; ++i) {
                for (int j = -1; j <= 1; ++j) {
                    if (!(i == 0 && j == 0)) {
                        dfs(board, new int[]{x+i, y+j});
                    }
                }
            }
        }
    }

    private int getCount(char[][] board, int x, int y) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if ((i == 0 && j == 0) ||!(x+i >= 0 && x+i < board.length && y+j >= 0 && y+j < board[0].length)) continue;
                if (board[x+i][y+j] == 'M') count++;
            }
        }
        return count;
    }
}
