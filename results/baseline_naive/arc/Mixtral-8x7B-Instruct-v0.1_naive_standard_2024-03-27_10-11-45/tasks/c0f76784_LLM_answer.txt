  const m = board.length;
  const n = board[0].length;

  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (board[i][j] === ".") continue;

      let count = 0;
      // check top, bottom, left and right neighbors
      if (i > 0 && board[i - 1][j] === "M") count++;

      if (j > 0 && board[i][j - 1] === "M") count++;
      if (j + 1 < n && board[i][j + 1] === "M") count++;

      if (count === 0) {
        board[i][j] = "B";
      } else if (count <= 9) {
        board[i][j] = String(count);
      }
    }
  }
