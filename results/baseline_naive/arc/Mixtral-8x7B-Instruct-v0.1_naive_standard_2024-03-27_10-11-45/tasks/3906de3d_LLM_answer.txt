class Solution {
    public List<List<String>> solve(List<List<String>> board) {
        if (board == null || board.size() < 3 || board.get(0).size() < 3) return board;
        int m = board.size(), n = board.get(0).size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if ((i == 0 || i == m - 1 || j == 0 || j == n - 1) && board.get(i).get(j).equals("O")) {
                    dfs(board, i, j);
                }
            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (!board.get(i).get(j).equals("X") &&!board.get(i).get(j).equals("#")) {
                    board.get(i).set(j, "O");
                } else if (board.get(i).get(j).equals("O")) {
                    board.get(i).set(j, "#");
                }
            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board.get(i).get(j).equals("#")) {
                    board.get(i).set(j, "O");
                }
            }
        }
        return board;
    }

    private void dfs(List<List<String>> board, int x, int y) {
        if (x < 0 || x >= board.size() || y < 0 || y >= board.get(0).size()) return;
        if (board.get(x).get(y).equals("X") || board.get(x).get(y).equals("#")) return;
        board.get(x).set(y, "#");
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
}
