public class Solution {
    public char[][] convert(char[][] board) {
        if (board == null || board.length == 0) return new char[0][];
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; ++i) {
            boolean isFirstE = true;
            for (int j = 0; j < n; ++j) {
                if (board[i][j]!= 'e') continue;
                if (isFirstE) {
                    dfs(board, i, j, m, n);
                    isFirstE = false;
                } else {
                    board[i][j] = 'o';
                }
            }
        }
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 'o') board[i][j] = 'x';
                else if (board[i][j] == '#') board[i][j] = 'o';
            }
        }
        return board;
    }

    private void dfs(char[][] board, int x, int y, int m, int n) {
        if (x < 0 || x >= m || y < 0 || y >= n || board[x][y]!= 'e') return;
        board[x][y] = '#';
        dfs(board, x + 1, y, m, n);
        dfs(board, x - 1, y, m, n);
        dfs(board, x, y + 1, m, n);
        dfs(board, x, y - 1, m, n);
    }
}
