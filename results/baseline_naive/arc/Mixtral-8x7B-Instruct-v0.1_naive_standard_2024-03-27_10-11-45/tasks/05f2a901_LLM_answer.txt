    def find_b(row, col):
        for r in range(len(board)):
            if board[r][col] == 'b':
                return r, col
        return -1, -1

    def find_h(row, col):

            if board[row][c] == 'h':
                return row, c
        return -1, -1

    # Find all bs and hs on the board
    bs = []
    hs = []
    for i in range(len(board)):
        for j in range(len(board[i])):
            if board[i][j] == 'b':
                bs.append((i, j))
            elif board[i][j] == 'h':
                hs.append((i, j))

    # Check each b against each h
    for b in bs:
        br, bc = b
        for h in hs:
            hr, hc = h
            if abs(br-hr) + abs(bc-hc) > 1:
                # Move the b towards the h
                dr = hr - br
                dc = hc - bc

                    new_r = br
                if dc!= 0:
                    new_c = bc + dc // abs(dc)
                else:
                    new_c = bc
                board[new_r][new_c] = 'b'

    # No more movement possible
