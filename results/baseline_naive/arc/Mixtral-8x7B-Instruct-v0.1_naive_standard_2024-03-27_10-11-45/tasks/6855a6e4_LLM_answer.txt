const removeIslands = (grid) => {
  const m = grid.length;
  if (!m) return [];
  const n = grid[0].length;
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      // If the current cell is a land, start DFS from it to mark all its connected lands as visited
      if (grid[i][j] === ".") continue;
      dfs(grid, i, j);
    }
  }
  return grid;
};

const dfs = (grid, x, y) => {
  const m = grid.length;
  const n = grid[0].length;
  if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y]!== ".") return;
  grid[x][y] = "#";
  dfs(grid, x + 1, y);
  dfs(grid, x - 1, y);
  dfs(grid, x, y + 1);
  dfs(grid, x, y - 1);
};
