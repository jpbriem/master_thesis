def solve(i, j, m, n, grid, ans):
    # base condition when i == len(grid)-1 and j==len(grid[0])-1
    if i == m - 1 and j == n - 1 and grid[i][j]!= 0:
        ans.append([grid[i][j]])
        return True

    if i >= m or j >= n or grid[i][j] == 0:
        return False

    temp = []
    # move down
    if i + 1 < m and solve(i+1, j, m, n, grid, temp):
        temp.append(grid[i][j])
        ans.append(temp)
        return True

    # move right
    if j + 1 < n and solve(i, j+1, m, n, grid, temp):
        temp.append(grid[i][j])
        ans.append(temp)
        return True

    return False
