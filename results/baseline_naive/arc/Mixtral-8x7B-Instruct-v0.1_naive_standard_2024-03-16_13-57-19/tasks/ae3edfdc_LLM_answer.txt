import numpy as np

def fill(arr):
    idx = np.argwhere(arr!= 0)
    return arr.copy().astype(bool).astype(int) \
       .where((~np.any(idx[:, None] == idx, axis=0)) & (~np.any(idx[None, :] == idx, axis=1)), 0)
