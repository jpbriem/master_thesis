class Solution {
public:
    vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
        int m = mat.size(), n = mat[0].size();
        queue<pair<int, int>> q;
        for (int i=0; i<m; ++i)
            for (int j=0; j<n; ++j)
                if (!mat[i][j])
                    q.push({i, j});
        while (!q.empty()) {
            auto p = q.front();
            q.pop();
            int x = p.first, y = p.second;
            if (x > 0 &&!mat[x-1][y]) {
                mat[x-1][y] = mat[x][y]+1;
                q.push({x-1, y});
            }
            if (x < m-1 &&!mat[x+1][y]) {
                mat[x+1][y] = mat[x][y]+1;
                q.push({x+1, y});
            }
            if (y > 0 &&!mat[x][y-1]) {
                mat[x][y-1] = mat[x][y]+1;
                q.push({x, y-1});
            }
            if (y < n-1 &&!mat[x][y+1]) {
                mat[x][y+1] = mat[x][y]+1;
                q.push({x, y+1});
            }
        }
        return mat;
    }
};
