void rotate(int** arr, int n) {
    for (int i = 0; i < n / 2; ++i) {
        for (int j = i; j < n - i - 1; ++j) {
            // save the current element in temp variable
            int temp = arr[i][j];

            arr[i][j] = arr[n-j-1][i];

            // move elements from left to bottom
            arr[n-i-1][n-j-1] = arr[j][n-i-1];

            // assign saved value to left

        }
    }
}
