```
def numIslands(grid):
    if len(grid) == 0: return []
    m = len(grid)
    n = len(grid[0])
    vis = [[False]*n for _ in range(m)]
    ans = [[-1]*n for _ in range(m)]
    d = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    def bfs(i, j):
        q = deque()
        q.append((i, j))
        grid[i][j] = 'v'
        while q:
            i, j = q.popleft()
            vis[i][j] = True
            for di, dj in d:
                ni, nj = i+di*grid[i][j], j+dj*grid[i][j]
                if 0 <= ni < m and 0 <= nj < n and grid[ni][nj]!= 'x':
                    if not vis[ni][nj]:
                        q.append((ni, nj))
                        grid[ni][nj] = 'v'

    for i in range(m):
        for j in range(n):
            if grid[i][j] == '1' and not vis[i][j]:
                cnt = 0
                bfs(i, j)
                for k in range(m):
                    for l in range(n):
                        if vis[k][l]:
                            cnt += 1
                for k in range(m):
                    for l in range(n):
                        if grid[k][l] == '1' and not vis[k][l]:
                            ans[k][l] = cnt

    for i in range(m):
        for j in range(n):
            if ans[i][j] == -1:
                ans[i][j] = grid[i][j]

    return ans
