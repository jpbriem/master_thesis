    vector<vector<char>> rotateTheBox(vector<vector<char>>& box) {
        int m = box.size(), n = box[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = n - 2, k = n - 1; j >= 0 && box[i][j]!= '#'; --j) {
                if (box[i][j] == '*') {
                    swap(box[i][k], box[i][j]);
                    --k;
                }
            }
        }
        vector<vector<char>> ans(n, vector<char>(m));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                ans[j][m-1-i] = box[i][j];
            }
        }
        return ans;
    }
