public class Solution {
    public int[][] shiftGrid(int[][] grid, int k) {
        if (grid == null || grid.length == 0 || grid[0].length == 0) return new int[0][];
        int m = grid.length;
        int n = grid[0].length;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                int pos = (i * n + j + k) % (m * n);
                grid[(pos / n)][pos % n] = grid[i][j];
            }
        }
        return grid;
    }
}
