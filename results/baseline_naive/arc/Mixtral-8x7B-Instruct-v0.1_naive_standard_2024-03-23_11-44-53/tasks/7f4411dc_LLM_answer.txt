    vector<vector<char>> rotateTheBox(vector<vector<char>>& box) {
        int m = box.size(), n = box[0].size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0, k = n - 1; j <= k; ++j) {
                if (box[i][j] == '#') {
                    while (k >= j && box[i][k]!= '*') --k;
                    swap(box[i][j], box[i][k]);
                }
            }
        }
        vector<vector<char>> ans(n, vector<char>(m));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                ans[j][m-1-i] = box[i][j];
            }
        }
        return ans;
    }
};

    vector<vector<char>> rotateTheBox(vector<vector<char>>& box) {
        int m = box.size(), n = box[0].size();
        // first step: move all the * to right side of each row
        for (auto &row : box) {
            int l = 0, r = n - 1;
            while (l < r) {
                if (row[r] == '*') {
                    while (l < r && row[l]!= '*') ++l;
                    swap(row[l], row[r]);
                }
                --r;
            }
        }
        // second step: reverse rows and columns
        auto res = vector<vector<char>>(n, vector<char>(m));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                res[j][m - 1 - i] = box[i][j];
            }
        }
        return res;
    }
