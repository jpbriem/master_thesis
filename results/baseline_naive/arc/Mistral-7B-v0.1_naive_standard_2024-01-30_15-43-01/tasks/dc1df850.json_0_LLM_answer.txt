public class Solution {
    public String[][] updateBoard(String[][] board, int[] click) {
        if (board == null || board.length == 0 || board[0].length == 0) return board;
        int m = board.length, n = board[0].length;
        boolean[][] visited = new boolean[m][n];
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j].equals("M")) continue;
                if (i == click[0] && j == click[1]) {
                    board[i][j] = "X";
                    queue.offer(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            int x = cur[0], y = cur[1];
            int count = 0;
            for (int dx = -1; dx <= 1; ++dx) {
                for (int dy = -1; dy <= 1; ++dy) {
                    if (dx == 0 && dy == 0) continue;
                    int nextx = x + dx, nexty = y + dy;
                    if (nextx >= 0 && nextx < m && nexty >= 0 && nexty < n &&!visited[nextx][nexty] && board[nextx][nexty].equals("E")) {
                        ++count;
                    }
                }
            }
            if (count > 0) {
                board[cur[0]][cur[1]] = Integer.toString(count);

                board[cur[0]][cur[1]] = "B";
