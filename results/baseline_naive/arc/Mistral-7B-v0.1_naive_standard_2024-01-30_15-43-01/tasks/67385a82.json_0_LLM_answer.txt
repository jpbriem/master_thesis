public class Solution {
    public String[][] updateBoard(String[][] board, int[] click) {
        if (board == null || board.length == 0 || board[0].length == 0) return new String[0][];
        int m = board.length;
        int n = board[0].length;
        boolean[][] visited = new boolean[m][n];
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j].equals("M")) continue;
                if (i == click[0] && j == click[1]) {
                    board[i][j] = "B";
                    queue.offer(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            int x = cur[0], y = cur[1];
            int count = getCount(x, y, board);
            if (count > 0) {
                board[x][y] = Integer.toString(count);
                for (int i = -1; i <= 1; ++i) {
                    for (int j = -1; j <= 1; ++j) {
                        if (i == 0 && j == 0) continue;
                        int nextX = x + i, nextY = y + j;
                        if (nextX >= 0 && nextX < m && nextY >= 0 && nextY < n &&!visited[nextX][nextY]) {
                            if (board[nextX][nextY].equals("E")) {
                                board[nextX][nextY] = "B";
                                queue.offer(new int[]{nextX, nextY});
                                visited[nextX][nextY] = true;
                            } else if (board[nextX][nextY].equals("M")) {
                                board[nextX][nextY] = "X";
                                queue.offer(new int[]{nextX, nextY});
                                visited[nextX][nextY] = true;
                            }
                        }
                    }
                }
            }
        }
        return board;
    }

    private int getCount(int x, int y, String[][] board) {
        int count = 0;
        for (int i = -1; i <= 1; ++i) {
            for (int j = -1; j <= 1; ++j) {
                if (i == 0 && j == 0) continue;
                int nextX = x + i, nextY = y + j;
                if (nextX >= 0 && nextX < board.length && nextY >= 0 && nextY < board[0].length) {
                    if (board[nextX][nextY].equals("M")) ++count;
                }
            }
        }
        return count;
    }
}
