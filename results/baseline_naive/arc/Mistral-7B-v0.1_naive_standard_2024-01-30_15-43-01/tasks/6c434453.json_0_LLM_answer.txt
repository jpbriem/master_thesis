class Solution(object):
    def findBall(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: List[int]
        """
        m = len(grid)
        n = len(grid[0])
        res = []
        for i in range(n):
            x = y = 0
            while True:
                if (x == m - 1 or y == n - 1) and grid[x][y]!= i:
                    break
                elif grid[x][y] == i:
                    x += 1
                    y += grid[x][y]
                else:
                    break
            res.append(y)
        return res
