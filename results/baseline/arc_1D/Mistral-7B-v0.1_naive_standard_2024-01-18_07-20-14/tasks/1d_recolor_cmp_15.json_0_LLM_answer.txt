public class Solution {
    public String[] updateBoard(String[] board, int[] click) {
        if (board == null || board.length == 0) return new String[]{};
        int m = board.length;
        int n = board[0].length();
        boolean[][] visited = new boolean[m][n];
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i].charAt(j) == 'M') continue;
                if (i == click[0] && j == click[1]) {
                    board[i] = board[i].substring(0, j) + "X" + board[i].substring(j+1);
                    queue.offer(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            int x = cur[0], y = cur[1];
            int count = 0;
            for (int dx = -1; dx <= 1; ++dx) {
                for (int dy = -1; dy <= 1; ++dy) {
                    if (dx == 0 && dy == 0) continue;
                    int nextx = x + dx, nexty = y + dy;
                    if (nextx >= 0 && nextx < m && nexty >= 0 && nexty < n &&!visited[nextx][nexty] && board[nextx].charAt(nexty) == 'E') {
                        ++count;
                    }
                }
            }
            if (count > 0) {
                board[cur[0]] = board[cur[0]].substring(0, cur[1]) + String.valueOf(count) + board[cur[0]].substring(cur[1]+1);
                queue.addAll(getNeighbors(board, cur));
            } else {
                board[cur[0]] = board[cur[0]].substring(0, cur[1]) + "B" + board[cur[0]].substring(cur[1]+1);
            }
        }
        return board;
    }

    private List<int[]> getNeighbors(String[] board, int[] cur) {
        List<int[]> neighbors = new ArrayList<>();
        int x = cur[0], y = cur[1];
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                if (dx == 0 && dy == 0) continue;
                int nextx = x + dx, nexty = y + dy;
                if (nextx >= 0 && nextx < board.length && nexty >= 0 && nexty < board[0].length() && board[nextx].charAt(nexty) == 'E') {
                    neighbors.add(new int[]{nextx, nexty});
                }
            }
        }
        return neighbors;
    }
