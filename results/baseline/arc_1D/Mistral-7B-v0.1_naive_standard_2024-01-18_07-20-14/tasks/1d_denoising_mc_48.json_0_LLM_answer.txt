    public List<String> removeInvalidParentheses(String s) {

        int right = 0;
        for (char c : s.toCharArray()) {

                ++left;

            }

                queue.offer(new int[]{index + 1, count - 1});
            }
        }
        return new ArrayList<>();
    }

    private boolean isValid(char[] chars) {

        for (char c : chars) {
