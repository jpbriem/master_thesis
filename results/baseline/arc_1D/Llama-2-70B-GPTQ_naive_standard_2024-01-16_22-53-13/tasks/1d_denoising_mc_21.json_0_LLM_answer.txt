class Solution:
    def islandPerimeter(self, grid: List[List[str]]) -> int:
        """
        思路：扫描每个格子，如果是海洋则计算周围的陆地数量。
            - 对于边界上的格子，只需要考虑一侧的邻居；
            - 对于内部的格子，则需要考虑四侧的邻居。
            最后将所有格子的结果相加得到答案。

        时间复杂度：O(n)，其中 n 为输入的长度。
        空间复杂度：O(1)。
        """
        # 记录每个格子的邻居数量
        perimeters = [[0 for _ in range(len(grid))] for _ in range(len(grid))]
        for i in range(len(perimeters)):
            for j in range(len(perimeters[i])):
                if grid[i][j] == "." or (i > 0 and grid[i - 1][j]!= ".") \
                        or (j > 0 and grid[i][j - 1]!= ".") \
                        or (i < len(grid) - 1 and grid[i + 1][j]!= ".") \
                        or (j < len(perimeters[i]) - 1 and grid[i][j + 1]!= "."):
                    perimeters[i][j] += 1
        return sum(sum(row) for row in perimeters)
