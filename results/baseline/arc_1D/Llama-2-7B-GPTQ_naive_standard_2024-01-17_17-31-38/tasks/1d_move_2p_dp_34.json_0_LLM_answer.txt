class Solution(object):
    def removeDuplicates(self, nums: List[int]) -> int:
        """
        :type nums: List[int]
        :rtype: int
        """
        if not nums or len(nums) == 0:
            return 0
        
        # 暴力解法，时间复杂度为 O(n^2), 空间复杂度为 O(n)。
        # 遍历数组，判断当前元素是否已经存在于列表中，如果不存在则添加到列表中，反之将其移除。
        for i in range(len(nums)):

                else:

                nums.pop(j - 1)
