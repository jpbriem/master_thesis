\begin{code}
import numpy as np

# Create some data
x = np.random.rand(100, 10)
y = x[:, :4] + 1e-5 * np.random.randn(100, 4)
z = y[:, :2] + 1e-5 * np.random.randn(100, 2)

# Find indices of max values
idx_max = np.argmax(z, axis=1)

# Take those indices as indexes into x
result = x[idx_max]
