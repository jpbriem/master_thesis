def get_description(description):
    instructions = []
    dimensions = []
    for line in description.split('\n'):
        parts = line.strip().split(' ')
        if len(parts) == 0: continue
        if parts[0].startswith('Input'):
            instructions.append('Input')
            dimensions.append(len(parts))
        elif parts[0].startswith('Output'):
            instructions.append('Output')
            dimensions.append(len(parts))
        else:
            instructions.append(parts[0])
    return instructions, dimensions

def transform_grid(grid, instructions):
    result = [[None]*len(grid[0]) for _ in range(len(grid))]
    for i, instruction in enumerate(instructions):
        if instruction == 'Input':
            print('Input:')
            print(grid)
        elif instruction == 'Output':
            print('Output:')
            print(result)
        else:
            if instruction == 'Fill':
                for j in range(len(grid[0])):
                    for k in range(len(grid)):
                        if grid[k][j] == 'b':
                            result[k][j] = 'b'
            print('Fill blue')
            print(result)
        elif instruction == 'Transform':
            for j in range(len(grid[0])):
                for k in range(len(grid)):
                    if grid[k][j] == 'b':
                        if k % 2 == 0:
                            result[k+1][j] = 'b'
                        else:
                            result[k+1][j] = 'c'
            print('Transform blue')
            print(result)
    return result
