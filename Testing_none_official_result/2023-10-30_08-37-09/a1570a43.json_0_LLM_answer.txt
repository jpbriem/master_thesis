```
df = pd.DataFrame({'A': [1, 2, 3], 'B': ['a', 'b', 'c']})

def f(x):
    return x['A'].sum(), x['B'].count()

result = df.groupby('A')[['B']].apply(f).reset_index()
print(result)
```
