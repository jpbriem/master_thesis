```

# create some data
data = np.array([[7, 7, 0, 1, 5, 0, 0], [0, 0, 0, 1, 0, 0, 0], [7, 7, 0, 1, 5, 0, 0]])

# replace zeros with -1
data = np.where(data == 0, -1, data)

print(data)
