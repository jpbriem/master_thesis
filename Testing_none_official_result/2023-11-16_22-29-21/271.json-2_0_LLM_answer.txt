class Solution {
    public String[] findTransformedArray(String[] arr, int[][] queries) {
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < arr.length; ++i) {
            char ch = arr[i].charAt(0);
            map.putIfAbsent(ch, i + 1);
        }
        List<Integer> list = new ArrayList<>(map.values());
        Collections.sort(list);
        int n = list.size();
        int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            ans[i] = list.get(i);
        }
        return Arrays.stream(ans).boxed().toArray(Integer[]::new);
    }
}
