class Solution {
    public String[][] generate(String[][] inputGrid, String description) {
        // Write your code here

		int rows = inputGrid.length;
		int columns = inputGrid[0].length;
		char[][] charArray = new char[rows][columns];
		for(int i=0; i<rows; i++){
			for(int j=0; j<columns; j++){
				charArray[i][j] = inputGrid[i][j].charAt(0);
			}
		}
		// Description
		String[] descArr = description.split(" ");
		String inputGridStr = "";
		for(int i=0; i<rows; i++){
			for(int j=0; j<columns; j++){
				if(charArray[i][j] == 'a'){
					inputGridStr += " ";
				}else{
					inputGridStr += charArray[i][j];
				}
			}
			inputGridStr += "\n";
		}
		System.out.println(inputGridStr);
		String instructions = "";
		for(int i=0; i<descArr.length; i+=4){
			instructions += descArr[i] + ": " + descArr[i+1] + ", " + descArr[i+2] + ", " + descArr[i+3] + "\n";
		}
		System.out.println(instructions);
		String outputDimension = "";
		for(int i=0; i<descArr.length; i+=8){
			outputDimension += descArr[i] + ": " + descArr[i+1] + ", " + descArr[i+2] + ", " + descArr[i+3] + ", " + descArr[i+4] + ", " + descArr[i+5] + ", " + descArr[i+6] + ", " + descArr[i+7] + "\n";
		}

				}else{
					testOutput += " ";
				}
			}
			testOutput += "\n";
		}
		System.out.println(testOutput);
		return null;
    }
}
