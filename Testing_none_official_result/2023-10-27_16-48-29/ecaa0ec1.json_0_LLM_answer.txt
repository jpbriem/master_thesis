```
## C[edit]

This is a simple implementation that uses the `rand()` function to generate random numbers for each cell in the grid. It then prints out the grid and asks the user if they want to continue or not. If yes, it generates another grid and continues until the user says no.

```
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void) {
    srand((unsigned int) time(NULL));
    int i, j;
    int grid[5][6];
    int testgrid[5][6];
    int answer = 1;
    while (answer == 1) {
        printf("Random Grid\n");
        for (i=0; i<5; i++) {
            for (j=0; j<6; j++) {
                grid[i][j] = rand();
                printf("%d ", grid[i][j]);
            }
            printf("\n");
        }
        printf("Test Random Grid\n");
        for (i=0; i<5; i++) {
            for (j=0; j<6; j++) {
                testgrid[i][j] = grid[i][j];
                printf("%d ", testgrid[i][j]);
            }
            printf("\n");
        }
        printf("Continue? (y/n): ");
        scanf("%c", &answer);
    }
}
