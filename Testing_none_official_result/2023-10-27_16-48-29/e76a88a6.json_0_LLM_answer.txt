```
## C[edit]

This is a simple implementation that uses the `rand()` function to generate random numbers between 0 and 9. It also uses the `srand(time(NULL))` function to seed the random number generator with the current time. The program takes two command line arguments: the first one is the number of rows in the matrix, and the second one is the number of columns in the matrix.

```
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main ( int argc, char *argv[] ) {
    if (argc!= 3) {
        fprintf(stderr,"Usage: %s N M\n", argv[0]);
        return EXIT_FAILURE;
    }
    srand(time(NULL));
    int n = atoi(argv[1]), m = atoi(argv[2]);
    int **A = calloc(n, sizeof(*A)), **B = calloc(m, sizeof(*B));
    for (int i=0; i<n; ++i) A[i] = calloc(m, sizeof(**A));
    for (int i=0; i<m; ++i) B[i] = calloc(n, sizeof(**B));
    for (int i=0; i<n; ++i)
        for (int j=0; j<m; ++j)
            A[i][j] = rand()%10;
    for (int i=0; i<m; ++i)
        for (int j=0; j<n; ++j)
            B[i][j] = rand()%10;
    printf("matrix A:\n");
    for (int i=0; i<n; ++i) {
        for (int j=0; j<m; ++j)
            printf("%d ", A[i][j]);
        putchar('\n');
    }
    printf("\nmartix B:\n");
    for (int i=0; i<m; ++i) {
        for (int j=0; j<n; ++j)
            printf("%d ", B[i][j]);
        putchar('\n');
    }
    free(A);
    free(B);
    return EXIT_SUCCESS;
}
