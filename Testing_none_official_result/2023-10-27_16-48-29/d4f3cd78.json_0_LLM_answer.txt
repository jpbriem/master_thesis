```
## C[edit]

This is a simple implementation that uses the `rand()` function to generate random numbers for each cell in the grid. It then checks if any cells are adjacent and adds them together. If they sum up to 10 or more, it sets the value to 10. Otherwise, it sets the value to zero. The program prints out the resulting matrix.

```
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void) {
    srand((unsigned int) time(NULL)); // seed rand with current time
    const size_t rows = 3;
    const size_t cols = 4;
    int mat[rows][cols];
    for (size_t i = 0; i < rows; ++i) {
        for (size_t j = 0; j < cols; ++j) {
            mat[i][j] = rand();
            printf("%d ", mat[i][j]);
        }
        putchar('\n');
    }
    for (size_t i = 0; i < rows; ++i) {
        for (size_t j = 0; j < cols; ++j) {
            for (size_t k = 0; k < rows; ++k) {
                for (size_t l = 0; l < cols; ++l) {
                    if ((mat[i][j] + mat[k][l]) >= 10) {
                        mat[i][j] = 10;
                    } else {
                        mat[i][j] = 0;
                    }
                }
            }
        }
    }
    for (size_t i = 0; i < rows; ++i) {
        for (size_t j = 0; j < cols; ++j) {
            printf("%d ", mat[i][j]);
        }
        putchar('\n');
    }
}
